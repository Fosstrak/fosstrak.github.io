51  /**
52   * A NotificationChannel is used to report notifications to a single host. It
53   * has a list of Sources associated with it. Only events coming from these
54   * Sources shall be reported. It also has a DataSelector that determines what
55   * data should be reported to the host. If no DataSelector is explicitly
56   * associated, then the Readers current DataSelector shall be used.
57   * Notification shall be sent whenever the associated Trigger fires.
58   * Alternatively, a host can also query the NotificationChannel for its
59   * contents, using the command NotificationChannel.readQueuedData().
60   * @author Markus Vitalini
61   */
62  public final class NotificationChannel {

171    /**
172     * Create a NotificationChannel object with a given name. If a
173     * NotificationChannel object with the same name exists already, an error is
174     * returned. This is a static method. The NotificationChannel will implicitly
175     * be added to the list of all NotificationChannels kept by the ReaderDevice
176     * object.
177     * @param name
178     *           The name of the notification channel
179     * @param addr
180     *           The host address
181     * @param readerDevice
182     *           The reader device this channel belongs to
183     * @return The instance of the new NotificationChannel
184     * @throws ReaderProtocolException
185     *            The ReaderProtocolException "ERROR_OBJECT_EXISTS" is thrown
186     */
187    public static NotificationChannel create(final String name,
188          final String addr, final ReaderDevice readerDevice)
189          throws ReaderProtocolException {
190 
191       // check if NotificationChannel with the same name exists
192       try {
193          readerDevice.getNotificationChannel(name);
194       } catch (ReaderProtocolException e) {
195          // create new NotificationChannel
196          NotificationChannel newNotificationChannel = new NotificationChannel(
197                name, addr, readerDevice);
198          readerDevice.getNotificationChannels().put(name,
199                newNotificationChannel);
200          return newNotificationChannel;
201       }
202 
203       throw new ReaderProtocolException("ERROR_OBJECT_EXISTS",
204             MessagingConstants.ERROR_OBJECT_EXISTS);
205    }

392    /**
393     * Adds the specified Triggers to the list of Notification Triggers currently
394     * associated with this NotificationChannel. If some of the Triggers to be
395     * added are already associated with this NotificationChannel, only the not
396     * yet associated Triggers shall be added and the command completes
397     * successfully. Once a Trigger has been added, it is immediately activated.
398     * @param triggerList
399     *           The triggers to add
400     * @throws ReaderProtocolException
401     *            "ERROR_TOO_MANY_TRIGGERS"
402     */
403    public void addNotificationTriggers(final Trigger[] triggerList)
404          throws ReaderProtocolException {
405 
406       Vector triggers = readerDevice.getVector(triggerList);
407 
408       if (readerDevice.getMaxTriggerNumber() <= triggers.size()) {
409          throw new ReaderProtocolException("ERROR_TOO_MANY_TRIGGERS",
410                MessagingConstants.ERROR_TOO_MANY_TRIGGERS);
411       }
412 
413       Enumeration iterator = triggers.elements();
414       Trigger cur;
415 
416       while (iterator.hasMoreElements()) {
417          cur = (Trigger) iterator.nextElement();
418          if (!this.notificationTriggers.containsKey(cur.getName())) {
419             this.notificationTriggers.put(cur.getName(), cur);
420             if (cur.getType().equals(TriggerType.CONTINUOUS)) {
421                // continuous trigger
422                if (continuousThread == null && timerThreads.size() == 0) {
423                   continuousThread = new ContinuousNotificationThread(this, cur);
424                   continuousThread.start();
425                }
426             } else if (cur.getType().equals(TriggerType.TIMER)) {
427                // timer trigger
428                if (continuousThread == null) {
429                   Timer timerThread = new Timer();
430                   timerThreads.put(cur.getName(), timerThread);
431                   final int num = 3;
432                   timerThread.schedule(
433                         new TimerNotificationThread(this, cur), 0, Integer
434                               .parseInt(cur.getValue().substring(num)));
435                }
436             } else if (cur.getType().equals(TriggerType.IO_EDGE)) {
437                // io edge trigger
438                if (continuousThread == null) {
439                   // get port
440                   final int num = 6;
441                   String port = cur.getValue().substring(
442                         cur.getValue().indexOf(';') + num,
443                         cur.getValue().lastIndexOf(';'));
444                   if (readerDevice.getEdgeTriggers().containsKey(port)) {
445                      IOEdgeTriggerPortManager manager = 
446                            (IOEdgeTriggerPortManager) readerDevice
447                            .getEdgeTriggers().get(port);
448                      manager.addListener(cur, this.getName());
449                      if (manager.getNumberOfTriggers() == 1) {
450                         manager.start();
451                      }
452                   } else {
453                      throw new ReaderProtocolException(
454                            "no trigger manager available",
455                            MessagingConstants.ERROR_UNKNOWN);
456                   }
457                }
458             } else if (cur.getType().equals(TriggerType.IO_VALUE)) {
459                // io value trigger
460                if (continuousThread == null) {
461                   // get port
462                   final int num = 5;
463                   String port = cur.getValue().substring(num,
464                         cur.getValue().indexOf(';'));
465                   if (readerDevice.getValueTriggers().containsKey(port)) {
466                      IOValueTriggerPortManager manager = 
467                            (IOValueTriggerPortManager) readerDevice
468                            .getValueTriggers().get(port);
469                      manager.addListener(cur, this.getName());
470                      if (manager.getNumberOfTriggers() == 1) {
471                         manager.start();
472                      }
473                   } else {
474                      throw new ReaderProtocolException(
475                            "no trigger manager available",
476                            MessagingConstants.ERROR_UNKNOWN);
477                   }
478                }
479             }
480          }
481       }
482 
483    }

485    /**
486     * Removes the specified Triggers from the list of Notification Triggers
487     * currently associated with this NotificationChannel. Once a Trigger isnt
488     * associated to any object anymore, it is deactivated.
489     * @param triggerList
490     *           The triggers to remove
491     */
492    public void removeNotificationTriggers(final Trigger[] triggerList) {
493 
494       Vector triggers = readerDevice.getVector(triggerList);
495 
496       Enumeration iterator = triggers.elements();
497       Trigger cur;
498 
499       while (iterator.hasMoreElements()) {
500          cur = (Trigger) iterator.nextElement();
501 
502          if (notificationTriggers.containsKey(cur.getName())) {
503             if (cur.getType().equals(TriggerType.CONTINUOUS)) {
504                // continuous trigger
505                continuousThread.stop();
506                continuousThread = null;
507             } else if (cur.getType().equals(TriggerType.TIMER)) {
508                // timer trigger
509                if (timerThreads.containsKey(cur.getName())) {
510                   Timer t = (Timer) timerThreads.get(cur.getName());
511                   t.cancel();
512                   timerThreads.remove(cur.getName());
513                }
514             } else if (cur.getType().equals(TriggerType.IO_EDGE)) {
515                // io edge trigger
516                // get port
517                final int num = 6;
518                String port = cur.getValue().substring(
519                      cur.getValue().indexOf(';') + num,
520                      cur.getValue().lastIndexOf(';'));
521                if (readerDevice.getEdgeTriggers().containsKey(port)) {
522                   IOEdgeTriggerPortManager manager = 
523                         (IOEdgeTriggerPortManager) readerDevice
524                         .getEdgeTriggers().get(port);
525                   manager.removeListener(cur, this.getName());
526                   if (manager.getNumberOfTriggers() <= 0) {
527                      manager.stop();
528                   }
529                }
530             } else if (cur.getType().equals(TriggerType.IO_VALUE)) {
531                // get port
532                final int num = 5;
533                String port = cur.getValue().substring(num,
534                      cur.getValue().indexOf(';'));
535                if (readerDevice.getValueTriggers().containsKey(port)) {
536                   IOValueTriggerPortManager manager = 
537                         (IOValueTriggerPortManager) readerDevice
538                         .getValueTriggers().get(port);
539                   manager.removeListener(cur, this.getName());
540                   if (manager.getNumberOfTriggers() <= 0) {
541                      manager.stop();
542                   }
543                }
544             }
545 
546             this.notificationTriggers.remove(cur.getName());
547          }
548       }
549 
550    }

593    /**
594     * This command returns the data currently queued for delivery in the report
595     * buffer. What data is reported depends on the DataSelector associated with
596     * the NotificationChannel.
597     * @param clearBuffer
598     *           An optional flag to indicate if the report buffer should be
599     *           cleared after the ReadReport is returned
600     * @return The resulting ReadReport
601     */
602    public ReadReport readQueuedData(final boolean clearBuffer) {
603 
604       Hashtable tempReport;
605       synchronized(this) {
606          if (report.isEmpty()) {
607             try {
608                wait();
609             } catch (InterruptedException e) {}
610          }
611          tempReport = report;
612          if (clearBuffer) {
613             report = new Hashtable();
614          }
615          notify();
616       }
617       return generateReadReport(tempReport, dataSelector, null);
618    }

650    /**
651     * Add a report to the actual report of this source.
652     * @param sourceReport
653     *           The report to add
654     */
655    public void addSourceReport(final SourceReport sourceReport) {
656 
657       if ((sourceReport == null) || (sourceReport.getSourceInfo() == null)) {
658     	  log.debug("Source Report or Source Info is null. " +
659     	  		"Source Report is not added to notification channel");
660          return;
661       }
662       
663       synchronized(this) {
664          SourceReport tempReport;
665    
666          if (!report.containsKey(sourceReport.getSourceInfo().getSourceName())) {
667        	  // if there is no source report from this source 
668        	  // currently stored for this notification channel, 
669        	  // add an empty report with the source name only to the report hashmap
670             tempReport = new SourceReport();
671             SourceInfoType tempSourceInfo = new SourceInfoType();
672             tempSourceInfo.setSourceName(sourceReport.getSourceInfo()
673                   .getSourceName());
674             tempReport.setSourceInfo(tempSourceInfo);
675             report.put(sourceReport.getSourceInfo().getSourceName(), tempReport);
676          } else {
677        	  // there is already a source report from this source stored, 
678        	  // get the corresponding source report 
679             tempReport = (SourceReport) report.get(sourceReport.getSourceInfo()
680                   .getSourceName());
681          }
682    
683          Hashtable fieldNames = dataSelector.getFieldNames();
684    
685          // SOURCE_NAME
686          if (fieldNames.containsKey(FieldName.SOURCE_NAME)
687                || fieldNames.containsKey(FieldName.ALL_SOURCE)
688                || fieldNames.containsKey(FieldName.ALL)) {
689             tempReport.getSourceInfo().setSourceName(
690                   sourceReport.getSourceInfo().getSourceName());
691          } else {
692             tempReport.getSourceInfo().setSourceName(null);
693          }  
694          log.debug("Setting source name in notification source report to " +
695         		 tempReport.getSourceInfo().getSourceName());
696          
697          // SOURCE_FREQUENCY
698          if (fieldNames.containsKey(FieldName.SOURCE_FREQUENCY)
699                || fieldNames.containsKey(FieldName.ALL_SOURCE)
700                || fieldNames.containsKey(FieldName.ALL)) {
701             tempReport.getSourceInfo().setSourceFrequency(
702                   sourceReport.getSourceInfo().getSourceFrequency());
703          } else {
704             tempReport.getSourceInfo().setSourceFrequency(-1);
705          }   
706          log.debug("Setting source frequency in notification source report to " +
707         		 tempReport.getSourceInfo().getSourceFrequency());
708          
709          // SOURCE_PROTOCOL
710          if (fieldNames.containsKey(FieldName.SOURCE_PROTOCOL)
711                || fieldNames.containsKey(FieldName.ALL_SOURCE)
712                || fieldNames.containsKey(FieldName.ALL)) {
713             // Not supported in HardwareAbstraction
714             tempReport.getSourceInfo().setSourceProtocol("not supported");
715          } else {
716             tempReport.getSourceInfo().setSourceProtocol(null);
717          }
718          log.debug("Setting source Protocol in notification source report to " +
719          		 tempReport.getSourceInfo().getSourceProtocol());
720          
721          // tags
722          Enumeration tagIterator = sourceReport.getAllTags().elements();
723          TagType curTag;
724          while (tagIterator.hasMoreElements()) {
725             curTag = (TagType) tagIterator.nextElement();
726             TagType tag;
727             if (tempReport.getTag(curTag.getId()) == null) {
728                tag = new TagType();
729                tag.setId(curTag.getId());
730                tempReport.addTag(tag);
