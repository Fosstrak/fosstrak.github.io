58  /**
59   * TextCommandParserHelper provides general utilities for the text parser.
60   * In particular it generates the tree of the intermediate representation (IR).<br />
61   * <br />
62   * How parameters are parsed:<br />
63   * During the parsing process the <code>TextCommandParser</code> puts parameters
64   * to the parameter queue (FIFO). While creating the IR the parameters are again read from
65   * the parameter queue. To be able to handle cases where there are nested parameter lists
66   * the parser distincts different parsing states which can be modified by <code>pushState()</code>, 
67   * <code>popState()</code> and <code>topState()</code>. If the parser is in the state 
68   * <code>STATE_LIST_PARAMETER</code> (parsing a parameter list) it works on a so called
69   * working list. Parameters are added to the working list and the whole working list is
70   * added to the local parameter queue. In this way it's possible to handle recursive (nested)
71   * lists of parameters.
72   * 
73   * @author Andreas Fürer, ETH Zurich Switzerland, Winter 2005/06
74   * 
75   * @see org.fosstrak.reader.rprm.core.msg.command.TextCommandParser
76   */

216 	 */
217 	public void pushState(int state) {

219 	}

221 	/**
222 	 * Reads top element of the parsing state stack without removing it.
223 	 * @return the top element of the parsing state.
224 	 */
225 	public int topState() {
226 		if (stateStack.isEmpty()) {
227 			return STATE_NONE;
228 		} else {
229 			Integer i = (Integer)stateStack.peek();
230 			return i.intValue();
231 		}
232 	}
233 	
234 	/**
235 	 * Pops the parsing state stack and removes the element.
236 	 * @return the top of the parsing stack
237 	 */
238 	public int popState() {
239 		if (stateStack.isEmpty()) {
240 			return STATE_NONE;
241 		} else {
242 			Integer i = (Integer)stateStack.pop();
243 			return i.intValue();
244 		}
245 	}

247 	/**
248 	 * Gets the head of the parameter queue and removes the element.
249 	 * @return the first parameter in the queue as a <code>ValueParameter</code>.
250 	 * @throws TextCommandParserException if no more parameters are available or the parameter is not of type <code>ValueParameter</code>
251 	 */
252 	public ValueParameter readValueParameter() throws TextCommandParserException {
253 		if (!hasParameters()) {
254 			throw new ParameterMissingException("No more parameters available. Parameter queue is empty.");
255 		}
256 		Parameter ret = (Parameter)parameters.get(0);
257 		if (ret instanceof ValueParameter) {
258 			parameters.remove(0);
259 			return (ValueParameter)ret;
260 		}
261 		throw new ParameterWrongTypeException("Wrong parameter type. Expected a ValueParameter instead of a " + ret.getClass());
262 	}

284 	 * @throws TextCommandParserException if no more parameters are available or the parameter is not of type <code>Parameter</code>

287 		if (!hasParameters()) {
