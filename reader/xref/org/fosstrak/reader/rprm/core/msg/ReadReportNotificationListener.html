144 	 * @see NotificationListener#notifyHost(ReadReport, String, DataSelector)
145 	 */
146 	public synchronized void notifyHost(ReadReport report, String channelName, DataSelector dataSelector) {
147 		try {
148 			String outMsg = null;
149 			Clients clients = Clients.getInstance();
150 
151 			Notification notification = createNotification(report, channelName, dataSelector);
152 
153 			// TODO IDs der notifications werden einfach raufgezählt. Korrekt so??
154 			notification.setId(Integer.toString(NotificationChannelConnections.nextMessageId()));
155 
156 			MessageFactory factory = MessageFactory.getInstance();
157 			MessageFormat format = notifyChannelConnections.getChannelFormat(channelName);
158 			serializer = factory.createSerializer(format, Context.NOTIFICATION);
159 			outMsg = serializer.serialize(notification);
160 
161 			log.debug("Notification message: " + outMsg);
162 
163 			NotificationChannel notifChan = null;
164 			try {
165 				notifChan = ReaderDevice.getInstance().getNotificationChannel(channelName);
166 			} catch (ReaderProtocolException rpe) {
167 				log.error("Unknown NotificationChannel: " + channelName);
168 			}
169 
170 			Date curTime = new Date();
171 
172 			if (notifChan != null) {
173 				notifChan.setLastNotificationAttempt(curTime);
174 			}
175 
176 			outMsgDispatcher.sendRequest(clients.getNotificationChannelClientID(channelName), outMsg);
177 
178 			if (notifChan != null) {
179 				notifChan.setLastSuccessfulNotification(curTime);
180 			}
181 
182 
183 		} catch (MessageSerializingException e) {
184 			log.error(e);
185 		}
186 	}

188 	/**
189 	 * Creates a JAXB notification using the ReadReport from the ReaderDevice.
190 	 *
191 	 * @param report
192 	 *            The <code>ReadReport</code> from the ReaderDevice
193 	 * @param channelName
194 	 *            The name of the <code>NotificationChannel</code> to use to
195 	 *            send the <code>ReadReport</code>.
196 	 * @param dataSelector
197 	 *            The DataSelector to check, which information properties are
198 	 *            expected.
199 	 * @return the JAXB notification to serialize in XML or Text format.
200 	 */
201 	private Notification createNotification(ReadReport report, String channelName, DataSelector dataSelector) {
202 		try {
203 			Notification notification = notificationFactory
204 					.createNotification();
205 
206 			// create a set with all fields from the data selector
207 			String[] fields = dataSelector.getAllFieldNames();
208 			CompareSet dataSelectorFields = new CompareSet(fields.length);
209 			for (int i = 0; i < fields.length; i++) {
210 				dataSelectorFields.add(fields[i]);
211 			}
212 
213 			log.debug("All data selector fields associated with this notification "
214 					+ dataSelectorFields.toString());
215 			//add the reader info
216 			//FIXME: @Markus: Bug in report -> containsReaderInfo auch true, wenn alle Werte von ReaderInfo auf null sind
217 			if (report.containsReaderInfo()) {
218 				ReaderInfoType readerInfo = report.getReaderInfo();
219 				ReaderType readerType = notificationFactory.createReaderType();
220 
221 				if (readerInfo.getEpc() != null) {
222 					EPC epc = notificationFactory.createEPC();
223 					epc.setValue(readerInfo.getEpc());
224 					readerType.setReaderEPC(epc);
225 				}
226 
227 				readerType.setReaderName(readerInfo.getName());
228 				if (readerInfo.getHandle() != -1) {
229 					readerType.setReaderHandle(readerInfo.getHandle());
230 				}
231 				readerType.setReaderRole(readerInfo.getRole());
232 
233 				org.fosstrak.reader.rprm.core.msg.notification.EventTimeType time = notificationFactory.createEventTimeType();
234 				if (readerInfo.getNowTick() >= 0) {
235 					time.setEventTimeTick(Long.toString(readerInfo.getNowTick()));
236 				}
237 				if (readerInfo.getNowUTC() != null) {
238 					time.setEventTimeUTC(calendarToXMLGregorianCalendar(toCalendar(readerInfo.getNowUTC())));
239 				}
240 
241 				notification.setReader(readerType);
242 			}
243 
244 			//Name of the Trigger that caught the notification
245 			if (report.containsNotificationInfo()) {
246 				String triggerName = null;
247 				Iterator it = report.getNotificationInfo().getChannelTriggers().keySet().iterator();
248 				/* there should be only one entry in the hashtable, so take the first one */
249 				if (it.hasNext()) {
250 					triggerName = (String)it.next();
251 				}
252 				notification.setNotifyTriggerName(triggerName);
253 			}
254 
255 			//Name of the NotificationChannel on which the Notification is sent on
256 			notification.setNotifyChannelName(channelName);
257 
258 			//create and add ReadReport
259 			ReadReportType reportItems = notificationFactory.createReadReportType();
260 			List sourceList = reportItems.getSourceReport();
261 
262 			//add the source reports
263 			Iterator sourceReportIt = report.getSourceReports().values().iterator();
264 			while (sourceReportIt.hasNext()) {
265 				org.fosstrak.reader.rprm.core.msg.notification.ReadReportType.SourceReport sourceReport = notificationFactory
266 						.createReadReportTypeSourceReport();
267 				SourceReport rdSourceReport = (SourceReport) sourceReportIt.next();
268 
269 				//set the source info
270 				if (rdSourceReport.containsSourceInfo()) {
271 					org.fosstrak.reader.rprm.core.msg.notification.SourceInfoType sourceInfo = notificationFactory
272 							.createSourceInfoType();
273 					org.fosstrak.reader.rprm.core.readreport.SourceInfoType rdSourceInfo = rdSourceReport.getSourceInfo();
274 					if (rdSourceInfo.getSourceFrequency() != -1) {
275 						sourceInfo.setSourceFrequency(Integer.toString(rdSourceInfo
276 								.getSourceFrequency()));
277 					}
278 					if (dataSelectorFields != null
279 							&& ((dataSelectorFields.contains(FieldName.SOURCE_NAME))
280 									|| (dataSelectorFields.contains(FieldName.ALL_SOURCE))
281 									|| (dataSelectorFields.contains(FieldName.ALL)) )) {
282 
283 						sourceInfo.setSourceName(rdSourceInfo.getSourceName());
284 						log.debug("Adding source name " + rdSourceInfo.getSourceName() + " when generating XML instance of notification");
285 					}
286 					else {
287 						log.debug("Neglecting source name when generating XML instance of notification");
288 					}
289 					sourceInfo.setSourceProtocol(rdSourceInfo.getSourceProtocol());
290 					sourceReport.setSourceInfo(sourceInfo);
291 
292 				}
293 
294 				//add all the tags
295 				List tagList = sourceReport.getTag();
296 				//Collection tags = report.getAllTags().values();
297 				Collection tags = rdSourceReport.getAllTags().values();
298 				Iterator tagIt = tags.iterator();
299 				while (tagIt.hasNext()) {
300 					org.fosstrak.reader.rprm.core.readreport.TagType rdTag = (org.fosstrak.reader.rprm.core.readreport.TagType) tagIt
301 							.next();
302 					org.fosstrak.reader.rprm.core.msg.notification.TagType tag = notificationFactory
303 							.createTagType();
304 					if (dataSelectorFields != null
305 							&& (dataSelectorFields.contains(FieldName.TAG_ID)
306 									|| dataSelectorFields.contains(FieldName.ALL)
307 									|| dataSelectorFields
308 											.contains(FieldName.ALL_TAG) || dataSelectorFields
309 									.contains(FieldName.ALL_SUPPORTED))) {
310 						tag.setTagID(HexUtil.hexToByteArray(rdTag.getId()));
311 					}
312 					tag.setTagIDAsPureURI(rdTag.getIdAsPureURI());
313 					tag.setTagIDAsTagURI(rdTag.getIdAsTagURI());
314 					tag.setTagType(rdTag.getTagType());
315 					getTagEvents(tag, rdTag);
316 					getTagFields(tag, rdTag);
317 					tagList.add(tag);
318 				}
319 				//add source report to the source list
320 				sourceList.add(sourceReport);
321 			}
322 
323 			notification.getReadReport().add(reportItems);
324 
325 			return notification;
326 		} catch (JAXBException e) {
327 			log.error(e);
328 			return null;
329 		}

335 	 * Insert all the ReadReports from the ReaderDevice ReadReport into the
336 	 * ReadReport of the JAXB classes.
337 	 *
338 	 * @param msgReadReport
339 	 *            The JAXB ReadReport
340 	 * @param rdReadReport
341 	 *            The ReaderDevice ReadReport
342 	 * @param dataSelectorFields
343 	 * @return List with all JABX ReadReports
344 	 * @throws JAXBException
345 	 */
346 	private List getReadReports(List msgReadReport, ReadReport rdReadReport, CompareSet dataSelectorFields) throws JAXBException {
347 		ReadReportType reportItems = notificationFactory.createReadReportType();
348 		List sourceList = reportItems.getSourceReport();
349 
350 		//get all Source reports
351 		Iterator sourceReportIt = rdReadReport.getSourceReports().values().iterator();
352 		while (sourceReportIt.hasNext()) {
353 			org.fosstrak.reader.rprm.core.msg.notification.ReadReportType.SourceReport sourceReport = notificationFactory.createReadReportTypeSourceReport();
354 			SourceReport rdSourceReport = (SourceReport) sourceReportIt.next();
355 
356 			//add the source info
357 			if (rdSourceReport.containsSourceInfo()) {
358 				SourceInfoType sourceInfo = notificationFactory.createSourceInfoType();
359 				org.fosstrak.reader.rprm.core.readreport.SourceInfoType rdSourceInfo = rdSourceReport.getSourceInfo();
360 				if (rdSourceInfo.getSourceFrequency() >= 0) {
361 					sourceInfo.setSourceFrequency(Integer.toString(rdSourceInfo.getSourceFrequency()));
362 				}
363 				if (dataSelectorFields.contains(FieldName.SOURCE_NAME)) {
364 					sourceInfo.setSourceName(rdSourceInfo.getSourceName());
365 				}
366 				sourceInfo.setSourceProtocol(rdSourceInfo.getSourceProtocol());
367 				sourceReport.setSourceInfo(sourceInfo);
368 				//sourceList.add(sourceInfo);
369 			}
370 
371 			//add all the tags
372 			List tagList = sourceReport.getTag();
373 			Collection tags = rdSourceReport.getAllTags().values();
374 			Iterator it = tags.iterator();
375 			while(it.hasNext()) {
376 				org.fosstrak.reader.rprm.core.readreport.TagType rdTag = (org.fosstrak.reader.rprm.core.readreport.TagType)it.next();
377 				TagType tag = notificationFactory.createTagType();
378 				if (dataSelectorFields != null && (dataSelectorFields.contains(FieldName.TAG_ID) || dataSelectorFields.contains(FieldName.ALL)  || dataSelectorFields.contains(FieldName.ALL_TAG)  || dataSelectorFields.contains(FieldName.ALL_SUPPORTED))) {
379 					tag.setTagID(HexUtil.hexToByteArray(rdTag.getId()));
380 				}
381 				tag.setTagIDAsPureURI(rdTag.getIdAsPureURI());
382 				tag.setTagIDAsTagURI(rdTag.getIdAsTagURI());
383 				tag.setTagType(rdTag.getTagType());
384 				tagList.add(tag);
385 			}
386 
387 			//add source report the the sourceList
388 			sourceList.add(sourceReport);
389 		}
390 		//add read report to the report list
391 		msgReadReport.add(reportItems);
392 		return msgReadReport;
393 	}

396 	 * Insert all the tag events from the ReaderDevice TagType into the TagType
397 	 * of the JAXB classes.
398 	 *
399 	 * @param msgTag
400 	 *            The JAXB TagType
401 	 * @param rdTag
402 	 *            The ReaderDevice TagType
403 	 * @return List with all JAXB TagEvents
404 	 * @throws JAXBException
405 	 */
406 	private List getTagEvents(org.fosstrak.reader.rprm.core.msg.notification.TagType msgTag, org.fosstrak.reader.rprm.core.readreport.TagType rdTag) throws JAXBException {
407 		List eventList = msgTag.getTagEvent();
408 		Collection tagEvents = rdTag.getAllTagEvents().values();
409 		Iterator it = tagEvents.iterator();
410 		while(it.hasNext()) {
411 			org.fosstrak.reader.rprm.core.msg.notification.TagEventType tagEvent = notificationFactory.createTagEventType();
412 			org.fosstrak.reader.rprm.core.readreport.TagEventType rdEvent = (org.fosstrak.reader.rprm.core.readreport.TagEventType)it.next();
413 			tagEvent.setEventType(rdEvent.getEventType());
414 
415 			org.fosstrak.reader.rprm.core.msg.notification.EventTimeType time = notificationFactory.createEventTimeType();
416 			if (rdEvent.getTimeTick() >= 0) {
417 				time.setEventTimeTick(Long.toString(rdEvent.getTimeTick()));
418 			}
419 			if (rdEvent.getTimeUTC() != null) {
420 				time.setEventTimeUTC(calendarToXMLGregorianCalendar(toCalendar(rdEvent.getTimeUTC())));
421 			}
422 			tagEvent.setTime(time);
423 
424 			EventTriggers eventTriggers = notificationFactory.createTagEventTypeEventTriggers();
425 			List eventTriggerList = eventTriggers.getTrigger();
426 			Collection evTriggers = rdEvent.getEventTriggers().values();
427 			Iterator trgIt = evTriggers.iterator();
428 
429 			if (!trgIt.hasNext()) {
430 				eventTriggerList.add("NoTrigger");
431 			}
432 
433 			while(trgIt.hasNext()) {
434 				Trigger rdTrigger = (Trigger)trgIt.next();
435 				eventTriggerList.add(rdTrigger.getName());
436 			}
437 
438 			tagEvent.setEventTriggers(eventTriggers);
439 			eventList.add(tagEvent);
440 		}
441 		return eventList;
442 	}
