474    /**
475     * Create a source object with a given name. If a source object with the same
476     * name exists, an error is returned ("ERROR_OBJECT_EXISTS"). This is a
477     * static method. The source shall implicitly be added to the list of all
478     * sources kept by the ReaderDevice object.
479     * @param name
480     *           The name of the source
481     * @param readerDevice
482     *           The reader device it belongs to
483     * @return The instance of the new source
484     * @throws ReaderProtocolException
485     *            "ERROR_UNKNOWN"
486     */
487    public static Source create(final String name,
488          final ReaderDevice readerDevice) throws ReaderProtocolException {
489 
490       if (readerDevice.getMaxSourceNumber() <= readerDevice.getSources()
491             .size()) {
492          throw new ReaderProtocolException("ERROR_UNKNOWN",
493                MessagingConstants.ERROR_UNKNOWN);
494       }
495 
496       // check if DataSelector with the same name exists
497       try {
498          readerDevice.getSource(name);
499       } catch (ReaderProtocolException e) {
500          // create new Source
501          Source newSource = new Source(name, readerDevice);
502          readerDevice.getSources().put(name, newSource);
503          return newSource;
504       }
505 
506       throw new ReaderProtocolException("ERROR_OBJECT_EXISTS",
507             MessagingConstants.ERROR_OBJECT_EXISTS);
508 
509    }

712    /**
713     * Add a list of read triggers to this source.
714     * @param triggerList
715     *           The list of read triggers
716     * @throws ReaderProtocolException
717     *            "ERROR_TOO_MANY_TRIGGERS"
718     */
719    public void addReadTriggers(final Trigger[] triggerList)
720          throws ReaderProtocolException {
721 
722       Vector triggers = readerDevice.getVector(triggerList);
723 
724       if (readerDevice.getMaxTriggerNumber() <= triggers.size()) {
725          throw new ReaderProtocolException("ERROR_TOO_MANY_TRIGGERS",
726                MessagingConstants.ERROR_TOO_MANY_TRIGGERS);
727       }
728 
729       Enumeration iterator = triggers.elements();
730       Trigger cur;
731 
732       while (iterator.hasMoreElements()) {
733          cur = (Trigger) iterator.nextElement();
734          if (!this.readTriggers.containsKey(cur.getName())) {
735             this.readTriggers.put(cur.getName(), cur);
736             if (cur.getType().equals(TriggerType.CONTINUOUS)) {
737                // conitnuous trigger
738                if (continuousThread == null && timerThreads.size() == 0) {
739                   continuousThread = new ContinuousReadThread(this, cur);
740                   continuousThread.start();
741                }
742             } else if (cur.getType().equals(TriggerType.TIMER)) {
743                // timer trigger
744                if (continuousThread == null) {
745                   Timer timerThread = new Timer();
746                   timerThreads.put(cur.getName(), timerThread);
747                   final int num = 3;
748                   timerThread.schedule(new TimerReadThread(this, cur), 0,
749                         Integer.parseInt(cur.getValue().substring(num)));
750                }
751             } else if (cur.getType().equals(TriggerType.IO_EDGE)) {
752                // io edge trigger
753                if (continuousThread == null) {
754                   // get port
755                   final int num = 6;
756                   String port = cur.getValue().substring(
757                         cur.getValue().indexOf(';') + num,
758                         cur.getValue().lastIndexOf(';'));
759                   if (readerDevice.getEdgeTriggers().containsKey(port)) {
760                      IOEdgeTriggerPortManager manager =
761                            (IOEdgeTriggerPortManager) readerDevice
762                            .getEdgeTriggers().get(port);
763                      manager.addListener(cur, this.getName());
764                      if (manager.getNumberOfTriggers() == 1) {
765                         manager.start();
766                      }
767                   } else {
768                      throw new ReaderProtocolException(
769                            "no trigger manager available",
770                            MessagingConstants.ERROR_UNKNOWN);
771                   }
772                }
773             } else if (cur.getType().equals(TriggerType.IO_VALUE)) {
774                // io value trigger
775                if (continuousThread == null) {
776                   // get port
777                   final int num = 5;
778                   String port = cur.getValue().substring(num,
779                         cur.getValue().indexOf(';'));
780                   if (readerDevice.getValueTriggers().containsKey(port)) {
781                      IOValueTriggerPortManager manager =
782                            (IOValueTriggerPortManager) readerDevice
783                            .getValueTriggers().get(port);
784                      manager.addListener(cur, this.getName());
785                      if (manager.getNumberOfTriggers() == 1) {
