21  package org.fosstrak.reader.rp.proxy;
22  
23  
24  public interface NotificationChannel {
25  
26  	/**
27  	 * Returns the name of the given NotificationChannel object.
28  	 * @return The name of the NotificationChannel
29  	 */
30  	String getName() throws RPProxyException;
31  
32  	/**
33  	 * Returns the address to which this NotificationChannel object sends its
34  	 * notifications as specified by either commands NotificationChannel.create
35  	 * or NotificationChannel.setAddress.
36  	 * @return The address of the host
37  	 */
38  	String getAddress() throws RPProxyException;
39  
40  	/**
41  	 * Returns the effective address to which this NotificationChannel object
42  	 * sends its notifications.
43  	 * @return The effective address of the host
44  	 */
45  	String getEffectiveAddress() throws RPProxyException;
46  
47  	/**
48  	 * Sets the address to which this NotificationChannel object sends its
49  	 * notifications.
50  	 * @param addr
51  	 *           The address of the host
52  	 * @return If connect mode was specified, then the return value does not
53  	 *         apply and the Reader returns zero. Otherwise, if listen mode was
54  	 *         specified, then the return returns the port number assigned by the
55  	 *         Reader to listen for host NotificationChannel connections.
56  	 */
57  	int setAddress(final String addr) throws RPProxyException;
58  
59  	/**
60  	 * Returns the DataSelector currently associated with this
61  	 * NotificationChannel object. If there is no DataSelector object associated,
62  	 * the error ERROR_DATASELECTOR_NOT_FOUND is raised.
63  	 * @return The dataSelector associated to this notification channel
64  	 * @throws RPProxyException
65  	 *            "ERROR_DATASELECTOR_NOT_FOUND"
66  	 */
67  	DataSelector getDataSelector() throws RPProxyException;
68  
69  	/**
70  	 * Sets the DataSelector object to be used with this NotificationChannel.
71  	 * @param dataSelector
72  	 *           The dataSelector to use
73  	 */
74  	void setDataSelector(final DataSelector dataSelector) throws RPProxyException;
75  
76  	/**
77  	 * Adds the specified Sources to the list of Sources currently associated
78  	 * with this NotificationChannel. If some of the Sources to be added are
79  	 * already associated with this NotificationChannel, only the not yet
80  	 * associated Sources shall be added and the command completes successfully.
81  	 * @param sourceList
82  	 *           The sources to add
83  	 */
84  	void addSources(final Source[] sourceList) throws RPProxyException;
85  
86  	/**
87  	 * Removes the specified Sources from the list of Sources currently
88  	 * associated with this NotificationChannel.
89  	 * @param sourceList
90  	 *           The sources to remove
91  	 */
92  	void removeSources(final Source[] sourceList) throws RPProxyException;
93  
94  	/**
95  	 * Removes all Sources currently associated with this NotificationChannel.
96  	 */
97  	void removeAllSources() throws RPProxyException;
98  
99  	/**
100 	 * Returns the Source with the specified name currently associated with this
101 	 * NotificationChannel object. If no Source object with the given name
102 	 * exists, the error ERROR_SOURCE_NOT_FOUND is raised.
103 	 * @param name
104 	 *           The name of the source
105 	 * @return The instance of the source
106 	 * @throws RPProxyException
107 	 *            "ERROR_SOURCE_NOT_FOUND"
108 	 */
109 	Source getSource(final String name) throws RPProxyException;
110 
111 	/**
112 	 * Returns all Sources currently associated with this NotificationChannel
113 	 * object. If no Sources are currently associated with this object, the
114 	 * command completes successfully and an empty list will be returned.
115 	 * @return A list of sources
116 	 */
117 	Source[] getAllSources() throws RPProxyException;
118 
119 	/**
120 	 * Adds the specified Triggers to the list of Notification Triggers currently
121 	 * associated with this NotificationChannel. If some of the Triggers to be
122 	 * added are already associated with this NotificationChannel, only the not
123 	 * yet associated Triggers shall be added and the command completes
124 	 * successfully. Once a Trigger has been added, it is immediately activated.
125 	 * @param triggerList
126 	 *           The triggers to add
127 	 * @throws RPProxyException
128 	 *            "ERROR_TOO_MANY_TRIGGERS"
129 	 */
130 	void addNotificationTriggers(final Trigger[] triggerList)
131 			throws RPProxyException;
132 
133 	/**
134 	 * Removes the specified Triggers from the list of Notification Triggers
135 	 * currently associated with this NotificationChannel. Once a Trigger isnt
136 	 * associated to any object anymore, it is deactivated.
137 	 * @param triggerList
138 	 *           The triggers to remove
139 	 */
140 	void removeNotificationTriggers(final Trigger[] triggerList) throws RPProxyException;
141 
142 	/**
143 	 * Removes all Triggers currently associated with this NotificationChannel.
144 	 */
145 	void removeAllNotificationTriggers() throws RPProxyException;
146 
147 	/**
148 	 * Returns the Trigger with the specified name currently associated with this
149 	 * NotificationChannel object. If no Trigger object with the given name
150 	 * exists, the error ERROR_TRIGGER_NOT_FOUND is raised.
151 	 * @param name
152 	 *           The name of the notification trigger
153 	 * @return The notification trigger
154 	 * @throws RPProxyException
155 	 *            "ERROR_TRIGGER_NOT_FOUND"
156 	 */
157 	Trigger getNotificationTrigger(final String name) throws RPProxyException;
158 
159 	/**
160 	 * Returns all Triggers currently associated with this NotificationChannel
161 	 * object. If no Triggers are currently associated with this object, the
162 	 * command will complete successfully and an empty list will be returned.
163 	 * @return A list of triggers
164 	 */
165 	Trigger[] getAllNotificationTriggers() throws RPProxyException;
166 
167 	/**
168 	 * This command returns the data currently queued for delivery in the report
169 	 * buffer. What data is reported depends on the DataSelector associated with
170 	 * the NotificationChannel.
171 	 * @param clearBuffer
172 	 *           An optional flag to indicate if the report buffer should be
173 	 *           cleared after the ReadReport is returned
174 	 * @return The resulting ReadReport
175 	 */
176 	ReadReport readQueuedData(final boolean clearBuffer) throws RPProxyException;
177 
178 	/**
179 	 * This command returns the data currently queued for delivery in the report
180 	 * buffer. What data is reported depends on the DataSelector associated with
181 	 * the NotificationChannel.
182 	 * @param trigger
183 	 *           The trigger that cause this readQueuedData request
184 	 * @param clearBuffer
185 	 *           An optional flag to indicate if the report buffer should be
186 	 *           cleared after the ReadReport is returned
187 	 * @return The resulting ReadReport
188 	 */
189 	ReadReport readQueuedData(final boolean clearBuffer, final Trigger trigger) throws RPProxyException;
190 
191 	int getMaxNumberSupported() throws RPProxyException;
192 
193 }
