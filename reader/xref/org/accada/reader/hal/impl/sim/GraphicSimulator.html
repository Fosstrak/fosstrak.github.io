120 	 * implements the initialize method of the SimulatorEngine
121      * 
122      * @param controller 
123      * @param file is only required for BatchSimulator
124 	 * @throws IOException 
125 	 */
126 	public void initialize(SimulatorController controller) throws IOException {
127 		this.controller = controller;
128 		mgmtSimDialogs = new Hashtable<String, MgmtSimDialog>();
129 
130 		// load propterties
131 		props = new Properties();
132 		props.load(this.getClass().getResourceAsStream(PROPERTIES_FILE_LOCATION));
133 		
134 		// load text
135 		InputStream languageStream = null;
136 		
137 		// try to get language form property file
138 		if (props.containsKey("Language")) {
139 			languageStream = this.getClass().getResourceAsStream("/props/GUIText_" + props.getProperty("Language") + ".properties");
140 		}
141 		
142 		// try system default language
143 		if (languageStream == null) {
144 			languageStream = this.getClass().getResourceAsStream("/props/GUIText_" + SYSTEM_DEFAULT_LOCALE.getLanguage() + ".properties");
145 		}
146 		
147 		// try default language
148 		if (languageStream == null) {
149 			languageStream = this.getClass().getResourceAsStream("/props/GUIText_" + DEFAULT_LOCALE.getLanguage() + ".properties");
150 		}
151 		guiText = new PropertyResourceBundle(languageStream);
152 		
153 		// initialize GUI
154 		initializeGUI();
155 		LOG.info("GraphicSimulator started");
156 	}

160 	 */
161 	private void initializeGUI() {
162 		if(!props.containsKey("WindowWidth")) {
163 			props.setProperty("WindowWidth", new Integer(2 * getProperty("FramePadding") + getProperty("TagWidth") + getProperty("HorizontalPadding") + getProperty("AntennasPerRow") * (getProperty("AntennaWidth") + getProperty("HorizontalInterAntennaPadding")) - getProperty("HorizontalInterAntennaPadding")).toString());
164 		}
165 		if(!props.containsKey("WindowHeight")) {
166 			try {
167 				props.setProperty("WindowHeight", new Integer(2 * getProperty("FramePadding") + getProperty("ReaderHeight") + getProperty("VerticalPadding") + (controller.getReadPointNames().length / getProperty("AntennasPerRow") + (controller.getReadPointNames().length % getProperty("AntennasPerRow") > 0 ? 1 : 0)) * (getProperty("AntennaHeight") + getProperty("VerticalInterAntennaPadding")) - getProperty("VerticalInterAntennaPadding") + 50).toString());
168 			} catch (HardwareException ignored) { }
169 		}
170 		if(!props.containsKey("AntennaPaneX")) {
171 			props.setProperty("AntennaPaneX", new Integer(getProperty("FramePadding") + getProperty("TagWidth") + getProperty("HorizontalPadding")).toString());
172 		}
173 		if(!props.containsKey("AntennaPaneY")) {
174 			props.setProperty("AntennaPaneY", new Integer(getProperty("FramePadding") + getProperty("ReaderHeight") + getProperty("VerticalPadding")).toString());
175 		}
176 		if(!props.containsKey("AntennaPaneWidth")) {
177 			props.setProperty("AntennaPaneWidth", new Integer(getProperty("WindowWidth") - getProperty("AntennaPaneX") - getProperty("FramePadding")).toString());
178 		}
179 		if(!props.containsKey("AntennaPaneHeight")) {
180 			props.setProperty("AntennaPaneHeight", new Integer(getProperty("WindowHeight") - getProperty("AntennaPaneY") - getProperty("FramePadding")).toString());
181 		}
182 		this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
183 		this.setSize(getProperty("WindowWidth"), getProperty("WindowHeight"));
184 		this.setLayeredPane(getJLayeredPane());
185 		this.setJMenuBar(getJJMenuBar());
186 		
187 		this.setTitle(guiText.getString("ApplicationTitle"));
188 		this.setVisible(true);
189 	}

297 	/**
298 	 * creates the layered pane if it does not already exists
299 	 * 
300 	 * @return layered pane
301 	 */
302 	private JLayeredPane getJLayeredPane() {
303 		if (jLayeredPane == null) {
304 			jLayeredPane = new JLayeredPane();
305 			jLayeredPane.setLayout(null);
306 			jLayeredPane.setOpaque(true);
307 			jLayeredPane.setBackground(Color.WHITE);
308 			jLayeredPane.add(getReader(), new Integer(0));
309 
310 			// add antennas
311 			String[] antennaIds = null;
312 			try {
313 				antennaIds = controller.getReadPointNames();
314 			} catch (HardwareException ignored) { }
315 			for (int i = 0; i < antennaIds.length; i++) {
316 				createNewAntenna(antennaIds[i]);
317 			}
318 			jLayeredPane.add(getContextMenu());
319 			jLayeredPane.add(getMgmtSimMenu());
320 
321 			// add mouse listener
322 			jLayeredPane.addMouseListener(new MouseAdapter() {
323 
324 				// context menu
325 				public void mouseReleased(MouseEvent e) {
326 					hideActiveContextMenuAndSelection();
327 					if (e.getButton() == MouseEvent.BUTTON3) {
328 						Point posOnScreen = getJLayeredPane().getLocationOnScreen();
329 						contextMenu.setLocation(posOnScreen.x + e.getX(), posOnScreen.y + e.getY());
330 						contextMenu.setVisible(true);
331 						mgmtSimMenu.setVisible(false);
332 						setActiveContextMenu(contextMenu);
333 					} else {
334 						contextMenu.setVisible(false);
335 						mgmtSimMenu.setVisible(false);
336 						if(e.getButton() == MouseEvent.BUTTON1) {
337 							selection.select(tags);
338 						}
339 					}
340 				}
341 				
342 				// selection
343 				public void mousePressed(final MouseEvent e) {
344 					hideActiveContextMenuAndSelection();
345 					if(e.getButton() == MouseEvent.BUTTON1) {
346 						selection.setBounds(0, 0, getProperty("WindowWidth"), getProperty("WindowHeight"));
347 						selection.setStartPoint(e.getPoint());
348 						jLayeredPane.add(selection, new Integer(2));
349 					}
350 				}
351 			});
352 			
353 			// add drag listener
354 			jLayeredPane.addMouseMotionListener(new MouseMotionAdapter() {
355 				public void mouseDragged(MouseEvent e) {
356 					if(selection.isActive()) {
357 						selection.setCurrentPoint(e.getPoint());
358 					}
359 				}
360 			});
361 		}
362 		return jLayeredPane;
363 	}

527 	/**
528 	 * creates a new antenna and adds it to the layered pane
529 	 * 
530 	 * @param name of the new antenna
531 	 * @param id of the new antenna
532 	 */
533 	private void createNewAntenna(String id) {
534 		if("".equals(id)) {
535 			try {
536 				id = controller.getReadPointNames()[antennas.size()];
537 			} catch (HardwareException ignored) { }
538 		}
539 		int x = (antennas.size() % getProperty("AntennasPerRow")) * (getProperty("AntennaWidth") + getProperty("HorizontalInterAntennaPadding")) + getProperty("AntennaPaneX");
540 		int y = (antennas.size() / getProperty("AntennasPerRow")) * (getProperty("AntennaHeight") + getProperty("VerticalInterAntennaPadding")) + (getProperty("AntennaPaneY"));
541 		Point pos = new Point(x, y);
542 		
543 		// antenna
544 		Antenna newAntenna = new Antenna(id, this);
545 		newAntenna.setPosition(pos);
546 		antennas.add(newAntenna);
547 		jLayeredPane.add(newAntenna, new Integer(0));
548 		
549 		// listeners
550 		getTranslationListener().add(newAntenna);
551 		newAntenna.addMouseListener(new AntennaMouseAdapter());
552 		
553 		//cable
554 		Cable newCable = new Cable(pos, this);
555 		jLayeredPane.add(newCable, new Integer(-1));
556 		jLayeredPane.repaint();
557 	}
