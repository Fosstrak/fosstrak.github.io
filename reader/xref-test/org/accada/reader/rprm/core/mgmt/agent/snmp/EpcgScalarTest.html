258 	public final void testSetValue() {
259 		boolean oldBoolean;
260 		OperationalStatus oldOperStatus;
261 		AlarmLevel oldAlarmLevel;
262 		int oldInterval;
263 		int oldThreshold;
264 
265 		String cont = readerDevice.getContact();
266 		String newCont = cont + "_new";
267 		readerDevice.setContact(newCont);
268 		assertEquals(newCont, readerDevice.getContact());
269 		epcgRdrDevReboot.setValue(new Integer32(1));
270 		assertEquals(cont, readerDevice.getContact());
271 
272 		AntennaReadPoint antReadPoint = null;
273 		ReadPoint[] readPoints = readerDevice.getAllReadPoints();
274 		for (int i = 0; i < readPoints.length; i++) {
275 			if (readPoints[i] instanceof AntennaReadPoint) {
276 				antReadPoint = (AntennaReadPoint)readPoints[i];
277 			}
278 		}
279 		if (antReadPoint != null) {
280 			antReadPoint.increaseWriteCount();
281 			assertTrue(antReadPoint.getWriteCount() > 0);
282 			epcgRdrDevResetStatistics.setValue(new Integer32(1));
283 			assertEquals(0, antReadPoint.getWriteCount());
284 		}
285 
286 		oldBoolean = readerDevice.getOperStatusAlarmControl().getEnabled();
287 		epcgRdrDevOperStateEnable.setValue(new Integer32(!oldBoolean ? 1 : 2));
288 		assertEquals(!oldBoolean, readerDevice.getOperStatusAlarmControl().getEnabled());
289 		readerDevice.getOperStatusAlarmControl().setEnabled(oldBoolean);
290 
291 		oldOperStatus = readerDevice.getOperStatusAlarmControl().getTriggerFromState();
292 		epcgRdrDevOperNotifFromState.setValue(SnmpUtil.operStateToBITS(OperationalStatus.UP));
293 		assertEquals(OperationalStatus.UP, readerDevice.getOperStatusAlarmControl().getTriggerFromState());
294 		epcgRdrDevOperNotifFromState.setValue(SnmpUtil.operStateToBITS(OperationalStatus.DOWN));
295 		assertEquals(OperationalStatus.DOWN, readerDevice.getOperStatusAlarmControl().getTriggerFromState());

297

302 		assertEquals(OperationalStatus.DOWN, readerDevice.getOperStatusAlarmControl().getTriggerToState());
303 		readerDevice.getOperStatusAlarmControl().setTriggerFromState(oldOperStatus);
304 
305 		oldAlarmLevel = readerDevice.getOperStatusAlarmControl().getLevel();
