44  /**
45   * this class runs some basic tests on the AdaptorManagement. the tests

83  	/**
84  	 * this method tests the initial setup is right. in detail:
85  	 * <ul>
86  	 * <li>test whether the default adaptor gets created upon reset.</li>
87  	 * <li>test that the shortcut for getDefaultAdaptor and getAdaptor(name) yields the same adaptor.</li>
88  	 * <li>test that the adaptor is empty (contains no readers).</li>
89  	 * <li>test to create the default adaptor twice (should trigger an exception).</li>
90  	 * <li>test that getAdaptorNames delivers the adaptor names correctly.</li>
91  	 * <li>test that the repository gets removed.</li>
92  	 * <li>test that the repository can be added and retrieved correctly.</li>
93  	 * </ul>
94  	 * @throws Exception
95  	 */
96  	public void testInitialSetup() throws Exception {
97  		initializeManagement();
98  		
99  		AdaptorManagement management = AdaptorManagement.getInstance();
100 		
101 		// we do not want the changes to be written back to the configuration file.
102 		management.setCommitChanges(false);
103 		
104 		String defaultAdaptorName = AdaptorManagement.DEFAULT_ADAPTOR_NAME;
105 
106 		// reset the adaptor management to just the default adaptor (local adaptor).
107 		resetManagement();
108 			
109 		// test whether the default adaptor gets created upon reset. 
110 		assertEquals(true, management.containsAdaptor(defaultAdaptorName));
111 		
112 		// test that the shortcut for getDefaultAdaptor and getAdaptor(name) yields the same adaptor.
113 		Adaptor getAdaptor = management.getAdaptor(defaultAdaptorName);
114 		Adaptor getDefaultAdaptor = management.getDefaultAdaptor();
115 		assertEquals(getAdaptor.getAdaptorName(), getDefaultAdaptor.getAdaptorName());
116 		
117 		
118 		
119 		// test that the adaptor is empty (contains no readers).
120 		Adaptor adaptor = management.getAdaptor(defaultAdaptorName);
121 		assertNotNull(adaptor.getReaderNames());
122 		assertEquals(0, adaptor.getReaderNames().size());
123 		
124 		
125 		
126 		// test to create the default adaptor twice (should trigger an exception).
127 		try {
128 			management.define(defaultAdaptorName, null);
129 			
130 			throw new Exception("Could define an adaptor with the same name twice!");
131 		} catch (LLRPDuplicateNameException e) {
132 			// everything is fine!
133 		}
134 		
135 		
136 		
137 		// test that getAdaptorNames delivers the adaptor names correctly.
138 		assertNotNull(management.getAdaptorNames());
139 		assertEquals(1, management.getAdaptorNames().size());
140 		assertEquals(defaultAdaptorName, management.getAdaptorNames().get(0));
141 			
142 		
143 		
144 		// test that the handler can be added and retrieved correctly.
145 		MessageHandler handler = new MessageHandler() {
146 
147 			public void handle(String adaptorName, String readerName,
148 					LLRPMessage message) {
149 				// do nothing
150 			}
151 			
152 		};
153 		
154 		management.registerFullHandler(handler);
155 		management.registerPartialHandler(handler, String.class);
156 		assertTrue(management.hasFullHandler(handler));
157 		assertTrue(management.hasPartialHandler(handler, String.class));
158 		
159 		management.deregisterFullHandler(handler);
160 		management.deregisterPartialHandler(handler, String.class);
161 		assertFalse(management.hasFullHandler(handler));
162 		assertFalse(management.hasPartialHandler(handler, String.class));
163 	}

165 	/**
166 	 * this test creates a remote adaptor through the AdaptorManagement.
167 	 * @throws Exception
168 	 */
169 	public void testCreateRemoteAdaptor() throws Exception {
170 		initializeManagement();
171 		
172 		AdaptorManagement management = AdaptorManagement.getInstance();
173 		
174 		// we do not want the changes to be written back to the configuration file.
175 		management.setCommitChanges(false);
176 		
177 		resetManagement();
178 	
179 		String adaptorName = Constants.adaptorNameInRegistry;
180 		
181 		/**
182 		 * helper class to create a remote server adaptor running on a thread.
183 		 * this thread will hold the adaptor accessible through rmi.
184 		 * @author sawielan
185 		 *
186 		 */
187 		class MyRunner implements Runnable {
188 
189 			public MyRunner(Adaptor adaptor) {
190 				try {
191 					try {
192 						LocateRegistry.createRegistry(Constants.registryPort);
193 					} catch (Exception e) {
194 						// the registry already exists...
195 					}
196 					
197 					Registry r = LocateRegistry.getRegistry(Constants.registryPort);
198 					r.bind(Constants.adaptorNameInRegistry, adaptor);					
199 				} catch (RemoteException e) {
200 					e.printStackTrace();
201 				} catch (AlreadyBoundException e) {
202 					e.printStackTrace();
203 				}				
204 			}
205 			
206 			public void run() {
207 				while (true) {
208 					try {
209 						Thread.sleep(1000);
210 					} catch (InterruptedException e) {
211 						break;
212 					}
213 				}
214 				// deregister the adaptor.
215 				Registry r;
216 				try {
217 					r = LocateRegistry.getRegistry(Constants.registryPort);
218 					r.unbind(Constants.adaptorNameInRegistry);
219 				} catch (RemoteException e) {
220 					e.printStackTrace();
221 				} catch (NotBoundException e) {
222 					e.printStackTrace();
223 				}				
224 			}
225 			
226 		}
227 		// create an adaptor an register it through rmi
228 		AdaptorImpl serverAdaptor = new AdaptorImpl(adaptorName);
229 		Thread thread = new Thread(new MyRunner(serverAdaptor));
230 		thread.start();
231 		
232 		
233 		// ----------------------------  test part ----------------------------
234 		try {
235 			management.getAdaptor(adaptorName);
236 		} catch (LLRPRuntimeException e) {
237 			// this is good!
238 		}
239 		
240 		// save the name of the adaptor that might get returned by adaptorManagement
241 		adaptorName = management.define(adaptorName, null);
242 		// retrieve the adaptor
243 		Adaptor adaptor = management.getAdaptor(adaptorName);
244 		
245 		assertNotNull(adaptor.getAdaptorName());
246 		assertEquals(adaptorName, adaptor.getAdaptorName());
247 		
248 		// stop the thread.
249 		thread.interrupt();
250 	}
