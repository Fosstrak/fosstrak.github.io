170     /**
171      * Loads the appropriate JDBC driver for this environment/framework. For
172      * example, if we are in an embedded environment, we load Derby's
173      * embedded Driver, <code>org.apache.derby.jdbc.EmbeddedDriver</code>.
174      */
175     private boolean loadDriver() {
176     	boolean isHealth = false;
177     	try {
178             Class.forName(DB_DRIVER).newInstance();
179             log.info("Loaded the appropriate driver");
180             isHealth = true;
181         } catch (ClassNotFoundException cnfe) {
182             log.warn("Unable to load the JDBC driver " + DB_DRIVER);
183             log.warn("Please check your CLASSPATH.");
184             cnfe.printStackTrace(System.err);
185         } catch (InstantiationException ie) {
186         	log.warn("Unable to instantiate the JDBC driver " + DB_DRIVER);
187             ie.printStackTrace(System.err);
188         } catch (IllegalAccessException iae) {
189         	log.warn("Not allowed to access the JDBC driver " + DB_DRIVER);
190             iae.printStackTrace(System.err);
191         }
192         
193         return isHealth;
194     }

197      * Load the Derby driver.
198      * And create the table in first time.\
199      */
200 	public void open() {
201   	
202 		/* load the desired JDBC driver */
203 		isHealth = loadDriver();
204 		
205 		log.info("Derby EmbeddedDriver Loaded.");
206 			
207 		String connectURL = DB_PROTOCOL + repoLocation + DB_CREATE;
208 		
209 		try {
210 			conn = DriverManager.getConnection(connectURL);
211 			log.info("Connection Established");
212 		} catch (Exception e) {
213 			isHealth = false;
214 			e.printStackTrace();
215 		}
216 		
217 		// wipe table if erroneous or if user requests it by preferences.
218 		if (!existsTable() || ResourceCenter.getInstance().wipeRepositoryOnStartup()) {
219 			dropTable();
220 			createTable();
221 		}
222 	}

225 	 * checks whether the required tables exist or not.
226 	 * @return true if everything is ok, false otherwise.
227 	 */
228 	private boolean existsTable() {
229 		// we try to make a SQL query. if it fails, we assume the table to be dead...
230 		try {
231 			DatabaseMetaData dbMeta = conn.getMetaData();
232 			ResultSet resultSet = dbMeta.getColumns(null, null, "LLRP_MSG", null);
233 			int n = 0;
234 			while (resultSet.next()) {
235 				n++;
236 			}
237 			if (n<NUM_TABLE_COLUMNS) {
238 				throw new SQLException("missing fields");
239 			}
240 			 
241 		} catch (SQLException e) {
242 			log.error("table erroneous or missing. therefore recreate it.");
243 			return false;
244 		}
245 		return true;
246 	}

298 	/**
299 	 * returns all the messages from the specified adaptor and the reader 
300 	 * limited by num. if you set num to RETRIEVE_ALL all messages get returned.
301 	 * if you set readerName to null, all the messages of all the readers with 
302 	 * adaptor adaptorName will be returned.
303 	 * @param adaptorName the name of the adaptor.
304 	 * @param readerName the name of the reader.
305 	 * @param num how many messages to retrieve.
306 	 * @param content if true retrieve the message content, false no content.
307 	 * @return a list of messages.
308 	 */
309 	public ArrayList<LLRPMessageItem> get(
310 			String adaptorName, String readerName, int num, boolean content) {
311 		
312 		ArrayList<LLRPMessageItem> msgs = new ArrayList<LLRPMessageItem> ();
313 		
314 		try {
315 			PreparedStatement st = null;
316 			ResultSet results = null;
317 			String sql = null;
318 			if ((null == adaptorName) || 
319 					(ReaderExplorerViewContentProvider.ROOT_NAME.equals(
320 							adaptorName))) {	
321 				sql = SQL_SELECT_ALL;
322 				if (content) {
323 					sql = SQL_SELECT_ALL_CONTENT;
324 				}
325 				st = conn.prepareStatement(sql);
326 			} else if (readerName != null) {
327 				sql = SQL_SELECT_BY_ADAPTOR_AND_READER;
328 				if (content) {
329 					sql = SQL_SELECT_BY_ADAPTOR_AND_READER_CONTENT;
330 				}
331 				st = conn.prepareStatement(sql);
332 				st.setString(1, adaptorName.trim());
333 				st.setString(2, readerName.trim());
334 			} else {
335 				sql = SQL_SELECT_BY_ADAPTOR;
336 				if (content) {
337 					sql = SQL_SELECT_BY_ADAPTOR_CONTENT;
338 				}
339 				
340 				st = conn.prepareStatement(sql);
341 				st.setString(1, adaptorName.trim());
342 			}
343 			// bound the number of items to retrieve
344 			if (num != Repository.RETRIEVE_ALL) {
345 				st.setMaxRows(num);
346 			}
347 			results = st.executeQuery();
348 			while (results.next()) {
349 				LLRPMessageItem item = new LLRPMessageItem();
350 				item.setAdapter(results.getString(SELECTOR_ADAPTOR));
351 				item.setComment(results.getString(SELECTOR_COMMENT));
352 				item.setId(results.getString(SELECTOR_ID));
353 				item.setMark(results.getInt(SELECTOR_MARK));
354 				item.setMessageType(results.getString(SELECTOR_MESSAGE_TYPE));
355 				item.setReader(results.getString(SELECTOR_READER));
356 				item.setStatusCode(results.getString(SELECTOR_STATUS));
357 				item.setTime(
358 						new Date(results.getTimestamp(
359 								SELECTOR_TIMESTAMP).getTime()));
360 				if (content) {
361 					item.setContent(results.getString(SELECTOR_CONTENT));
362 				}
363 				msgs.add(item);
364 			}
365 			
366 		} catch (Exception e) {
367 			log.error(
368 					String.format(
369 							"could not retrieve from database: %s\n", 
370 							e.getMessage())
371 					);
372 		}
373 			
374 		return msgs;
375 	}

377 	public LLRPMessageItem get(String aMsgSysId) {
378 		
379 		LLRPMessageItem msg = new LLRPMessageItem();
380 		
381 		try {
382 			psSelect = conn.prepareStatement(SQL_SELECT_MSG);
383 			psSelect.setString(1, aMsgSysId);
384 			ResultSet results = psSelect.executeQuery();
385 			
386 			if (results.next()) {
387 				msg.setId(results.getString(SELECTOR_ID));
388 				msg.setMessageType(results.getString(SELECTOR_MESSAGE_TYPE));
389 				msg.setReader(results.getString(SELECTOR_READER));
390 				msg.setAdapter(results.getString(SELECTOR_ADAPTOR));
391 				msg.setTime(new Date(
392 						results.getTimestamp(SELECTOR_TIMESTAMP).getTime()));
393 				msg.setContent(results.getString(SELECTOR_CONTENT));
394 				msg.setComment(results.getString(SELECTOR_COMMENT));
395 				msg.setMark(results.getInt(SELECTOR_MARK));
396 				msg.setStatusCode(results.getString(SELECTOR_STATUS));
397 				
398 				log.debug("Get Message (ID=" + results.getString(1) + ") from JavaDB.");
399 			}
400 			
401 			psSelect.close();
402 			results.close();
403 			
404 		} catch (SQLException sqle) {
405             sqle.printStackTrace();
406 		}
407 		
408 		return msg;

411 	public void put(LLRPMessageItem aMessage) {
412 		try {
413 			psInsert = conn.prepareStatement(SQL_INSERT_MSG);
414 
415 			psInsert.setString(SELECTOR_ID, aMessage.getId());
416 			psInsert.setString(SELECTOR_MESSAGE_TYPE, aMessage.getMessageType());
417 			psInsert.setString(SELECTOR_READER, aMessage.getReader());
418 			String adaptor = aMessage.getAdapter();
419 			if (adaptor == null) {
420 				adaptor = AdaptorManagement.DEFAULT_ADAPTOR_NAME;
421 			}
422 			psInsert.setString(SELECTOR_ADAPTOR, adaptor);
423 			psInsert.setTimestamp(
424 					SELECTOR_TIMESTAMP, 
425 					new Timestamp(aMessage.getTime().getTime()));
426 			psInsert.setString(SELECTOR_CONTENT, aMessage.getContent());
427 			psInsert.setString(SELECTOR_COMMENT, aMessage.getComment());
428 			psInsert.setString(SELECTOR_MARK, "" + aMessage.getMark());
429 			psInsert.setString(SELECTOR_STATUS, aMessage.getStatusCode());
430 			
431 			psInsert.executeUpdate();
432 			psInsert.close();
433 			
434 			ResourceCenter.getInstance().addToMessageMetadataList(aMessage);
435 			
436 			if (aMessage.getMessageType().equals("GET_READER_CONFIG_RESPONSE")) {
437 				log.debug("Receiving GET_READER_CONFIG_RESPONSE, set the flag...");
438 				ResourceCenter.getInstance().addReaderConfig(aMessage.getAdapter(), aMessage.getReader(), aMessage.getId());
439 			}
440 			
441 			if (aMessage.getMessageType().equals("GET_ROSPECS_RESPONSE")) {
442 				log.debug("Receiving GET_ROSPECS_RESPONSE, set the flag...");
443 				ResourceCenter.getInstance().addReaderROSpec(aMessage.getAdapter(), aMessage.getReader(), aMessage.getId());
444 			}
445 			
446 			log.debug("Put Message (ID=" + aMessage.getId() + ") into JavaDB.");
447 		} catch (SQLException sqle) {
448             sqle.printStackTrace();
449 		}
450 	}

466 	public int count(String adaptor, String reader) {
467 				
468 		int rowcount = 0;
469 		try {
470 			Statement stmt = conn.createStatement();
471 			String query = "SELECT COUNT(*) FROM LLRP_MSG";
472 			if (null == adaptor) {
473 				// all ok
474 			} else if (null == reader) {
475 				// restrict to adaptor
476 				query = String.format("%s WHERE ADAPTER='%s'", query, adaptor);
477 			} else {
478 				query = String.format("%s WHERE ADAPTER='%s' AND READER='%s'", 
479 						query, adaptor, reader);
480 			}
481 			ResultSet resultSet = stmt.executeQuery(query);
482 	    
483 	        // Get the number of rows from the result set
484 	        resultSet.next();
485 	        rowcount = resultSet.getInt(1);
486 	        stmt.close();
487 	        resultSet.close();
488 		} catch (SQLException e) {
489 			log.error("Could not retrieve the number of messages: " + 
490 					e.getMessage());
491 		}
492 		
493 		return rowcount;
494 	}
