25  /**
26   * ByteBlock represents a block of bytes of arbitrary length.
27   * 
28   * The bytes are stored internally in a byte array, howerver
29   * the bytes are numbered from left to right in contrary to 
30   * numbers in hexadecimal format. The bytes in the array are
31   * also considered unsigned in contrast to Java. This allows
32   * the usage of ByteBlock to store output of hardware devices.
33   * 
34   * The following is an example of the String representation of 
35   * a byte block of length 12:
36   * <code>10FF024E6A0000A1611D6400</code>.
37   * 
38   * In addition the class provides several static byte array support
39   * functions, which allow the usage of ByteBlock without creating
40   * objects of the class.
41   * 
42   * @version	1.0, 03/2003
43   * @author 	Matthias Lampe, lampe@acm.org
44   */

54  	/**
55  	 * creates a byte block from the given byte array.
56  	 *
57  	 * @param bytes the byte array, which is copied into the
58  	 * 					internal representation.
59  	 */
60  	public ByteBlock(byte[] bytes) {
61  		// copy source to internal byte array
62  		this.bytes = new byte[bytes.length];
63  		for (int i = 0; i < bytes.length; i++)
64  			this.bytes[i] = bytes[i];
65  	}

196 	/**
197 	 * returns a string with the byte of the input array in hexadecimal
198 	 * representation trimed/padded to a specific length.
199 	 * 
200 	 * If length is less than the length of the input array, the most
201 	 * significant bits (low array indices) are cut.
202 	 * 
203 	 * If length is greater than the length of the input array, "00"
204 	 * bytes are added in front of the resulting string.
205 	 * 
206 	 * @param byteArray
207 	 * @param length
208 	 * @return a string with the bytes of the input array in hexadecimal
209 	 *          representation.
210 	 */
211 	static public String byteArrayToHexString(byte[] byteArray, int length) {
212 		byte[] data = new byte[length];
213 		int pad = length - byteArray.length;
214 		if (pad > 0) {
215 			System.arraycopy(byteArray, 0, data, pad, byteArray.length);
216 		} else {
217 			System.arraycopy(byteArray, -pad, data, 0, length);
218 		}
219 		return byteArrayToHexString(data);
220 	}

222 	/**
223 	 * converts a String of hex bytes to a byte array.
224 	 *
225 	 * @param hexString the byte array as a String.
226 	 */
227 	static public byte[] hexStringToByteArray(String hexString)
228 	{
229 		byte[] byteArray;
230 
231 		int len = hexString.length();
232 		byteArray = new byte[len/2];
233 		if (len % 2 != 0) throw new IllegalArgumentException();
234 		for (int i = 0; i < len/2; i++){
235 			//String byteString = "0x"+hexString.substring(i*2, i*2+2);
236 			//Integer intValue = Integer.decode(byteString);
237 			String byteString = hexString.substring(i*2, i*2+2);
238 			Integer intValue = new Integer(Integer.parseInt(byteString, 16));
239 			byte byteValue = intValue.byteValue();
240 			byteArray[i] = byteValue;
241 		}
242 
243 		return byteArray;
244 	}

247 	 * converts a String of hex byte to a byte array of a specific length.
248 	 * 
249 	 * If length is less than the converted String, the most significant
250 	 * bits (low array indices) are cut.
251 	 * 
252 	 * If length is greater than the converted String, the resulting array
253 	 * is padded with 0 bytes at the beginning (low array indices).
254 	 * 
255 	 * @param hexString
256 	 * @param length
257 	 * @return a byte array derived from the hexadecimal representation given
258 	 */
259 	static public byte[] hexStringToByteArray(String hexString, int length) {
260 		byte[] data = new byte[length];
261 		
262 		int pad = length - hexString.length()/2;
263 		byte[] byteArray = hexStringToByteArray(hexString);
264 		if (pad > 0) {
265 			System.arraycopy(byteArray, 0, data, pad, byteArray.length);
266 		} else {
267 			System.arraycopy(byteArray, -pad, data, 0, length);
268 		}
269 		
270 		return data;
271 	}

285 	 * converts an unsigned byte array (MSB first, max first four bytes) to a
286     * (signed) integer.
287 	 * 
288 	 * @param b
289 	 * @return the value representated by the input byte array
290 	 */
291 	public static int bytesToNumber(byte[] b) {
292 		int val = 0;
293       int len = b.length;
294       if (len > 4) {
295          len = 4;
296       }
297 		for (int i = 0; i < len; i++) {
298 			val |= byteToNumber(b[i]) << (8 * (len - i - 1));
299 		}
300 		return val;
301 	}

303 	/**
304 	 * creates a byte array out of the given number. The numbering
305 	 * of the bytes starts from the left to the right in contrary
306 	 * to usual numbering e.g. 
307 	 * <code>n=0134A0FF002EC0B1</code> returns the array
308 	 * <code>{ 01,34,A0,FF,00,2E,C0,B1 }</code>
309 	 * 
310 	 * @param n the number
311 	 * 
312 	 * @return the byte array representing the number.
313 	 */
314 	public static byte[] numberToByteArray(long n) 
315 	{
316 		byte[] ba = new byte[8];
317 		byte b;
318 
319 		// go through all bytes and filter them out
320 		for (int s = 0; s < 8; s++) {
321 			b = (byte) ((n >> ((8 - 1 - s) * 8)) & 0xFF);
322 			ba[s] = b;
323 		}
324 		
325 		return ba;
326 	}

377 	 * checks if two given byte block regions are equal or not.
378 	 * 
379 	 * @param firstArray the first byte array.
380 	 * @param firstOffset the offset of the first block in the array.
381 	 * @param secondArray the second byte array.
382 	 * @param secondOffset the offset of the second block in the array.
383 	 * @param length the length of the blocks in the arrays.
384 	 * 
385 	 * @return true, if both byte block regions are equal.
386 	 */
387 	public static boolean equalsRegion(
388 				byte[] firstArray, int firstOffset,
389 				byte[] secondArray, int secondOffset, 
390 				int length) 
391 	{
392 		int firstIndex = firstOffset;
393 		int secondIndex = secondOffset;
394 		
395 		//-- compare byte blocks in arrays
396 		while (firstIndex < firstArray.length
397 				&& secondIndex < secondArray.length
398 				&& firstIndex < firstOffset + length
399 				&& secondIndex < secondOffset + length
400 				&& firstArray[firstIndex] == secondArray[secondIndex]) {
401 			firstIndex++;
402 			secondIndex++;
403 		}
404 
405 		//-- check if end reached, i.e. blocks are equal
406 		if (firstIndex == firstOffset + length
407 			&& secondIndex == secondOffset + length) 
408 			return true;
409 		else
410 			return false;
411 	}

431 	 * compares two given byte block regions. 
432 	 * 
433 	 * @param firstArray the first byte array.
434 	 * @param firstOffset the offset of the first block in the array.
435 	 * @param secondArray the second byte array.
436 	 * @param secondOffset the offset of the second block in the array.
437 	 * @param length the length of the blocks in the arrays.
438 	 * 
439 	 * @return -1, if first block < second block,
440 	 * 			 0, if first block == second block,
441 	 * 			 1, if first block > second block.
442 	 */
443 	public static int compareRegion(
444 				byte[] firstArray, int firstOffset,
445 				byte[] secondArray, int secondOffset, 
446 				int length) 
447 	{
448 		int firstIndex = firstOffset;
449 		int secondIndex = secondOffset;
450 		
451 		//-- compare byte blocks in arrays
452 		while (firstIndex < firstArray.length
453 				&& secondIndex < secondArray.length
454 				&& firstIndex < firstOffset + length
455 				&& secondIndex < secondOffset + length) {
456 			int res = compare(firstArray[firstIndex], secondArray[secondIndex]);
457 			if (res != 0)
458 				return res;
459 			
460 			firstIndex++;
461 			secondIndex++;
462 		}
463 		
464 		// end of blocks reached => blocks equal
465 		return 0;
466 	}

481 		int s = second & 0x000000FF;

483 		if (f < s)
