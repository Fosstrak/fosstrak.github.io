40  /**
41   * ISOProtocol is an interface to the ISO 15693 compliant RFID-Reader FEIG ID
42   * ISC.MR100. It allows to send a request to the reader. The sendRequest method
43   * then locks until a response from the reader has arrived or a timeout occured.
44   * A request consists of a comAdr, which is the bus-address of the reader, a
45   * control byte, which specifies the command that the reader should execute and
46   * some data-bytes, which state more precisely, what's asked for. When
47   * terminating the session with the reader one should close the serial-port with
48   * the closePort method.
49   *
50   * @author Simon Keel, skeel@student.ethz.ch
51   * @version 1.0
52   */
53  public class ISOProtocol implements SerialPortEventListener {

133 	/**
134 	 * opens the com-port.
135 	 *
136 	 * @param portName
137 	 *            string representing the port name for the current os
138 	 * @param baud
139 	 *            the baud rate for the serial port communication.
140 	 * @param databits
141 	 *            the databits for the serial port communication.
142 	 * @param stopbits
143 	 *            the stopbits for the serial port communication.
144 	 * @param parity
145 	 *            then parity for the serial port communication.
146 	 * @throws UnsupportedCommOperationException
147 	 *             if setting port parameters fails
148 	 * @throws PortInUseException
149 	 *             if opening port fails
150 	 * @throws IOException
151 	 *             if geting in-/outputstream fails or if there are too many
152 	 *             listeners for this port
153 	 */
154 	private void openPort(String portName, int baud, int databits,
155 			int stopbits, int parity) throws UnsupportedCommOperationException,
156 			PortInUseException, IOException {
157 		Enumeration portList = CommPortIdentifier.getPortIdentifiers();
158 		CommPortIdentifier portId;
159 
160 		while (portList.hasMoreElements()) {
161 			portId = (CommPortIdentifier) portList.nextElement();
162 			if (portId.getPortType() == CommPortIdentifier.PORT_SERIAL) {
163 				if (portId.getName().equals(portName)) {
164 
165 					// Port found. Open it.
166 					// (maybe throws PortInUseException)
167 					serialPort = (SerialPort) portId.open("RFID ISOProtocol",
168 							openPortTimeout);
169 
170 					// setting default port parameters
171 					// (maybe throws UnsupportedCommOperationException)
172 					serialPort.setSerialPortParams(baud, databits, stopbits,
173 							parity);
174 
175 					// add EventListener
176 					try {
177 						serialPort.addEventListener(this);
178 					} catch (TooManyListenersException e) {
179 						throw new IOException(e.getMessage());
180 					}
181 
182 					serialPort.notifyOnDataAvailable(true);
183 					serialPort.notifyOnBreakInterrupt(true);
184 					serialPort.notifyOnFramingError(true);
185 					serialPort.notifyOnOverrunError(true);
186 					serialPort.notifyOnParityError(true);
187 
188 					// assign Input- and OutputStream
189 					// (maybe throws IOException)
190 					out = serialPort.getOutputStream();
191 					in = serialPort.getInputStream();
192 
193 					// return
194 					return;
195 
196 				}
197 			}
198 		}
199 
200 		// port not found
201 		throw new IOException("Serial port " + portName
202 				+ " not found (maybe it is locked).");
203 		// NoSuchPortException not yet implemented???
204 
205 	}

222 	 * sends the request (COM-ADR, CONTROLBYTE and some DATA) to the reader.
223 	 * This method method locks until a response from the reader has arrived or
224 	 * a timeout occured.
225 	 *
226 	 * @param requestRecord
227 	 *            a record containing of COM-ADR, CONTROLBYTE and some DATA to
228 	 *            be sent to the reader
229 	 * @return the response from the reader. STATUSBYTE and DATA encapsulated in
230 	 *         a ResponseRecord.
231 	 * @throws IOException
232 	 *             in case of a serial port timeout or any other io exception.
233 	 */
234 	public synchronized ResponseRecord sendRequest(RequestRecord requestRecord)
235 			throws IOException {
236 
237 		// create new send-frame
238 		byte[] sendFrame = composeReaderSendFrame(requestRecord);
239 
240 		return sendFrame(sendFrame);
241 	}

243 	/**
244 	 * Sends a request to a function unit (multiplexer, antenna tuner)
245 	 * attached to the reader.
246 	 *
247 	 * @param requestRecord
248 	 * @return the response
249 	 * @throws IOException
250 	 */
251 	public synchronized ResponseRecord sendFunctionUnitRequest(
252 			RequestRecord requestRecord) throws IOException {
253 
254 	   // create request data
255 		byte[] fuSendFrame = composeFUSendFrame(requestRecord);
256       byte[] transparent = new byte[] { 0x01, 0x00, 0x20, 0x02, 0x1F};
257       byte[] data = new byte[transparent.length + fuSendFrame.length];
258       System.arraycopy(transparent, 0, data, 0, 5);
259       System.arraycopy(fuSendFrame, 0, data, 5, fuSendFrame.length);
260 
261       RequestRecord req = new RequestRecord();
262       req.comAdr = (byte) 0xFF;
263       req.controlByte = (byte) 0xBF; // (ISO transparent command)
264       req.data = data;
265 
266 		return sendRequest(req);
267 	}

269 	private synchronized ResponseRecord sendFrame(byte[] frame) throws IOException {
270 
271 		while (sendRequestActive) {
272 			try {
273 				wait();
274 			} catch (InterruptedException e) {
275 			}
276 		}
277 		sendRequestActive = true;
278 		// send request (may throw IOException)
279 		out.write(frame);
280 		serialEventExpected = true;
281 		// for debugging only...
282 		// System.out.println(" sendFrame: " +
283 		// ByteBlock.byteArrayToHexString(sendFrame));
284 
285 		errorMsg = null;
286 
287 		// wait for serialEvent to return a response record
288 		long t0 = System.currentTimeMillis();
289 		long time = 0;
290 		while ((responseRecord == null) && (time < timeout)) {
291 			try {
292 				wait(timeout - time);
293 			} catch (InterruptedException e) {
294 			}
295 			time = System.currentTimeMillis() - t0;
296 		}
297 		serialEventExpected = false;
298 		if (responseRecord == null) {
299 			throw new IOException("Serial port timeout.");
300 		}
301 
302 		// Did an IOException occur?
303 		if (errorMsg != null) {
304 			responseRecord = null;
305 			throw new IOException(errorMsg);
306 		}
307 
308 		// transfer responseRecord
309 		ResponseRecord resp = new ResponseRecord();
310 		resp.status = responseRecord.status;
311 		resp.data = responseRecord.data;
312 		responseRecord = null;
313 
314 		// reader needs ~50 ms before it is ready for the next command
315 		try {
316 			Thread.sleep(50);
317 		} catch (InterruptedException e) {
318 		}
319 
320 		// allow another sendRequest
321 		sendRequestActive = false;
322 		notifyAll();
323 		return resp;
324 	}

326 	/**
327 	 * creates a new send-frame.
328 	 *
329 	 * @param requestRecord,
330 	 *            the data to be sent to the reader.
331 	 * @return the composed frame ready for sending to the reader.
332 	 * @throws IOException
333 	 *             if the send frame is longer then 255 bytes.
334 	 */
335 	static byte[] composeReaderSendFrame(RequestRecord requestRecord)
336 			throws IOException {
337 
338 		byte[] frame;
339 
340 		// calculating LENGTH
341 		int dataLen;
342 		if (requestRecord.data != null)
343 			dataLen = requestRecord.data.length;
344 		else
345 			dataLen = 0;
346 
347 		int length = 1 + 1 + 1 + dataLen + 2; // Wahrscheinlich: Header 3Bytes
348 												// + Daten + Checksumme 2Bytes
349 		if (length > 255) {
350 			throw new IOException("ISOProtocol send frame too long. (>255)");
351 		}
352 		byte[] len = ByteBlock.numberToByteArray(length);
353 
354 		// constructing send-frame
355 		frame = new byte[length];
356 		frame[0] = len[3];
357 		frame[1] = requestRecord.comAdr;
358 		frame[2] = requestRecord.controlByte;
359 		for (int i = 0; i < dataLen; i++) {
360 			frame[3 + i] = requestRecord.data[i];
361 		}
362 		// calculating crc16-checksum
363 		byte[] temp = new byte[length - 2];
364 		for (int i = 0; i < length - 2; i++)
365 			temp[i] = frame[i];
366 		byte[] cs = CRC16.crc16(temp);
367 		frame[length - 2] = cs[0];
368 		frame[length - 1] = cs[1];
369 
370 		return frame;
371 	}

373 	/**
374 	 * creates a new send-frame.
375 	 *
376 	 * @param requestRecord,
377 	 *            the data to be sent to the reader.
378 	 * @return the composed frame ready for sending to the reader.
379 	 * @throws IOException
380 	 *             if the send frame is longer then 255 bytes.
381 	 */
382 	static byte[] composeFUSendFrame(RequestRecord requestRecord)
383 			throws IOException {
384 
385 		byte[] frame;
386 
387 		// calculating LENGTH
388 		int dataLen;
389 		if (requestRecord.data != null)
390 			dataLen = requestRecord.data.length;
391 		else
392 			dataLen = 0;
393 
394 		int length = 3 + dataLen;
395 
396 		if (length > 255) {
397 			throw new IOException("ISOProtocol send frame too long. (>255)");
398 		}
399 
400 		// constructing send-frame
401 		frame = new byte[length];
402 		frame[0] = 0x02;
403 		frame[1] = requestRecord.controlByte;
404 		frame[2] = (byte) 0xfe;
405 
406 		for (int i = 0; i < dataLen; i++) {
407 			frame[3 + i] = requestRecord.data[i];
408 		}
409 
410 		return frame;
411 	}

413 	/**
414 	 * is called by the underlying driver when an event occurs. This methods
415 	 * gets the answer from the reader and creates a responseRecord which is
416 	 * delivered to the sendRequest method
417 	 *
418 	 * @param event
419 	 *            an instance representing the serial event, containing data
420 	 *            about the type of the event.
421 	 */
422 	public synchronized void serialEvent(SerialPortEvent event) {
423 
424 		boolean functionUnitResponse = false;
425 
426 		// Ignore "spontaneous" events
427 		if (!serialEventExpected) {
428 			return;
429 		}
430 
431 		byte[] buf = null;
432 		try {
433 
434 			// receiving data...
435 			switch (event.getEventType()) {
436 			case SerialPortEvent.BI:
437 				throw new IOException("Serial port break interrupt.");
438 			case SerialPortEvent.OE:
439 				throw new IOException("Serial port overrun error.");
440 			case SerialPortEvent.FE:
441 				throw new IOException("Serial port framing error.");
442 			case SerialPortEvent.PE:
443 				throw new IOException("Serial port parity error.");
444 			case SerialPortEvent.DATA_AVAILABLE:
445 
446 				// read first byte
447 				byte[] lenByte = new byte[1];
448 				int stat = in.read(lenByte, 0, 1);
449 				if (stat != 1) {
450 					throw new IOException("No data available at serial port.");
451 				}
452 
453 				// get length of response frame
454 				int len = Integer.parseInt(ByteBlock
455 						.byteToHexString(lenByte[0]), 16);
456 
457 
458 				if (len == 0) {
459 					// Handle events from function units (read until eof)
460 					functionUnitResponse = true;
461 
462 					// TODO Determine buffer size (8 would probably be enough)
463 					buf = new byte[16];
464 					stat = in.read(buf);
465 
466 					if (stat <= 0) {
467 						throw new IOException(
468 						"Unexpected \"stream at end of file\" while reading ");
469 					}
470 
471 					if (in.available() > 0) {
472 						throw new IOException(
473 								"ISOProtocol response frame too long.");
474 					}
475 					byte[] newBuf = new byte[stat];
476 					System.arraycopy(buf, 0, newBuf, 0, stat);
477 					buf = newBuf;
478 
479 				} else {
480 					buf = new byte[len - 1];
481 					// read len- 1 bytes
482 					stat = 0;
483 					int temp;
484 					while (stat != len - 1) {
485 						temp = in.read(buf, stat, len - 1 - stat);
486 						if (temp == -1) {
487 							throw new IOException(
488 									"Unexpected \"stream at end of file\" while reading ");
489 						}
490 						stat += temp;
491 					}
492 					if (in.available() > 0) {
493 						throw new IOException(
494 								"ISOProtocol response frame too long.");
495 					}
496 
497 					// verify checksum
498 					byte[] frame = new byte[len];
499 					frame[0] = lenByte[0];
500 					for (int i = 1; i < len; i++)
501 						frame[i] = buf[i - 1];
502 					byte[] cs = CRC16.crc16(frame);
503 					if ((cs[0] != 0) || (cs[1] != 0)) {
504 						throw new IOException(
505 								"ISOProtocol response frame checksum error.");
506 					}
507 				}
508 
509 				break;
510 			}
511 
512 		} catch (IOException e) {
513 			errorMsg = e.getMessage();
514 		}
515 
516 		// create responseRecord
517 		responseRecord = new ResponseRecord();
518 		if (errorMsg == null) {
519 			if (!functionUnitResponse) {
520 				// Reader ISO responses
521 				responseRecord.status = buf[2];
522 				byte[] data = new byte[buf.length - 5];
523 				for (int i = 3, j = 0; i < buf.length - 2; i++) {
524 					data[j++] = buf[i];
525 				}
526 				responseRecord.data = data;
527 			} else {
528 				// Function unit responses
529 				responseRecord.status = buf[0];
530 				byte[] data = new byte[buf.length - 1];
531 				for (int i = 1, j = 0; i < buf.length; i++) {
532 					data[j++] = buf[i];
533 				}
534 				responseRecord.data = data;
535 			}
536 
537 		}
538 
539 		// for debugging only...
540 		// System.out.println(" buf: (without length) " +
541 		// ByteBlock.byteArrayToHexString(buf));
542 		// System.out.println();
543 
544 		// allow sendRequest-method
545 		notifyAll();
546 
547 	}

549 	/**
550 	 * returns the port-name for a given serial-port-number for the running OS.
551 	 *
552 	 * @param portNr
553 	 *            the port nnumber for which the portname should be generated
554 	 * @return a string representing the port name
555 	 * @throws UnsupportedCommOperationException
556 	 *             if the Operating System is not known by the method
557 	 */
558 	static public String getSerialPortNameForOS(int portNr)
559 			throws UnsupportedCommOperationException {
560 		char letter[] = { 'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j',
561 				'k', 'l', 'm', 'n', 'o', 'p', 'q', 'r', 's', 't', 'u', 'v',
562 				'w', 'x', 'y', 'z' };
563 		String os = System.getProperty("os.name");
564 
565 		if ((portNr > 26) || (portNr < 1)) {
566 			throw new UnsupportedCommOperationException(
567 					"Serial port number must be between 1 and 26.");
568 		}
569 
570 		if (os.equals("Linux")) {
571 			return "/dev/ttyS" + (portNr - 1);
572 		} else if (os.equals("SunOS")) {
573 			return "/dev/term/" + letter[portNr - 1];
574 		} else if (os.startsWith("Windows")) {
575 			return "COM" + portNr;
576 		} else {
577 			throw new UnsupportedCommOperationException(
578 					"ISOProtocol: Unknown Operating System.");
579 		}
580 	}
