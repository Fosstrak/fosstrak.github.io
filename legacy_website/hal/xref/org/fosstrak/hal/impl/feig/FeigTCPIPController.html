118 	/**
119 	 * Initialize the reader hardware.
120 	 *
121 	 * @throws HardwareException
122 	 */
123 	public void initialize() throws HardwareException {
124 
125       // read parameters from configuration file
126       config = new XMLConfiguration();
127       config.setListDelimiter(',');
128       URL fileurl = ResourceLocator.getURL(configFile, defaultConfigFile,
129          this.getClass());
130       try {
131          config.load(fileurl);
132          // get parameters
133          this.address = config.getString("address");
134          this.port = config.getInt("port");
135          this.timeout = config.getInt("timeout");
136          this.epcTransponderModelsConfig = config.getString(
137             "epcTransponderModelsConfig");
138          readPoints = new HashMap<String, Integer>();
139          antennaNames = new HashMap<Integer, String>();
140          numberOfReadPoints = config.getMaxIndex("readpoint") + 1;
141          if (numberOfReadPoints > 4) {
142             numberOfReadPoints = 4;
143          }
144          for (int i = 0; i < numberOfReadPoints; i++) {
145             // key to current read point
146             String key = "readpoint(" + i + ")";
147             // read point name
148             String readpointName = config.getString(key + ".name");
149             log.debug("Property found: " + key + ".name = " + readpointName);
150             // read point connector
151             int readpointConnector = config.getInt(key + ".connector");
152             log.debug("Property found: " + key + ".connector = " + readpointConnector);
153             if ((readpointConnector < 1) || (readpointConnector > 4)) {
154                log.error("Readpoint connector value out of range, readpoint '" +
155                   readpointName + "' not added to read points.");
156             } else {
157                readPoints.put(readpointName, readpointConnector);
158                antennaNames.put(readpointConnector, readpointName);
159             }
160          }
161       } catch (ConfigurationException e) {
162          log.error("Properties file not found.");
163          throw new HardwareException("Properties file not found.");
164       }
165 
166 		// Initialize the reader hardware
167 		try {
168 			log.info("  trying address: " + address + " port: " + port + " ...");
169 			initReader(address, port);
170 			log.info("reader initialized.");
171 			return;
172 		} catch (HardwareException e) {
173 			String message = "Error initializing reader";
174 			log.error("initialize: " + message, e);
175 			throw new HardwareException(message, e);
176 		}
177 
178 	}

185 	public Observation[] identify(String[] readPointNames)
186 			throws HardwareException {
187 
188 		currentInventory.clear();
189 
190       Observation[] observations = new Observation[readPointNames.length];
191       for (int i = 0; i < readPointNames.length; i++) {
192    		observations[i] = new Observation();
193    		observations[i].setHalName(getHALName());
194    		observations[i].setReadPointName(readPointNames[i]);
195 
196          // Select channel
197          if (!selectReadPoint(readPointNames[i])) {
198             log.error("readBytes: Read point not available");
199             throw new HardwareException("Read point not available");
200          }
201 
202    		Vector<InventoryItem> items = this.getInventory();
203          Vector<String> ids = new Vector<String>();
204          Vector<TagDescriptor> tds = new Vector<TagDescriptor>();
205 
206    		for (InventoryItem item : items) {
207    			String id = item.id;
208 
209             if (item.transponderType == TransponderType.EPCclass1Gen2) {
210 					IDType idType = IDType.getIdType("EPC",
211 							config.getString("idTypesConfig"));
212                EPCTransponderModel tagModel = item.epcTransponderModel;
213                MemoryBankDescriptor[] memoryBankDescriptors =
214                   new MemoryBankDescriptor[4];
215                memoryBankDescriptors[0] = new MemoryBankDescriptor(
216                   tagModel.getReservedSize(), tagModel.getReservedReadable(),
217                   tagModel.getReservedWriteable());
218                memoryBankDescriptors[1] = new MemoryBankDescriptor(
219                      tagModel.getEpcSize(), tagModel.getEpcReadable(),
220                      tagModel.getEpcWriteable());
221                memoryBankDescriptors[2] = new MemoryBankDescriptor(
222                      tagModel.getTidSize(), tagModel.getTidReadable(),
223                      tagModel.getTidWriteable());
224                memoryBankDescriptors[3] = new MemoryBankDescriptor(
225                      tagModel.getUserSize(), tagModel.getUserReadable(),
226                      tagModel.getUserWriteable());
227                MemoryDescriptor memoryDescriptor = new MemoryDescriptor(
228                      memoryBankDescriptors);
229                TagDescriptor td = new TagDescriptor(idType, memoryDescriptor);
230                tds.add(td);
231             }
232 
233    			item.readPoint = readPointNames[i];
234    			ids.add(id);
235    			currentInventory.put(id, item);
236    		}
237 
238    		int len = ids.size();
239    		String[] ids_arr = new String[len];
240    		ids_arr = ids.toArray(ids_arr);
241          if (tds.size() == len) {
242             TagDescriptor[] tds_arr = new TagDescriptor[len];
243             tds_arr = tds.toArray(tds_arr);
244             observations[i].setTagDescriptors(tds_arr);
245          }
246          observations[i].setIds(ids_arr);
247    		observations[i].setTimestamp(System.currentTimeMillis());
248       }
249       // turn readpoints off after inventories
250       selectAntennaConnector(0);
251 
252 		return observations;
253 	}

255    /**
256     * Select a read point
257     *
258     * @param readpointName
259     *          the name of the read point
260     * @return
261     *          true if read point selected
262     */
263    protected boolean selectReadPoint(String readpointName) {
264 
265       boolean selected = false;
266 
267       if (readpointName.equals(selectedReadPoint)) {
268          selected = true;
269       }
270 
271       if (!selected) {
272          Integer connector = readPoints.get(readpointName);
273          if (connector != null) {
274             try {
275                selectAntennaConnector(connector);
276                selected = true;
277             } catch (Exception e) {}
278          }
279       }
280 
281       return selected;
282    }

284    /**
285     * Selects an antenna connector
286     *
287     * @param connector
288     *          the number of the connector to select (1-4)
289     * @throws HardwareException
290     */
291    synchronized protected void selectAntennaConnector(int connector)
292          throws HardwareException {
293 
294       // construct request record...
295       RequestRecord req = new RequestRecord();
296       req.comAdr = (byte) 0xff;
297       req.controlByte = (byte) 0x6a; // (RF Output ON/OFF)
298       switch (connector) {
299          case 0:
300             req.data = ByteBlock.hexStringToByteArray("80");
301             break;
302          case 1:
303             req.data = ByteBlock.hexStringToByteArray("81");
304             break;
305          case 2:
306             req.data = ByteBlock.hexStringToByteArray("82");
307             break;
308          case 3:
309             req.data = ByteBlock.hexStringToByteArray("83");
310             break;
311          case 4:
312             req.data = ByteBlock.hexStringToByteArray("84");
313             break;
314          default:
315             log.error("Can not select connector " + connector + ", switching "
316                + "all antennas off.");
317             req.data = ByteBlock.hexStringToByteArray("80");
318       }
319 
320       ResponseRecord resp = null;
321 
322       try {
323          resp = tcpip.sendRequest(req);
324       } catch (IOException e) {
325          log.error("selectAntennaConnector: Communication error with reader", e);
326          throw new HardwareException("Communication error with reader");
327       }
328 
329       StatusByte status = StatusByte.getStatus(resp.status);
330       String statusMessage = getStatusMessage(status);
331       if (resp.status != (byte) 0x00) {
332          log.error(statusMessage);
333          throw new HardwareException();
334       }
335    }

358 	 * (non-Javadoc)
359 	 *
360 	 * @see org.fosstrak.hal.HardwareAbstraction#readBytes(java.lang.String,
361 	 *      java.lang.String, int, int, int, java.lang.String[])
362 	 */
363 	public UnsignedByteArray readBytes(String readPointName, String id,
364          int memoryBank, int offset, int length, String[] passwords)
365 			throws HardwareException {
366 
367 		// Select channel
368 		if (!selectReadPoint(readPointName)) {
369 			log.error("readBytes: Read point not available");
370 			throw new HardwareException("Read point not available");
371 		}
372 
373 		int blockSize = getBlockSize();
374 
375 		if (blockSize == 0) {
376 			log.error("readBytes: Detected transponder blocksize is not valid");
377 			return new UnsignedByteArray(new byte[0]);
378 		}
379 
380 		int blockStart = offset / blockSize;
381 		int blockEnd = (offset + length - 1) / blockSize;
382 		int blockCount = blockEnd - blockStart + 1;
383 
384       byte[] idbytes = ByteBlock.hexStringToByteArray(id);
385 
386       // read words and extract requested bytes
387       int tries = (passwords == null) ? 2 : (passwords.length + 2);
388       byte[] password;
389       byte[] data = null;
390       boolean success = false;
391       for (int i = 0; i < tries; i++) {
392          // get password for this try
393          if (i == (tries - 1)) {
394             password = new byte[] { 0x00, 0x00, 0x00, 0x00 };
395          } else if (i == (tries - 2)) {
396             password = null;
397          } else {
398             password = ByteBlock.hexStringToByteArray(passwords[i]);
399          }
400 
401          // read data
402          try {
403             data = readMemoryBlocks(idbytes, memoryBank, blockStart,
404                blockCount, password);
405             success = true;
406             break;
407          } catch (Exception e) {}
408       }
409 		if (!success) {
410 			log.error("readBytes: Communication error with reader");
411 			throw new HardwareException("Communication error with reader");
412 		}
413 
414 		byte[] dataTrim = new byte[length];
415 		System.arraycopy(data, offset % blockSize, dataTrim, 0, length);
416 
417 		return new UnsignedByteArray(dataTrim);
418 	}

420 	private byte[] readMemoryBlocks(byte[] id, int bank, int address,
421          int length, byte[] password) throws IOException, HardwareException {
422 
423 		// log.debug("readMemoryBlocks: started");
424 
425 		RequestRecord req = new RequestRecord();
426 		req.comAdr = (byte) 0xff;
427 		// Set ISO15693 host command
428 		req.controlByte = (byte) 0xb0;
429 
430       int datalen = 7 + id.length;
431       if (password != null) {
432          datalen = datalen + 1 + password.length;
433       }
434 		byte[] readCommand = new byte[datalen];
435 		// Set read command (0x23)
436 		readCommand[0] = (byte) 0x23;
437 		// Set mode (addressed, UID_LF set, EXT_ADDR set)
438 		readCommand[1] = (byte) 0x31;
439       // Set UID length
440       readCommand[2] = (byte) id.length;
441 		// Set tag address (UID)
442 		System.arraycopy(id, 0, readCommand, 3, id.length);
443       // Set bank
444       readCommand[3 + id.length] = (byte) bank;
445       if (password != null) {
446          // Set A_FLAG
447          readCommand[3 + id.length] |= (byte) 0x80;
448          // Set password length
449          readCommand[4 + id.length] = (byte) password.length;
450          // Set password
451          System.arraycopy(password, 0, readCommand, 5 + id.length,
452                password.length);
453          readCommand[5 + id.length + password.length] =
454             (byte) ((address >> 8) & 0xFF);
455          readCommand[6 + id.length + password.length] =
456             (byte) (address & 0xFF);
457          readCommand[7 + id.length + password.length] = (byte) (length & 0xFF);
458       } else {
459          readCommand[4 + id.length] = (byte) ((address >> 8) & 0xFF);
460          readCommand[5 + id.length] = (byte) (address & 0xFF);
461          readCommand[6 + id.length] = (byte) (length & 0xFF);
462       }
463 		req.data = readCommand;
464 
465 		ResponseRecord resp = tcpip.sendRequest(req);
466 
467 		StatusByte status = StatusByte.getStatus(resp.status);
468 		String statusMessage = getStatusMessage(status);
469 		if (status != StatusByte.OK) {
470 			log.error(statusMessage);
471 			throw new HardwareException("Reading data blocks failed\n" + " "
472 			      + statusMessage + "/n");
473 		}
474 
475       byte[] data = resp.data;
476       int dbn = data[0];
477       int dbsize = data[1];
478 
479       byte[] readBytes = new byte[dbn * dbsize];
480       for (int i = 0; i < dbn; i++) {
481          for (int j = 0; j < dbsize; j++) {
482             readBytes[i * dbsize + j] = data[3 + i * (dbsize + 1) + j];
483          }
484       }
485 
486 		// log.debug("readMemoryBlocks: finished successfuly");
487 
488 		return readBytes;
489 	}

491 	/** TODO: test
492 	 * (non-Javadoc)
493 	 *
494 	 * @see org.fosstrak.hal.HardwareAbstraction#writeBytes(java.lang.String,
495 	 *      java.lang.String, int, int, UnsignedByteArray, java.lang.String[])
496 	 */
497 	public void writeBytes(String readPointName, String id, int memoryBank,
498 			int offset, UnsignedByteArray data, String[] passwords)
499 			throws HardwareException, UnsupportedOperationException {
500 
501       byte[] idbytes = ByteBlock.hexStringToByteArray(id);
502       byte[] databytes = data.toByteArray();
503       int datalen = databytes.length;
504 
505 		// Select channel
506 		if (!selectReadPoint(readPointName)) {
507 			throw new HardwareException("Read point not available");
508 		}
509 
510 		int blockSize = getBlockSize();
511 
512 		int blockStart = offset / blockSize;
513 		int blockEnd = (offset + datalen - 1) / blockSize;
514 		int blockCount = blockEnd - blockStart + 1;
515 
516       // write
517       int tries = (passwords == null) ? 2 : (passwords.length + 2);
518       byte[] password;
519       int offsetStart = offset % blockSize;
520       int offsetEnd = (offset + datalen -1) / blockSize;
521       boolean success = false;
522       for (int i = 0; i < tries; i++) {
523          // get password for this try
524          if (i == (tries - 1)) {
525             password = new byte[] { 0x00, 0x00, 0x00, 0x00 };
526          } else if (i == (tries - 2)) {
527             password = null;
528          } else {
529             password = ByteBlock.hexStringToByteArray(passwords[i]);
530          }
531 
532          // read startblock if write does not start at block address
533          byte[] backupStart = null;
534          if (offsetStart != 0) {
535             try {
536                backupStart = readMemoryBlocks(idbytes, memoryBank, blockStart,
537                      1, password);
538             } catch (Exception e) {
539                continue;
540             }
541          }
542 
543          // read endblock if write does not end at block end
544          byte[] backupEnd = null;
545          if (blockCount > 1 && offsetEnd != 0) {
546             try {
547                backupEnd = readMemoryBlocks(idbytes, memoryBank, blockEnd, 1,
548                      password);
549             } catch (Exception e) {
550                continue;
551             }
552          }
553 
554          // assemble block data to write
555          byte[] blockData = new byte[blockSize * blockCount];
556          if (backupStart != null) {
557             System.arraycopy(backupStart, 0, blockData, 0, blockSize);
558          }
559          if (backupEnd != null) {
560             System.arraycopy(backupEnd, 0, blockData,
561                   blockData.length - blockSize, blockSize);
562          }
563          System.arraycopy(data.toByteArray(), 0, blockData, offsetStart, datalen);
564 
565          // write blocks
566          try {
567             writeMemoryBlocks(idbytes, memoryBank, blockStart, blockData,
568                   password);
569             success = true;
570          } catch (Exception e) {
571             continue;
572          }
573       }
574 
575       if (!success) {
576          throw new HardwareException("Write failed.");
577       }
578 
579 	}

581 	private void writeMemoryBlocks(byte[] id, int memoryBank, int address,
582          byte[] data, byte[] password) throws HardwareException,
583 			UnsupportedOperationException, IOException {
584 
585 		// log.debug("writeMemoryBlocks: started");
586 
587       RequestRecord req = new RequestRecord();
588       req.comAdr = (byte) 0xff;
589       // Set ISO15693 host command
590       req.controlByte = (byte) 0xb0;
591 
592       int datalen = 8 + id.length + data.length;
593       if (password != null) {
594          datalen = datalen + 1 + password.length;
595       }
596       byte[] writeCommand = new byte[datalen];
597       // Set write command (0x24)
598       writeCommand[0] = (byte) 0x24;
599       // Set mode (addressed, UID_LF set, EXT_ADDR set)
600       writeCommand[1] = (byte) 0x31;
601       // Set UID length
602       writeCommand[2] = (byte) id.length;
603       // Set tag address (UID)
604       System.arraycopy(id, 0, writeCommand, 3, id.length);
605       // Set bank
606       writeCommand[3 + id.length] = (byte) memoryBank;
607       int dbAddrIndex;
608       if (password != null) {
609          // Set A_FLAG
610          writeCommand[3 + id.length] |= (byte) 0x80;
611          // Set password length
612          writeCommand[4 + id.length] = (byte) password.length;
613          // Set password
614          System.arraycopy(password, 0, writeCommand, 5 + id.length,
615                password.length);
616          dbAddrIndex = 5 + id.length + password.length;
617       } else {
618          dbAddrIndex = 4 + id.length;
619       }
620       writeCommand[dbAddrIndex] = (byte) ((address >> 8) & 0xFF);
621       writeCommand[dbAddrIndex + 1] = (byte) (address & 0xFF);
622       int dbsize = getBlockSize();
623       int dbn = data.length / dbsize;
624       writeCommand[dbAddrIndex + 2] = (byte) (dbn & 0xFF);
625       writeCommand[dbAddrIndex + 3] = (byte) (dbsize & 0xFF);
626       System.arraycopy(data, 0, writeCommand, dbAddrIndex + 4, data.length);
627       req.data = writeCommand;
628 
629 		ResponseRecord resp = tcpip.sendRequest(req);
630 
631 		// Handle response
632 		StatusByte status = StatusByte.getStatus(resp.status);
633 		String statusMessage = getStatusMessage(status);
634 		if (status == StatusByte.WRITE_ERROR) {
635 			log.error(statusMessage);
636 			throw new HardwareException("Writing data blocks failed\n" + " "
637 			      + statusMessage + " at block " + Integer.toHexString(resp.data[0]));
638 		} else if (status != StatusByte.OK) {
639 			log.error(statusMessage);
640 			throw new HardwareException("Writing data blocks failed\n" + " "
641 			      + statusMessage + "/n");
642 		}
643 
644 		// log.debug("writeMemoryBlocks: finished successfuly");
645 	}

678 	public void reset() throws HardwareException {
679 		// construct request record...
680 		RequestRecord req = new RequestRecord();
681 		req.comAdr = (byte) 0xff;
682 		// Set reader RF reset command
683 		req.controlByte = (byte) 0x69;
684 		req.data = null;
685 
686 		// send record...
687 		ResponseRecord resp = null;
688 
689 		// may throw IOException
690 		try {
691 			resp = tcpip.sendRequest(req);

783 	/**
784 	 * Throws RFIDException if error occurs. (returns empty Vector if error
785 	 * occurs!)
786 	 *
787 	 * @return a vector with all the inventory items
788 	 */
789 	synchronized protected Vector<InventoryItem> getInventory()
790 			throws HardwareException {
791 
792 		Vector<InventoryItem> inventory = new Vector<InventoryItem>();
793 		boolean moreTags = false;
794 
795 		do {
796 			// construct request record...
797 			RequestRecord req = new RequestRecord();
798 			req.comAdr = (byte) 0xff;
799 			// control byte = 0xb0 (ISO host command)
800 			req.controlByte = (byte) 0xb0;
801 
802 			if (moreTags) {
803 				// 1. byte: inventory request (01)
804 				// 2. byte: mode: more data request (80)
805 				req.data = ByteBlock.hexStringToByteArray("0180");
806 			} else {
807 				// 1. byte: inventory request (01)
808 				// 2. byte: mode: new inventory request (00)
809 				req.data = ByteBlock.hexStringToByteArray("0100");
810 			}
811 
812 			ResponseRecord resp = null;
813 
814 			try {
815 				resp = tcpip.sendRequest(req);
816 			} catch (IOException e) {
817 				log.error("readBytes: Communication error with reader", e);
818 				throw new HardwareException("Communication error with reader");
819 			}
820 
821 			StatusByte status = StatusByte.getStatus(resp.status);
822 			String statusMessage = getStatusMessage(status);
823 			if (status == StatusByte.NO_TRANSPONDER) {
824 				log.info(statusMessage);
825 				return inventory;
826 			} else if (status == StatusByte.MORE_DATA) {
827 				moreTags = true;
828 			} else if (resp.status == (byte) 0x00) {
829 				moreTags = false;
830 			} else {
831 				log.error(statusMessage);
832 				throw new HardwareException();
833 			}
834 			log.info(statusMessage);
835 
836 			int datasets = ByteBlock.byteToNumber(resp.data[0]);
837          int index = 1;
838          int datasetlen;
839 
840 			// Process each transponder
841 			for (int i = 0; i < datasets; i++) {
842 				InventoryItem item = new InventoryItem();
843 
844             datasetlen = ByteBlock.byteToNumber(resp.data[index + 2]) + 3;
845 
846 				byte trType = resp.data[index];
847             item.transponderType = TransponderType.getType(trType);
848 				item.rfTechnology = RFTechnology.getType(trType);
849 
850             if (item.transponderType == TransponderType.EPCclass1Gen2) {
851                // TODO: read TID
852                item.tid = new byte[] { (byte) 0xE2, 0x00, 0x00, 0x00 };
853                item.epcTransponderModel = EPCTransponderModel.
854                   getEpcTrasponderModel(item.tid, epcTransponderModelsConfig);
855             }
856 
857             byte[] idbytes;
858             if (item.transponderType == TransponderType.EPCclass1Gen2) {
859                // PC bits are sent with the ID of EPCclass1Gen2
860                idbytes = new byte[datasetlen - 5];
861                System.arraycopy(resp.data, index + 5, idbytes, 0,
862                   idbytes.length);
863             } else {
864                idbytes = new byte[datasetlen - 3];
865                System.arraycopy(resp.data, index + 3, idbytes, 0,
866                   idbytes.length);
867             }
868             item.id = ByteBlock.byteArrayToHexString(idbytes);
869 				inventory.add(item);
870 
871             index += datasetlen;
872 			}
873 		} while (moreTags);
874 
875 		return inventory;
876 	}

1001 	private void killEPC(String id, String password)
1002 			throws HardwareException {
1003 		// construct request record...
1004 		RequestRecord req = new RequestRecord();
1005 		req.comAdr = (byte) 0xff;
1006 		// Set special host command (I-Code UID)
1007 		req.controlByte = (byte) 0x1c;
1008 
1009 		byte[] destroyCommand = new byte[24];
1010 		// Set destroy command
1011 		destroyCommand[0] = (byte) 0x18;
1012 		// Set UID mode
1013 		destroyCommand[1] = (byte) 0x1;
1014 		// Set tag address (IDD)
1015 		byte[] idb = ByteBlock.hexStringToByteArray(id, 19);
1016 		System.arraycopy(idb, 0, destroyCommand, 2, 19);
1017 		if (password.length() != 4) {
1018 			log.warn("destroyIcodeUID: Password is " + password.length()
1019 					+ " bytes but should be 4");
1020 		}
1021 		byte[] pswb = ByteBlock.hexStringToByteArray(password.substring(0, 2));
1022 		System.arraycopy(pswb, 0, destroyCommand, 21, 3);
1023 
1024 		req.data = destroyCommand;
1025 
1026 		ResponseRecord resp = null;
1027 
1028 		try {
1029 			resp = tcpip.sendRequest(req);
1030 		} catch (IOException e) {
1031 			String message = "Communication error with reader";
1032 			log.error("destroyIcodeUID: " + message, e);
1033 			throw new HardwareException(message);
1034 		}
1035 
1036 		// Handle response
1037 		StatusByte status = StatusByte.getStatus(resp.status);
1038 		String statusMessage = getStatusMessage(status);
1039 		if (status == StatusByte.RF_COMMUNICATION_ERROR) {
1040 			log.error(statusMessage);
1041 			log
1042 					.error("destroyIcodeUID: Destroying tag failed. More than one transponder in range or ID not matching.");
1043 			throw new HardwareException("Destroying tag failed. More than one transponder in range or ID not matching.");
1044 		} else if (status == StatusByte.WRITE_ERROR) {
1045 			log.error(statusMessage);
1046 			log
1047 					.error("destroyIcodeUID: Destroying tag not successful. Tag may still be read.");
1048 			throw new HardwareException("Destroying tag not successful. Tag may still be read.");
1049 		}
1050 	}
