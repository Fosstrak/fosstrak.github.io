116 	 *
117 	 * @throws HardwareException

119 	public void initialize() throws HardwareException {
120 		this.props = new ControllerProperties(propFile, defaultPropFile);
121 
122 		try {
123 			// Read parameters from property file
124 			this.baudRate = this.props.getParameter("baudRate");
125 			this.comPort = Integer.parseInt(this.props.getParameter("comPort"));
126 			this.frame = this.props.getParameter("frame");
127 			this.timeout = Integer.parseInt(this.props.getParameter("timeout"));
128 
129 			// Initialize multiplexer configuration
130 			String multiplexerConfig = this.props.getParameter("multiplexerConfig");
131          if ((multiplexerConfig == null)) {
132             multiplexConfig = new FeigMultiplexConfiguration(this,
133                   propFile, defaultPropFile);
134          } else {
135             multiplexConfig = new FeigMultiplexConfiguration(this,
136                   multiplexerConfig, null);
137          }
138 			multiplexConfig.initialize();
139 			transponderModel = TransponderModel.getTransponderSpec(this.props.getParameter("transponderModel"));
140 		} catch (Exception e) {
141 			String message = "Error in reader property file";
142 			log.error("initialize: " + message, e);
143 			throw new HardwareException(message, e);
144 		}
145 
146 		// Initialize the reader hardware
147 		try {
148 			log.info("  trying port: " + comPort + " baud: " + baudRate
149 					+ " frame: " + frame + " ...");
150 			initReader(comPort, baudRate, frame);
151 			log.info("reader initialized.");
152 			return;
153 		} catch (HardwareException e) {
154 			String message = "Error initializing reader";
155 			log.error("initialize: " + message, e);
156 			throw new HardwareException(message, e);
157 		}

164 	 * @see org.fosstrak.hal.HardwareAbstraction#identify(java.lang.String[])

166 	public Observation[] identify(String[] readPointNames)
167 			throws HardwareException {
168 		Observation[] observations = new Observation[readPointNames.length];
169 
170 		currentInventory.clear();
171 		for (int i = 0; i < readPointNames.length; i++) {
172 			observations[i] = new Observation();
173 			observations[i].setHalName(getHALName());
174 			observations[i].setReadPointName(readPointNames[i]);
175 
176 			if (!this.selectReadpoint(readPointNames[i])) {
177 				String message = "Channel not valid!";
178 				throw new HardwareException(message);
179 			}
180 
181 			Vector<String> ids = new Vector<String>();
182 	         Vector<TagDescriptor> tds = new Vector<TagDescriptor>();
183 			Vector<InventoryItem> items = this.getInventory();
184 
185 			for (InventoryItem item : items) {
186 				String id;
187 				if (item.transponderType == TransponderType.ICodeUID) {
188 					id = item.id.substring(item.id.length() - 5, item.id.length());
189 				} else {
190 					id = item.id;
191 				}
192 
193 				int memSize = detectTransponderBlockSize(id)* transponderModel.getBlocks();
194  					 TransponderType trt = transponderModel.getType();
195  					 IDType idType = null;
196 					 if (trt == TransponderType.ISO15693) {
197 	                idType = IDType.getIdType("ISO15693", null);
198 					 } else if (trt == TransponderType.ICode1) {
199 						 idType = IDType.getIdType("ICode1", null);
200 					 } else if (trt == TransponderType.ICodeUID) {
201 						 idType = IDType.getIdType("ICodeUID", null);
202 					 } else {
203 						 idType = IDType.getIdType("Unknown", null);
204 					 }
205                 MemoryBankDescriptor[] memoryBankDescriptors =
206                    new MemoryBankDescriptor[1];
207                 memoryBankDescriptors[0] = new MemoryBankDescriptor(memSize,
208                 	true, false);
209                 MemoryDescriptor memoryDescriptor = new MemoryDescriptor(
210                       memoryBankDescriptors);
211                 TagDescriptor td = new TagDescriptor(idType, memoryDescriptor);
212                 tds.add(td);
213 
214 				item.readPoint = readPointNames[i];
215 				ids.add(id);
216 				currentInventory.put(id, item);
217 			}
218 
219 			int len = ids.size();
220 			String ids_arr[] = new String[len];
221 			ids_arr = ids.toArray(ids_arr);
222 	         if (tds.size() == len) {
223 	             TagDescriptor[] tds_arr = new TagDescriptor[len];
224 	             tds_arr = tds.toArray(tds_arr);
225 	             observations[i].setTagDescriptors(tds_arr);
226 	          }
227 			observations[i].setIds(ids_arr);
228 			observations[i].setTimestamp(System.currentTimeMillis());
229 		}
230 		return observations;
231 	}

248 	private int detectTransponderBlockSize(String id) throws HardwareException {
249 		if (transponderModel == TransponderModel.UNKNOWN) {
250 			byte[] data;
251 
252 			try {
253 				data = readMemoryBlocks(id, 0, 1);
254 			} catch (IOException e) {
255 				String message = "Communication error with reader";
256 				log.error("readBytes: Communication error with reader", e);
257 				throw new HardwareException(message);
258 			}
259 
260 			// Handle response
261 			int blockSize = data[1];
262 
263 			if (blockSize == 0) {
264 				log.warn("detectTransponderBlockSize: Blocksize is 0");
265 			}
266 
267 			return blockSize;
268 		} else {
269 			return transponderModel.getBlocksize();
270 		}
271 	}

274 	 * (non-Javadoc)
275 	 *
276 	 * @see org.fosstrak.hal.HardwareAbstraction#readBytes(java.lang.String,
277 	 *      java.lang.String, int, int, int, java.lang.String[])
278 	 */
279 	public UnsignedByteArray readBytes(String readPointName, String id, int memoryBank,
280 			int offset, int length, String[] passwords)
281 			throws HardwareException, UnsupportedOperationException {
282 
283 		// Select channel
284 		if (!selectReadpoint(readPointName)) {
285 			log.error("readBytes: Read point not available");
286 			throw new HardwareException("Read point not available");
287 		}
288 
289 		int blockSize = detectTransponderBlockSize(id);
290 
291 		if (blockSize == 0) {
292 			log.error("readBytes: Detected transponder blocksize is not valid");
293 			return new UnsignedByteArray(new byte[0]);
294 		}
295 
296 		int blockStart = offset / blockSize;
297 		int blockEnd = (offset + length - 1) / blockSize;
298 		int blockCount = blockEnd - blockStart + 1;
299 
300 		if (blockCount > transponderModel.getBlocks()) {
301 			// TODO Correct error semantics?
302 			log.error("readBytes: Data length exceeds transponder capacity");
303 			return new UnsignedByteArray(new byte[0]);
304 		}
305 
306 		byte[] data;
307 		try {
308 			data = readMemoryBlocks(id, blockStart, blockCount);
309 		} catch (IOException e) {
310 			String message = "Communication error with reader";
311 			log.error("readBytes: " + message , e);
312 			throw new HardwareException(message);
313 		}
314 
315 		byte[] dataTrim = new byte[length];
316 		System.arraycopy(data, offset % blockSize, dataTrim, 0, length);
317 
318 		return new UnsignedByteArray(dataTrim);
319 	}

321 	private byte[] readMemoryBlocks(String id, int address, int number)
322 			throws IOException, HardwareException {
323 
324 		log.debug("readMemoryBlocks: started");
325 		RequestRecord req = new RequestRecord();
326 		req.comAdr = (byte) 0xff;
327 		// Set ISO15693 host command
328 		req.controlByte = (byte) 0xb0;
329 
330 		byte[] readCommand = new byte[12];
331 		// Set read command (0x23)
332 		readCommand[0] = (byte) 0x23;
333 		// Set adressed mode (MODE)
334 		readCommand[1] = (byte) 0x1;
335 		// Set tag address (UID)
336 		byte[] idb = ByteBlock.hexStringToByteArray(id, 8);
337 		System.arraycopy(idb, 0, readCommand, 2, 8);
338 		// Set first block number to be read (DB-ADDR)
339 		if (address < 0 || address > 255) {
340 			log.warn("readMemoryBlocks: Offset exceeds range");
341 		}
342 		readCommand[10] = (byte) address;
343 		// Set no of data blocks to be read (DB-N)
344 		if (number < 0 || number > 255) {
345 			log.warn("readMemoryBlocks: Read length exceeds range.");
346 		}
347 		readCommand[11] = (byte) number;
348 		req.data = readCommand;
349 
350 		ResponseRecord resp = ip.sendRequest(req);
351 
352 		StatusByte status = StatusByte.getStatus(resp.status);
353 		String statusMessage = getStatusMessage(status);
354 		if (status == StatusByte.ISO_15693_ERROR) {
355 			ISOTransponderResponseErrorCode errorcode = ISOTransponderResponseErrorCode
356 					.getResponseError(resp.data[0]);
357 			String errorMessage = getTransponderErrorMessage(errorcode);
358 
359 			log.error(statusMessage);
360 			log.error(errorMessage);
361 			throw new HardwareException("Reading data blocks failed\n" + " "
362 			      + statusMessage + "/n" + " " + errorMessage + "" + " at block "
363 					+ Integer.toHexString(resp.data[1]));
364 		} else if (status != StatusByte.OK) {
365 			log.error(statusMessage);
366 			throw new HardwareException("Reading data blocks failed\n" + " "
367 			      + statusMessage + "/n");
368 		}
369 		log.debug("readMemoryBlocks: finished successfuly");
370 
371 		return resp.data;
372 	}

374 	/**
375 	 * (non-Javadoc)
376 	 *
377 	 * @see org.fosstrak.hal.HardwareAbstraction#writeBytes(java.lang.String,
378 	 *      java.lang.String, int, int, UnsignedByteArray, java.lang.String[])
379 	 */
380 	public void writeBytes(String readPointName, String id, int memoryBank,
381 			int offset, UnsignedByteArray data, String[] passwords)
382 			throws HardwareException, UnsupportedOperationException {
383 
384 		int length = data.toByteArray().length;
385 
386 		// Select channel
387 		if (!selectReadpoint(readPointName)) {
388 			throw new HardwareException("Read point not available");
389 		}
390 
391 		int blockSize = detectTransponderBlockSize(id);
392 
393 		if (blockSize == 0) {
394 			// TODO How to notify failure? Throw exception?
395 			log.error("readBytes: Detected transponder blocksize is not valid");
396 			return;
397 		}
398 
399 		int blockStart = offset / blockSize;
400 		int blockEnd = (offset + length - 1) / blockSize;
401 		int blockCount = blockEnd - blockStart + 1;
402 
403 		if (blockCount > transponderModel.getBlocks()) {
404 			// TODO How to notify failure? Throw exception?
405 			log.error("writeBytes: Data length exceeds transponder capacity");
406 			return;
407 		}
408 
409 		byte[] backupStart = null;
410 		int offsetStart = offset % blockSize;
411 		if (offsetStart != 0) {
412 			try {
413 				backupStart = readMemoryBlocks(id, blockStart, 1);
414 			} catch (IOException e) {
415 				String message = "Communication error with reader";
416 				log.error("writeBytes: " + message , e);
417 				throw new HardwareException(message);
418 			}
419 		}
420 
421 		byte[] backupEnd = null;
422 		int offsetEnd = (offset + length -1) / blockSize;
423 		if (blockCount > 1 && offsetEnd != 0) {
424 			try {
425 				backupEnd = readMemoryBlocks(id, blockEnd, 1);
426 			} catch (IOException e) {
427 				String message = "Communication error with reader";
428 				log.error("writeBytes: " + message , e);
429 				throw new HardwareException(message);
430 			}
431 		}
432 
433 		byte[] blockData = new byte[blockSize * blockCount];
434 		if (backupStart != null) {
435 			System.arraycopy(backupStart, 0, blockData, 0, blockSize);
436 		}
437 
438 		if (backupEnd != null) {
439 			System.arraycopy(backupEnd, 0, blockData, blockData.length - blockSize, blockSize);
440 		}
441 
442 		System.arraycopy(data.toByteArray(), 0, blockData, offsetStart, length);
443 
444 		try {
445 			writeMemoryBlocks(id, blockStart, blockSize, blockData);
446 		} catch (IOException e) {
447 			log.error("writeBytes: Communication error with reader", e);
448 			throw new HardwareException("Communication error with reader");
449 		}
450 	}

452 	private void writeMemoryBlocks(String id, int address, int dbsize,
453 			byte[] data) throws HardwareException,
454 			UnsupportedOperationException, IOException {
455 
456 		log.debug("writeMemoryBlocks: started");
457 
458 		RequestRecord req = new RequestRecord();
459 		byte[] writeCommand = new byte[13 + data.length];
460 		// Set write command
461 		writeCommand[0] = (byte) 0x24;
462 		// Set adressed mode (MODE)
463 		writeCommand[1] = (byte) 0x1;
464 		// Set tag address (UID)
465 		byte[] idb = ByteBlock.hexStringToByteArray(id, 8);
466 		System.arraycopy(idb, 0, writeCommand, 2, 8);
467 		// Set first block number to be read (DB-ADDR)
468 		if (address < 0 || address > 255) {
469 			log.warn("writeMemoryBlocks: Address exeeds range");
470 		}
471 		writeCommand[10] = (byte) address;
472 		// Set no of data blocks to be read (DB-N)
473 		int dbn = data.length;
474 		if (dbn < 0 || dbn > 255) {
475 			log.warn("writeMemoryBlocks: Data size exceeds exceeds range");
476 		}
477 		writeCommand[11] = (byte) dbn;
478 		writeCommand[12] = (byte) dbsize;
479 		System.arraycopy(data, 0, writeCommand, 13, dbn);
480 		req.data = writeCommand;
481 
482 		ResponseRecord resp = ip.sendRequest(req);
483 
484 		// Handle response
485 		StatusByte status = StatusByte.getStatus(resp.status);
486 		String statusMessage = getStatusMessage(status);
487 		if (status == StatusByte.ISO_15693_ERROR) {
488 			ISOTransponderResponseErrorCode errorcode = ISOTransponderResponseErrorCode
489 					.getResponseError(resp.data[0]);
490 			String errorMessage = getTransponderErrorMessage(errorcode);
491 
492 			log.error(statusMessage);
493 			log.error(errorMessage);
494 			throw new HardwareException("Writing data blocks failed\n" + " "
495 			      + statusMessage + "/n" + " " + errorMessage + "" + " at block "
496 					+ Integer.toHexString(resp.data[1]));
497 		} else if (status == StatusByte.WRITE_ERROR) {
498 			log.error(statusMessage);
499 			throw new HardwareException("Writing data blocks failed\n" + " "
500 			      + statusMessage + " at block " + Integer.toHexString(resp.data[1]));
501 		} else if (status != StatusByte.OK) {
502 			log.error(statusMessage);
503 			throw new HardwareException("Writing data blocks failed\n" + " "
504 			      + statusMessage + "/n");
505 		}
506 
507 		log.debug("writeMemoryBlocks: finished successfuly");
508 	}

548 		// construct request record...
549 		RequestRecord req = new RequestRecord();
550 		req.comAdr = (byte) 0xff;
551 		// Set reader RF reset command
552 		req.controlByte = (byte) 0x69;
553 		req.data = null;
554 
555 		// send record...
556 		ResponseRecord resp = null;
557 
558 		// may throw IOException
559 		try {
560 			resp = ip.sendRequest(req);
561 		} catch (IOException e) {
562 			log.error("readBytes: Communication error with reader", e);
563 			throw new HardwareException("Communication error with reader");
564 		}
565 		// Handle response
566 		StatusByte status = StatusByte.getStatus(resp.status);
567 		String statusMessage = getStatusMessage(status);
568 		if (status != StatusByte.OK) {
569 			log.error(statusMessage);
570 			throw new HardwareException("Reset of reader reader failed\n"
571 							+ " " + statusMessage);
572 		}
573 		log.info(statusMessage);
574 	}

641 	synchronized protected void populateSystemInformation(
642 			Vector<InventoryItem> inventory) throws HardwareException {
643 
644 		for (InventoryItem item : inventory) {
645 			if (item.transponderType == TransponderType.ISO15693) {
646 				RequestRecord req = new RequestRecord();
647 				req.comAdr = (byte) 0xff;
648 				// Set ISO15693 host command
649 				req.controlByte = (byte) 0xb0;
650 
651 				byte[] sysInfCommand = new byte[10];
652 				// Set system information command
653 				sysInfCommand[0] = (byte) 0x2b;
654 				// Set mode to addressed (MODE)
655 				sysInfCommand[1] = (byte) 0x1;
656 				// Set UID

693 	 * Throws RFIDException if error occurs. (returns empty Vector if error
694 	 * occurs!)
695 	 *
696 	 * @return a vector with all inventory items
697 	 */
698 	synchronized protected Vector<InventoryItem> getInventory()
699 			throws HardwareException {
700 
701 		Vector<InventoryItem> inventory = new Vector<InventoryItem>();
702 		boolean moreTags = false;
703 
704 		do {
705 			// construct request record...
706 			RequestRecord req = new RequestRecord();
707 			req.comAdr = (byte) 0xff;
708 			// control byte = 0xb0 (ISO host command)
709 			req.controlByte = (byte) 0xb0;
710 
711 			if (moreTags) {
712 				// 1. byte: inventory request (01)
713 				// 2. byte: mode: more data request (80)
714 				req.data = ByteBlock.hexStringToByteArray("0180");
715 			} else {
716 				// 1. byte: inventory request (01)
717 				// 2. byte: mode: new inventory request (00)
718 				req.data = ByteBlock.hexStringToByteArray("0100");
719 			}
720 
721 			ResponseRecord resp = null;
722 
723 			try {
724 				resp = ip.sendRequest(req);
725 			} catch (IOException e) {
726 				log.error("readBytes: Communication error with reader", e);
727 				throw new HardwareException("Communication error with reader");
728 			}
729 
730 			StatusByte status = StatusByte.getStatus(resp.status);
731 			String statusMessage = getStatusMessage(status);
732 			if (status == StatusByte.NO_TRANSPONDER) {
733 				log.info(statusMessage);
734 				return inventory;
735 			} else if (status == StatusByte.MORE_DATA) {
736 				moreTags = true;
737 			} else if (resp.status == (byte) 0x00) {
738 				moreTags = false;
739 			} else {
740 				log.error(statusMessage);
741 				throw new HardwareException();
742 			}
743 			log.info(statusMessage);
744 
745 			int datasets = ByteBlock.byteToNumber(resp.data[0]);
746 			TransponderType tType = TransponderType.getType(resp.data[1]);
747 
748 			int dataSize = tType.dataSize();
749 
750 			// Process each transponder
751 			for (int i = 0; i < datasets; i++) {
752 				InventoryItem item = new InventoryItem();
753 
754 				byte[] data = new byte[dataSize];
755 				System.arraycopy(resp.data, 1 + i * dataSize, data, 0, dataSize);
756 
757 				byte trType = data[0];
758 				item.transponderType = TransponderType.getType(trType);
759 				item.rfTechnology = RFTechnology.getType(trType);
760 
761 				if (item.transponderType == TransponderType.ICodeEPC) {
762 					item.id = ByteBlock.byteArrayToHexString(copyOfRange(data,
763 							1, item.transponderType.dataSize()));
764 				} else if (item.transponderType == TransponderType.ICodeUID) {
765 					item.id =
766 						ByteBlock.byteArrayToHexString(copyOfRange(data,
767 						1, item.transponderType.dataSize()));
768 				} else {
769 					item.dsfid = data[1];
770 					item.id = ByteBlock.byteArrayToHexString(copyOfRange(data,
771 							2, item.transponderType.dataSize()));
772 				}
773 				inventory.add(item);
774 			}
775 		} while (moreTags);
776 
777 		return inventory;
778 	}

780 	private byte[] copyOfRange(byte[] data, int from, int to) {
781 		if (to < from) {
782 			return null;
783 		}
784 		byte[] res = new byte[to - from];
785 		for (int i = 0; i < res.length; i++) {
786 			if ((from + i) < data.length) {
787 				res[i] = data[from + i];
788 			} else {
789 				res[i] = (byte) 0x00;
790 			}
791 		}
792 		return res;

809 	 *            the port number the reader is connected to.
810 	 * @param baudRate
811 	 *            the baud rate used for communication with the reader
812 	 * @param frame
813 	 *            a frame constisting of databit, parity and stopbit
814 	 * @throws RFIDException
815 	 *             if the port is not found, the frame is not supported, an
816 	 *             io-error occurs...

818 	protected void initReader(int comPort, String baudRate, String frame)
819 			throws HardwareException {
820 
821 		// create new ISOProtocol that opens the serial port
822 
823 		int baud = Integer.parseInt(baudRate);
824 		int databits = Integer.parseInt(frame.substring(0, 1));
825 		int parity;
826 		String p = frame.substring(1, 2);
827 		if (p.equalsIgnoreCase("e")) {
828 			parity = SerialPort.PARITY_EVEN;
829 		} else if (p.equalsIgnoreCase("o")) {
830 			parity = SerialPort.PARITY_ODD;
831 		} else {
832 			parity = SerialPort.PARITY_NONE;
833 		}
834 		int stopbits = Integer.parseInt(frame.substring(2, 3));
835 		try {
836 			if (ip != null)
837 				ip.closePort();
838 			ip = new ISOProtocol(comPort, timeout, baud, databits, stopbits,
839 					parity);
840 		} catch (UnsupportedCommOperationException e) {
841 			// serial port parameters (baud, databits, stopbits, parity) not
842 			// supported
843 			// or port not found (maybe locked)
844 			String message = "initReader: Port configuration error";
845 			log.error(message, e);
846 			throw new HardwareException(message, e);
847 		} catch (PortInUseException e) {
848 			// port is already open
849 			String message = "initReader: Port already in use: ";
850 			log.error(message, e);
851 			throw new HardwareException(message, e);
852 		} catch (IOException e) {
853 			// input/output error or too many event listeners for this
854 			// serial
855 			// port
856 			String message = "initReader: Unknown port error";
857 			log.error(message,e);
858 			throw new HardwareException(message, e);
859 		}
860 
861 		reset();
862 	}
