111 	/**
112 	 * implements the initialize method of the SimulatorServerEngine
113      * 
114      * @param controller 
115 	 * @throws SimulatorServerException 
116 	 */
117 	public void initialize(SimulatorServerController controller, String propFile,
118          String defaultPropFile) throws SimulatorServerException {
119 		this.controller = controller;
120 
121       // load language
122       String prefix = propFile.substring(0, propFile.lastIndexOf("/")) + "/GUIText_";
123       String postfix = ".xml";
124       String language = null;
125       guiTextConfig = new XMLConfiguration();
126       boolean loaded = false;
127       
128       // try default language
129       if (!loaded) {
130          language = LOCALE.getLanguage();
131          String langFile = prefix + language + postfix;
132          URL fileurl = ResourceLocator.getURL(langFile, langFile, this.getClass());
133          try {
134             guiTextConfig.load(fileurl);
135             loaded = true;
136          } catch (ConfigurationException ce) {
137             throw new SimulatorServerException("Graphic simulator server language file not found.");
138          }
139       }
140 
141 		// load properties
142 		try {
143 		   propsConfig = new XMLConfiguration();
144          URL fileurl = ResourceLocator.getURL(propFile, defaultPropFile, this.getClass());
145 		   propsConfig.load(fileurl);
146 		} catch (ConfigurationException ce) {
147 		   throw new SimulatorServerException("Could not load properties file."); 
148 		}
149 		
150 		// initialize GUI
151 		initializeGUI();
152 		LOG.info("GraphicSimulatorServer started");
153 	}

172 	/**
173 	 * computes some properties using other properties
174 	 */
175 	private void computeProperties() {
176 		int nbrOfReader = controller.getReaderIds().size();
177 		int maxNbrOfAntennas = 0;
178 		Iterator readerIt = controller.getReaderIds().iterator();
179 		while (readerIt.hasNext()) {
180 			maxNbrOfAntennas = Math.max(controller.getAntennaIds(((String)readerIt.next())).size(), maxNbrOfAntennas);
181 		}
182 		
183 		// window size, reader pane and antenna pane
184 
185 	    String value;
186       if(!propsConfig.containsKey("WindowWidth")) {
187          value = new Integer(2 * getProperty("FramePadding")
188                + getProperty("TagWidth") + 2 * getProperty("HorizontalPadding")
189                + getProperty("ReaderWidth") + maxNbrOfAntennas * (getProperty("AntennaWidth")
190                      + getProperty("HorizontalInterAntennaPadding"))
191                      - getProperty("HorizontalInterAntennaPadding")).toString();
192          propsConfig.addProperty("WindowWidth", value);
193       }
194       if(!propsConfig.containsKey("WindowHeight")) {
195          value = new Integer(Math.max(getProperty("MinimumWindowHeight"),
196                2 * getProperty("FramePadding") + (nbrOfReader * (getProperty("ReaderHeight")
197                      + getProperty("InterReaderPadding"))
198                      - getProperty("InterReaderPadding")) + 50)).toString();
199          propsConfig.addProperty("WindowHeight", value);
200       }
201       if(!propsConfig.containsKey("ReaderPaneX")) {
202          value = new Integer(getProperty("FramePadding") + getProperty("TagWidth")
203                + getProperty("HorizontalPadding")).toString();
204          propsConfig.addProperty("ReaderPaneX", value);
205       }
206       if(!propsConfig.containsKey("ReaderPaneY")) {
207          value = new Integer(getProperty("FramePadding")).toString();
208          propsConfig.addProperty("ReaderPaneY", value);
209       }
210       if(!propsConfig.containsKey("AntennaPaneX")) {
211          value = new Integer(getProperty("ReaderPaneX") + getProperty("ReaderWidth")
212                + getProperty("HorizontalPadding")).toString();
213          propsConfig.addProperty("AntennaPaneX", value);
214       }
215       if(!propsConfig.containsKey("AntennaPaneY")) {
216          value = new Integer(getProperty("FramePadding")).toString();
217          propsConfig.addProperty("AntennaPaneY", value);
218       }
219 	}

329 	 * creates the layered pane if it does not already exists
330 	 * 
331 	 * @return layered pane
332 	 */
333 	private JLayeredPane getJLayeredPane() {
334 		if (jLayeredPane == null) {
335 			jLayeredPane = new JLayeredPane();
336 			jLayeredPane.setLayout(null);
337 			jLayeredPane.setOpaque(true);
338 			jLayeredPane.setBackground(Color.WHITE);
339 			//jLayeredPane.add(getReader(), new Integer(0));
340 			
341 			// add readers
342 			createReaders(controller.getReaderIds());
343 			
344 			// add context menu
345 			jLayeredPane.add(getContextMenu());
346 
347 			// add mouse listener
348 			jLayeredPane.addMouseListener(new MouseAdapter() {
349 
350 				// context menu
351 				public void mouseReleased(MouseEvent e) {
352 					hideActiveContextMenuAndSelection();
353 					if (e.getButton() == MouseEvent.BUTTON3) {
354 						Point posOnScreen = getJLayeredPane().getLocationOnScreen();
355 						contextMenu.setLocation(posOnScreen.x + e.getX(), posOnScreen.y + e.getY());
356 						contextMenu.setVisible(true);
357 						setActiveContextMenu(contextMenu);
358 					} else {
359 						contextMenu.setVisible(false);
360 						if(e.getButton() == MouseEvent.BUTTON1) {
361 							selection.select(tags);
362 						}
363 					}
364 				}
365 				
366 				// selection
367 				public void mousePressed(final MouseEvent e) {
368 					hideActiveContextMenuAndSelection();
369 					if(e.getButton() == MouseEvent.BUTTON1) {
370 						selection.setBounds(0, 0, getProperty("WindowWidth"), getProperty("WindowHeight"));
371 						selection.setStartPoint(e.getPoint());
372 						jLayeredPane.add(selection, new Integer(2));
373 					}
374 				}
375 			});
376 			
377 			// add drag listener
378 			jLayeredPane.addMouseMotionListener(new MouseMotionAdapter() {
379 				public void mouseDragged(MouseEvent e) {
380 					if(selection.isActive()) {
381 						selection.setCurrentPoint(e.getPoint());
382 					}
383 				}
384 			});
385 		}
386 		return jLayeredPane;
387 	}

389 	/**
390 	 * update the layered pane with new reader information
391 	 * 
392 	 * @return layered pane
393 	 */
394 	private void updateJLayeredPane() {
395 		Set controllerReaderIds = controller.getReaderIds();
396 		Set guiReaderIds = readers.keySet();
397 		
398 		// get new readers
399 		Set newReaderIds = new TreeSet(controllerReaderIds);
400 		newReaderIds.removeAll(guiReaderIds);
401 		
402 		// get update readers
403 		Set updateReaderIds = new TreeSet(controllerReaderIds);
404 		updateReaderIds.retainAll(guiReaderIds);
405 		
406 		// get old readers
407 		Set oldReaderIds = new TreeSet(guiReaderIds);
408 		oldReaderIds.removeAll(controllerReaderIds);
409 
410 		// create, update and delete readers
411 		createReaders(newReaderIds);
412 		updateReaders(updateReaderIds);
413 		removeReaders(oldReaderIds);
414 		
415 		jLayeredPane.repaint();
416 	}

441 	 * shows the dialog menu to add a new tag
442 	 */
443 	private void showAddTagDialog() {
444 		// compute position of the dialog
445 		final Point pos = new Point();
446 		pos.x = jLayeredPane.getLocationOnScreen().x + (jLayeredPane.getWidth() - getProperty("DialogWindowWidth")) / 2;
447 		pos.y = jLayeredPane.getLocationOnScreen().y + (jLayeredPane.getHeight() - getProperty("DialogWindowHeight")) / 2;
448 		
449 		// set default id
450 		String id = "A47033F";
451 		for(int i = 0; i < 4 - new Integer(tags.size()).toString().length(); i++) {
452 			id += "0";
453 		}
454 		id += tags.size();
455 		if(tagIdField == null) {
456 			tagIdField = new JTextField();
457 		}
458 		tagIdField.setText(id);
459 		
460 		// create tag dialog if it does not already exists
461 		if(newTagDialog == null) {
462 			newTagDialog = new JDialog(this, guiTextConfig.getString("AddNewTagDialogTitle"), true);
463 			newTagDialog.setSize(getProperty("DialogWindowWidth"), getProperty("DialogWindowHeight"));
464 			newTagDialog.setLayout(new BorderLayout());
465 			
466 			// input fields panel
467 			JLabel epcLabel = new JLabel(guiTextConfig.getString("TagIdLabel") + ": ");
468 			JPanel inputFields = new JPanel();
469 			inputFields.setLayout(new GridLayout(2, 2));
470 			inputFields.add(epcLabel);
471 			inputFields.add(tagIdField);
472 			
473 			// cancel button
474 			JButton cancelButton = new JButton(guiTextConfig.getString("CancelButton"));
475 			cancelButton.addMouseListener(new MouseAdapter() {
476 				public void mouseReleased(MouseEvent e) {
477 					newTagDialog.setVisible(false);
478 				}
479 			});
480 			
481 			// add button
482 			JButton addButton = new JButton(guiTextConfig.getString("AddButton"));
483 			addButton.addMouseListener(new MouseAdapter() {
484 				public void mouseReleased(MouseEvent e) {
485 					newTagDialog.setVisible(false);
486 					createNewTag(tagIdField.getText());
487 				}
488 			});
489 			
490 			// buttons panel
491 			JPanel buttons = new JPanel();
492 			buttons.add(addButton);
493 			buttons.add(cancelButton);
494 			
495 			// compose all together
496 			newTagDialog.add(inputFields, BorderLayout.CENTER);
497 			newTagDialog.add(buttons, BorderLayout.SOUTH);
498 			newTagDialog.getRootPane().setDefaultButton(addButton);
499 		}
500 		newTagDialog.setLocation(pos);
501 		newTagDialog.setVisible(true);
502 	}

584 	/**
585 	 * deletes a reader and removes it from the layered pane
586 	 * 
587 	 * @param id to remove
588 	 */
589 	private void removeReader(String id) {
590 		Reader reader = (Reader)readers.get(id);
591 		
592 		// remove antennas
593 		Antenna[] antennas = (Antenna[])reader.getAntennas().values().toArray(new Antenna[0]);
594 		for (int i = 0; i < antennas.length; i++) {
595 			removeAntenna(antennas[i], reader);
596 		}
597 		
598 		// remove reader;
599 		readers.remove(id);
600 		jLayeredPane.remove(reader);
601 	}
