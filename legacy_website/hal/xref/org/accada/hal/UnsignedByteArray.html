26   * UnsignedByteArray represents an array of unsigned bytes of arbitrary length.
27   * 
28   * Since Java does not support unsigned byte arrays this class can be used to 
29   * send or receive bytes from hardware devices such as RFID readers.
30   * 
31   * In addition, the class provides several static byte array support functions,
32   * which allow the usage of UnsignedByteArray without creating objects of the class.
33   * 
34   * @author 	Matthias Lampe, lampe@acm.org
35   */

159 	 * 
160 	 * @return the hashcode of the byte block.
161 	 * 
162 	 * @see java.lang.Object#hashCode()
163 	 */
164 	public int hashCode() {
165 		long v = 0;
166 		for (int i = bytes.length-1; i >= 0; i--) {
167 			v = 256 * v + ((int)bytes[i] & 0xFF);
168 		}
169 		return (int)v;

209 	 * converts a String of hex bytes to a byte array. The hex bytes are interpreted
210 	 * as unsigned bytes.
211 	 *
212 	 * @param hexString the byte array as a String.
213 	 */
214 	static public byte[] hexStringToByteArray(String hexString)
215 	{
216 		byte[] byteArray;
217 
218 		int len = hexString.length();
219 		byteArray = new byte[len/2];
220 		if (len % 2 != 0) throw new IllegalArgumentException();
221 		for (int i = 0; i < len/2; i++){
222 			//String byteString = "0x"+hexString.substring(i*2, i*2+2);
223 			//Integer intValue = Integer.decode(byteString);
224 			String byteString = hexString.substring(i*2, i*2+2);
225 			Integer intValue = new Integer(Integer.parseInt(byteString, 16));
226 			byte byteValue = intValue.byteValue();
227 			byteArray[i] = byteValue;
228 		}
229 
230 		return byteArray;
231 	}

235 	/**
236 	 * creates a byte array out of the given long number. The numbering
237 	 * of the bytes starts from the left to the right in contrary
238 	 * to usual numbering e.g. 
239 	 * <code>n=0134A0FF002EC0B1</code> returns the array
240 	 * <code>{ 01,34,A0,FF,00,2E,C0,B1 }</code>. The bytes in the array 
241 	 * should be interpreted as unsigned bytes. 
242 	 * 
243 	 * @param n the number
244 	 * 
245 	 * @return the byte array representing the number.
246 	 */
247 	public static byte[] numberToByteArray(long n) 
248 	{
249 		byte[] ba = new byte[8];
250 		byte b;
251 
252 		// go through all bytes and filter them out
253 		for (int s = 0; s < 8; s++) {
254 			b = (byte) ((n >> ((8 - 1 - s) * 8)) & 0xFF);
255 			ba[s] = b;
256 		}
257 		
258 		return ba;
259 	}

312 	 * checks if two given UnsignedByteArray regions are equal or not.
313 	 * 
314 	 * @param firstArray the first UnsignedByteArray.
315 	 * @param firstOffset the offset of the first region in the array.
316 	 * @param secondArray the second UnsignedByteArray.
317 	 * @param secondOffset the offset of the second region in the array.
318 	 * @param length the length of the regions in the arrays.
319 	 * 
320 	 * @return true, if both UnsignedByteArray regions are equal.
321 	 */
322 	public static boolean equalsRegion(
323 				byte[] firstArray, int firstOffset,
324 				byte[] secondArray, int secondOffset, 
325 				int length) 
326 	{
327 		int firstIndex = firstOffset;
328 		int secondIndex = secondOffset;
329 		
330 		//-- compare byte blocks in arrays
331 		while (firstIndex < firstArray.length
332 				&& secondIndex < secondArray.length
333 				&& firstIndex < firstOffset + length
334 				&& secondIndex < secondOffset + length
335 				&& firstArray[firstIndex] == secondArray[secondIndex]) {
336 			firstIndex++;
337 			secondIndex++;
338 		}
339 
340 		//-- check if end reached, i.e. blocks are equal
341 		if (firstIndex == firstOffset + length
342 			&& secondIndex == secondOffset + length) 
343 			return true;
344 		else
345 			return false;
346 	}

366 	 * compares two given UnsignedByteArray regions. 
367 	 * 
368 	 * @param firstArray the first UnsignedByteArray.
369 	 * @param firstOffset the offset of the first region in the array.
370 	 * @param secondArray the second UnsignedByteArray.
371 	 * @param secondOffset the offset of the second region in the array.
372 	 * @param length the length of the regions in the arrays.
373 	 * 
374 	 * @return -1, if first region < second region,
375 	 * 			 0, if first region == second region,
376 	 * 			 1, if first region > second region.
377 	 */
378 	public static int compareRegion(
379 				byte[] firstArray, int firstOffset,
380 				byte[] secondArray, int secondOffset, 
381 				int length) 
382 	{
383 		int firstIndex = firstOffset;
384 		int secondIndex = secondOffset;
385 		
386 		//-- compare byte regions in arrays
387 		while (firstIndex < firstArray.length
388 				&& secondIndex < secondArray.length
389 				&& firstIndex < firstOffset + length
390 				&& secondIndex < secondOffset + length) {
391 			int res = compare(firstArray[firstIndex], secondArray[secondIndex]);
392 			if (res != 0)
393 				return res;
394 			
395 			firstIndex++;
396 			secondIndex++;
397 		}
398 		
399 		// end of blocks reached => blocks equal
400 		return 0;
401 	}

405 	 * compares two bytes which are interpreted as unsigned bytes.
406 	 * 
407 	 * @param first the first byte.
408 	 * @param second the second byte.
409 	 * @return -1, if first < second,
410 	 * 			 0, if first == second,
411 	 * 			 1, if first > second 
412 	 */
413 	public static int compare(byte first, byte second)
414 	{
415 		int f = first & 0x000000FF;
416 		int s = second & 0x000000FF;
417 		
418 		if (f < s) 
419 			return -1;
420 		else if (f == s)
421 			return 0;
422 		else
423 			return 1;
424 	}
