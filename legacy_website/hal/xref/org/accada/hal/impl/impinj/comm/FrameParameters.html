21  package org.accada.hal.impl.impinj.comm;

23  import java.util.HashMap;

25  /**
26   * This class represents ... of a Mach1 protocol frame.
27   *
28   * <p>Copyright: Copyright (c) 2007</p>
29   * @author Jonas Haller
30   */
31  public class FrameParameters {
32  
33     // -------- Constants --------------------------------------------------- //
34     /**  */
35     public static final boolean C = true;
36  
37     // -------- Fields ------------------------------------------------------ //
38     /** hash map containing the parameters */
39     private HashMap<Integer, byte[]> params = new HashMap<Integer, byte[]>();
40  
41     // -------- Public Methods ---------------------------------------------- //
42     /**
43      * Constructor.
44      *
45      * @param aData
46      *          the data field of the frame
47      * @param aSpec
48      *          the specification of the parameters contained in the data field
49      * @throws FrameException
50      *          if data does not comply to the spec or other error occures
51      */
52     public FrameParameters(byte[] aData, FrameParametersSpec aSpec)
53           throws FrameException {
54        int index = 0;
55        int datalen = aData.length;
56        int[] paramspec;
57        int id, len;
58        byte[] param;
59  
60        // get mandantory parameters
61        aSpec.resetMandantory();
62        while (true) {
63           try {
64              paramspec = aSpec.getNextMandantory();
65           } catch (FrameException fe) {
66              break;
67           }
68           id = paramspec[0];
69           len = paramspec[1];
70           if (len == 0) {
71              // get variable length
72              checkEnough(index, datalen, 2);
73              len = (((aData[index] & 0xFF) << 8) | (aData[index + 1] & 0xFF));
74              index += 2;
75           }
76           checkEnough(index, datalen, len);
77           param = new byte[len];
78           System.arraycopy(aData, index, param, 0, len);
79           params.put(id, param);
80           index += len;
81        }
82  
83        // get optional parameters
84        while (index < datalen) {
85           id = aData[index] & 0xFF;
86           index++;
87           len = aSpec.getOptionalLength(id);
88           if (len == 0) {
89              // get variable length
90              checkEnough(index, datalen, 2);
91              len = (((aData[index] & 0xFF) << 8) | (aData[index + 1] & 0xFF));
92              index += 2;
93           }
94           checkEnough(index, datalen, len);
95           param = new byte[len];
96           System.arraycopy(aData, index, param, 0, len);
97           params.put(id, param);
98           index += len;
99        }
100    }
101 
102    /**
103     * Get a parameter value.
104     *
105     * @param aId
106     *          the ID of the parameter
107     * @return
108     *          the value of the parameter
109     * @throws FrameException
110     *          if parameter not available
111     */
112    public byte[] getParameter(int aId) throws FrameException {
113       byte[] param = params.get(aId);
114       if (param == null) {
115          throw new FrameException("Parameter '" + aId + "' not available.");
116       }
117       return param;
118    }
119 
120    // -------- Private Methods --------------------------------------------- //
121    /**
122     * Check if enough elements of data left
123     *
124     * @param aIndex
125     *          the index (in the array)
126     * @param aDataLen
127     *          the total length of data
128     * @param aLen
129     *          the length to check if it is available
130     * @throws FrameException
131     *          if less than aLen elements available
132     */
133    private void checkEnough(int aIndex, int aDataLen, int aLen)
134          throws FrameException {
135       if ((aDataLen - aIndex) < aLen) {
136          throw new FrameException("Not enough data available in array.");
137       }
138    }
