234 	/**
235 	 * rearranges the selected tags
236 	 */
237 	private void arrangeTags() {
238 		// calculate position 
239 		Tag curTag;
240 		int x = 0;
241 		int y = 0;
242 		Iterator tagsIt = ((TreeSet)selectedTags.clone()).iterator();
243 		while(tagsIt.hasNext()) {
244 			curTag = (Tag)tagsIt.next();
245 			x += curTag.getX();
246 			y += curTag.getY();
247 		}
248 		x /= selectedTags.size();
249 		y = y / selectedTags.size() - (((selectedTags.size() - 1) * (simulator.getProperty("TagHeight") + simulator.getProperty("InterTagPadding"))) / 2);
250 		
251 		// arrange tags
252 		int counter = 0;
253 		tagsIt = ((TreeSet)selectedTags.clone()).iterator();
254 		while(tagsIt.hasNext()) {
255 			curTag = (Tag)tagsIt.next();
256 			curTag.setLocation(x, y + counter++ * (simulator.getProperty("TagHeight") + simulator.getProperty("InterTagPadding")));
257 		}
258 		simulator.hideActiveContextMenu();
259 	}

262 	 * groups the tag to the tags in the grouped tags hash map
263 	 * 
264 	 * @param groupedTags tags in the same group
265 	 */
266 	public void group(TreeSet groupedTags) {
267 		if(isGrouped()) this.groupedTags.remove(this);
268 		this.groupedTags = groupedTags;
269 	}

275 		groupedTags.clear();
276 	}

335    /**
336     * shows the memory banks dialog window
337     * 
338     * @param x the horizontal position of the dialog window
339     * @param y the vertical position of the dialog window
340     */
341    private void showMemoryBanksDialog(int x, int y) {
342       final JDialog memoryBanksDialog = new JDialog();
343       memoryBanksDialog.setName(simulator.getGuiText().getString("MemoryBanksDialogTitle"));
344       memoryBanksDialog.setTitle(simulator.getGuiText().getString("MemoryBanksDialogTitle"));
345       memoryBanksDialog.setSize(simulator.getProperty("DialogWindowWidth"),
346             simulator.getProperty("DialogWindowHeight") * 3 / 2);
347       memoryBanksDialog.setLayout(new BorderLayout());
348       
349       // create memory fields if not existent yet
350       JTextField reservedInputField = new JTextField();
351       JTextField crcpcInputField = new JTextField();
352       JTextField tidInputField = new JTextField();
353       if(epcInputField == null) {
354          epcInputField = new JTextField();
355       }
356       if(userInputField == null) {
357          userInputField = new JTextField();
358       }
359       
360       // set text
361       byte[] data;
362       reservedInputField.setText("read protected");
363       reservedInputField.setEditable(false);
364       try {
365          data = tag.readData(1, 0, 0);
366          if (data.length < 4) {
367             throw new HardwareException("epc memory bank too short");
368          }
369          byte[] crcpc = new byte[4];
370          System.arraycopy(data, 0, crcpc, 0, crcpc.length);
371          int epclen = ((((int) crcpc[2]) & 0xFF) >> 3) * 2;
372          crcpcInputField.setText(ByteBlock.byteArrayToHexString(crcpc));
373          crcpcInputField.setEditable(false);
374          if (epclen > (data.length - 4)) {
375             throw new HardwareException("epc shorter than declared in PC bits");
376          }
377          byte[] epc = new byte[epclen];
378          System.arraycopy(data, 4, epc, 0, epc.length);
379          epcInputField.setText(ByteBlock.byteArrayToHexString(epc));
380       } catch (HardwareException he) {
381          crcpcInputField.setEditable(false);
382          epcInputField.setText("read failed");
383       }
384       epcInputFieldBefore = epcInputField.getText();
385       try {
386          data = tag.readData(2, 0, 0);
387          tidInputField.setText(ByteBlock.byteArrayToHexString(data));
388          tidInputField.setEditable(false);
389       } catch (HardwareException he) {
390          tidInputField.setText("read failed");
391          tidInputField.setEditable(false);
392       }
393       try {
394          data = tag.readData(3, 0, 0);
395          userInputField.setText(ByteBlock.byteArrayToHexString(data));
396       } catch (HardwareException he) {
397          userInputField.setText("read failed");
398       }
399       userInputFieldBefore = userInputField.getText();
400 
401       // memory banks input fields
402       final int FIELDS_HEIGHT = 25;
403       final Dimension LABEL_DIMENSION = new Dimension(120, 25);
404       JPanel inputFields = new JPanel();
405       inputFields.setLayout(new BoxLayout(inputFields, BoxLayout.Y_AXIS));
406       // glue
407       inputFields.add(Box.createGlue());
408       // reserved memory
409       JPanel reservedFields = new JPanel();
410       reservedFields.setLayout(new BoxLayout(reservedFields, BoxLayout.X_AXIS));
411       reservedFields.setMaximumSize(new Dimension(simulator.getProperty("DialogWindowWidth") - 20, FIELDS_HEIGHT));
412       JPanel reservedLabelBox = new JPanel();
413       reservedLabelBox.setLayout(new BoxLayout(reservedLabelBox, BoxLayout.X_AXIS));
414       reservedLabelBox.setPreferredSize(LABEL_DIMENSION);
415       JLabel reservedMemoryLabel = new JLabel(simulator.getGuiText().getString(
416          "ReservedMemoryLabel"));
417       reservedLabelBox.add(reservedMemoryLabel);
418       reservedFields.add(reservedLabelBox);
419       JPanel reservedInputBox = new JPanel();
420       reservedInputBox.setLayout(new BoxLayout(reservedInputBox, BoxLayout.X_AXIS));
421       reservedInputBox.add(reservedInputField);
422       reservedFields.add(reservedInputBox);
423       inputFields.add(reservedFields);
424       // glue
425       inputFields.add(Box.createGlue());
426       // epc memory
427       JPanel epcFields = new JPanel();
428       epcFields.setLayout(new BoxLayout(epcFields, BoxLayout.X_AXIS));
429       epcFields.setMaximumSize(new Dimension(simulator.getProperty("DialogWindowWidth") - 20, FIELDS_HEIGHT));
430       JPanel epcLabelBox = new JPanel();
431       epcLabelBox.setLayout(new BoxLayout(epcLabelBox, BoxLayout.X_AXIS));
432       epcLabelBox.setPreferredSize(LABEL_DIMENSION);
433       JLabel epcMemoryLabel = new JLabel(simulator.getGuiText().getString(
434          "EPCMemoryLabel"));
435       epcLabelBox.add(epcMemoryLabel);
436       epcFields.add(epcLabelBox);
437       JPanel epcInputBox = new JPanel();
438       epcInputBox.setLayout(new BoxLayout(epcInputBox, BoxLayout.X_AXIS));
439       JPanel crcpcInputBox = new JPanel();
440       crcpcInputBox.setLayout(new BoxLayout(crcpcInputBox, BoxLayout.X_AXIS));
441       crcpcInputBox.setMaximumSize(new Dimension(FIELDS_HEIGHT * 4, FIELDS_HEIGHT));
442       crcpcInputBox.add(crcpcInputField);
443       epcInputBox.add(crcpcInputBox);
444       epcInputBox.add(epcInputField);
445       epcFields.add(epcInputBox);
446       inputFields.add(epcFields);
447       // glue
448       inputFields.add(Box.createGlue());
449       // tid memory
450       JPanel tidFields = new JPanel();
451       tidFields.setLayout(new BoxLayout(tidFields, BoxLayout.X_AXIS));
452       tidFields.setMaximumSize(new Dimension(simulator.getProperty("DialogWindowWidth") - 20, FIELDS_HEIGHT));
453       JPanel tidLabelBox = new JPanel();
454       tidLabelBox.setLayout(new BoxLayout(tidLabelBox, BoxLayout.X_AXIS));
455       tidLabelBox.setPreferredSize(LABEL_DIMENSION);
456       JLabel tidMemoryLabel = new JLabel(simulator.getGuiText().getString(
457          "TIDMemoryLabel"));
458       tidLabelBox.add(tidMemoryLabel);
459       tidFields.add(tidLabelBox);
460       JPanel tidInputBox = new JPanel();
461       tidInputBox.setLayout(new BoxLayout(tidInputBox, BoxLayout.X_AXIS));
462       tidInputBox.add(tidInputField);
463       tidFields.add(tidInputBox);
464       inputFields.add(tidFields);
465       // glue
466       inputFields.add(Box.createGlue());
467       // user memory
468       JPanel userFields = new JPanel();
469       userFields.setLayout(new BoxLayout(userFields, BoxLayout.X_AXIS));
470       userFields.setMaximumSize(new Dimension(simulator.getProperty("DialogWindowWidth") - 20, FIELDS_HEIGHT));
471       JPanel userLabelBox = new JPanel();
472       userLabelBox.setLayout(new BoxLayout(userLabelBox, BoxLayout.X_AXIS));
473       userLabelBox.setPreferredSize(LABEL_DIMENSION);
474       JLabel userMemoryLabel = new JLabel(simulator.getGuiText().getString(
475          "UserMemoryLabel"));
476       userLabelBox.add(userMemoryLabel);
477       userFields.add(userLabelBox);
478       JPanel userInputBox = new JPanel();
479       userInputBox.setLayout(new BoxLayout(userInputBox, BoxLayout.X_AXIS));
480       userInputBox.add(userInputField);
481       userFields.add(userInputBox);
482       inputFields.add(userFields);
483       // glue
484       inputFields.add(Box.createGlue());
485       
486       // cancel button
487       JButton cancelButton = new JButton(simulator.getGuiText().getString("CancelButton"));
488       cancelButton.addMouseListener(new MouseAdapter() {
489          public void mouseReleased(MouseEvent e) {
490             memoryBanksDialog.setVisible(false);
491          }
492       });
493       
494       // ok button
495       JButton okButton = new JButton(simulator.getGuiText().getString("OkButton"));
496       okButton.addMouseListener(new MouseAdapter() {
497          public void mouseReleased(MouseEvent e) {
498             memoryBanksDialog.setVisible(false);
499             if (!epcInputField.getText().equals(epcInputFieldBefore)) {
500                tag.setTagID(epcInputField.getText());
501             }
502             if (userInputField.getText().length() % 2 != 0) {
503                userInputField.setText("0" + userInputField.getText());
504             }
505             if (!userInputField.getText().equals(userInputFieldBefore)) {
506                try {
507                   tag.setData(ByteBlock.hexStringToByteArray(userInputField.getText()), 3);
508                } catch (NumberFormatException nfe) {}
509             }
510          }
511       });
512       
513       // buttons panel
514       JPanel buttons = new JPanel();
515       buttons.add(okButton);
516       buttons.add(cancelButton);
517       
518       // add elements
519       memoryBanksDialog.add(inputFields, BorderLayout.CENTER);
520       memoryBanksDialog.add(buttons, BorderLayout.SOUTH);
521       memoryBanksDialog.getRootPane().setDefaultButton(okButton);
522       memoryBanksDialog.setLocation(x, y);
523       memoryBanksDialog.setVisible(true);
524    }
