54   * The SimulatorController is a HAL controller that enables the implementation
55   * of different kinds of RFID reader simulators/emulators.
56   * 
57   * From the HAL point of view the SimulatorController supports 
58   * the same functionality as hardware controllers. Additionally it 
59   * provides methods that allow a Simulator (i.e. SimulatorEngine) to 
60   * (virtually) add and remove RFID tags. 
61   * 
62   * @author Roland Schneider
63   */

186 	public SimulatorController(String halName, String propFile) {
187 		this.halName = halName;
188 		this.propFile = propFile;
189 		this.initialize();
190 		
191 		log.debug("Simulator: " + simType);
192 		reader = new HashMap();
193 		
194 		for (int i = 0; i < nOfReadPoints; i++){
195 			reader.put( readPointNames[i] , new HashSet() );
196 		}
197 		
198 		
199 		try{
200 			//Dynamic class loading
201 			//1.) The full package path 
202 			Class clazz = Class.forName(simType);
203 			//2.) State the types of the constructors arguments. here: only one argument, of type String
204 			java.lang.reflect.Constructor co = clazz.getConstructor(new Class[] {});
205 			//3.) Create new instance with the argument's values
206 			simulator = (SimulatorEngine) co.newInstance(null);
207 			simulator.initialize(this, simTypePropFile);	
208 		}
209 		catch(ClassNotFoundException cnfe){
210 			cnfe.printStackTrace();
211 			System.exit(1);
212 		}
213 		catch(Exception e){
214 			e.printStackTrace();
215 			System.exit(1);
216 		}		
217 	}

219 	/**
220 	 * Initialize a reader.
221 	 */
222 	public void initialize(){
223 		
224 		this.props = new ControllerProperties(propFile, defaultPropFile);
225 		
226 		//sets the parameters according to the properties file
227 		try{
228 			simType = props.getParameter("simType");
229 			simTypePropFile = props.getParameter("simTypePropFile");
230          epcTransponderModelsConfig = props.getParameter(
231          "epcTransponderModelsConfig");
232 			
233 			nOfReadPoints = Integer.parseInt(props.getParameter("numberOfReadPoints"));
234 			readPointNames = new String[nOfReadPoints];
235 			readyReadPoints = new Vector<String>();
236 			for (int i = 0; i < nOfReadPoints; i++){
237 				readPointNames[i] = this.props.getParameter("readPoint_" + (i + 1));
238 				readyReadPoints.add(readPointNames[i]);
239 			}
240 		}	
241 		catch (Exception e){
242 			String message="Couldn't initialize the reader:" + e.getMessage();
243 			log.debug(message);
244 			e.printStackTrace();
245 		}
246 	}

356 	 * @see org.accada.hal.HardwareAbstraction#identify(java.lang.String[])
357 	 */
358 	public Observation[] identify(String[] readPointNames)
359          throws HardwareException {
360 		// Each read point gets its own Observation
361 		Observation[] observations = new Observation[readPointNames.length];
362 		
363 		for (int i = 0; i < readPointNames.length; i++){
364 			
365 			observations[i] = new Observation();
366 			observations[i].setHalName(this.halName);
367 			observations[i].setReadPointName(readPointNames[i]);
368 			
369 			//Get the tag IDs on this antenna
370 			Vector<String> v = new Vector<String>();
371 			Iterator it = ((HashSet)reader.get(readPointNames[i])).iterator();
372 			while (it.hasNext()){
373 				Tag t = (Tag) it.next();
374 				v.add(t.getTagID());
375 			}
376 			
377          int len = v.size();
378          String v_arr[] = new String[len];
379          v_arr = v.toArray(v_arr);
380          TagDescriptor[] td_arr = new TagDescriptor[v_arr.length];
381          String idTConf;
382 			try {
383 				idTConf = props.getParameter("idTypesConfig");
384 			} catch (Exception e) {
385 				idTConf = null;
386 			}
387          for (int j = 0; j < td_arr.length; j++) {
388 				IDType idType = IDType.getIdType("EPC", idTConf);
389             byte[] accada_sim_tid_bytes = ByteBlock.hexStringToByteArray("E2FFF000");
390             EPCTransponderModel tagModel = EPCTransponderModel.getEpcTrasponderModel(
391                   accada_sim_tid_bytes, epcTransponderModelsConfig);
392             MemoryBankDescriptor[] memoryBankDescriptors =
393                new MemoryBankDescriptor[4];
394             memoryBankDescriptors[0] = new MemoryBankDescriptor(
395                tagModel.getReservedSize(), tagModel.getReservedReadable(),
396                tagModel.getReservedWriteable());
397             memoryBankDescriptors[1] = new MemoryBankDescriptor(
398                   tagModel.getEpcSize(), tagModel.getEpcReadable(),
399                   tagModel.getEpcWriteable());
400             memoryBankDescriptors[2] = new MemoryBankDescriptor(
401                   tagModel.getTidSize(), tagModel.getTidReadable(),
402                   tagModel.getTidWriteable());
403             memoryBankDescriptors[3] = new MemoryBankDescriptor(
404                   tagModel.getUserSize(), tagModel.getUserReadable(),
405                   tagModel.getUserWriteable());
406             MemoryDescriptor memoryDescriptor = new MemoryDescriptor(
407                   memoryBankDescriptors); 
408             td_arr[j] = new TagDescriptor(idType, memoryDescriptor);
409          }
410          observations[i].setTagDescriptors(td_arr);
411 			observations[i].setIds(v_arr);
412 			observations[i].setTimestamp(System.currentTimeMillis());
413 			
414 			if (!readyReadPoints.contains(readPointNames[i])) {
415 				log.error("Read point \"" + readPointNames[i] + "\" is not ready.");
416 				observations[i].setIds(new String[0]);
417             observations[i].setTagDescriptors(new TagDescriptor[0]);
418 				observations[i].successful = false;
419 			} else if (continuousIdentifyErrors.contains(readPointNames[i])) {
420 				observations[i].successful = false;
421 			} else if (identifyError.contains(readPointNames[i])) {
422 				observations[i].successful = false;
423 				identifyError.remove(readPointNames[i]);
424 			} else {
425 				observations[i].successful = true;
426 			}
427 			
428 			log.debug(observations[i].toString());
429 		}
430 		return observations;
431 	}

489 	 * @see org.accada.hal.HardwareAbstraction#readBytes(java.lang.String, java.lang.String, int, int, int, java.lang.String[])
490 	 */
491 	public UnsignedByteArray readBytes(String readPointName, String id, int memoryBank,
492 			int offset, int length, String[] passwords)
493 			throws HardwareException, UnsupportedOperationException {
494 		if (readyReadPoints.contains(readPointName)) {
495 			
496 			if (continuousReadErrors.contains(readPointName)) {
497 				throw new HardwareException();
498 			} else if (readError.contains(readPointName)) {
499 				readError.remove(readPointName);
500 				throw new HardwareException();
501 			}
502 			
503 			UnsignedByteArray byteData;
504 			Tag tag = null;
505 			if (contains(readPointName, id)) {
506 				Iterator it = ((HashSet)reader.get(readPointName)).iterator();
507 				while (it.hasNext()){
508 					Tag curTag = (Tag)it.next();
509 					if (curTag.getTagID().equalsIgnoreCase(id)) {
510 						tag = curTag;
511 						break;
512 					}
513 				}
514 			}
515 			
516 			if(tag != null){
517 				byteData = new UnsignedByteArray(tag.readData(memoryBank, offset, length));
518 			}
519 			else{
520 				String message="Specified tag not in range";
521 				throw new HardwareException(message);	
522 			}
523 			
524 			return byteData;
525 		} else {
526 			throw new HardwareException("Read point is not ready.");
527 		}
528 	}

534 	public void writeBytes(String readPointName, String id, int memoryBank,
535 			int offset, UnsignedByteArray data, String[] passwords)
536 			throws HardwareException, UnsupportedOperationException {
537 		if (readyReadPoints.contains(readPointName)) {
538

540 				throw new HardwareException();
541 			} else if (writeError.contains(readPointName)) {
542 				writeError.remove(readPointName);
543 				throw new HardwareException();
544 			}
545 			
546 			Tag tag = null;
547 			if (contains(readPointName, id)) {
548 				Iterator it = ((HashSet)reader.get(readPointName)).iterator();
549 				while (it.hasNext()){
550 					Tag curTag = (Tag)it.next();
551 					if (curTag.getTagID().equalsIgnoreCase(id)) {
552 						tag = curTag;
553 						break;
554 					}
555 				}
556 			}
557 			
558 			if(tag != null){
559 				tag.writeData(memoryBank, offset, data.toByteArray());
560 			}
561 			else{
562 				String message="Specified tag not in range";
563 				throw new HardwareException(message);	
564 			}
565 		} else {
566 			throw new HardwareException("Read point is not ready.");
567 		}
568 	}
