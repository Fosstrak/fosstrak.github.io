41   * This class represents a subscriber of an ec specification.
42   * There are three types of such subscribers: http, tcp, file

53  	/** number representing a http uri */
54  	private static final int HTTP = 0;
55  	/** number representing a tcp uri */
56  	private static final int TCP = 1;
57  	/** number representing a file uri */
58  	private static final int FILE = 2;
59  	
60  	/** http uri prefix */
61  	private static final String HTTP_PREFIX = "http";
62  	/** tcp uri prefix */
63  	private static final String TCP_PREFIX = "tcp";
64  	/** file uri prefix */
65  	private static final String FILE_PREFIX = "file";

87  	/**
88  	 * Constructor parses and validates the notification uri and creates the corresponding subscriber.
89  	 * 
90  	 * @param notificationURI of the subscriber 
91  	 * @throws InvalidURIException if the notification uri is invalid
92  	 */
93  	public Subscriber(String notificationURI) throws InvalidURIExceptionResponse {
94  		
95  		this.uri = notificationURI;
96  		
97  		String[] parts = notificationURI.split(":");
98  		
99  		if (parts.length < 2 || parts.length > 3 || parts[1].length() < 3 || !parts[1].startsWith("//")) {
100 			throw new InvalidURIExceptionResponse(INVALID_URI_EXCEPTION_TEXT);
101 		}
102 		
103 		if (HTTP_PREFIX.equals(parts[0])) {
104 			
105 			// http url
106 			protocol = HTTP;
107 			
108 			if (parts.length == 2) {
109 				
110 				// default port
111 				port = DEFAULT_PORT;
112 				int slashPos = parts[1].indexOf("/", 2);
113 				host = slashPos < 2 ? parts[1].substring(2) : parts[1].substring(2, slashPos);
114 				path = slashPos < 2 ? "" : parts[1].substring(slashPos + 1);
115 				
116 			} else {
117 				
118 				// explicit port
119 				host = parts[1].substring(2);
120 				int slashPos = parts[2].indexOf("/");
121 				try {
122 					port = Integer.parseInt(slashPos < 0 ? parts[2] : parts[2].substring(0, slashPos));
123 				} catch (NumberFormatException e) {
124 					throw new InvalidURIExceptionResponse("Invalid port. " + INVALID_URI_EXCEPTION_TEXT);
125 				}
126 				path = slashPos < 0 ? "" : parts[2].substring(slashPos + 1);
127 				
128 			}
129 		} else if (TCP_PREFIX.equals(parts[0])) {
130 			
131 			// tcp url
132 			protocol = TCP;
133 			
134 			if (parts.length != 3) {
135 				throw new InvalidURIExceptionResponse(INVALID_URI_EXCEPTION_TEXT);
136 			}
137 			
138 			host = parts[1].substring(2);
139 			try {
140 				port = Integer.parseInt(parts[2]);
141 			} catch (NumberFormatException e) {
142 				throw new InvalidURIExceptionResponse("Invalid port. " + INVALID_URI_EXCEPTION_TEXT);
143 			}
144 			
145 			
146 		} else if (FILE_PREFIX.equals(parts[0])) {
147 			
148 			// file url
149 			protocol = FILE;
150 			
151 			if (parts.length == 3) {
152 				if (parts[1].startsWith("//") && parts[1].length() < 5) {
153 					parts[1] = parts[1] + ":" + parts[2];
154 				} else {
155 					throw new InvalidURIExceptionResponse(INVALID_URI_EXCEPTION_TEXT);
156 				}
157 			}
158 			
159 			int slashPos = parts[1].indexOf("/", 2);
160 			
161 			if (slashPos < 0) {
162 				throw new InvalidURIExceptionResponse("Invalid path. " + INVALID_URI_EXCEPTION_TEXT);
163 			}
164 			
165 			if (slashPos == 2) {
166 				host = LOCALHOST;
167 			} else {
168 				host = parts[1].substring(2, slashPos);
169 			}
170 			
171 			path = slashPos == parts[1].length() ? parts[1] : parts[1].substring(slashPos + 1);
172 			
173 		} else {
174 			
175 			// invalid url
176 			throw new InvalidURIExceptionResponse("Invalid protocol. " + INVALID_URI_EXCEPTION_TEXT);
177 			
178 		}
179 		
180 	}

259 	/**
260 	 * This method notifies the subscriber about the ec reports.
261 	 * 
262 	 * @param reports to notify the subscriber about
263 	 * @throws ImplementationException if an implementation exception occures
264 	 */
265 	public void notify(ECReports reports) throws ImplementationExceptionResponse {
266 		
267 		if (isHttp()) {
268 			
269 			// write reports as post request to http socket
270 			LOG.debug("Write reports '" + reports.getSpecName() + "' as post request to http socket '" + host + ":" + port + "'.");
271 			writeToSocket(getPostRequest(reports));
272 			
273 		} else if (isTcp()) {
274 			
275 			// write reports as xml to tcp socket
276 			LOG.debug("Write reports '" + reports.getSpecName() + "' as xml to tcp socket '" + host + ":" + port + "'.");
277 			writeToSocket(getXml(reports));
278 			
279 			
280 		} else if (isFile()) {
281 			
282 			writeNotificationToFile(reports);
283 			
284 		}
285 		
286 	}

292 	/**
293 	 * This method writes data to a socket with host name and port number of this subscriber.
294 	 * 
295 	 * @param data to write to the socket
296 	 * @throws ImplementationException if an implementation exception occures
297 	 */
298 	private void writeToSocket(String data) throws ImplementationExceptionResponse {
299 	
300 		Socket socket;
301 		try {
302 			
303 			// open socket and stream
304 			socket = new Socket(host, port);
305 			DataOutputStream dataOutputStream = new DataOutputStream(socket.getOutputStream());
306 			
307 			// write reports
308 			dataOutputStream.writeBytes(data);
309 			dataOutputStream.write("\n".getBytes());
310 			dataOutputStream.flush();
311 			
312 			// close socket and stream
313 			dataOutputStream.close();
314 			socket.close();
315 			
316 		} catch (UnknownHostException e) {
317 			throw new ImplementationExceptionResponse("Host '" + host + "' not found.");
318 		} catch (IOException e) {
319 			throw new ImplementationExceptionResponse("Could not write data to socket at '" + host + ":" + port + "'.");
320 		}
321 		
322 	}

324 	/**
325 	 * This method writes ec reports to a file.
326 	 * 
327 	 * @param reports to write to the file
328 	 * @throws ImplementationException if an implementation exception occures
329 	 */
330 	private void writeNotificationToFile(ECReports reports) throws ImplementationExceptionResponse {
331 		
332 		// append reports as xml to file
333 		LOG.debug("Append reports '" + reports.getSpecName() + "' as xml to file '" + path + "'.");
334 
335 		// can only write to local files
336 		if (LOCALHOST.equals(host)) {
337 			
338 			File file = new File(path);
339 			
340 			// create file if it does not already exists
341 			if (!file.exists() || !file.isFile()) {
342 				try {
343 					file.createNewFile();
344 				} catch (IOException e) {
345 					throw new ImplementationExceptionResponse("Could not create new file '" + path + "'.");
346 				}
347 			}
348 			
349 			try {
350 				
351 				// open streams
352 				FileOutputStream fileOutputStream = new FileOutputStream(file, true);
353 				DataOutputStream dataOutputStream = new DataOutputStream(fileOutputStream);
354 
355 				// append reports as xml to file
356 				dataOutputStream.writeBytes(getPrettyXml(reports));
357 				dataOutputStream.writeBytes("\n\n");
358 				dataOutputStream.flush();
359 				
360 				// close streams
361 				dataOutputStream.close();
362 				fileOutputStream.close();
363 				
364 			} catch (IOException e) {
365 				throw new ImplementationExceptionResponse("Could not write to file '" + path + "'.");
366 			}
367 		
368 		} else {
369 			throw new ImplementationExceptionResponse("This implementation can not write reports to a remote file.");
370 		}
371 		
372 	}

412 	/**
413 	 * This method creates a post request from ec reports containing an xml representation of the reports.
414 	 * 
415 	 * @param reports to transform into a post request
416 	 * @return post request containing an xml representation of the reports
417 	 * @throws ImplementationException if an implementation exception occurs
418 	 */
419 	private String getPostRequest(ECReports reports) throws ImplementationExceptionResponse {
420 		
421 		LOG.debug("Create POST request with reports '" + reports.getSpecName() + "'.");
422 		
423 		// create body
424 		String body = getXml(reports);
425 		
426 		// create header
427 		StringBuffer header = new StringBuffer();
428 		
429 		// append request line
430 		header.append("POST ");
431 		// add the trimmed / again together with a white space
432 		String p = "/" + path + " ";
433 		header.append(p);
434 			
435 		header.append("HTTP/1.1");
436 		header.append("\n");
437 		
438 		// append host
439 		header.append("Host: ");
440 		header.append(host);
441 		header.append("\n");
442 		
443 		// append content type
444 		header.append("Content-Type: ");
445 		header.append("text/xml; charset=\"utf-8\"");
446 		header.append("\n");
447 		
448 		// append content length
449 		header.append("Content-Length: ");
450 		header.append(body.length());
451 		header.append("\n");
452 		
453 		// terminate body
454 		header.append("\n");
455 
456 		return header + body;
457 		
458 	}
