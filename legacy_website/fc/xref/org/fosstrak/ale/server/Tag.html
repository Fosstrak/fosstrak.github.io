197 	/**
198 	 * comparator to check whether two tags are the same.
199 	 * @param tag the other tag to be checked.
200 	 * @return boolean value flagging whether equal or not
201 	 */
202 	public boolean equalsTag(Tag tag) {
203 		// if the origin in both tags is null, then do not take the 
204 		// origin into account when comparing
205 		if ((getOrigin() != null) && (tag.getOrigin() != null)) { 
206 			// compare the origin
207 			if (!tag.getOrigin().equalsIgnoreCase(getOrigin())) {
208 				return false;
209 			}
210 		} else if ((getOrigin() == null) || (tag.getOrigin() == null)) {
211 			return false;
212 		}
213 		
214 		// if the reader in both tags is null, then do not take the
215 		// reader into account when comparing
216 		if ((getReader() != null) && (tag.getReader() != null)) {
217 			// compare the reader
218 			if (!tag.getReader().equalsIgnoreCase(getReader()))  {
219 				return false;
220 			}
221 		} else if ((getReader() == null) || (tag.getReader() == null)) {
222 			return false;
223 		}
224 		
225 		// compare the tag id
226 		if ((null != tag.getTagIDAsPureURI()) && (null != getTagIDAsPureURI())) {
227 			// both not null, so compare
228 			if (!tag.getTagIDAsPureURI().equalsIgnoreCase(getTagIDAsPureURI())) {
229 				return false;
230 			}				
231 		} else if ((null != tag.getTagIDAsPureURI()) || (null != getTagIDAsPureURI())) {
232 			// only one null, therefore not equals
233 			return false;
234 		}
235 		
236 		// try to compare the binary value
237 		if ((null != tag.getTagAsBinary()) && (null != getTagAsBinary())) {
238 			// both not null, so compare
239 			if (!tag.getTagAsBinary().equalsIgnoreCase(getTagAsBinary())) {
240 				return false;
241 			}				
242 		} else if ((null != tag.getTagAsBinary()) || (null != getTagAsBinary())) {
243 			// only one null, therefore not equals
244 			return false;
245 		}
246 		
247 		return true;
248 	}

394 	 * @param tagLength the inbound taglength must be specified as "64" or "96".
395 	 * @param filter the inbound filter value must be specified - range 
396 	 * depends on coding scheme.

398 	 * specified for GS1 coding schemes. if set to null paramter is ignored.

406 			String tag) {

408 		LevelTypeList outputLevel = LevelTypeList.LEGACY;

410 		if (null != tagLength) extraparms.put("taglength", tagLength);
