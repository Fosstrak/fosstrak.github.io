138 	/**
139 	 * This method initializes the class, by loading properties, texts and setting the endpoint.
140 	 * 
141 	 * @throws IOException
142 	 */
143 	public void initialize() throws IOException {
144 
145 		// load properties
146 		InputStream inputStream = this.getClass().getResourceAsStream(PROPERTIES_FILE_LOCATION);
147 		if (inputStream == null) {
148 			throw new IllegalArgumentException("Could not load properties from classpath (" + PROPERTIES_FILE_LOCATION + ")");
149 		}
150 		props.load(inputStream);
151 		
152 		// load text
153 		InputStream languageStream = null;
154 		
155 		// try to get language form property file
156 		if (props.containsKey("Language")) {
157 			languageStream = this.getClass().getResourceAsStream("/props/ALEClient_" + props.getProperty("Language") + ".lang");
158 		}
159 		
160 		// try system default language
161 		if (languageStream == null) {
162 			languageStream = this.getClass().getResourceAsStream("/props/ALEClient_" + SYSTEM_DEFAULT_LOCALE.getLanguage() + ".lang");
163 		}
164 		
165 		// try default language
166 		if (languageStream == null) {
167 			languageStream = this.getClass().getResourceAsStream("/props/ALEClient_" + DEFAULT_LOCALE.getLanguage() + ".lang");
168 		}
169 		
170 		if (languageStream == null) {
171 			throw new IllegalArgumentException("Could not load language package from classpath (" + DEFAULT_LOCALE + ")");
172 		}
173 		guiText = new PropertyResourceBundle(languageStream);
174 
175 		//--------------------
176 		JaxWsProxyFactoryBean factory = new JaxWsProxyFactoryBean();
177         factory.setServiceClass(ALEServicePortType.class);
178         String address = props.getProperty("EndPoint");
179         if (address == null) {
180             System.out.println("using default address " + address);
181         	address = "http://localhost:8080/fc-server-0.3.0-SNAPSHOT/services/ALEService";
182         }
183         factory.setAddress(address);
184         aleProxy = (ALEServicePortType) factory.create();
185         
186 		initializeGUI();
187 		
188 	}

248 	 * This method sets the command selection panel depending on the command id.
249 	 * 
250 	 * @param command id
251 	 */
252 	private void setCommandPanel(int command) {
253 		
254 		if (command == -1) {
255 			commandPanel.removeAll();
256 			commandPanel.setBorder(null);
257 			this.setVisible(false);
258 			this.setVisible(true);
259 			return;
260 		}
261 		
262 		commandPanel.setBorder(BorderFactory.createCompoundBorder(BorderFactory.createTitledBorder(guiText.getString("Command" + command)), BorderFactory.createEmptyBorder(5,5,5,5)));
263 		commandPanel.removeAll();
264 
265 		switch(command) {
266 		
267 			case 4: //getECSpecNames
268 			case 10: // getStandardVersion
269 			case 11: // getVendorVersion
270 				commandPanel.setLayout(new GridLayout(1, 1, 5, 0));
271 				break;
272 				
273 			case 2: // undefine
274 			case 3: // getECSpec
275 			case 7: // poll
276 			case 9: // getSubscribers
277 				commandPanel.setLayout(new GridLayout(5, 1, 5, 0));
278 				addECSpecNameComboBox(commandPanel);
279 				addSeparator(commandPanel);
280 				break;
281 				
282 			case 5: // subscribe
283 			case 6: // unsubscribe
284 				commandPanel.setLayout(new GridLayout(7, 1, 5, 0));
285 				addECSpecNameComboBox(commandPanel);
286 				addNotificationURIField(commandPanel);
287 				addSeparator(commandPanel);
288 				break;
289 				
290 			case 8: // immediate
291 				commandPanel.setLayout(new GridLayout(6, 1, 5, 0));
292 				addChooseFileField(commandPanel);
293 				addSeparator(commandPanel);
294 				break;
295 				
296 			case 1: // define
297 				commandPanel.setLayout(new GridLayout(8, 1, 5, 0));
298 				addECSpecNameComboBox(commandPanel);
299 				addChooseFileField(commandPanel);
300 				addSeparator(commandPanel);
301 				break;
302 				
303 		}
304 		
305 		addExecuteButton(commandPanel);
306 		
307 		this.setVisible(false);
308 		this.setVisible(true);
309 		
310 	}

434 	/**
435 	 * This method executes the command which is selected in the command selection combobox
436 	 * with the parameters which are set in the corresponding fields.
437 	 * To execute the commands, the methods of the ale proxy will be invoked. 
438 	 */
439 	private void executeCommand() {
440 		
441 		Object result = null;
442 		String specName = null;
443 		String notificationURI = null;
444 		
445 		try {
446 			
447 			switch(commandSelection.getSelectedIndex()) {
448 			
449 				case 4: //getECSpecNames
450 					result = aleProxy.getECSpecNames(new EmptyParms());
451 					break;
452 					
453 				case 10: // getStandardVersion
454 					result = aleProxy.getStandardVersion(new EmptyParms());
455 					break;
456 					
457 				case 11: // getVendorVersion
458 					result = aleProxy.getVendorVersion(new EmptyParms());
459 					break;
460 			
461 				case 2: // undefine
462 				case 3: // getECSpec
463 				case 7: // poll
464 				case 9: // getSubscribers
465 					// get specName
466 					specName = (String)specNameComboBox.getSelectedItem();
467 					if (specName == null || "".equals(specName)) {
468 						showExcpetionDialog(guiText.getString("SpecNameNotSpecifiedDialog"));
469 						break;
470 					}
471 					
472 					switch(commandSelection.getSelectedIndex()) {
473 					
474 					case 2: // undefine
475 						Undefine undefineParms = new Undefine();
476 						undefineParms.setSpecName(specName);
477 						aleProxy.undefine(undefineParms);
478 						result = guiText.getString("SuccessfullyUndefinedMessage");
479 						break;
480 						
481 					case 3: // getECSpec
482 						GetECSpec getECSpecParms = new GetECSpec();
483 						getECSpecParms.setSpecName(specName);
484 						result = aleProxy.getECSpec(getECSpecParms);
485 						break;
486 						
487 					case 7: // poll
488 						Poll pollParms = new Poll();
489 						pollParms.setSpecName(specName);
490 						result = aleProxy.poll(pollParms);
491 						break;
492 						
493 					case 9: // getSubscribers
494 						GetSubscribers getSubscribersParms = new GetSubscribers();
495 						getSubscribersParms.setSpecName(specName);
496 						result = aleProxy.getSubscribers(getSubscribersParms);
497 						break;
498 						
499 					}
500 					
501 					break;
502 					
503 				case 5: // subscribe
504 				case 6: // unsubscribe
505 					// get specName
506 					specName = (String)specNameComboBox.getSelectedItem();
507 					if (specName == null || "".equals(specName)) {
508 						showExcpetionDialog(guiText.getString("SpecNameNotSpecifiedDialog"));
509 						break;
510 					}
511 					
512 					// get notificationURI
513 					notificationURI = notificationUriField.getText();
514 					if (notificationURI == null || "".equals(notificationURI)) {
515 						showExcpetionDialog(guiText.getString("NotificationUriNotSpecifiedDialog"));
516 						break;
517 					}
518 					
519 					switch (commandSelection.getSelectedIndex()) {
520 					
521 					case 5:
522 						Subscribe subscribeParms = new Subscribe();
523 						subscribeParms.setSpecName(specName);
524 						subscribeParms.setNotificationURI(notificationURI);
525 						aleProxy.subscribe(subscribeParms);
526 						result = guiText.getString("SuccessfullySubscribedMessage");
527 						break;
528 						
529 					case 6:
530 						Unsubscribe unsubscribeParms = new Unsubscribe();
531 						unsubscribeParms.setSpecName(specName);
532 						unsubscribeParms.setNotificationURI(notificationURI);
533 						aleProxy.unsubscribe(unsubscribeParms);
534 						result = guiText.getString("SuccessfullyUnsubscribedMessage");
535 						break;
536 						
537 					}
538 					
539 					break;
540 					
541 				case 1: // define
542 				case 8: // immediate
543 
544 					if (commandSelection.getSelectedIndex() == 1) {
545 						// get specName
546 						specName = (String)specNameComboBox.getSelectedItem();
547 						if (specName == null || "".equals(specName)) {
548 							showExcpetionDialog(guiText.getString("SpecNameNotSpecifiedDialog"));
549 							break;
550 						}
551 					}
552 					
553 					// get filePath
554 					String filePath = filePathField.getText();
555 					if (filePath == null || "".equals(filePath)) {
556 						showExcpetionDialog(guiText.getString("FilePathNotSpecifiedDialog"));
557 						break;
558 					}
559 					
560 					// get ecSpec
561 					ECSpec ecSpec;
562 					try {
563 						ecSpec = getECSpecFromFile(filePath);
564 					} catch(FileNotFoundException e) {
565 						showExcpetionDialog(guiText.getString("FileNotFoundDialog"));
566 						break;
567 					} catch(Exception e) {
568 						showExcpetionDialog(guiText.getString("UnexpectedFileFormatDialog"));
569 						break;
570 					}
571 					
572 					if (commandSelection.getSelectedIndex() == 1) {
573 						Define defineParms = new Define();
574 						defineParms.setSpecName(specName);
575 						defineParms.setSpec(ecSpec);
576 						aleProxy.define(defineParms);
577 						result = guiText.getString("SuccessfullyDefinedMessage");
578 					} else {
579 						Immediate immediateParms = new Immediate();
580 						immediateParms.setSpec(ecSpec);
581 						result = aleProxy.immediate(immediateParms);
582 					}
583 					break;
584 				
585 			}
586 			
587 		} catch (Exception e) {
588 			String reason = e.getMessage();
589 			if (e instanceof DuplicateNameExceptionResponse) {
590 				showExcpetionDialog(guiText.getString("DuplicateNameExceptionDialog"), reason);
591 			} else if (e instanceof DuplicateSubscriptionExceptionResponse) {
592 				showExcpetionDialog(guiText.getString("DuplicateSubscriptionExceptionDialog"), reason);
593 			} else if (e instanceof ECSpecValidationExceptionResponse) {
594 				showExcpetionDialog(guiText.getString("ECSpecValidationExceptionDialog"), reason);
595 			} else if (e instanceof ImplementationExceptionResponse) {
596 				showExcpetionDialog(guiText.getString("ImplementationExceptionDialog"), reason);
597 			} else if (e instanceof InvalidURIExceptionResponse) {
598 				showExcpetionDialog(guiText.getString("InvalidURIExceptionDialog"), reason);
599 			} else if (e instanceof NoSuchNameExceptionResponse) {
600 				showExcpetionDialog(guiText.getString("NoSuchNameExceptionDialog"), reason);
601 			} else if (e instanceof NoSuchSubscriberExceptionResponse) {
602 				showExcpetionDialog(guiText.getString("NoSuchSubscriberExceptionDialog"), reason);
603 			} else if (e instanceof SecurityExceptionResponse) {
604 				showExcpetionDialog(guiText.getString("SecurityExceptionDialog"), reason);
605 			}
606 		}
607 		
608 		showResult(result);
609 		
610 		// update spec name combobox
611 		List<String> ecSpecNames = null;
612 		try {
613 			ecSpecNames = aleProxy.getECSpecNames(new EmptyParms()).getString();
614 		} catch (Exception e) {}
615 		if (ecSpecNames != null && specNameComboBox != null && specNameComboBox.getSelectedObjects() != null && specNameComboBox.getSelectedObjects().length > 0) {
616 			String current = (String)specNameComboBox.getSelectedObjects()[0];
617 			specNameComboBox.removeAllItems();
618 			if (ecSpecNames != null && ecSpecNames.size() > 0) {
619 				for (String name : ecSpecNames) {
620 					specNameComboBox.addItem(name);
621 				}
622 			}
623 			specNameComboBox.setSelectedItem(current);
624 		}
625 
626 	}
