177 	/**
178 	 * This method removes all objects on the reader device which are created by this input generator.
179 	 */
180 	public void remove() {
181 		
182 		LOG.debug("Try to remove InputGenerator.");
183 		
184 		// remove Triggers, NotificationChannel and DataSelector
185 		if (readerDevice != null) {
186 			try {
187 				readerDevice.removeTriggers(new Trigger[] {readTrigger});
188 				LOG.debug("ReadTrigger '" + readTriggerName + "' successfully removed.");
189 			} catch (RPProxyException e) {
190 				e.printStackTrace();
191 			}
192 			try {
193 				readerDevice.removeTriggers(new Trigger[] {notificationTrigger});
194 				LOG.debug("NotificationTrigger '" + notificationTriggerName + "' successfully removed.");
195 			} catch (RPProxyException e) {
196 				e.printStackTrace();
197 			}
198 			try {
199 				readerDevice.removeNotificationChannels(new NotificationChannel[] {notificationChannel});
200 				LOG.debug("NotificationChannel '" + notificationChannelName + "' successfully removed.");
201 			} catch (RPProxyException e) {
202 				e.printStackTrace();
203 			}
204 			try {
205 				readerDevice.removeDataSelectors(new DataSelector[] {dataSelector});
206 				LOG.debug("DataSelector '" + dataSelectorName + "' successfully removed.");
207 			} catch (RPProxyException e) {
208 				e.printStackTrace();
209 			}
210 		}
211 		
212 		// stop NotificationChannelEndPoint
213 		if (notificationChannelEndPoint != null) {
214 			notificationChannelEndPoint.stop();
215 			LOG.debug("NotificationChannelEndPoint stopped.");
216 		}
217 		
218 		LOG.debug("InputGenerator removed");
219 		
220 	}

222 	/**
223 	 * polls the rp-proxy for all tags available.
224 	 * @throws RPProxyException when the polling failed
225 	 */
226 	public void poll() throws RPProxyException {
227 		
228 		//LOG.debug("Polling the rp-proxy");
229 		
230 		// get all the sources from the device
231 		Source[] sources = readerDevice.getAllSources();
232 		if (sources != null) {
233 			for (Source source : sources) {
234 				// remove all data-selectors on the source 
235 				// this is important !!!!!!!!!!!!! 
236 				// otherwise you will not get any tags
237 				source.removeAllTagSelectors();
238 				
239 				// read all ids on the source
240 				// just for safety pass an empty dataSelector
241 				ReadReport readReport = source.rawReadIDs(null);
242 
243 				
244 				if (readReport != null) {
245 					// get the readReport holding the sourceReports
246 					ReadReportType report = readReport.getReport();
247 					// get the sourceReports
248 					List<SourceReport> sourceReports = report.getSourceReport();
249 					
250 					List<Tag> tagsToNotify = new LinkedList<Tag>();
251 					// for each sourceReport process the tags
252 					for (SourceReport sourceReport : sourceReports) {
253 						List<TagType> tags = sourceReport.getTag();
254 						if (tags != null) {
255 							// for each tag create a new reader api tag
256 							for (TagType tag : tags) {
257 								List<TagEventType> tagEvents = tag.getTagEvent();
258 								if (tagEvents != null) {
259 									for (TagEventType tagEvent : tagEvents) {
260 										String eventType = tagEvent.getEventType();
261 										LOG.debug("Tag '" + tag.getTagIDAsPureURI() + "' fired Event of type '" + eventType + "'");
262 										if (EventType.EV_OBSERVED.equals(eventType)) {
263 											tagsToNotify.add(new Tag(adaptor.getName(), tag.getTagID(), tag.getTagIDAsPureURI(), System.currentTimeMillis()));
264 										}
265 									}
266 								}
267 							}
268 						}
269 					}
270 					
271 					if (tagsToNotify.size() > 0) {
272 						adaptor.addTags(tagsToNotify);
273 					}
274 					
275 				} else {
276 					LOG.debug("readReport null");
277 				}
278 			}
279 		} else {
280 			LOG.debug("sources null");
281 		}
282 	}

294 		/**

297 		 * @param generator the inputGenerator for this reader.

305 		/**
306 		 * This method contains the main loop.
307 		 */
308 		public void run() {
309 			
310 			try {
311 				initialize();
312 			} catch (RPProxyException e) {
313 				isFailed(e);
314 			}
315 			
316 		}

318 		/**
319 		 * This method creates objects on the reader device using the proxies.
320 		 * 
321 		 * @throws RPProxyException if a proxy operation fails
322 		 */
323 		private void initialize() throws RPProxyException {
324 			
325 			LOG.debug("-----------------------------------------------------------");
326 			LOG.debug("Start initializing InputGenerator...");
327 			LOG.info("Try connecting to reader devices...");
328 			
329 			// create ReaderDevice
330 			LOG.debug("Create ReaderDevice Proxy");
331 			Handshake handshake = new Handshake();
332 			handshake.setTransportProtocol(Handshake.HTTP);
333 			handshake.setMessageFormat(Handshake.FORMAT_XML);
334 			readerDevice = ReaderDeviceFactory.getReaderDevice(adaptor.getCommandChannelHost(),
335 					adaptor.getCommandChannelPort(), handshake);
336 			readerName = readerDevice.getName();
337 
338 			// create notification channel endpoint and add listener
339 			LOG.debug("Try to create NotificationChannelEndpoint at port '" + adaptor.getNotificationChannelPort() + "'...");
340 			notificationChannelEndPoint = new NotificationChannelEndPoint(adaptor.getNotificationChannelPort());
341 			notificationChannelEndPoint.addListener(generator);
342 			LOG.debug("NotificationChannelEndpoint at port '" + adaptor.getNotificationChannelPort() + "' created.");			
343 			
344 			
345 			// create read trigger
346 			readTrigger = TriggerFactory.createTrigger(readTriggerName, Trigger.TIMER, 
347 						String.format("ms=%d", adaptor.getReadTimeInterval()), readerDevice);
348 
349 			LOG.debug("created read trigger: " + readTriggerName);
350 
351 			// create notification trigger
352 			notificationTrigger = TriggerFactory.createTrigger(notificationTriggerName, Trigger.CONTINUOUS,
353 						"", readerDevice);
354 			LOG.debug("created notification trigger: " + notificationTriggerName);
355 
356 			// create the data selector
357 			LOG.debug("create DataSelector");
358 			dataSelector = DataSelectorFactory.createDataSelector(dataSelectorName, readerDevice);
359 			LOG.debug("adding Fieldnames to dataSelector");
360 			dataSelector.addFieldNames(new String[] {FieldName.EVENT_TYPE, 
361 								FieldName.READER_NAME, 
362 								FieldName.TAG_ID,  
363 								FieldName.SOURCE_NAME});
364 			
365 			// create notification channel
366 			String notificationAddress = "tcp://" + adaptor.getNotificationChannelHost() + ":" + adaptor.getNotificationChannelPort()
367 						+ "?mode=" + NOTIFICATION_CHANNEL_MODE;
368 
369 			LOG.debug("create NotificationChannel" + notificationChannelName);
370 			notificationChannel = NotificationChannelFactory.createNotificationChannel(
371 							notificationChannelName, 
372 							notificationAddress, readerDevice);
373 			LOG.debug("NotificationChannel " + notificationChannelName + " created");
374 			
375 			LOG.debug("adding dataSelector to notificationChannel");
376 			notificationChannel.setDataSelector(dataSelector);
377 			LOG.debug("adding notificationTrigger to notificationChannel");
378 			notificationChannel.addNotificationTriggers(new Trigger[] {notificationTrigger});
379 
380 
381 			// add sources to notification channel
382 			LOG.debug("Add Sources to NotificationChannel...");
383 			Source[] sources = readerDevice.getAllSources();
384 			notificationChannel.addSources(sources);
385 			LOG.debug("Sources were added to NotificationChannel.");
386 			
387 			LOG.debug("add readTriggers to the sources");
388 			for (Source source : sources) {
389 				if (adaptor.getSources().contains(source.getName())) {
390 					LOG.debug("adding physical source " + source.getName());
391 					source.addReadTriggers(new Trigger[] {readTrigger});
392 				}
393 			}
394 
395 			// set isReady
396 			isReady = true;
397 						
398 			// notify all when InputGenerator is ready
399 			synchronized (generator) {
400 				generator.notifyAll();
401 			}
402 		
403 			LOG.debug("InputGenerator initialized.");
404 			LOG.info("Connection to reader devices established.");
405 			LOG.debug("-----------------------------------------------------------");			
406 		}
407 
408 		/**
409 		 * This method is invoked if the initialization is failed.
410 		 * An error message will be displayed in log.
411 		 * 
412 		 * @param exception exception thrown
413 		 */
414 		private void isFailed(Exception exception) {
415 			
416 			isFailed = true;
417 			LOG.error("InputGenerator '" + adaptor.getCommandChannelHost() + ":" + adaptor.getCommandChannelPort() + "' initialization failed. (" + exception.getMessage() + ")");
418 			
419 			// notify all when InputGenerator is failed
420 			synchronized (generator) {
421 				generator.notifyAll();
422 			}
423 			
424 		}		
425 	}
