156 	 * This method sets the state of this report generator.
157 	 * If the state changes from UNREQUESTED to REQUESTED, the report generators main loop will be started.
158 	 * If the state changes from REQUESTED to UNREQUESTED, the report generatots main loop will be stoped.

160 	 * @param state to set
161 	 */
162 	public void setState(ReportsGeneratorState state) {
163 		
164 		ReportsGeneratorState oldState = this.state;
165 		this.state = state;
166 		synchronized (this.state) {
167 			this.state.notifyAll();
168 		}
169 		LOG.debug("ReportGenerator '" + name + "' change state from '" + oldState + "' to '" + state + "'");
170 		
171 		if (state == ReportsGeneratorState.REQUESTED && !isRunning) {
172 			start();
173 		} else if (state == ReportsGeneratorState.UNREQUESTED && isRunning) {
174 			stop();
175 		}
176 		
177 	}

192 	 * 
193 	 * @param notificationURI to subscribe
194 	 * @throws DuplicateSubscriptionException if the specified notifiction uri is already subscribed
195 	 * @throws InvalidURIException if the notification uri is invalid
196 	 */
197 	public void subscribe(String notificationURI) throws DuplicateSubscriptionExceptionResponse, InvalidURIExceptionResponse {
198 		
199 		Subscriber uri = new Subscriber(notificationURI);
200 		if (subscribers.containsKey(notificationURI)) {
201 			throw new DuplicateSubscriptionExceptionResponse();
202 		} else {
203 			subscribers.put(notificationURI, uri);
204 			LOG.debug("NotificationURI '" + notificationURI + "' subscribed to spec '" + name + "'.");
205 			if (state == ReportsGeneratorState.UNREQUESTED) {
206 				setState(ReportsGeneratorState.REQUESTED);
207 			}
208 		}
209 		
210 	}

223 			subscribers.remove(notificationURI);
224 			LOG.debug("NotificationURI '" + notificationURI + "' unsubscribed from spec '" + name + "'.");
225 			if (subscribers.isEmpty() && !isPolling) {
226 				setState(ReportsGeneratorState.UNREQUESTED);
227 			}

342 	 * This method contains the main loop of the reports generator.
343 	 * Here the event cycles will be generated and started.
344 	 */
345 	public void run() {
346 		
347 		try {
348 			eventCycle = new EventCycle(this);
349 		} catch (ImplementationExceptionResponse e) {
350 			// TODO Auto-generated catch block
351 			e.printStackTrace();
352 		}
353 		
354 		if (startTriggerValue != null) {
355 			if (repeatPeriodValue == -1) {
356 				// startTrigger is specified and repeatPeriod is not specified
357 				// eventCycle is started when:
358 				// state is REQUESTED and startTrigger is received
359 				
360 			}
361 		} else {
362 			if (repeatPeriodValue != -1) {
363 						
364 				// startTrigger is not specified and repeatPeriod is specified
365 				// eventCycle is started when:
366 				// state transitions from UNREQUESTED to REQUESTED or
367 				// repeatPeriod has elapsed from start of the last eventCycle and
368 				// in that interval the state was never UNREQUESTED
369 				while (isRunning) {
370 					
371 					// wait until state is REQUESTED
372 					while (state != ReportsGeneratorState.REQUESTED) {
373 						try {
374 							synchronized (state) {
375 								state.wait();
376 							}
377 						} catch (InterruptedException e) {
378 							return;
379 						}
380 					}
381 					
382 					// while state is REQUESTED start every repeatPeriod a new EventCycle
383 					while (state == ReportsGeneratorState.REQUESTED) {
384 						
385 						if (eventCycle == null) {
386 							LOG.error("eventCycle is null");
387 						} else {
388 							eventCycle.launch();
389 						}
390 
391 						try {
392 							synchronized (state) {
393 								state.wait(repeatPeriodValue);
394 							}
395 						} catch (InterruptedException e) {
396 							return;
397 						}
398 					}
399 					
400 				}
401 			} else {
402 				
403 				// neither startTrigger nor repeatPeriod are specified
404 				// eventCycle is started when:
405 				// state transitions from UNREQUESTED to REQUESTED or
406 				// immediately after the previous event cycle, if the state is still REQUESTED
407 				while (isRunning) {
408 					
409 					// wait until state is REQUESTED
410 					while (state != ReportsGeneratorState.REQUESTED) {
411 						try {
412 							synchronized (state) {
413 								state.wait();
414 							}
415 						} catch (InterruptedException e) {
416 							return;
417 						}
418 					}
419 					
420 					// while state is REQUESTED start one EventCycle after the other
421 					while (state == ReportsGeneratorState.REQUESTED) {
422 						eventCycle.launch();
423 						
424 						while (!eventCycle.isTerminated()) {
425 							try {
426 								synchronized (eventCycle) {
427 									eventCycle.wait();
428 								}
429 							} catch (InterruptedException e) {
430 								return;
431 							}
432 						}
433 					}
434 				}
435 			}
436 		}

509 	/**
510 	 * This method validates an ec specification under criterias of chapter 8.2.11 of ALE specification version 1.0.
511 	 * 
512 	 * @param spec to validate
513 	 * @throws ECSpecValidationException if the specification is invalid
514 	 * @throws ImplementationException if an implementation exception occurs
515 	 */
516 	private void validateSpec(ECSpec spec) throws ECSpecValidationExceptionResponse, ImplementationExceptionResponse {
517 				
518 		// check if the logical readers are known to the implementation
519 		List<String> logicalReaders = spec.getLogicalReaders().getLogicalReader();
520 		if (logicalReaders != null) {
521 			for (String logicalReaderName : logicalReaders) {
522 				if (!LogicalReaderManager.contains(logicalReaderName)) {
523 					throw new ECSpecValidationExceptionResponse("LogicalReader '" + logicalReaderName + "' is unknown.");
524 				}
525 			}
526 		}
527 		
528 		// boundaries parameter of ECSpec is null or omitted
529 		ECBoundarySpec boundarySpec = spec.getBoundarySpec();
530 		if (boundarySpec == null) {
531 			throw new ECSpecValidationExceptionResponse("The boundaries parameter of ECSpec is null.");
532 		}
533 		
534 		// start and stop tiggers
535 		checkTrigger(boundarySpec.getStartTrigger());
536 		checkTrigger(boundarySpec.getStopTrigger());
537 		
538 		// check if duration, stableSetInterval or repeatPeriod is negative
539 		ECTime time;
540 		if ((time = boundarySpec.getDuration()) != null) {
541 			if (time.getValue() < 0) {
542 				throw new ECSpecValidationExceptionResponse("The duration field of ECBoundarySpec is negative.");
543 			}
544 		}
545 		if ((time = boundarySpec.getStableSetInterval()) != null) {
546 			if (time.getValue() < 0) {
547 				throw new ECSpecValidationExceptionResponse("The stableSetInterval field of ECBoundarySpec is negative.");
548 			}
549 		}
550 		if ((time = boundarySpec.getRepeatPeriod()) != null) {
551 			if (time.getValue() < 0) {
552 				throw new ECSpecValidationExceptionResponse("The repeatPeriod field of ECBoundarySpec is negative.");
553 			}
554 		}
555 		
556 		// check if start trigger is non-empty and repeatPeriod is non-zero
557 		if (boundarySpec.getStartTrigger() != null && boundarySpec.getRepeatPeriod().getValue() != 0) {
558 			throw new ECSpecValidationExceptionResponse("The startTrigger field of ECBoundarySpec is non-empty and " +
559 					"the repeatPeriod field of ECBoundarySpec is non-zero.");
560 		}
561 		
562 		// check if a stopping condition is specified
563 		if (boundarySpec.getStopTrigger() == null && boundarySpec.getDuration() == null &&
564 				boundarySpec.getStableSetInterval() == null) {
565 			throw new ECSpecValidationExceptionResponse("No stopping condition is specified in ECBoundarySpec.");
566 		}
567 		
568 		// check if there is a ECReportSpec instance
569 		if (spec.getReportSpecs() == null || spec.getReportSpecs() == null ||
570 				spec.getReportSpecs().getReportSpec().size() == 0) {
571 			throw new ECSpecValidationExceptionResponse("List of ECReportSpec instances is empty.");
572 		}
573 		
574 		// check if two ECReportSpec instances have identical names
575 		Set<String> reportSpecNames = new HashSet<String>();
576 		for (ECReportSpec reportSpec : spec.getReportSpecs().getReportSpec()) {
577 			if (reportSpecNames.contains(reportSpec.getReportName())) {
578 				throw new ECSpecValidationExceptionResponse("Two ReportSpecs instances have identical names '" +
579 						reportSpec.getReportName() + "'.");
580 			} else {
581 				reportSpecNames.add(reportSpec.getReportName());
582 			}
583 		}
584 		
585 		// check filters
586 		for (ECReportSpec reportSpec : spec.getReportSpecs().getReportSpec()) {
587 			ECFilterSpec filterSpec = reportSpec.getFilterSpec();
588 			
589 			if (filterSpec != null) {
590 				// check include patterns
591 				if (filterSpec.getIncludePatterns() != null) {
592 					for (String pattern : filterSpec.getIncludePatterns().getIncludePattern()) {
593 						new Pattern(pattern, PatternUsage.FILTER);
594 					}
595 				}
596 				
597 				// check exclude patterns
598 				if (filterSpec.getExcludePatterns() != null) {
599 					for (String pattern : filterSpec.getExcludePatterns().getExcludePattern()) {
600 						new Pattern(pattern, PatternUsage.FILTER);
601 					}
602 				}
603 			}
604 			
605 		}
606 		
607 		// check grouping patterns
608 		for (ECReportSpec reportSpec : spec.getReportSpecs().getReportSpec()) {
609 			if (reportSpec.getGroupSpec() != null) {
610 				for (String pattern1 : reportSpec.getGroupSpec().getPattern()) {
611 					Pattern pattern = new Pattern(pattern1, PatternUsage.GROUP);
612 					for (String pattern2 : reportSpec.getGroupSpec().getPattern()) {
613 						if (pattern1 != pattern2 && !pattern.isDisjoint(pattern2)) {
614 							throw new ECSpecValidationExceptionResponse("The two grouping patterns '" + pattern1 +
615 									"' and '" + pattern2 + "' are not disjoint.");
616 						}
617 					}
618 				}
619 			}
620 		}
621 		
622 		// check if there is a output type specified for each ECReportSpec
623 		for (ECReportSpec reportSpec : spec.getReportSpecs().getReportSpec()) {
624 			ECReportOutputSpec outputSpec = reportSpec.getOutput();
625 			if (!outputSpec.isIncludeEPC() && !outputSpec.isIncludeTag() && !outputSpec.isIncludeRawHex() &&
626 					!outputSpec.isIncludeRawDecimal() && !outputSpec.isIncludeCount()) {
627 				throw new ECSpecValidationExceptionResponse("The ECReportOutputSpec of ReportSpec '" +
628 						reportSpec.getReportName() + "' has no output type specified.");
629 			}
630 		}
631 		
632 	}
