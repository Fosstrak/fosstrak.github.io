114 	/**
115 	 * Constructor sets parameter and starts thread.
116 	 * 
117 	 * @param generator to which this event cycle belongs to
118 	 * @throws ImplementationException if an implementation exception occurs
119 	 */
120 	public EventCycle(ReportsGenerator generator) throws ImplementationExceptionResponse {
121 		
122 		// set name
123 		name = generator.getName() + "_" + number++;
124 		
125 		// set ReportGenerator
126 		this.generator = generator;
127 		
128 		// set spec
129 		spec = generator.getSpec();
130 		
131 		// get report specs and create a report for each spec
132 		for (ECReportSpec reportSpec : spec.getReportSpecs().getReportSpec()) {
133 			
134 			// add report spec and report to reports
135 			reports.add(new Report(reportSpec, this));
136 			
137 		}
138 		
139 		// init BoundarySpec values
140 		durationValue = getDurationValue();
141 		
142 		LOG.debug(String.format("durationValue: %s\n",
143 				durationValue));
144 		
145 		setAcceptTags(false);
146 		
147 		LOG.debug("adding logicalReaders to EventCycle");
148 		// get LogicalReaderStubs
149 		if (spec.getLogicalReaders() != null) {
150 			List<String> logicalReaderNames = spec.getLogicalReaders().getLogicalReader();
151 			for (String logicalReaderName : logicalReaderNames) {
152 				LOG.debug("retrieving logicalReader " + logicalReaderName);
153 				LogicalReader logicalReader = LogicalReaderManager.getLogicalReader(logicalReaderName);
154 				
155 				if (logicalReader != null) {
156 					LOG.debug("adding logicalReader " + logicalReader.getName() + " to EventCycle " + name);
157 					logicalReaders.add(logicalReader);
158 				}
159 			}
160 		} else {
161 			LOG.error("ECSpec contains no readers");
162 		}
163 		
164 		for (LogicalReader logicalReader : logicalReaders) {
165 			
166 			// subscribe this event cycle to the logical readers
167 			LOG.debug("registering EventCycle " + name + " on reader " + logicalReader.getName());
168 			logicalReader.addObserver(this);
169 		}
170 		
171 		rounds = 0;
172 		
173 		// create and start Thread
174 		thread = new Thread(this);
175 		thread.start();
176 		
177 		LOG.debug("New EventCycle  '" + name + "' created.");
178 
179 	}

235 	 * This method adds a tag to this event cycle.
236 	 * 
237 	 * @param tag to add
238 	 * @throws ImplementationException if an implementation exception occures
239 	 * @throws ECSpecValidationException if the tag is not valid
240 	 */
241 	public synchronized void addTag(Tag tag) throws ImplementationExceptionResponse, ECSpecValidationExceptionResponse {
242 		if (!isAcceptingTags()) {
243 			return;
244 		}
245 		
246 		// add event only if EventCycle is still running
247 		if (thread.isAlive()) {
248 			LOG.debug("EventCycle '" + name + "' add Tag '" + tag.getTagIDAsPureURI() + "'.");
249 			
250 			for (Tag atag : tags) {
251 				// do not add the tag it is already in the list
252 				if (atag.equals(tag)) {
253 					//LOG.debug("Tag " + tag.getTagIDAsPureURI() + "already in the list");
254 					return;
255 				}
256 			}
257 			
258 			// add tag to tags
259 			tags.add(tag);			
260 		}
261 	}

293 	/**
294 	 * implementation of the observer interface for tags.
295 	 * @param o an observable object that triggered the update
296 	 * @param arg the arguments passed by the observable
297 	 */
298 	public synchronized void update(Observable o, Object arg) {
299 		LOG.debug("EventCycle "+ getName() + ": Update notification received. ");
300 		if (!isAcceptingTags()) {
301 			LOG.debug("EventCycle "+ getName() + ": Not accepting notification.");
302 			return;
303 		}
304 		if (arg instanceof Tag) {
305 			
306 			// process one tag
307 			Tag tag = (Tag) arg;
308 			//tag.prettyPrint(LOG);
309 			try {
310 				addTag(tag);
311 			} catch (ImplementationExceptionResponse ie) {
312 				ie.printStackTrace();
313 			} catch (ECSpecValidationExceptionResponse ive) {
314 				ive.printStackTrace();
315 			}
316 		} else if (arg instanceof List) {
317 			// process multiple tags at once
318 			
319 			List<Tag> tagList = (List<Tag>) arg;
320 			LOG.debug("EventCycle "+ getName() + ": Received list of tags :");
321 			for (Tag tag : tagList) {
322 				try {
323 					addTag(tag);
324 				} catch (ImplementationExceptionResponse ie) {
325 					ie.printStackTrace();
326 				} catch (ECSpecValidationExceptionResponse ive) {
327 					ive.printStackTrace();
328 				}
329 			}
330 		}
331 	}

334 	 * This method stops the thread.
335 	 */
336 	public void stop() {
337 
338 		// unsubscribe this event cycle from logical readers
339 		for (LogicalReader logicalReader : logicalReaders) {
340 			//logicalReader.unsubscribeEventCycle(this);
341 			logicalReader.deleteObserver(this);
342 		}
343 
344 		
345 		if (thread.isAlive()) {
346 			thread.interrupt();
347 			
348 			// stop EventCycle
349 			LOG.debug("EventCycle '" + name + "' stopped.");
350 		}
351 		
352 		isTerminated = true;
353 		
354 		synchronized (this) {
355 			this.notifyAll();
356 		}
357 		
358 	}

382 	/**
383 	 * This method is the main loop of the event cycle in which the tags will be collected.
384 	 * At the end the reports will be generated and the subscribers will be notified.
385 	 */
386 	public void run() {
387 		
388 		// wait for the start
389 		// running will be set by the ReportsGenerator when the EventCycle
390 		// has a subscriber
391 		if (!running) {
392 			synchronized (this) {
393 				try {
394 					this.wait();
395 				} catch (InterruptedException e) {
396 					e.printStackTrace();
397 				}
398 			}
399 		}
400 		
401 		while (running) {
402 			rounds ++;
403 			LOG.info("EventCycle "+ getName() + ": Starting (Round " + rounds + ").");
404 			
405 			// set start time
406 			long startTime = System.currentTimeMillis();
407 	
408 			// accept tags
409 			setAcceptTags(true);
410 			
411 			//------------------------------ run for the specified time
412 			try {
413 				
414 				if (durationValue > 0) {
415 					
416 					// if durationValue is specified and larger than zero, wait for notify or durationValue elapsed.
417 					synchronized (this) {
418 						this.wait(Math.max(1, durationValue - (System.currentTimeMillis() - startTime)));
419 						terminationCondition = ECTerminationCondition.DURATION;
420 					}
421 				} else {
422 					
423 					// if durationValue is not specified or smaller than zero, wait for notify.
424 					synchronized (this) {
425 						this.wait();
426 					}
427 				}
428 			
429 			} catch (InterruptedException e) {
430 				
431 				// if Thread is stopped with method stop(), then return without notify subscribers.
432 				return;
433 				
434 			}
435 			
436 			// dont accept tags anymore
437 			setAcceptTags(false);
438 			//-------------------------------------------------- generate the reports
439 			
440 			// get reports
441 			try {
442 				// compute total time
443 				totalTime = System.currentTimeMillis() - startTime;
444 				
445 				LOG.info("EventCycle "+ getName() + ": Number of Tags read in the current EventCyle.java: " + tags.size());
446 				
447 				ECReports ecReports = getECReports();
448 				
449 				// notifySubscribers
450 				generator.notifySubscribers(ecReports);
451 				
452 				// store the current tags into the old tags
453 				// explicitly clear the tags
454 				if (lastEventCycleTags != null) {
455 					lastEventCycleTags.clear();
456 				}
457 				lastEventCycleTags = tags;
458 				tags = new HashSet<Tag>();
459 				
460 			} catch (ECSpecValidationExceptionResponse e) {
461 				LOG.error("EventCycle "+ getName() + ": Could not create ECReports (" + e.getMessage() + ")");
462 			} catch (ImplementationExceptionResponse e) {
463 				LOG.error("EventCycle "+ getName() + ": Could not create ECReports (" + e.getMessage() + ")");
464 			} catch (Exception e) {
465 				e.printStackTrace();
466 			}
467 			
468 			
469 			LOG.info("EventCycle "+ getName() + ": EventCycle finished (Round " + rounds + ").");
470 			try {
471 				synchronized (this) {
472 					this.wait();
473 				}
474 			} catch (InterruptedException e) {
475 			}			
476 		}
477 			
478 			
479 		// stop EventCycle
480 		stop();
481 		
482 	}

516 	/**
517 	 * This method returns the duration value extracted from the ec specification.
518 	 * 
519 	 * @return duration value in milliseconds
520 	 * @throws ImplementationException if an implementation exception occurs
521 	 */
522 	private long getDurationValue() throws ImplementationExceptionResponse {
523 		if (spec.getBoundarySpec() != null) {
524 			ECTime duration = spec.getBoundarySpec().getDuration();
525 			if (duration != null) {
526 				if (duration.getUnit().compareToIgnoreCase(ECTimeUnit.MS) == 0) {
527 					return duration.getValue();
528 				} else {
529 					throw new ImplementationExceptionResponse("The only ECTimeUnit allowed is milliseconds (MS).");
530 				}
531 			}
532 		}
533 		return -1;
534 		
535 	}
