110 	 * implements the initialize method of the SimulatorServerEngine
111      * 
112      * @param controller 
113 	 * @throws SimulatorServerException 
114 	 */
115 	public void initialize(SimulatorServerController controller) throws SimulatorServerException {
116 		this.controller = controller;
117 
118 		// load gui text
119 		String guiTextFileName = "/props/GUIText_" + LOCALE.getLanguage() + ".properties";
120 		try {
121 			guiText = new PropertyResourceBundle(this.getClass().getResourceAsStream(guiTextFileName));
122 		} catch (IOException e) {
123 			throw new SimulatorServerException("Could not load gui text file.");
124 		}
125 
126 		// load properties
127 		try {
128 			props = new Properties();
129 			props.load(this.getClass().getResourceAsStream(PROPERTIES_FILE_LOCATION));
130 		} catch (IOException e) {
131 			throw new SimulatorServerException("Could not load properties from properties file.");
132 		}
133 		
134 		// initialize GUI
135 		initializeGUI();
136 		LOG.info("GraphicSimulatorServer started");
137 	}

156 	/**
157 	 * computes some properties using other properties
158 	 */
159 	private void computeProperties() {
160 		int nbrOfReader = controller.getReaderIds().size();
161 		int maxNbrOfAntennas = 0;
162 		Iterator readerIt = controller.getReaderIds().iterator();
163 		while (readerIt.hasNext()) {
164 			maxNbrOfAntennas = Math.max(controller.getAntennaIds(((String)readerIt.next())).size(), maxNbrOfAntennas);
165 		}
166 		
167 		// window size
168 		props.setProperty("WindowWidth", new Integer(2 * getProperty("FramePadding") + getProperty("TagWidth") + 2 * getProperty("HorizontalPadding") + getProperty("ReaderWidth") + maxNbrOfAntennas * (getProperty("AntennaWidth") + getProperty("HorizontalInterAntennaPadding")) - getProperty("HorizontalInterAntennaPadding")).toString());
169 		props.setProperty("WindowHeight", new Integer(Math.max(getProperty("MinimumWindowHeight"), 2 * getProperty("FramePadding") + (nbrOfReader * (getProperty("ReaderHeight") + getProperty("InterReaderPadding")) - getProperty("InterReaderPadding")) + 50)).toString());
170 
171 		// reader pane
172 		props.setProperty("ReaderPaneX", new Integer(getProperty("FramePadding") + getProperty("TagWidth") + getProperty("HorizontalPadding")).toString());
173 		props.setProperty("ReaderPaneY", new Integer(getProperty("FramePadding")).toString());
174 		
175 		// antenna pane
176 		props.setProperty("AntennaPaneX", new Integer(getProperty("ReaderPaneX") + getProperty("ReaderWidth") + getProperty("HorizontalPadding")).toString());
177 		props.setProperty("AntennaPaneY", new Integer(getProperty("FramePadding")).toString());
178 	}

288 	 * creates the layered pane if it does not already exists
289 	 * 
290 	 * @return layered pane
291 	 */
292 	private JLayeredPane getJLayeredPane() {
293 		if (jLayeredPane == null) {
294 			jLayeredPane = new JLayeredPane();
295 			jLayeredPane.setLayout(null);
296 			jLayeredPane.setOpaque(true);
297 			jLayeredPane.setBackground(Color.WHITE);
298 			//jLayeredPane.add(getReader(), new Integer(0));
299 			
300 			// add readers
301 			createReaders(controller.getReaderIds());
302 			
303 			// add context menu
304 			jLayeredPane.add(getContextMenu());
305 
306 			// add mouse listener
307 			jLayeredPane.addMouseListener(new MouseAdapter() {
308 
309 				// context menu
310 				public void mouseReleased(MouseEvent e) {
311 					hideActiveContextMenuAndSelection();
312 					if (e.getButton() == MouseEvent.BUTTON3) {
313 						Point posOnScreen = getJLayeredPane().getLocationOnScreen();
314 						contextMenu.setLocation(posOnScreen.x + e.getX(), posOnScreen.y + e.getY());
315 						contextMenu.setVisible(true);
316 						setActiveContextMenu(contextMenu);
317 					} else {
318 						contextMenu.setVisible(false);
319 						if(e.getButton() == MouseEvent.BUTTON1) {
320 							selection.select(tags);
321 						}
322 					}
323 				}
324 				
325 				// selection
326 				public void mousePressed(final MouseEvent e) {
327 					hideActiveContextMenuAndSelection();
328 					if(e.getButton() == MouseEvent.BUTTON1) {
329 						selection.setBounds(0, 0, getProperty("WindowWidth"), getProperty("WindowHeight"));
330 						selection.setStartPoint(e.getPoint());
331 						jLayeredPane.add(selection, new Integer(2));
332 					}
333 				}
334 			});
335 			
336 			// add drag listener
337 			jLayeredPane.addMouseMotionListener(new MouseMotionAdapter() {
338 				public void mouseDragged(MouseEvent e) {
339 					if(selection.isActive()) {
340 						selection.setCurrentPoint(e.getPoint());
341 					}
342 				}
343 			});
344 		}
345 		return jLayeredPane;
346 	}

348 	/**
349 	 * update the layered pane with new reader information
350 	 * 
351 	 * @return layered pane
352 	 */
353 	private void updateJLayeredPane() {
354 		Set controllerReaderIds = controller.getReaderIds();
355 		Set guiReaderIds = readers.keySet();
356 		
357 		// get new readers
358 		Set newReaderIds = new TreeSet(controllerReaderIds);
359 		newReaderIds.removeAll(guiReaderIds);
360 		
361 		// get update readers
362 		Set updateReaderIds = new TreeSet(controllerReaderIds);
363 		updateReaderIds.retainAll(guiReaderIds);
364 		
365 		// get old readers
366 		Set oldReaderIds = new TreeSet(guiReaderIds);
367 		oldReaderIds.removeAll(controllerReaderIds);
368 
369 		// create, update and delete readers
370 		createReaders(newReaderIds);
371 		updateReaders(updateReaderIds);
372 		removeReaders(oldReaderIds);
373 		
374 		jLayeredPane.repaint();
375 	}

399 	/**
400 	 * shows the dialog menu to add a new tag
401 	 */
402 	private void showAddTagDialog() {
403 		// compute position of the dialog
404 		final Point pos = new Point();
405 		pos.x = jLayeredPane.getLocationOnScreen().x + (jLayeredPane.getWidth() - getProperty("DialogWindowWidth")) / 2;
406 		pos.y = jLayeredPane.getLocationOnScreen().y + (jLayeredPane.getHeight() - getProperty("DialogWindowHeight")) / 2;
407 		
408 		// set default id
409 		String id = "A47033F";
410 		for(int i = 0; i < 4 - new Integer(tags.size()).toString().length(); i++) {
411 			id += "0";
412 		}
413 		id += tags.size();
414 		if(tagIdField == null) {
415 			tagIdField = new JTextField();
416 		}
417 		tagIdField.setText(id);
418 		
419 		// create tag dialog if it does not already exists
420 		if(newTagDialog == null) {
421 			newTagDialog = new JDialog(this, guiText.getString("AddNewTagDialogTitle"), true);
422 			newTagDialog.setSize(getProperty("DialogWindowWidth"), getProperty("DialogWindowHeight"));
423 			newTagDialog.setLayout(new BorderLayout());
424 			
425 			// input fields panel
426 			JLabel epcLabel = new JLabel(guiText.getString("TagIdLabel") + ": ");
427 			JPanel inputFields = new JPanel();
428 			inputFields.setLayout(new GridLayout(2, 2));
429 			inputFields.add(epcLabel);
430 			inputFields.add(tagIdField);
431 			
432 			// cancel button
433 			JButton cancelButton = new JButton(guiText.getString("CancelButton"));
434 			cancelButton.addMouseListener(new MouseAdapter() {
435 				public void mouseReleased(MouseEvent e) {
436 					newTagDialog.setVisible(false);
437 				}
438 			});
439 			
440 			// add button
441 			JButton addButton = new JButton(guiText.getString("AddButton"));
442 			addButton.addMouseListener(new MouseAdapter() {
443 				public void mouseReleased(MouseEvent e) {
444 					newTagDialog.setVisible(false);
445 					createNewTag(tagIdField.getText());
446 				}
447 			});
448 			
449 			// buttons panel
450 			JPanel buttons = new JPanel();
451 			buttons.add(addButton);
452 			buttons.add(cancelButton);
453 			
454 			// compose all together
455 			newTagDialog.add(inputFields, BorderLayout.CENTER);
456 			newTagDialog.add(buttons, BorderLayout.SOUTH);
457 			newTagDialog.getRootPane().setDefaultButton(addButton);
458 		}
459 		newTagDialog.setLocation(pos);
460 		newTagDialog.setVisible(true);
461 	}

543 	/**
544 	 * deletes a reader and removes it from the layered pane
545 	 * 
546 	 * @param id to remove
547 	 */
548 	private void removeReader(String id) {
549 		Reader reader = (Reader)readers.get(id);
550 		
551 		// remove antennas
552 		Antenna[] antennas = (Antenna[])reader.getAntennas().values().toArray(new Antenna[0]);
553 		for (int i = 0; i < antennas.length; i++) {
554 			removeAntenna(antennas[i], reader);
555 		}
556 		
557 		// remove reader;
558 		readers.remove(id);
559 		jLayeredPane.remove(reader);
560 	}

672 	/**
673 	 * gets a property value as integer
674 	 * 
675 	 * @param key of the property
676 	 * @return property as integer value
677 	 */
678 	public int getProperty(String key) {
679 		String error;
680 		String value = props.getProperty(key);
681 		if(value == null) {
682 			error =  "Value '" + key + "' not found in properties !";
