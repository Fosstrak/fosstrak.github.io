48  /**
49   * This class can be used to process <code>Alarm</code> objects using SNMP.
50   */
51  public class SnmpAlarmProcessor implements AlarmProcessor {
52  	
53  	/**
54  	 * The SNMP agent.
55  	 */
56  	private SnmpAgent snmpAgent;
57  	
58  	/**
59  	 * The <code>EpcglobalReaderMib</code>.
60  	 */
61  	private EpcglobalReaderMib epcglobalReaderMib;
62  	
63  	/**
64  	 * Constructor.
65  	 * 
66  	 * @param snmpAgent
67  	 *            The SNMP agent
68  	 */
69  	public SnmpAlarmProcessor(SnmpAgent snmpAgent) {
70  		this.snmpAgent = snmpAgent;
71  		epcglobalReaderMib = EpcglobalReaderMib.getInstance();
72  	}

75  	 * Processes an <code>Alarm</code> object.

83  		if (notifOrig != null) {

86  				ReaderDeviceOperStatusAlarm readerDeviceOperStatusAlarm = (ReaderDeviceOperStatusAlarm) alarm;

88  						.findMOScalar(EpcglobalReaderMib.oidTrapVarEpcgRdrDevOperStatusPrior);

90  						readerDeviceOperStatusAlarm.getFromState().toInt()));

112 										new VariableBinding(

114 												operStatePrior.getValue()),
115 										new VariableBinding(
116 												EpcglobalReaderMib.oidTrapVarEpcgRdrDevOperStatus,
117 												new Integer32(
118 														readerDeviceOperStatusAlarm
119 																.getToState()
120 																.toInt())) });
