154 	/**
155 	 * Creates and returns the singleton instance of the <code>SnmpAgent</code>.
156 	 * If there already exists an instance it returns that instance.
157 	 * 
158 	 * @param bootCounterFile
159 	 *            The boot counter file
160 	 * @param configFile
161 	 *            The config file
162 	 * @param address
163 	 *            The SNMP agent's address
164 	 * @return The singleton instance of the <code>SnmpAgent</code>
165 	 */
166 	public static SnmpAgent create(File bootCounterFile, File configFile,
167 			String address) {
168 		if (SnmpAgent.instance == null) {
169 			try {
170 				SnmpAgent.instance = new SnmpAgent(bootCounterFile, configFile,
171 						address);
172 			} catch (IOException ioe) {
173 				log.error("Unable to create SnmpAgent.");
174 			}
175 		} else {
176 			log
177 					.warn("SnmpAgent already exists: only one SnmpAgent object can be created.");
178 		}
179 		return SnmpAgent.instance;
180 	}

188 		if (SnmpAgent.instance == null) log.warn("Call create() method first -> null returned");
189 		return SnmpAgent.instance;

303 	 * @param vacm

306 	 */
307 	protected void addViews(VacmMIB vacm) {
308 		vacm.addGroup(SecurityModel.SECURITY_MODEL_SNMPv1, new OctetString(
309 				"public"), new OctetString("v1v2group"),
310 				StorageType.nonVolatile);
311 		vacm.addGroup(SecurityModel.SECURITY_MODEL_SNMPv2c, new OctetString(
312 				"public"), new OctetString("v1v2group"),
313 				StorageType.nonVolatile);
314 		vacm.addAccess(new OctetString("v1v2group"), new OctetString(),
315 				SecurityModel.SECURITY_MODEL_ANY, SecurityLevel.NOAUTH_NOPRIV,
316 				VacmMIB.vacmExactMatch, new OctetString("fullReadView"),
317 				new OctetString("fullWriteView"), new OctetString(
318 						"fullNotifyView"), StorageType.nonVolatile);
319 		vacm.addViewTreeFamily(new OctetString("fullReadView"), new OID("1.3"),
320 				new OctetString(), VacmMIB.vacmViewIncluded,
321 				StorageType.nonVolatile);
322 		vacm.addViewTreeFamily(new OctetString("fullWriteView"),
323 				new OID("1.3"), new OctetString(), VacmMIB.vacmViewIncluded,
324 				StorageType.nonVolatile);
325 		vacm.addViewTreeFamily(new OctetString("fullNotifyView"),
326 				new OID("1.3"), new OctetString(), VacmMIB.vacmViewIncluded,
327 				StorageType.nonVolatile);

362 	 * Adds community to security name mappings needed for SNMPv1 and SNMPv2c.
363 	 * 
364 	 * @param communityMIB
365 	 *            The <code>SnmpCommunityMIB</code> holding coexistence
366 	 *            configuration for community based security models
367 	 */
368 	protected void addCommunities(SnmpCommunityMIB communityMIB) {
369 		Variable[] com2sec = new Variable[] { new OctetString("public"), // community name
370 				new OctetString("public"), // security name
371 				getAgent().getContextEngineID(), // local engine ID
372 				new OctetString(), // default context name
373 				new OctetString(), // transport tag
374 				new Integer32(StorageType.nonVolatile), // storage type
375 				new Integer32(RowStatus.active) // row status
376 		};
377 		MOTableRow row = communityMIB.getSnmpCommunityEntry().createRow(
378 				new OctetString("public2public").toSubIndex(true), com2sec);
379 		communityMIB.getSnmpCommunityEntry().addRow(row);
380 	}

785 	 * Adds the given <code>AlarmChannel</code>s.
786 	 * 
787 	 * @param alarmChannels
788 	 *            The <code>AlarmChannel</code>s to add
789 	 */
790 	public void addAlarmChannels(AlarmChannel[] alarmChannels) {
791 		for (int i = 0; i < alarmChannels.length; i++) {
792 			AlarmChannel curChan = alarmChannels[i];
793 			Address curAddr = curChan.getAddress();
794 			
795 			if (snmpTargetMIB != null) {
796 				snmpTargetMIB.addTargetAddress(new OctetString(curChan
797 						.getName()), TransportDomains.transportDomainUdpIpv4,
798 						new OctetString(new UdpAddress(curAddr.getHost() + "/"
799 								+ curAddr.getPort()).getValue()), 200, 1,
800 						new OctetString("notify"), new OctetString("v2c"),
801 						StorageType.permanent);
802 			} else {
803 				log.error("Run the init() method first.");
804 				return;
805 			}
806 		}
807 	}
