485   private void createIpAddrEntry() {
486     MOColumn[] ipAddrEntryColumns = new MOColumn[5];
487     ipAddrEntryColumns[idxIpAdEntAddr] = 
488       moFactory.createColumn(colIpAdEntAddr, 
489                              SMIConstants.SYNTAX_IPADDRESS,
490                              MOAccessImpl.ACCESS_READ_ONLY);
491     ipAddrEntryColumns[idxIpAdEntIfIndex] = 
492       moFactory.createColumn(colIpAdEntIfIndex, 
493                              SMIConstants.SYNTAX_INTEGER,
494                              MOAccessImpl.ACCESS_READ_ONLY);
495     ipAddrEntryColumns[idxIpAdEntNetMask] = 
496       moFactory.createColumn(colIpAdEntNetMask, 
497                              SMIConstants.SYNTAX_IPADDRESS,
498                              MOAccessImpl.ACCESS_READ_ONLY);
499     ipAddrEntryColumns[idxIpAdEntBcastAddr] = 
500       moFactory.createColumn(colIpAdEntBcastAddr, 
501                              SMIConstants.SYNTAX_INTEGER,
502                              MOAccessImpl.ACCESS_READ_ONLY);
503     ipAddrEntryColumns[idxIpAdEntReasmMaxSize] = 
504       moFactory.createColumn(colIpAdEntReasmMaxSize, 
505                              SMIConstants.SYNTAX_INTEGER,
506                              MOAccessImpl.ACCESS_READ_ONLY);
