164 	/**
165 	 * Serializes the command.
166 	 */
167 	public String serializeCommand() {
168 		try {
169 			StringWriter sw = new StringWriter();
170 			sw.write(EXCLAMATION);
171 			sw.write(id + " ");
172 			if(objectTypeName != null) {
173 				sw.write(objectTypeName);
174 				if(targetName != null && !targetName.equals("")) {
175 					sw.write(SHARP);
176 					sw.write(targetName);
177 				}
178 				sw.write(DOT);
179 			}
180 			sw.write(commandName);
181 			if(parameterString != null) {
182 				sw.write(" ");
183 				sw.write(parameterString);
184 			}
185 			return sw.getBuffer().toString();
186 		} catch (Exception e) {
187 			return "";
188 		}
189 	}

319 	/**
320 	 * Creates a copy of the internal list of parameters. The list items are
321 	 * represented as <code>byte[]</code> used in <source>source.writeID(...)</code>.
322 	 * 
323 	 * @param strArray
324 	 *            Array of Strings with hex values (passwords, tag ids, ...)
325 	 * @return List of <code>byte[]</code>
326 	 */
327 	public List toHexStringList(String[] strArray) {
328 
329 		List hexList = new ArrayList(strArray.length);
330 		for (int i = 0; i < strArray.length; i++) {
331 			byte[] hexArray = HexUtil.hexToByteArray(strArray[i]);
332 			hexList.add(hexArray);
333 		}
334 		return hexList;
335 
336 	}

338 	/**
339 	 * Creates a copy of the internal list of parameters. The list items are
340 	 * represented as <code>TagFieldValueParamType</code> used in
341 	 * <source>source.write(...)</code>.
342 	 * 
343 	 * @param tfArray
344 	 *            TagFieldValue array
345 	 * @return List of <code>TagFieldValueParamType</code>
346 	 * @throws JAXBException
347 	 * @see org.accada.reader.msg.command.TagFieldValueParamType
348 	 */
349 	public List toTagFieldValueList(TagFieldValue[] tfArray)
350 			throws JAXBException {
351 		List tfvList = new ArrayList(tfArray.length);
352 		for (int i = 0; i < tfArray.length; i++) {
353 			TagFieldValueParamType tfvParam = TextCommandParserHelper.cmdFactory
354 					.createTagFieldValueParamType();
355 			tfvParam.setTagFieldName(tfArray[i].getTagFieldName());
356 			tfvParam.setTagFieldValue(HexUtil.hexToByteArray(tfArray[i]
357 					.getValue()));
358 			tfvList.add(tfvParam);
359 		}
360 		return tfvList;
361 
362 	}
