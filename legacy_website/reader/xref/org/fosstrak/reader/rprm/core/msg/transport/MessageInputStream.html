25  package org.fosstrak.reader.rprm.core.msg.transport;
26  
27  import java.io.BufferedReader;
28  import java.io.IOException;
29  import java.io.InputStream;
30  import java.io.InputStreamReader;
31  
32  import org.fosstrak.reader.rprm.core.msg.MessageFormat;
33  import org.fosstrak.reader.rprm.core.msg.SenderHandshakeMessage;
34  import org.apache.log4j.Logger;
35  
36  /**
37   * An input Stream that streams messages.
38   * 
39   * The <code>MessageInputStream</code> is based on a simple InputStream which
40   * it processes in order to extract messages. The processing of the InputStream
41   * is done in the abstrace method <code>processMessage()</code> which should
42   * be implemented by each concrete instance of this abstract class.
43   * 
44   * 
45   * @author Andreas Fürer, ETH Zurich Switzerland, Winter 2005/06
46   */
47  public abstract class MessageInputStream {
48  
49  	/** The logger. */
50  	static protected Logger log;
51  
52  	/** The bufferedReader. */
53  	protected BufferedReader reader = null;
54  
55  	/** Message buffer. */
56  	protected StringBuffer messageBuffer = null;
57  
58  	/** The message. */
59  	protected String message = null;
60  
61  	/** Flag that indicates, whether a message has been read or not. */
62  	protected boolean messageCompleted;
63  
64  	/**
65  	 * Creates a new buffered input stream to read data from the specified
66  	 * underlying input stream with a default 512-byte buffer size.
67  	 * 
68  	 * @param input
69  	 *            The input that has to be examined.
70  	 * 
71  	 */
72  	public MessageInputStream(InputStream input) {
73  		log = Logger.getLogger(getClass().getName());
74  		log.debug("MessageInputStream created");
75  		reader = new BufferedReader(new InputStreamReader(input));
76  	}
77  
78  	// ---------------------------methods------------------------------------------
79  
80  	/**
81  	 * Parses the handshake from the reader
82  	 */
83  	public abstract SenderHandshakeMessage readHandshake() throws IOException;
84  
85  	/**
86  	 * Reads a plain message from the internal buffer.
87  	 * 
88  	 * @return The message.
89  	 */
90  	public String readMessage() throws IOException {
91  		this.message = null;
92  		this.messageCompleted = false;
93  		processMessage();
94  		return message;
95  	}
96  
97  	/**
98  	 * Processes the incomming stream. The method reads from the stream and
99  	 * extracts the message
100 	 * 
101 	 */
102 	public abstract void processMessage() throws IOException;
103 
104 	/**
105 	 * Close the associated stream.
106 	 */
107 	public void close() {
108 		try {
109 			this.reader.close();
110 		} catch (IOException e) {
111 			log.error(e);
112 
113 		}
114 	}
115 }
