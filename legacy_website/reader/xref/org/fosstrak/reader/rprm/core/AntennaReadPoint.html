37   * <code>AntennaReadPoint</code> extends <code>ReadPoint</code>. It is used
38   * to track details about the Radio Frequency characteristics and RFID
39   * statistics for the reader device.

245 	/**
246 	 * The static method to create an instance of a
247 	 * <code>AntennaReadPoint</code>. If this method terminates successfully,
248 	 * the <code>AntennaReadPoint</code> is added to the list of read points
249 	 * in the reader device. If a read point with the same name exists, an
250 	 * exception ("ERROR_OBJECT_EXISTS") is thrown.
251 	 * 
252 	 * @param name
253 	 *            The name of the <code>AntennaReadPoint</code>
254 	 * @param readerDevice
255 	 *            The reader device the source belongs to
256 	 * @param reader
257 	 *            The <code>HardwareAbstraction</code> which contains the read
258 	 *            point
259 	 * @return The instance of the created readpoint
260 	 * @throws ReaderProtocolException
261 	 *             "ERROR_OBJECT_EXISTS"
262 	 */
263 	public static AntennaReadPoint create(final String name,
264 			final ReaderDevice readerDevice, final HardwareAbstraction reader)
265 			throws ReaderProtocolException {
266 		// check if ReadPoint with the same name exists
267 		try {
268 			readerDevice.getReadPoint(name);
269 		} catch (ReaderProtocolException e) {
270 			// create new ReadPoint
271 			AntennaReadPoint newReadPoint = new AntennaReadPoint(name, readerDevice, reader);
272 			readerDevice.getReadPoints().put(name, newReadPoint);
273 			return newReadPoint;
274 		}
275 		throw new ReaderProtocolException("ERROR_OBJECT_EXISTS",
276 				MessagingConstants.ERROR_OBJECT_EXISTS);
277 	}

587 		failedMemReadCount++;
588 		FailedMemReadAlarm alarm = new FailedMemReadAlarm(name
589 				+ "_FailedMemReadAlarm", failedMemReadAlarmControl.getLevel(),
590 				readerDevice, this);
591 		readerDevice.getAlarmManager().fireAlarm(alarm,
592 				failedMemReadAlarmControl);
593 	}
