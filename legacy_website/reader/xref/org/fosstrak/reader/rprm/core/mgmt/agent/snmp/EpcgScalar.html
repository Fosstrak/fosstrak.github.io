229 	 * Returns the actual value of this scalar managed object.
230 	 * 
231 	 * @return A non <code>null</code> <code>Variable</code> with the same
232 	 *         syntax defined for this scalar object
233 	 */
234 	@Override
235 	public Variable getValue() {
236 		
237 		boolean refresh = ((System.currentTimeMillis() - lastRefreshTime) > refreshTimeInMs);
238 		if (refresh) {
239 			log.debug("refreshing " + type);
240 			lastRefreshTime = System.currentTimeMillis();
241 		} else {
242 			log.debug("no need to refresh " + type);
243 		}
244 		
245 		if (refresh) {
246 			
247 			ReaderDevice readerDevice = null;
248 			
249 			switch (type) {
250 				case EPCG_RDR_DEV_DESCRIPTION:
251 					setValue(new OctetString(((ReaderDevice)(objects[0])).getDescription()));
252 					break;
253 				case EPCG_RDR_DEV_ROLE:
254 					setValue(new OctetString(((ReaderDevice)(objects[0])).getRole()));
255 					break;
256 				case EPCG_RDR_DEV_EPC:
257 					setValue(new OctetString(((ReaderDevice)(objects[0])).getEPC()));
258 					break;
259 				case EPCG_RDR_DEV_SERIAL_NUMBER:
260 					setValue(new OctetString(((ReaderDevice)(objects[0])).getSerialNumber()));
261 					break;
262 				case EPCG_RDR_DEV_TIME_UTC:
263 					setValue(SnmpUtil.dateToOctetString(((ReaderDevice)(objects[0])).getTimeUTC()));
264 					break;
265 				case EPCG_RDR_DEV_CURRENT_SOURCE:
266 					SnmpTable sourceTable = (SnmpTable)SnmpUtil.getSnmpTable(TableTypeEnum.EPCG_SOURCE_TABLE);
267 					String sourceName = ((ReaderDevice)objects[0]).getCurrentSource().getName();
268 					OID index = sourceTable.getTableRowIndexByValue(new OctetString(sourceName), EpcglobalReaderMib.idxEpcgSrcName);
269 					OID oid = new OID(EpcglobalReaderMib.oidEpcgSourceEntry);
270 					oid.append(EpcglobalReaderMib.colEpcgSrcName);
271 					oid.append(index);
272 					setValue(oid);
273 					break;
274 				case EPCG_RDR_DEV_REBOOT:
275 					setValue(new Integer32(2));
276 					break;
277 				case EPCG_RDR_DEV_RESET_STATISTICS:
278 					setValue(new Integer32(2));
279 					break;
280 				case EPCG_RDR_DEV_RESET_TIMESTAMP:
281 					// do nothing
282 					break;
283 				case EPCG_RDR_DEV_NORMALIZE_POWER_LEVEL:
284 					setValue(new Integer32(1));
285 					break;
286 				case EPCG_RDR_DEV_NORMALIZE_NOISE_LEVEL:
287 					setValue(new Integer32(1));
288 					break;
289 				case EPCG_RDR_DEV_OPER_STATUS:
290 					setValue(new Integer32(((ReaderDevice)objects[0]).getOperStatus().toInt()));
291 					break;
292 				case EPCG_RDR_DEV_OPER_STATUS_PRIOR:
293 					// do nothing
294 					break;
295 				case EPCG_RDR_DEV_OPER_STATE_ENABLE:
296 					setValue(new Integer32(((ReaderDevice)objects[0]).getOperStatusAlarmControl().getEnabled() ? 1 : 2));
297 					break;
298 				case EPCG_RDR_DEV_OPER_NOTIF_FROM_STATE:
299 					setValue(SnmpUtil.operStateToBITS(((ReaderDevice)objects[0]).getOperStatusAlarmControl().getTriggerFromState()));
300 					break;
301 				case EPCG_RDR_DEV_OPER_NOTIF_TO_STATE:
302 					setValue(SnmpUtil.operStateToBITS(((ReaderDevice)objects[0]).getOperStatusAlarmControl().getTriggerToState()));
303 					break;
304 				case EPCG_RDR_DEV_OPER_NOTIF_STATE_LEVEL:
305 					setValue(new Integer32(((ReaderDevice)objects[0]).getOperStatusAlarmControl().getLevel().toInt()));
306 					break;
307 				case EPCG_RDR_DEV_OPER_STATE_SUPPRESS_INTERVAL:
308 					setValue(new UnsignedInteger32(((ReaderDevice)objects[0]).getOperStatusAlarmControl().getSuppressInterval()));
309 					break;
310 				case EPCG_RDR_DEV_OPER_STATE_SUPPRESSIONS:
311 					setValue(new Counter32(ReaderDevice.getOperStateSuppressions()));
312 					break;
313 				case EPCG_RDR_DEV_FREE_MEMORY:
314 					setValue(new Gauge32(((ReaderDevice)objects[0]).getFreeMemory()));
315 					break;
316 				case EPCG_RDR_DEV_FREE_MEMORY_NOTIF_ENABLE:
317 					setValue(new Integer32(((ReaderDevice)objects[0]).getFreeMemoryAlarmControl().getEnabled() ? 1 : 2));
318 					break;
319 				case EPCG_RDR_DEV_FREE_MEMORY_NOTIF_LEVEL:
320 					setValue(new Integer32(((ReaderDevice)objects[0]).getFreeMemoryAlarmControl().getLevel().toInt()));
321 					break;
322 				case EPCG_RDR_DEV_FREE_MEMORY_ONSET_THRESHOLD:

324 					break;

331 					break;
332 				case EPCG_RDR_DEV_MEM_STATE_SUPPRESS_INTERVAL:
333 					setValue(new UnsignedInteger32(((ReaderDevice)objects[0]).getFreeMemoryAlarmControl().getSuppressInterval()));
334 					break;
335 				case EPCG_RDR_DEV_MEM_STATE_SUPPRESSIONS:
336 					setValue(new Counter32(ReaderDevice.getMemStateSuppressions()));
337 					break;
338 //				case EPCG_READ_POINT_PRIOR_OPER_STATUS:
339 //					// do nothing
340 //					break;
341 //				case EPCG_READ_POINT_OPER_STATE_SUPPRESS_INTERVAL:
342 //					// ...
343 //					break;
344 //				case EPCG_READ_POINT_OPER_STATE_SUPPRESSIONS:
345 //					setValue(new Counter32(ReadPoint.getOperStateSuppressions()));
346 //					break;
347 //				case EPCG_ANT_RD_PNT_SUPPRESS_INTERVAL:
348 //					// ...
349 //					break;
350 //				case EPCG_IO_PORT_OPER_STATUS_PRIOR:
351 //					// do nothing
352 //					break;
353 //				case EPCG_IO_PORT_OPER_STATE_SUPPRESS_INTERVAL:
354 //					// ...
355 //					break;
356 //				case EPCG_IO_PORT_OPER_STATE_SUPPRESSIONS:
357 //					setValue(new Counter32(IOPort.getOperStateSuppressions()));
358 //					break;
359 //				case EPCG_SRC_OPER_PRIOR_STATUS:
360 //					// do nothing
361 //					break;

364 //					break;
365 //				case EPCG_SRC_OPER_STATE_SUPPRESSIONS:
366 //					setValue(new Counter32(Source.getOperStateSuppressions()));

368 //				case EPCG_NOTIF_CHAN_OPER_STATUS_PRIOR:
369 //					// do nothing
370 //					break;
371 //				case EPCG_NOTIF_CHAN_OPER_STATE_SUPPRESS_INTERVAL:
372 //					// ...
373 //					break;

375 //					setValue(new Counter32(NotificationChannel.getOperStateSuppressions()));
376 //					break;
377 				case SYS_DESCR:
378 					readerDevice = (ReaderDevice)objects[0];
379 					String descr =

382 						+ ", Model: "

386 					setValue(new OctetString(descr));
387 					break;
388 				case SYS_LOCATION:
389 					setValue(new OctetString(((ReaderDevice)objects[0]).getLocationDescription()));
390 					break;
391 				case SYS_CONTACT:
392 					setValue(new OctetString(((ReaderDevice)objects[0]).getContact()));
393 					break;
394 				case SYS_UP_TIME:
395 					setValue(new TimeTicks(((ReaderDevice)objects[0]).getTimeTicks()));
396 					break;
397 				case SYS_NAME:
398 					setValue(new OctetString(((ReaderDevice)objects[0]).getName()));
399 					break;
400 			}
401 		}

405 	/**
406 	 * Sets the value of this scalar managed object without checking it for the
407 	 * correct syntax.
408 	 * 
409 	 * @param value
410 	 *            A <code>Variable</code> with the same syntax defined for
411 	 *            this scalar object (not checked)
412 	 * @return A SNMP error code (zero indicating success by default)
413 	 */
414 	@Override
415 	public int setValue(Variable value) {
416 		switch (type) {
417 			case EPCG_RDR_DEV_REBOOT:
418 				boolean reboot = ((Integer32)value).toInt() == 1;
419 				if (reboot) {
420 					try {
421 						((ReaderDevice)(objects[0])).reboot();
422 					} catch (ReaderProtocolException rpe) {
423 						log.error(rpe.getMessage());
424 					}
425 				}
426 				break;
427 			case EPCG_RDR_DEV_RESET_STATISTICS:
428 				boolean resetStatistics = ((Integer32) value).toInt() == 1;
429 				if (resetStatistics) {
430 					((ReaderDevice) (objects[0])).resetStatistics();
431 					MOScalar resetTimestamp = SnmpUtil.findMOScalar(EpcglobalReaderMib.oidEpcgRdrDevResetTimestamp);
432 					if (resetTimestamp != null) {
433 						resetTimestamp.setValue(SnmpUtil.dateToOctetString(new Date()));
434 					}
435 				}
436 				break;
437 			case EPCG_RDR_DEV_OPER_STATE_ENABLE:
438 				((ReaderDevice) (objects[0])).getOperStatusAlarmControl().setEnabled(((Integer32)value).toInt() == 1);
439 				break;
440 			case EPCG_RDR_DEV_OPER_NOTIF_FROM_STATE:
441 				((ReaderDevice) (objects[0])).getOperStatusAlarmControl().setTriggerFromState(SnmpUtil.bitsToOperState((OctetString)value));
442 				break;
443 			case EPCG_RDR_DEV_OPER_NOTIF_TO_STATE:
444 				((ReaderDevice) (objects[0])).getOperStatusAlarmControl().setTriggerToState(SnmpUtil.bitsToOperState((OctetString)value));
445 				break;
446 			case EPCG_RDR_DEV_OPER_NOTIF_STATE_LEVEL:
447 				((ReaderDevice) (objects[0])).getOperStatusAlarmControl().setLevel(AlarmLevel.intToEnum(((Integer32)value).toInt()));
448 				break;
449 			case EPCG_RDR_DEV_OPER_STATE_SUPPRESS_INTERVAL:
450 				((ReaderDevice) (objects[0])).getOperStatusAlarmControl().setSuppressInterval(((UnsignedInteger32)value).toInt());
451 				break;
452 			case EPCG_RDR_DEV_FREE_MEMORY_NOTIF_ENABLE:
453 				((ReaderDevice) (objects[0])).getFreeMemoryAlarmControl().setEnabled(((Integer32)value).toInt() == 1);
454 				break;
455 			case EPCG_RDR_DEV_FREE_MEMORY_NOTIF_LEVEL:
456 				((ReaderDevice) (objects[0])).getFreeMemoryAlarmControl().setLevel(AlarmLevel.intToEnum(((Integer32)value).toInt()));
457 				break;
458 			case EPCG_RDR_DEV_FREE_MEMORY_ONSET_THRESHOLD:
459 				((ReaderDevice) (objects[0])).getFreeMemoryAlarmControl().setAlarmThreshold(((UnsignedInteger32)value).toInt());
460 				break;
461 			case EPCG_RDR_DEV_FREE_MEMORY_ABATE_THRESHOLD:
462 				((ReaderDevice) (objects[0])).getFreeMemoryAlarmControl().setRearmThreshold(((UnsignedInteger32)value).toInt());
463 				break;
464 			case EPCG_RDR_DEV_MEM_STATE_SUPPRESS_INTERVAL:
465 				((ReaderDevice) (objects[0])).getFreeMemoryAlarmControl().setSuppressInterval(((UnsignedInteger32)value).toInt());
466 				break;
467 			case SYS_CONTACT:
468 				((ReaderDevice) (objects[0])).setContact(((OctetString)value).toString());
469 				break;
470 			case SYS_LOCATION:
471 				((ReaderDevice) (objects[0])).setLocationDescription(((OctetString)value).toString());
