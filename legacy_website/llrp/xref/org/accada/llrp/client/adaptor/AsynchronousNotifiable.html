21  package org.accada.llrp.client.adaptor;
22  
23  import java.rmi.Remote;
24  import java.rmi.RemoteException;
25  
26  import org.accada.llrp.client.exception.LLRPRuntimeException;
27  
28  /**
29   * if you want to receive asynchronous message notifications 
30   * you can implement this interface.
31   * @author sawielan
32   *
33   */
34  public interface AsynchronousNotifiable extends Remote {
35  	/**
36  	 * when an asynchronous message arrived, this method will be invoked.
37  	 * @param message the LLRPMessage arrived asynchronously.
38  	 * @param readerName the nam eof the reader that read the message.
39  	 * @throws RemoteException when there has been an error in the communication.
40  	 */
41  	public void notify(byte[] message, String readerName) throws RemoteException;
42  	
43  	/**
44  	 * this method can be called asynchronously when 
45  	 * there occurs an asynchronous error or exception.
46  	 * @param e the exception that was triggered.
47  	 * @param readerName the name of the reader that triggered the error.
48  	 * @throws RemoteException when there has been an error in the communication.
49  	 */
50  	public void notifyError(LLRPRuntimeException e, String readerName) throws RemoteException;
51  }
