204     /**
205      * Constructs a new QueryClientGui which sends its queries to the given
206      * endpoint address. If no such address is provided, the properties file is
207      * checked; if there is an error with reading the properties, a default url
208      * will be provided.
209      * 
210      * @param address
211      *            The address to send the queries to.
212      */
213     public QueryClientGui(final String address) {
214         generateParamHashMap();
215 
216         // setup client GUI
217         javax.swing.SwingUtilities.invokeLater(new Runnable() {
218             public void run() {
219                 createMainWindow(address);
220             }
221         });
222 
223         // set up query client. The supplied JTextArea is used for debug output
224         createDebugWindow();
225         client = new QueryClientGuiHelper(address, dwOutputTextArea);
226 
227         // update queryUrl, in case the provided address parameter was null
228         queryUrl = client.getEndpointAddress();
229         mwServiceUrlTextField.setText(queryUrl);
230     }

921      * Dispatcher for the various events. Some simple cases may be handled
922      * directly within this method.
923      * 
924      * @see java.awt.event.ActionListener#actionPerformed(java.awt.event.ActionEvent)
925      * @param e
926      *            The ActionEvent.
927      */
928     public final void actionPerformed(final ActionEvent e) {
929         if (e.getSource() == mwRunQueryButton) {
930             mwQueryButtonPressed();
931             return;
932         }
933         if (e.getSource() == mwServiceInfoButton) {
934             mwInfoButtonPressed();
935             return;
936         }
937         if (e.getSource() == mwSubscriptionIdButton) {
938             client.querySubscriptionIDs();
939             return;
940         }
941         if (e.getSource() == mwFillInExampleButton) {
942             createExampleWindow();
943             return;
944         }
945         if (e.getSource() == dwClearButton) {
946             dwOutputTextArea.setText("");
947         }
948         if (e.getSource() == mwUnsubscribeQueryButton) {
949             client.unsubscribeQuery(mwUnsubscribeQueryTextField.getText());
950         }
951         if (e.getSource() == ewOkButton) {
952             examplesChanged();
953             return;
954         }
955         if (e.getSource() == mwShowDebugWindowCheckBox) {
956             debugWindow.setVisible(mwShowDebugWindowCheckBox.isSelected());
957             return;
958         }
959         int i = mwQuerySelectComboBoxes.indexOf(e.getSource());
960         if (i >= 0) {
961             mwQuerySelectComboBoxesChanged(i);
962             return;
963         }
964     }

966     /**
967      * Handler for pressed "Run query" button.
968      */
969     private void mwQueryButtonPressed() {
970         dwOutputTextArea.setText("");
971         if (resultsWindow != null) {
972             resultsWindow.dispose();
973         }
974         try {
975             checkQueryUrl(mwServiceUrlTextField.getText());
976             client.clearParameters();
977 
978             /* get event type selection from GUI */
979             ArrayOfString events = new ArrayOfString();
980             if (mwObjectEventsCheckBox.isSelected()) {
981                 events.getString().add("ObjectEvent");
982             }
983             if (mwAggregationEventsCheckBox.isSelected()) {
984                 events.getString().add("AggregationEvent");
985             }
986             if (mwQuantityEventsCheckBox.isSelected()) {
987                 events.getString().add("QuantityEvent");
988             }
989             if (mwTransactionEventsCheckBox.isSelected()) {
990                 events.getString().add("TransactionEvent");
991             }
992             if (!events.getString().isEmpty()) {
993                 QueryParam queryParam = new QueryParam();
994                 queryParam.setName("eventType");
995                 queryParam.setValue(events);
996                 client.addParameter(queryParam);
997             }
998 
999             String name;
1000             for (int i = 0; i < mwQueryArgumentTextFields.size() - 1; i++) {
1001                 name = ((JTextFieldEnhanced) mwQueryArgumentTextFields.get(i)).queryItem.getQueryText();
1002                 QueryParam param = new QueryParam();
1003                 param.setName(name);
1004                 switch (((JTextFieldEnhanced) mwQueryArgumentTextFields.get(i)).queryItem.getParamType()) {
1005                 case ListOfString:
1006                     ArrayOfString valueArray = client.stringListToArray(((JTextField) mwQueryArgumentTextFields.get(i)).getText());
1007                     param.setValue(valueArray);
1008                     client.addParameter(param);
1009                     break;
1010                 case Int:
1011                     Integer valueInteger = Integer.decode(((JTextField) mwQueryArgumentTextFields.get(i)).getText());
1012                     param.setValue(valueInteger);
1013                     client.addParameter(param);
1014                     break;
1015                 case Time:
1016                     // parse given ISO8601 date string into a calendar
1017                     String dateStr = ((JTextField) mwQueryArgumentTextFields.get(i)).getText();
1018                     Calendar cal = TimeParser.parseAsCalendar(dateStr);
1019                     param.setValue(cal);
1020                     client.addParameter(param);
1021                     break;
1022                 default:
1023                     String value = ((JTextField) mwQueryArgumentTextFields.get(i)).getText();
1024                     param.setValue(value);
1025                     client.addParameter(param);
1026                     break;
1027                 }
1028             }
1029 
1030             if (isSubscribed.isSelected()) {
1031                 if (mwSubIdField.getText().equals("")) {
1032                     JFrame frame = new JFrame();
1033                     JOptionPane.showMessageDialog(frame, "Please specify a SubscriptionID", "Missing input",
1034                             JOptionPane.PLAIN_MESSAGE);
1035                     return;
1036                 }
1037                 Subscribe subcr = new Subscribe();
1038                 subcr.setDest(mwDestUriTextField.getText());
1039                 subcr.setQueryName("SimpleEventQuery");
1040                 subcr.setSubscriptionID(mwSubIdField.getText());
1041                 SubscriptionControls controls = new SubscriptionControls();
1042                 String dateStr = mwInitRecTimeField.getText();
1043                 DatatypeFactory factory = DatatypeFactory.newInstance();
1044                 controls.setInitialRecordTime(factory.newXMLGregorianCalendar(dateStr));
1045                 controls.setReportIfEmpty(reportIf.isSelected());
1046                 QuerySchedule sched = new QuerySchedule();
1047 
1048                 if (!triggerIf.isSelected()) {
1049                     sched.setSecond(mwScheduleSecField.getText());
1050                     sched.setMinute(mwScheduleMinuteField.getText());
1051                     sched.setHour(mwScheduleHourField.getText());
1052                     sched.setDayOfMonth(mwScheduleDayField.getText());
1053                     sched.setMonth(mwScheduleMonthField.getText());
1054                     sched.setDayOfWeek(mwScheduleWeekField.getText());
1055                     controls.setSchedule(sched);
1056                 } else {
1057                     controls.setTrigger(mwDestUriTextField.getText());
1058                 }
1059                 subcr.setControls(controls);
1060                 client.subscribeQuery(subcr);
1061                 JFrame frame = new JFrame();
1062                 JOptionPane.showMessageDialog(frame, "Query subscription successful.", "Service invocation successful",
1063                         JOptionPane.INFORMATION_MESSAGE);
1064             } else {
1065                 data = client.runQuery();
1066                 createResultsWindow();
1067             }
1068         } catch (ParseException e) {
1069             String msg = "Unable to parse a Time value.";
1070             dwOutputTextArea.append("\n" + msg + "\n");
1071             StringWriter sw = new StringWriter();
1072             PrintWriter pw = new PrintWriter(sw);
1073             e.printStackTrace(pw);
1074             dwOutputTextArea.append(sw.toString());
1075             JFrame frame = new JFrame();
1076             JOptionPane.showMessageDialog(frame, msg + "\n" + e.getMessage(), "Service invocation error",
1077                     JOptionPane.ERROR_MESSAGE);
1078         } catch (Exception e) {
1079             String msg = "Unexpected error while invoking EPCIS Query Interface.";
1080             dwOutputTextArea.append("\n" + msg + "\n");
1081             dwOutputTextArea.append(e.toString());
1082             StringWriter sw = new StringWriter();
1083             PrintWriter pw = new PrintWriter(sw);
1084             e.printStackTrace(pw);
1085             dwOutputTextArea.append(sw.toString());
1086             JFrame frame = new JFrame();
1087             JOptionPane.showMessageDialog(frame, msg + "\n" + e.toString(), "Service invocation error",
1088                     JOptionPane.ERROR_MESSAGE);
1089         }
1090     }

1094      * Will add or remove JComboBoxes as necessary and resize the window.
1095      * 
1096      * @param i
1097      *            The index of the combo box.
1098      */
1099     private void mwQuerySelectComboBoxesChanged(final int i) {
1100         JComboBox cb = (JComboBox) mwQuerySelectComboBoxes.get(i);
1101 
1102         if ((cb.getSelectedIndex() == 0) && (cb != mwQuerySelectComboBoxes.getLast())) {
1103             /*
1104              * the user selected "ignore" and this is not the last row, so
1105              * remove it
1106              */
1107             removeArgumentRow(i);
1108         } else if ((cb.getSelectedIndex() != 0) && (cb == mwQuerySelectComboBoxes.getLast())) {
1109             /* the user changed the value of the last row, so add a new row */
1110             addArgumentRow(i);
1111         } else {
1112             /* the user changed an existing row, so just update description */
1113             ((JTextFieldEnhanced) mwQueryArgumentTextFields.get(i)).setQueryItem(queryParamsUserText.get(queryParameterUsertext[cb.getSelectedIndex()]));
1114         }
1115     }

1154      * Checks if the provided queryUrl is still a valid URL and updates the
1155      * query client in case it has changed.
1156      */
1157     private void checkQueryUrl(String url) {
1158         if (url == null || "".equals(url)) {
1159             // query url has been deleted ... replace with previous url
1160             mwServiceUrlTextField.setText(queryUrl);
1161         } else if (!url.equals(queryUrl)) {
1162             // query url has changed
1163             System.out.println("Query service URL has changed, updating query client");
1164             queryUrl = client.updateEndpointAddress(mwServiceUrlTextField.getText());
1165             mwServiceUrlTextField.setText(queryUrl);
1166         }
