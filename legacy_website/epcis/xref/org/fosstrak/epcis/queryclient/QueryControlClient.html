129     /**
130      * Configures the service to communicate with the given endpoint address
131      * using the desired authentication method.
132      * 
133      * @param endpointAddress
134      *            The endpoint address this client will communicate to.
135      * @param authenticationOptions
136      *            The authentication options:
137      *            <p>
138      *            <table border="1">
139      *            <tr>
140      *            <td><code>authenticationOptions[0]</code></td>
141      *            <td><code>[1]</code></td>
142      *            <td><code>[2]</code></td>
143      *            </tr>
144      *            <tr>
145      *            <td><code>AuthenticationType.BASIC</code></td>
146      *            <td>username</td>
147      *            <td>password</td>
148      *            </tr>
149      *            <tr>
150      *            <td><code>AuthenticationType.HTTPS_WITH_CLIENT_CERT</code></td>
151      *            <td>keystore file</td>
152      *            <td>password</td>
153      *            </tr>
154      *            </table>
155      * @throws Exception
156      */
157     public void configureService(URL endpointAddress, Object[] authenticationOptions) throws Exception {
158         // logger.debug("Configuring service to communicate with endpoint: " +
159         // endpointAddress);
160         serviceConfigured = false;
161 
162         // setup the CXF bus
163         setUpBus();
164 
165         // instantiates a client proxy object from the EPCISServicePortType
166         // interface using the JAX-WS API
167         Service service = Service.create(SERVICE);
168         service.addPort(PORT, SOAPBinding.SOAP11HTTP_BINDING, endpointAddress.toString());
169         servicePort = service.getPort(PORT, EPCISServicePortType.class);
170 
171         // turn off chunked transfer encoding
172         Client client = ClientProxy.getClient(servicePort);
173         HTTPConduit httpConduit = (HTTPConduit) client.getConduit();
174         HTTPClientPolicy httpClientPolicy = new HTTPClientPolicy();
175         httpClientPolicy.setAllowChunking(false);
176         httpConduit.setClient(httpClientPolicy);
177 
178         // set up any authentication
179         if (authenticationOptions != null) {
180             if (AuthenticationType.BASIC.equals(authenticationOptions[0])) {
181                 // logger.debug("Authenticating via Basic as: " +
182                 // authenticationOptions[1]);
183 
184             	String username = (String)authenticationOptions[1];
185             	String password = (String)authenticationOptions[2];
186             	
187                 if (isEmpty(username) || isEmpty(password)) {
188                     throw new Exception("Authentication method " + authenticationOptions[0]
189                             + " requires a valid user name and password");
190                 }
191 
192                 AuthorizationPolicy ap = httpConduit.getAuthorization();
193                 ap.setUserName(username);
194                 ap.setPassword(password);
195             } else if (AuthenticationType.HTTPS_WITH_CLIENT_CERT.equals(authenticationOptions[0])) {
196                 // logger.debug("Authenticating with certificate in file: " +
197                 // authenticationOptions[1]);
198 
199                 if (!"HTTPS".equalsIgnoreCase(endpointAddress.getProtocol())) {
200                     throw new Exception("Authentication method " + authenticationOptions[0]
201                             + " requires the use of HTTPS");
202                 }
203                 
204     			String keyStoreFile = (String)authenticationOptions[1];
205     			String password = (String)authenticationOptions[2];
206 
207                 if (isEmpty(keyStoreFile) || isEmpty(password)) {
208                     throw new Exception("Authentication method " + authenticationOptions[0]
209                             + " requires a valid keystore (PKCS12 or JKS) and password");
210                 }
211 
212                 KeyStore keyStore = KeyStore.getInstance(keyStoreFile.endsWith(".p12") ? "PKCS12" : "JKS");
213                 keyStore.load(new FileInputStream(new File(keyStoreFile)), password.toCharArray());
214                 KeyManagerFactory keyManagerFactory = KeyManagerFactory.getInstance("SunX509");
215                 keyManagerFactory.init(keyStore, password.toCharArray());
216 
217                 TLSClientParameters tlscp = new TLSClientParameters();
218                 tlscp.setKeyManagers(keyManagerFactory.getKeyManagers());
219                 tlscp.setSecureRandom(new SecureRandom());
220                 tlscp.setDisableCNCheck(true);
221                 tlscp.setTrustManagers(new TrustManager[] { this });
222 
223                 httpConduit.setTlsClientParameters(tlscp);
224             }
225         }
226 
227         /*
228          * For instantiating a client proxy object using reflection (CXF simple
229          * frontend), use the ClientProxyFactoryBean and provide it with the
230          * service interface. The setter methods on the factory object can be
231          * used to configure the client proxy.
232          */
233         // ClientProxyFactoryBean factory = new ClientProxyFactoryBean();
234         // factory.setServiceClass(EPCISServicePortType.class);
235         // factory.setAddress(endpointAddress);
236         // factory.setBindingId(SOAPBinding.SOAP11HTTP_BINDING);
237         // servicePort = (EPCISServicePortType) factory.create();
238         /*
239          * For instantiating a client proxy object using the CXF-generated
240          * service implementation, uncomment the following lines. This will
241          * create the proxy from the WSDL file, using the endpointAddress from
242          * the <wsdlsoap:address> element inside the WSDL document.
243          */
244         // EPCglobalEPCISService s = new EPCglobalEPCISService(wsdlLocation,
245         // SERVICE);
246         // servicePort = s.getEPCglobalEPCISServicePort();
247         serviceConfigured = true;
248     }

250     private void setUpBus() {
251         Bus bus = CXFBusFactory.getDefaultBus();
252         ClientOnlyHTTPTransportFactory httpTransport = new ClientOnlyHTTPTransportFactory();
253         // httpTransport = new ServletTransportFactory();
254         httpTransport.setBus(bus);
255         List<String> transportIds = Arrays.asList(new String[] {
256                 "http://schemas.xmlsoap.org/wsdl/soap/http", "http://schemas.xmlsoap.org/soap/http",
257                 "http://www.w3.org/2003/05/soap/bindings/HTTP/", "http://schemas.xmlsoap.org/wsdl/http/",
258                 "http://cxf.apache.org/transports/http/configuration", "http://cxf.apache.org/bindings/xformat", });
259         httpTransport.setTransportIds(transportIds);
260         httpTransport.registerWithBindingManager();
261         // httpTransport.register();
262     }
