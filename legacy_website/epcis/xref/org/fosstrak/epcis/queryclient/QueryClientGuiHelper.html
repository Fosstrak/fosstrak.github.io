110     /**
111      * @return The query client properties.
112      */
113     private Properties loadProperties() {
114         Properties props = new Properties();
115         InputStream is = getClass().getResourceAsStream(PROPERTY_FILE);
116         if (is != null) {
117 	        try {
118 	            props.load(is);
119 	            is.close();
120 	        } catch (IOException e) {
121 	            System.out.println("Unable to load queryclient properties from "
122 	                    + QueryControlClient.class.getResource(PROPERTY_FILE).toString() + ". Using defaults.");
123 	        }
124         }
125         else {
126             System.out.println("Unable to load queryclient properties from "
127                     + PROPERTY_FILE + ". Using defaults.");
128         }
129         return props;
130     }

132     /**
133      * Converts the values in a calendar object into a nicely formatted string.
134      * 
135      * @param cal
136      *            with the Calendar-Date
137      * @return String
138      */
139     private String prettyStringCalendar(final Calendar cal) {
140         if (cal == null) {
141             return null;
142         }
143         // set to current timezone
144         cal.setTimeZone(TimeZone.getDefault());
145         return TimeParser.format(cal);
146     }

382     /**
383      * Run the query with the currently set query arguments Returns the results
384      * in a format that is suitable for JTable.
385      * 
386      * @return The pretty-printed query results.
387      * @throws Exception
388      *             If any Exception occurred while invoking the query service.
389      */
390     public Object[][] runQuery() throws Exception {
391         configureServiceIfNecessary();
392         QueryParams queryParams = new QueryParams();
393         queryParams.getParam().addAll(internalQueryParams);
394         debug("Number of query parameters: " + queryParams.getParam().size() + "\n");
395         for (QueryParam queryParam : internalQueryParams) {
396         	if (queryParam.getValue() instanceof ArrayOfString) {
397         		debug(queryParam.getName() + " " + ((ArrayOfString) queryParam.getValue()).getString() + "\n");
398         	} else {
399         		debug(queryParam.getName() + " " + queryParam.getValue() + "\n");
400         	}
401         }
402 
403         Poll poll = new Poll();
404         poll.setQueryName("SimpleEventQuery");
405         poll.setParams(queryParams);
406 
407         debug("running query...\n");
408         QueryResults results = queryClient.poll(poll);
409         debug("done\n");
410 
411         // print to debug window and return result
412         if (results != null && results.getResultsBody() != null && results.getResultsBody().getEventList() != null) {
413             return processEvents(results.getResultsBody().getEventList().getObjectEventOrAggregationEventOrQuantityEvent());
414         } else {
415             return null;
416         }
417     }

421      * 
422      * @param subscribe

424      * @throws Exception
425      *             If any Exception occurred while invoking the query service.

427     public void subscribeQuery(final Subscribe subscribe) throws Exception {
428         configureServiceIfNecessary();
429         QueryParams queryParams = new QueryParams();
430         queryParams.getParam().addAll(internalQueryParams);
431         debug("Number of query parameters: " + queryParams.getParam().size() + "\n");
432         for (QueryParam queryParam : internalQueryParams) {
433             debug(queryParam.getName() + " " + queryParam.getValue() + "\n");
434         }
435         subscribe.setParams(queryParams);
436         queryClient.subscribe(subscribe);
437     }

441      * 
442      * @param subscriptionID
443      *            The ID of the query to be unsubscribed.

445     public void unsubscribeQuery(final String subscriptionID) throws Exception {
446         configureServiceIfNecessary();
447         JFrame frame = new JFrame();
448         if (subscriptionID.equals("")) {
449             JOptionPane.showMessageDialog(frame, "Please specify a SubscriptionID", "Service is responding",

453         Unsubscribe parms = new Unsubscribe();

472     /**
473      * Query the service for subscriptions.
474      */
475     public void querySubscriptionIDs() throws Exception {
476         configureServiceIfNecessary();
477         String title = "Service is responding";
478         StringBuilder msg = new StringBuilder();
479         GetSubscriptionIDs parms = new GetSubscriptionIDs();
480         parms.setQueryName("SimpleEventQuery");
481         List<String> subscriptionIDs = queryClient.getSubscriptionIds(parms.getQueryName());
482         if (subscriptionIDs != null && !subscriptionIDs.isEmpty()) {
483             msg.append("The following subscription IDs were found in the repository:\n");
484             for (String s : subscriptionIDs) {
485                 msg.append("- ").append(s).append("\n");
486             }
487         } else {
488             msg.append("There are no subscribed queries.");
489         }
490         JFrame frame = new JFrame();
491         JOptionPane.showMessageDialog(frame, msg.toString(), title, JOptionPane.INFORMATION_MESSAGE);
492     }

532     /**
533      * Forces the client to reconfigure the service if the any of its parameters
534      * (authentication, endpoint address) have been changed.
535      * 
536      * @throws Exception
537      */
538     private void configureServiceIfNecessary() throws Exception {
539         if (!queryClient.isServiceConfigured() || configurationChanged) {
540             // validate the URL and get auth options from the main window
541             queryClient.configureService(new URL(mainWindow.getAddress()), mainWindow.getAuthenticationOptions());
542             configurationChanged = false;
543         }
