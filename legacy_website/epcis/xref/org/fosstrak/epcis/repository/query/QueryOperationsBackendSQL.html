160     private PreparedStatement prepareSimpleEventQuery(final QueryOperationsSession session, SimpleEventQueryDTO seQuery)
161             throws SQLException, ImplementationExceptionResponse {
162 
163         StringBuilder sqlSelectFrom;
164         StringBuilder sqlWhereClause = new StringBuilder(" WHERE 1");
165         List<Object> sqlParams = new ArrayList<Object>();
166 
167         String eventType = seQuery.getEventType();
168         if (EpcisConstants.AGGREGATION_EVENT.equals(eventType)) {
169             sqlSelectFrom = new StringBuilder(SQL_SELECT_FROM_AGGREGATIONEVENT);
170         } else if (EpcisConstants.OBJECT_EVENT.equals(eventType)) {
171             sqlSelectFrom = new StringBuilder(SQL_SELECT_FROM_OBJECTEVENT);
172         } else if (EpcisConstants.QUANTITY_EVENT.equals(eventType)) {
173             sqlSelectFrom = new StringBuilder(SQL_SELECT_FROM_QUANTITYEVENT);
174         } else if (EpcisConstants.TRANSACTION_EVENT.equals(eventType)) {
175             sqlSelectFrom = new StringBuilder(SQL_SELECT_FROM_TRANSACTIONEVENT);
176         } else {
177             String msg = "Unknown event type: " + eventType;
178             LOG.error(msg);
179             ImplementationException ie = new ImplementationException();
180             ie.setReason(msg);
181             throw new ImplementationExceptionResponse(msg, ie);
182         }
183 
184         boolean joinedEpcs = false;
185         boolean joinedBizTransacitions = false;
186 
187         // construct the SQL query dynamically
188         List<EventQueryParam> eventQueryParams = seQuery.getEventQueryParams();
189         int nofEventFieldExtensions = 0;
190         for (EventQueryParam queryParam : eventQueryParams) {
191             String eventField = queryParam.getEventField();
192             Operation op = queryParam.getOp();
193             Object value = queryParam.getValue();
194 
195             // check if we need to do any JOINs
196             if ("epcList".equals(eventField) || "childEPCs".equals(eventField) || "anyEPC".equals(eventField)) {
197                 // we have a query on EPCs, so we need to join the appropriate
198                 // "_EPCs" table
199                 if (!joinedEpcs) {
200                     sqlSelectFrom.append(" JOIN event_").append(eventType).append("_EPCs AS epc");
201                     sqlSelectFrom.append(" ON event_").append(eventType).append(".id=epc.event_id");
202                     joinedEpcs = true;
203                 }
204                 // update the event field to search in
205                 eventField = "epc.epc";
206             } else if (eventField.startsWith("extension")) {
207                 // we have a query on an extension field, so we need to join the
208                 // appropriate "_extensions" table
209 
210                 /*
211                  * For every extension condition there are two EventQueryParams,
212                  * one for the name of the parameter and another one for the
213                  * value. Example: extension.intValue extension.fieldname
214                  * Therefore, the JOINs will be created once from every two
215                  * extension conditions (the even ones)
216                  */
217                 nofEventFieldExtensions++;
218                 if (nofEventFieldExtensions % 2 == 0) {
219                     sqlSelectFrom.append(" JOIN event_").append(eventType).append("_extensions AS extension").append(
220                             nofEventFieldExtensions / 2);
221                     sqlSelectFrom.append(" ON event_").append(eventType).append(".id=extension").append(
222                             nofEventFieldExtensions / 2).append(".event_id");
223                 }
224             } else if (eventField.startsWith("bizTrans")) {
225                 // we have a query on business transactions, so we need to join
226                 // the appropriate "_bizTrans" and "bizTransList" tables
227                 if (!joinedBizTransacitions) {
228                     sqlSelectFrom.append(" JOIN event_").append(eventType).append("_bizTrans AS bizTransList");
229                     sqlSelectFrom.append(" ON event_").append(eventType).append(".id=bizTransList.event_id");
230                     sqlSelectFrom.append(" JOIN BizTransaction ON bizTransList.bizTrans_id=BizTransaction.id");
231                     sqlSelectFrom.append(" JOIN voc_BizTrans AS bizTrans ON BizTransaction.bizTrans=bizTrans.id");
232                     sqlSelectFrom.append(" JOIN voc_BizTransType AS bizTransType ON BizTransaction.type=bizTransType.id");
233                     joinedBizTransacitions = true;
234                 }
235                 if ("bizTransList.bizTrans".equals(eventField)) {
236                     eventField = "bizTrans";
237                 } else if ("bizTransList.type".equals(eventField)) {
238                     eventField = "bizTransType";
239                 }
240             } else if (eventField.endsWith(".attribute")) {
241                 String attrTable = attributeTablenameMap.get(eventField);
242                 if (attrTable != null) {
243                     String vocAlias = eventField.substring(0, eventField.indexOf("."));
244                     sqlSelectFrom.append(" JOIN ").append(attrTable);
245                     sqlSelectFrom.append(" ON ").append(attrTable).append(".id=").append(vocAlias).append(".id");
246                     eventField = attrTable + ".attribute";
247                 }
248             } else if (eventField.endsWith(".attribute.value")) {
249                 String attrTable = attributeTablenameMap.get(eventField.substring(0, eventField.length() - 6));
250                 eventField = attrTable + ".value";
251             }
252             String vocField = vocabularyTypeMap.get(eventField);
253             if (vocField != null) {
254                 eventField = vocField;
255             }
256 
257             // now check the provided event field, operation, and value and
258             // update the SQL strings accordingly
259             if (value == null && op == Operation.EXISTS) {
260                 if (eventField.startsWith("epc") || eventField.startsWith("bizTransList")) {
261                     // EXISTS-query already coped with by JOIN - nothing to do
262                 } else {
263                     // check if the given event field exists
264                     sqlWhereClause.append(" AND ?");
265                     sqlParams.add(eventField);
266                 }
267             } else if (value != null) {
268                 if (value instanceof List) {
269                     // we have a multi-value query parameter, e.g. action, EPCs,
270                     // vocabulary types
271                     List<?> paramValues = (List<?>) value;
272                     if (!paramValues.isEmpty()) {
273                         if (op == Operation.MATCH || op == Operation.WD) {
274                             // this results in a SQL "LIKE" query
275                             sqlWhereClause.append(" AND (0");
276                             for (Object paramValue : paramValues) {
277                                 String strValue = (String) paramValue;
278 
279                                 // MATCH-params might be 'pure identity' EPC
280                                 // patterns
281                                 if (op == Operation.MATCH && !eventField.startsWith("epcClass")) {
282                                     if (strValue.startsWith("urn:epc:idpat:")) {
283                                         strValue = strValue.replace("urn:epc:idpat:", "urn:epc:id:");
284                                     }
285                                 }
286                                 strValue = strValue.replaceAll("\\*", "%");
287 
288                                 sqlWhereClause.append(" OR ").append(eventField).append(" LIKE ?");
289                                 sqlParams.add(strValue);
290                                 if (seQuery.isAnyEpc() && "epc.epc".equals(eventField)) {
291                                     sqlWhereClause.append(" OR parentID LIKE ?");
292                                     sqlParams.add(strValue);
293                                 }
294                             }
295                             sqlWhereClause.append(")");
296                         } else {
297                             // this results in a SQL "IN" query
298                             sqlWhereClause.append(" AND ").append(eventField).append(" IN (?");
299                             sqlParams.add(paramValues.get(0));
300                             for (int i = 1; i < paramValues.size(); i++) {
301                                 sqlWhereClause.append(",?");
302                                 sqlParams.add(paramValues.get(i));
303                             }
304                             sqlWhereClause.append(")");
305                         }
306                     }
307                 } else {
308                     // we have a single-value parameter, e.g. eventTime,
309                     // recordTime, parentID
310                     String sqlOp = operationMap.get(op);
311                     sqlWhereClause.append(" AND ").append(eventField).append(" ").append(sqlOp).append(" ?");
312                     sqlParams.add(value);
313                 }
314             }
315         }
316 
317         // construct the final SQL query string
318         StringBuilder sql = sqlSelectFrom.append(sqlWhereClause);
319         if (seQuery.getOrderBy() != null) {
320             sql.append(" ORDER BY ").append(seQuery.getOrderBy());
321             if (seQuery.getOrderDirection() != null) {
322                 sql.append(" ").append(seQuery.getOrderDirection().name());
323             }
324         }
325         if (seQuery.getLimit() != -1) {
326             sql.append(" LIMIT ").append(seQuery.getLimit());
327         } else if (seQuery.getMaxEventCount() != -1) {
328             sql.append(" LIMIT ").append(seQuery.getMaxEventCount() + 1);
329         }
330         String sqlSelect = sql.toString();
331 
332         PreparedStatement selectEventsStmt = session.getConnection().prepareStatement(sqlSelect);
333         LOG.debug("SQL: " + sqlSelect);
334         for (int i = 0; i < sqlParams.size(); i++) {
335             selectEventsStmt.setObject(i + 1, sqlParams.get(i));
336             if (LOG.isDebugEnabled()) {
337                 LOG.debug("     param" + i + " = " + sqlParams.get(i));
338             }
339         }
340         return selectEventsStmt;
341     }

343     /**

345      */
346     public void runSimpleEventQuery(final QueryOperationsSession session, final SimpleEventQueryDTO seQuery,
347             final List<Object> eventList) throws SQLException, ImplementationExceptionResponse,
348             QueryTooLargeExceptionResponse {
349         PreparedStatement selectEventsStmt = prepareSimpleEventQuery(session, seQuery);
350         ResultSet rs = selectEventsStmt.executeQuery();
351 
352         String eventType = seQuery.getEventType();
353 
354         // prepare the required remaining SQL queries
355         String selectExtensions = null;
356         String selectEpcs = null;
357         String selectBizTrans = null;
358         if (EpcisConstants.AGGREGATION_EVENT.equals(eventType)) {
359             selectExtensions = SQL_SELECT_AGGREGATIONEVENT_EXTENSIONS;
360             selectEpcs = SQL_SELECT_AGGREGATIONEVENT_EPCS;
361             selectBizTrans = SQL_SELECT_AGGREGATIONEVENT_BIZTRANS;
362         } else if (EpcisConstants.OBJECT_EVENT.equals(eventType)) {
363             selectExtensions = SQL_SELECT_OBJECTEVENT_EXTENSIONS;
364             selectEpcs = SQL_SELECT_OBJECTEVENT_EPCS;
365             selectBizTrans = SQL_SELECT_OBJECTEVENT_BIZTRANS;
366         } else if (EpcisConstants.QUANTITY_EVENT.equals(eventType)) {
367             selectExtensions = SQL_SELECT_QUANTITYEVENT_EXTENSIONS;
368             selectEpcs = SQL_SELECT_QUANTITYEVENT_EPCS;
369             selectBizTrans = SQL_SELECT_QUANTITYEVENT_BIZTRANS;
370         } else if (EpcisConstants.TRANSACTION_EVENT.equals(eventType)) {
371             selectExtensions = SQL_SELECT_TRANSACTIONEVENT_EXTENSIONS;
372             selectEpcs = SQL_SELECT_TRANSACTIONEVENT_EPCS;
373             selectBizTrans = SQL_SELECT_TRANSACTIONEVENT_BIZTRANS;
374         }
375         PreparedStatement selectExtensionsStmt = session.getPreparedStatement(selectExtensions);
376         PreparedStatement selectEpcsStmt = session.getPreparedStatement(selectEpcs);
377         PreparedStatement selectBizTransStmt = session.getPreparedStatement(selectBizTrans);
378 
379         // cycle through result set and fill an event list
380         int actEventCount = 0;
381         while (rs.next()) {
382             actEventCount++;
383             int eventId = rs.getInt(1);
384             Timestamp eventTime = rs.getTimestamp(2);
385             Timestamp recordTime = rs.getTimestamp(3);
386             String eventTimeZoneOffset = rs.getString(4);
387             String readPointId = rs.getString(5);
388             ReadPointType readPoint = null;
389             if (readPointId != null) {
390                 readPoint = new ReadPointType();
391                 readPoint.setId(readPointId);
392             }
393             String bizLocationId = rs.getString(6);
394             BusinessLocationType bizLocation = null;
395             if (bizLocationId != null) {
396                 bizLocation = new BusinessLocationType();
397                 bizLocation.setId(bizLocationId);
398             }
399             String bizStep = rs.getString(7);
400             String disposition = rs.getString(8);
401             // fetch biz transactions
402             if (LOG.isDebugEnabled()) {
403                 LOG.debug("SQL: " + selectBizTrans);
404                 LOG.debug("     param1 = " + eventId);
405             }
406             selectBizTransStmt.setInt(1, eventId);
407             BusinessTransactionListType bizTransList = readBizTransactionsFromResult(selectBizTransStmt.executeQuery());
408 
409             EPCISEventType event = null;
410             if (EpcisConstants.AGGREGATION_EVENT.equals(eventType)) {
411                 AggregationEventType aggrEvent = new AggregationEventType();
412                 aggrEvent.setReadPoint(readPoint);
413                 aggrEvent.setBizLocation(bizLocation);
414                 aggrEvent.setBizStep(bizStep);
415                 aggrEvent.setDisposition(disposition);
416                 aggrEvent.setAction(ActionType.valueOf(rs.getString(9)));
417                 aggrEvent.setParentID(rs.getString(10));
418                 aggrEvent.setBizTransactionList(bizTransList);
419                 // fetch EPCs
420                 if (LOG.isDebugEnabled()) {
421                     LOG.debug("SQL: " + selectEpcs);
422                     LOG.debug("     param1 = " + eventId);
423                 }
424                 selectEpcsStmt.setInt(1, eventId);
425                 aggrEvent.setChildEPCs(readEpcsFromResult(selectEpcsStmt.executeQuery()));
426                 // fetch and fill extensions
427                 if (LOG.isDebugEnabled()) {
428                     LOG.debug("SQL: " + selectExtensions);
429                     LOG.debug("     param1 = " + eventId);
430                 }
431                 selectExtensionsStmt.setInt(1, eventId);
432                 readExtensionsFromResult(selectExtensionsStmt.executeQuery(), aggrEvent.getAny());
433                 event = aggrEvent;

524     private PreparedStatement prepareMasterDataQuery(final QueryOperationsSession session, String vocType,
525             MasterDataQueryDTO mdQuery) throws SQLException {
526 
527         StringBuilder sqlSelectFrom = new StringBuilder("SELECT uri FROM");
528         StringBuilder sqlWhereClause = new StringBuilder(" WHERE 1");
529         List<Object> sqlParams = new ArrayList<Object>();
530 
531         // get the values from the query DTO
532         List<String> attributeNames = mdQuery.getAttributeNames();
533         Map<String, List<String>> attributeNameAndValues = mdQuery.getAttributeNameAndValues();
534         List<String> vocabularyEqNames = mdQuery.getVocabularyEqNames();
535         List<String> vocabularyWdNames = mdQuery.getVocabularyWdNames();
536 
537         boolean joinedAttribute = false;
538         String vocTablename = getVocabularyTablename(vocType);
539         sqlSelectFrom.append(" ").append(vocTablename).append(",");
540         if ("voc_Any".equals(vocTablename)) {
541             // this is not a standard vocabulary, we need to restrict by vtype
542             // in the voc_Any table
543             sqlWhereClause.append(" AND voc_Any.vtype=?");
544             sqlParams.add(vocType);
545         }
546 
547         // filter by attribute names
548         if (attributeNames != null && !attributeNames.isEmpty()) {
549             if (!joinedAttribute) {
550                 sqlSelectFrom.append(" ").append(vocTablename).append("_attr,");
551                 sqlWhereClause.append(" AND ").append(vocTablename).append(".id=");
552                 sqlWhereClause.append(vocTablename).append("_attr.id");
553             }
554 
555             sqlWhereClause.append(" AND ").append(vocTablename).append("_attr.attribute IN (?");
556             sqlParams.add(attributeNames.get(0));
557             for (int i = 1; i < attributeNames.size(); i++) {
558                 sqlWhereClause.append(",?");
559                 sqlParams.add(attributeNames.get(i));
560             }
561             sqlWhereClause.append(")");
562         }
563 
564         // filter by attribute names and values
565         if (attributeNameAndValues != null && !attributeNameAndValues.isEmpty()) {
566             if (!joinedAttribute) {
567                 sqlSelectFrom.append(" ").append(vocTablename).append("_attr,");
568                 sqlWhereClause.append(" AND ").append(vocTablename).append(".id=");
569                 sqlWhereClause.append(vocTablename).append("_attr.id");
570             }
571             for (String attrName : attributeNameAndValues.keySet()) {
572                 sqlWhereClause.append(" AND ").append(vocTablename).append("_attr.attribute=?");
573                 sqlParams.add(attrName);
574                 sqlWhereClause.append(" AND ").append(vocTablename).append("_attr.value IN (?");
575                 List<String> attrValues = attributeNameAndValues.get(attrName);
576                 sqlParams.add(attrValues.get(0));
577                 for (int i = 1; i < attrValues.size(); i++) {
578                     sqlWhereClause.append(",?");
579                     sqlParams.add(attrValues.get(i));
580                 }
581                 sqlWhereClause.append(")");
582             }
583         }
584 
585         // filter by vocabulary names
586         if (vocabularyEqNames != null && !vocabularyEqNames.isEmpty()) {
587             sqlWhereClause.append(" AND ").append(vocTablename).append(".uri IN (?");
588             sqlParams.add(vocabularyEqNames.get(0));
589             for (int i = 1; i < vocabularyEqNames.size(); i++) {
590                 sqlWhereClause.append(",?");
591                 sqlParams.add(vocabularyEqNames.get(i));
592             }
593             sqlWhereClause.append(")");
594         }
595         if (vocabularyWdNames != null && !vocabularyWdNames.isEmpty()) {
596             sqlWhereClause.append(" AND (0");
597             for (String vocWdName : vocabularyWdNames) {
598                 sqlWhereClause.append(" OR ").append(vocTablename).append(".uri LIKE ?");
599                 sqlParams.add(vocWdName + "%");
600             }
601             sqlWhereClause.append(")");
602         }
603 
604         // remove last comma
605         sqlSelectFrom.delete(sqlSelectFrom.length() - 1, sqlSelectFrom.length());
606 
607         // set the complete query and pass it back to the caller
608         String sqlSelect = sqlSelectFrom.append(sqlWhereClause).toString();
609 
610         PreparedStatement ps = session.getConnection().prepareStatement(sqlSelect);
611         LOG.debug("SQL: " + sqlSelect);
612         for (int i = 0; i < sqlParams.size(); i++) {
613             ps.setObject(i + 1, sqlParams.get(i));
614             if (LOG.isDebugEnabled()) {
615                 LOG.debug("     param" + i + " = " + sqlParams.get(i));
616             }
617         }
618         return ps;
619     }

622      * {@inheritDoc}
623      */
624     public void runMasterDataQuery(final QueryOperationsSession session, final MasterDataQueryDTO mdQuery,
625             final List<VocabularyType> vocList) throws SQLException, ImplementationExceptionResponse,
626             QueryTooLargeExceptionResponse {
627         // create and run a separate query for each vocabulary
628         List<String> vocabularyTypes = mdQuery.getVocabularyTypes();
629         for (String vocType : vocabularyTypes) {
630             PreparedStatement ps = prepareMasterDataQuery(session, vocType, mdQuery);
631             ResultSet rs = ps.executeQuery();
632 
633             int maxElementCount = mdQuery.getMaxElementCount();
634             boolean includeAttributes = mdQuery.getIncludeAttributes();
635             boolean includeChildren = mdQuery.getIncludeChildren();
636 
637             // fetch matching vocabulary element uris
638             List<String> vocElemUris = new ArrayList<String>();
639             int actVocElemCount = 0;
640             while (rs.next()) {
641                 actVocElemCount++;
642                 if (maxElementCount > -1 && actVocElemCount > maxElementCount) {
643                     // according to spec, this must result in a
644                     // QueryTooLargeException
645                     String msg = "The query returned more results than specified by 'maxElementCount'";
646                     LOG.info("USER ERROR: " + msg);
647                     QueryTooLargeException e = new QueryTooLargeException();
648                     e.setReason(msg);
649                     throw new QueryTooLargeExceptionResponse(msg, e);
650                 }
651                 vocElemUris.add(rs.getString(1));
652             }
653             rs.close();
654             if (LOG.isDebugEnabled()) {
655                 LOG.debug("Masterdata query returned " + actVocElemCount + " vocabularies (maxElementCount is "
656                         + maxElementCount + ")");
657             }
658 
659             // populate the VocabularyElementList
660             VocabularyElementListType vocElems = new VocabularyElementListType();
661             for (String vocElemUri : vocElemUris) {
662                 VocabularyElementType vocElem = new VocabularyElementType();
663                 vocElem.setId(vocElemUri);
664                 if (includeAttributes) {
665                     fetchAttributes(session, vocType, vocElemUri, mdQuery.getIncludedAttributeNames(),
666                             vocElem.getAttribute());
667                 }
668                 if (includeChildren) {
669                     IDListType children = fetchChildren(session, vocType, vocElemUri);
670                     vocElem.setChildren(children);
671                 }
672                 vocElems.getVocabularyElement().add(vocElem);
673             }
674 
675             // add the vocabulary element to the vocabulary list
676             if (!vocElems.getVocabularyElement().isEmpty()) {
677                 VocabularyType voc = new VocabularyType();
678                 voc.setType(vocType);
679                 voc.setVocabularyElementList(vocElems);
680                 vocList.add(voc);
681             }
682         }
683     }

826      * Fetches the qualified XML elements representing extensions for event
827      * fields from the given result set and populates the given List.
828      * 
829      * @param rs
830      *            The result of the SQL query.
831      * @throws SQLException
832      *             If a database access error occurred.
833      */
834     private void readExtensionsFromResult(final ResultSet rs, final List<Object> extensions) throws SQLException {
835         while (rs.next()) {
836             String fieldname = rs.getString(1);
837             String[] parts = fieldname.split("#");
838             if (parts.length != 2) {
839                 throw new SQLException(
840                         "Fieldname extension has invalid format: required 'namespace#localname' but was " + fieldname);
841             }
842             String namespace = parts[0];
843             String localPart = parts[1];
844             String prefix = rs.getString(2);
845             String value = rs.getString(3);
846             if (value == null) {
847                 value = rs.getString(4);
848                 if (value == null) {
849                     value = rs.getString(5);
850                     if (value == null) {
851                         value = rs.getString(6);
852                         if (value == null) {
853                             throw new SQLException("No valid extension value found");
854                         }
855                     }
856                 }
857             }
858             JAXBElement<String> elem = new JAXBElement<String>(new QName(namespace, localPart, prefix), String.class,
859                     value);
860             extensions.add(elem);
861         }
862     }
