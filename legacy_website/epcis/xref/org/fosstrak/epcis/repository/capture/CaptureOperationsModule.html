149     /**
150      * Initializes the EPCIS schema used for validating incoming capture
151      * requests. Loads the WSDL and XSD files from the classpath (the schema is
152      * bundled with epcis-commons.jar).
153      * 
154      * @return An instantiated schema validation object.
155      */
156     private Schema initEpcisSchema(String xsdFile) {
157         InputStream is = this.getClass().getResourceAsStream(xsdFile);
158         if (is != null) {
159             try {
160                 SchemaFactory schemaFactory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
161                 Source schemaSrc = new StreamSource(is);
162                 schemaSrc.setSystemId(CaptureOperationsServlet.class.getResource(xsdFile).toString());
163                 Schema schema = schemaFactory.newSchema(schemaSrc);
164                 LOG.debug("EPCIS schema file initialized and loaded successfully");
165                 return schema;
166             } catch (Exception e) {
167                 LOG.warn("Unable to load or parse the EPCIS schema", e);
168             }
169         } else {
170             LOG.error("Unable to load the EPCIS schema file from classpath: cannot find resource " + xsdFile);
171         }
172         LOG.warn("Schema validation will not be available!");
173         return null;
174     }

176     /**
177      * Resets the database.
178      * 
179      * @throws SQLException
180      *             If something goes wrong resetting the database.
181      * @throws IOException
182      *             If something goes wrong reading the reset script.
183      * @throws UnsupportedOperationsException
184      *             If database resets are not allowed.
185      */
186     public void doDbReset() throws SQLException, IOException {
187         if (dbResetAllowed) {
188             Session session = null;
189             try {
190                 session = sessionFactory.openSession();
191                 Transaction tx = null;
192                 try {
193                     tx = session.beginTransaction();
194                     Connection connection = session.connection();
195                     LOG.info("Running db reset script from file " + dbResetScript);
196                     Statement stmt = connection.createStatement();
197                     BufferedReader reader = new BufferedReader(new FileReader(dbResetScript));
198                     String line;
199                     while ((line = reader.readLine()) != null) {
200                         if (!line.startsWith("--")) {
201                             LOG.debug("SQL: " + line);
202                             stmt.addBatch(line);
203                         }
204                     }
205                     stmt.executeBatch();
206                     tx.commit();
207                 } catch (Exception e) {
208                     LOG.error("dbReset failed: " + e.toString(), e);
209                     if (tx != null) {
210                         tx.rollback();
211                     }
212                     throw new SQLException(e.toString());
213                 }
214             } finally {
215                 if (session != null) {
216                     session.close();
217                 }
218             }
219         } else {
220             throw new UnsupportedOperationException();
221         }
222     }

224     /**
225      * Implements the EPCIS capture operation. Takes an input stream, extracts
226      * the payload into an XML document, validates the document against the
227      * EPCIS schema, and captures the EPCIS events given in the document.
228      * 
229      * @throws IOException
230      *             If an error occurred while validating the request or writing
231      *             the response.
232      * @throws ParserConfigurationException
233      * @throws SAXException
234      *             If the XML document is malformed or invalid
235      * @throws InvalidFormatException
236      */
237     public void doCapture(InputStream in, Principal principal) throws SAXException, IOException, InvalidFormatException {
238 
239         // parse the payload as XML document
240         Document document;
241         try {
242             DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
243             factory.setNamespaceAware(true);
244             DocumentBuilder builder = factory.newDocumentBuilder();
245             document = builder.parse(in);
246             LOG.debug("Payload successfully parsed as XML document");
247             if (LOG.isDebugEnabled()) {
248                 try {
249                     TransformerFactory tfFactory = TransformerFactory.newInstance();
250                     Transformer transformer = tfFactory.newTransformer();
251                     StringWriter writer = new StringWriter();
252                     transformer.transform(new DOMSource(document), new StreamResult(writer));
253                     String xml = writer.toString();
254                     if (xml.length() > 100 * 1024) {
255                         // too large, do not log
256                         xml = null;
257                     } else {
258                         LOG.debug("Incoming contents:\n\n" + writer.toString() + "\n");
259                     }
260                 } catch (Exception e) {
261                     // never mind ... do not log
262                 }
263             }
264 
265             // validate the XML document against the EPCISDocument schema
266             if (schema != null) {
267                 Validator validator = schema.newValidator();
268                 try {
269                     validator.validate(new DOMSource(document), null);
270                 } catch (SAXParseException e) {
271                     // TODO: we need to ignore XML element order, the following
272                     // is only a hack to pass some of the conformance tests
273                     if (e.getMessage().contains("parentID")) {
274                         LOG.warn("Ignoring XML validation exception: " + e.getMessage());
275                     } else {
276                         throw e;
277                     }
278                 }
279                 LOG.info("Incoming capture request was successfully validated against the EPCISDocument schema");
280             } else {
281                 LOG.warn("Schema validator unavailable. Unable to validate EPCIS capture event against schema!");
282             }
283 
284         } catch (ParserConfigurationException e) {
285             throw new SAXException(e);
286         }
287 
288         // handle the capture operation
289         Session session = null;
290         try {
291             session = sessionFactory.openSession();
292             Transaction tx = null;
293             try {
294                 tx = session.beginTransaction();
295                 LOG.debug("DB connection opened.");
296                 handleDocument(session, document);
297                 tx.commit();
298                 // return OK
299                 LOG.info("EPCIS Capture Interface request succeeded");
300             } catch (SAXException e) {
301                 LOG.error("EPCIS Capture Interface request failed: " + e.toString());
302                 if (tx != null) {
303                     tx.rollback();
304                 }
305                 throw e;
306             } catch (InvalidFormatException e) {
307                 LOG.error("EPCIS Capture Interface request failed: " + e.toString());
308                 if (tx != null) {
309                     tx.rollback();
310                 }
311                 throw e;
312             } catch (Exception e) {
313                 // Hibernate throws RuntimeExceptions, so don't let them
314                 // (or anything else) escape without clean up
315                 LOG.error("EPCIS Capture Interface request failed: " + e.toString(), e);
316                 if (tx != null) {
317                     tx.rollback();
318                 }
319                 throw new IOException(e.toString());
320             }
321         } finally {
322             if (session != null) {
323                 session.close();
324             }
325             // sessionFactory.getStatistics().logSummary();
326             LOG.debug("DB connection closed");
327         }
328     }

330     /**
331      * Parses the entire document and handles the supplied events.
332      * 
333      * @throws Exception
334      * @throws DOMException
335      */
336     private void handleDocument(Session session, Document document) throws DOMException, SAXException,
337             InvalidFormatException {
338         NodeList eventList = document.getElementsByTagName("EventList");
339         NodeList events = eventList.item(0).getChildNodes();
340 
341         // walk through all supplied events
342         int eventCount = 0;
343         for (int i = 0; i < events.getLength(); i++) {
344             Node eventNode = events.item(i);
345             String nodeName = eventNode.getNodeName();
346 
347             if (nodeName.equals(EpcisConstants.OBJECT_EVENT) || nodeName.equals(EpcisConstants.AGGREGATION_EVENT)
348                     || nodeName.equals(EpcisConstants.QUANTITY_EVENT)
349                     || nodeName.equals(EpcisConstants.TRANSACTION_EVENT)) {
350                 LOG.debug("processing event " + i + ": '" + nodeName + "'.");
351                 handleEvent(session, eventNode, nodeName);
352                 eventCount++;
353                 if (eventCount % 50 == 0) {
354                     session.flush();
355                     session.clear();
356                 }
357             } else if (!nodeName.equals("#text") && !nodeName.equals("#comment")) {
358                 throw new SAXException("Encountered unknown event '" + nodeName + "'.");
359             }
360         }
361     }

363     /**
364      * Takes an XML document node, parses it as EPCIS event and inserts the data
365      * into the database. The parse routine is generic for all event types; the
366      * query generation part has some if/elses to take care of different event
367      * parameters.
368      * 
369      * @param eventNode
370      *            The current event node.
371      * @param eventType
372      *            The current event type.
373      * @throws Exception
374      * @throws DOMException
375      */
376     private void handleEvent(Session session, final Node eventNode, final String eventType) throws DOMException,
377             SAXException, InvalidFormatException {
378         if (eventNode == null) {
379             // nothing to do
380             return;
381         } else if (eventNode.getChildNodes().getLength() == 0) {
382             throw new SAXException("Event element '" + eventNode.getNodeName() + "' has no children elements.");
383         }
384         Node curEventNode = null;
385 
386         // A lot of the initialized variables have type URI. This type isn't to
387         // compare with the URI-Type of the standard. In fact, most of the
388         // variables having type URI are declared as Vocabularies in the
389         // Standard. Commonly, we use String for the Standard-Type URI.
390 
391         Timestamp eventTime = null;
392         Timestamp recordTime = new Timestamp(System.currentTimeMillis());
393         String eventTimeZoneOffset = null;
394         String action = null;
395         String parentId = null;
396         Long quantity = null;
397         String bizStepUri = null;
398         String dispositionUri = null;
399         String readPointUri = null;
400         String bizLocationUri = null;
401         String epcClassUri = null;
402 
403         List<String> epcs = null;
404         List<BusinessTransaction> bizTransList = null;
405         List<EventFieldExtension> fieldNameExtList = new ArrayList<EventFieldExtension>();
406 
407         for (int i = 0; i < eventNode.getChildNodes().getLength(); i++) {
408             curEventNode = eventNode.getChildNodes().item(i);
409             String nodeName = curEventNode.getNodeName();
410 
411             if (nodeName.equals("#text") || nodeName.equals("#comment")) {
412                 // ignore text or comments
413                 LOG.debug("  ignoring text or comment: '" + curEventNode.getTextContent().trim() + "'");
414                 continue;
415             }
416 
417             LOG.debug("  handling event field: '" + nodeName + "'");
418             if (nodeName.equals("eventTime")) {
419                 String xmlTime = curEventNode.getTextContent();
420                 LOG.debug("    eventTime in xml is '" + xmlTime + "'");
421                 try {
422                     eventTime = TimeParser.parseAsTimestamp(xmlTime);
423                 } catch (ParseException e) {
424                     throw new SAXException("Invalid date/time (must be ISO8601).", e);
425                 }
426                 LOG.debug("    eventTime parsed as '" + eventTime + "'");
427             } else if (nodeName.equals("recordTime")) {
428                 // ignore recordTime
429             } else if (nodeName.equals("eventTimeZoneOffset")) {
430                 eventTimeZoneOffset = checkEventTimeZoneOffset(curEventNode.getTextContent());
431             } else if (nodeName.equals("epcList") || nodeName.equals("childEPCs")) {
432                 epcs = handleEpcs(eventType, curEventNode);
433             } else if (nodeName.equals("bizTransactionList")) {
434                 bizTransList = handleBizTransactions(session, curEventNode);
435             } else if (nodeName.equals("action")) {
436                 action = curEventNode.getTextContent();
437                 if (!action.equals("ADD") && !action.equals("OBSERVE") && !action.equals("DELETE")) {
438                     throw new SAXException("Encountered illegal 'action' value: " + action);
439                 }
440             } else if (nodeName.equals("bizStep")) {
441                 bizStepUri = curEventNode.getTextContent();
442             } else if (nodeName.equals("disposition")) {
443                 dispositionUri = curEventNode.getTextContent();
444             } else if (nodeName.equals("readPoint")) {
445                 Element attrElem = (Element) curEventNode;
446                 Node id = attrElem.getElementsByTagName("id").item(0);
447                 readPointUri = id.getTextContent();
448             } else if (nodeName.equals("bizLocation")) {
449                 Element attrElem = (Element) curEventNode;
450                 Node id = attrElem.getElementsByTagName("id").item(0);
451                 bizLocationUri = id.getTextContent();
452             } else if (nodeName.equals("epcClass")) {
453                 epcClassUri = curEventNode.getTextContent();
454             } else if (nodeName.equals("quantity")) {
455                 quantity = Long.valueOf(curEventNode.getTextContent());
456             } else if (nodeName.equals("parentID")) {
457                 checkEpcOrUri(curEventNode.getTextContent(), false);
458                 parentId = curEventNode.getTextContent();
459             } else {
460                 String[] parts = nodeName.split(":");
461                 if (parts.length == 2) {
462                     LOG.debug("    treating unknown event field as extension.");
463                     String prefix = parts[0];
464                     String localname = parts[1];
465                     // String namespace =
466                     // document.getDocumentElement().getAttribute("xmlns:" +
467                     // prefix);
468                     String namespace = curEventNode.lookupNamespaceURI(prefix);
469                     String value = curEventNode.getTextContent();
470                     EventFieldExtension evf = new EventFieldExtension(prefix, namespace, localname, value);
471                     fieldNameExtList.add(evf);
472                 } else {
473                     // this is not a valid extension
474                     throw new SAXException("    encountered unknown event field: '" + nodeName + "'.");
475                 }
476             }
477         }
478         if (eventType.equals(EpcisConstants.AGGREGATION_EVENT)) {
479             // for AggregationEvents, the parentID is only optional for
480             // action=OBSERVE
481             if (parentId == null && ("ADD".equals(action) || "DELETE".equals(action))) {
482                 throw new InvalidFormatException("'parentID' is required if 'action' is ADD or DELETE");
483             }
484         }
485 
486         String nodeName = eventNode.getNodeName();
487         VocabularyElement bizStep = bizStepUri != null ? getOrInsertVocabularyElement(session,
488                 EpcisConstants.BUSINESS_STEP_ID, String.valueOf(bizStepUri)) : null;
489         VocabularyElement disposition = dispositionUri != null ? getOrInsertVocabularyElement(session,
490                 EpcisConstants.DISPOSITION_ID, String.valueOf(dispositionUri)) : null;
491         VocabularyElement readPoint = readPointUri != null ? getOrInsertVocabularyElement(session,
492                 EpcisConstants.READ_POINT_ID, String.valueOf(readPointUri)) : null;
493         VocabularyElement bizLocation = bizLocationUri != null ? getOrInsertVocabularyElement(session,
494                 EpcisConstants.BUSINESS_LOCATION_ID, String.valueOf(bizLocationUri)) : null;
495         VocabularyElement epcClass = epcClassUri != null ? getOrInsertVocabularyElement(session,
496                 EpcisConstants.EPC_CLASS_ID, String.valueOf(epcClassUri)) : null;
497 
498         BaseEvent be;
499         if (nodeName.equals(EpcisConstants.AGGREGATION_EVENT)) {
500             AggregationEvent ae = new AggregationEvent();
501             ae.setParentId(parentId);
502             ae.setChildEpcs(epcs);
503             ae.setAction(Action.valueOf(action));
504             be = ae;
505         } else if (nodeName.equals(EpcisConstants.OBJECT_EVENT)) {
506             ObjectEvent oe = new ObjectEvent();
507             oe.setAction(Action.valueOf(action));
508             if (epcs != null && epcs.size() > 0) {
509                 oe.setEpcList(epcs);
510             }
511             be = oe;
512         } else if (nodeName.equals(EpcisConstants.QUANTITY_EVENT)) {
513             QuantityEvent qe = new QuantityEvent();
514             qe.setEpcClass((EPCClass) epcClass);
515             if (quantity != null) {
516                 qe.setQuantity(quantity.longValue());
517             }
518             be = qe;
519         } else if (nodeName.equals(EpcisConstants.TRANSACTION_EVENT)) {
520             TransactionEvent te = new TransactionEvent();
521             te.setParentId(parentId);
522             te.setEpcList(epcs);
523             te.setAction(Action.valueOf(action));
524             be = te;
525         } else {
526             throw new SAXException("Encountered unknown event element '" + nodeName + "'.");
527         }
528 
529         be.setEventTime(eventTime);
530         be.setRecordTime(recordTime);
531         be.setEventTimeZoneOffset(eventTimeZoneOffset);
532         be.setBizStep((BusinessStepId) bizStep);
533         be.setDisposition((DispositionId) disposition);
534         be.setBizLocation((BusinessLocationId) bizLocation);
535         be.setReadPoint((ReadPointId) readPoint);
536         if (bizTransList != null && bizTransList.size() > 0) {
537             be.setBizTransList(bizTransList);
538         }
539         if (!fieldNameExtList.isEmpty()) {
540             be.setExtensions(fieldNameExtList);
541         }
542 
543         session.save(be);
544     }

546     /**
547      * Parses the xml tree for epc nodes and returns a list of EPC URIs.
548      * 
549      * @param eventType
550      * @param epcNode
551      *            The parent Node from which EPC URIs should be extracted.
552      * @return An array of vocabularies containing all the URIs found in the
553      *         given node.
554      * @throws SAXException
555      *             If an unknown tag (no &lt;epc&gt;) is encountered.
556      * @throws InvalidFormatException
557      * @throws DOMException
558      */
559     private List<String> handleEpcs(final String eventType, final Node epcNode) throws SAXException, DOMException,
560             InvalidFormatException {
561         List<String> epcList = new ArrayList<String>();
562 
563         boolean isEpc = false;
564         boolean epcRequired = false;
565         boolean atLeastOneNonEpc = false;
566         for (int i = 0; i < epcNode.getChildNodes().getLength(); i++) {
567             Node curNode = epcNode.getChildNodes().item(i);
568             if (curNode.getNodeName().equals("epc")) {
569                 isEpc = checkEpcOrUri(curNode.getTextContent(), epcRequired);
570                 if (isEpc) {
571                     // if one of the values is an EPC, then all of them must be
572                     // valid EPCs
573                     epcRequired = true;
574                 } else {
575                     atLeastOneNonEpc = true;
576                 }
577                 epcList.add(curNode.getTextContent());
578             } else {
579                 if (curNode.getNodeName() != "#text" && curNode.getNodeName() != "#comment") {
580                     throw new SAXException("Unknown XML tag: " + curNode.getNodeName(), null);
581                 }
582             }
583         }
584         if (atLeastOneNonEpc && isEpc) {
585             throw new InvalidFormatException(
586                     "One of the provided EPCs was a 'pure identity' EPC, so all of them must be 'pure identity' EPCs");
587         }
588         return epcList;
589     }

591     /**
592      * @param epcOrUri
593      *            The EPC or URI to check.
594      * @param epcRequired
595      *            <code>true</code> if an EPC is required (will throw an
596      *            InvalidFormatException if the given <code>epcOrUri</code> is
597      *            an invalid EPC, but might be a valid URI), <code>false</code>
598      *            otherwise.
599      * @return <code>true</code> if the given <code>epcOrUri</code> is a
600      *         valid EPC, <code>false</code> otherwise.
601      * @throws InvalidFormatException
602      */
603     protected boolean checkEpcOrUri(String epcOrUri, boolean epcRequired) throws InvalidFormatException {
604         boolean isEpc = false;
605         if (epcOrUri.startsWith("urn:epc:id:")) {
606             // check if it is a valid EPC
607             checkEpc(epcOrUri);
608             isEpc = true;
609         } else {
610             // childEPCs in AggregationEvents, and epcList in
611             // TransactionEvents might also be simple URIs
612             if (epcRequired) {
613                 throw new InvalidFormatException(
614                         "One of the provided EPCs was a 'pure identity' EPC, so all of them must be 'pure identity' EPCs");
615             }
616             checkUri(epcOrUri);
617         }
618         return isEpc;
619     }

633         List<BusinessTransaction> bizTransactionList = new ArrayList<BusinessTransaction>();

635         for (int i = 0; i < bizNode.getChildNodes().getLength(); i++) {
636             Node curNode = bizNode.getChildNodes().item(i);
637             if (curNode.getNodeName().equals("bizTransaction")) {
638                 String bizTransTypeUri = curNode.getAttributes().item(0).getTextContent();
639                 String bizTransUri = curNode.getTextContent();
640                 BusinessTransactionId bizTrans = (BusinessTransactionId) getOrInsertVocabularyElement(session,
641                         EpcisConstants.BUSINESS_TRANSACTION_ID, bizTransUri.toString());
642                 BusinessTransactionTypeId type = (BusinessTransactionTypeId) getOrInsertVocabularyElement(session,
643                         EpcisConstants.BUSINESS_TRANSACTION_TYPE_ID, bizTransTypeUri.toString());
644 
645                 Criteria c0 = session.createCriteria(BusinessTransaction.class);
646                 c0.add(Restrictions.eq("bizTransaction", bizTrans));
647                 c0.add(Restrictions.eq("type", type));
648                 BusinessTransaction bizTransaction = (BusinessTransaction) c0.uniqueResult();
649 
650                 if (bizTransaction == null) {
651                     bizTransaction = new BusinessTransaction();
652                     bizTransaction.setBizTransaction(bizTrans);
653                     bizTransaction.setType(type);
654                     session.save(bizTransaction);
655                 }
656 
657                 bizTransactionList.add(bizTransaction);
658 
659             } else {
660                 if (!curNode.getNodeName().equals("#text") && !curNode.getNodeName().equals("#comment")) {

668     /**
669      * Inserts vocabulary into the database by searching for already existing
670      * entries; if found, the corresponding ID is returned. If not found, the
671      * vocabulary is extended if "insertmissingvoc" is true; otherwise an
672      * SQLException is thrown
673      * 
674      * @param tableName
675      *            The name of the vocabulary table.
676      * @param uri
677      *            The vocabulary adapting the URI to be inserted into the
678      *            vocabulary table.
679      * @return The ID of an already existing vocabulary table with the given
680      *         uri.
681      * @throws UnsupportedOperationException
682      *             If we are not allowed to insert a missing vocabulary.
683      */
684     public VocabularyElement getOrInsertVocabularyElement(Session session, String vocabularyType,
685             String vocabularyElement) throws SAXException {
686         Class<?> c = vocClassMap.get(vocabularyType);
687         Criteria c0 = session.createCriteria(c);
688         c0.setCacheable(true);
689         c0.add(Restrictions.eq("uri", vocabularyElement));
690         VocabularyElement ve;
691         try {
692             ve = (VocabularyElement) c0.uniqueResult();
693         } catch (ObjectNotFoundException e) {
694             ve = null;
695         }
696         if (ve == null) {
697             // the uri does not yet exist: insert it if allowed. According to
698             // the specs, some vocabulary is not allowed to be extended; this is
699             // currently ignored here
700             if (!insertMissingVoc) {
701                 throw new UnsupportedOperationException("Not allowed to add new vocabulary - use existing vocabulary");
702             } else {
703                 // VocabularyElement subclasses should always have public
704                 // zero-arg constructor to avoid problems here
705                 try {
706                     ve = (VocabularyElement) c.newInstance();
707                 } catch (InstantiationException e) {
708                     throw new RuntimeException(e);
709                 } catch (IllegalAccessException e) {
710                     throw new RuntimeException(e);
711                 }
712 
713                 ve.setUri(vocabularyElement);
714                 session.save(ve);
715                 session.flush();
716             }
717         }
718         return ve;

731         Matcher m = p.matcher(textContent);

733         if (matches) {

744         }
745         if (matches) {

748             throw new InvalidFormatException("'eventTimeZoneOffset' has invalid format: " + textContent);

768     /**
769      * Check EPC according to 'pure identity' URI as specified in Tag Data
770      * Standard.
771      * 
772      * @param textContent
773      * @throws InvalidFormatException
774      */
775     protected void checkEpc(String textContent) throws InvalidFormatException {
776         String uri = textContent;
777         if (!uri.startsWith("urn:epc:id:")) {
778             throw new InvalidFormatException("Invalid 'pure identity' EPC format: must start with \"urn:epc:id:\"");
779         }
780         uri = uri.substring("urn:epc:id:".length());
781 
782         // check the patterns for the different EPC types
783         String epcType = uri.substring(0, uri.indexOf(":"));
784         uri = uri.substring(epcType.length() + 1);
785         LOG.debug("Checking pattern for EPC type " + epcType + ": " + uri);
786         Pattern p;
787         if ("gid".equals(epcType)) {
788             p = Pattern.compile("((0|[1-9][0-9]*)\\.){2}(0|[1-9][0-9]*)");
789         } else if ("sgtin".equals(epcType) || "sgln".equals(epcType) || "grai".equals(epcType)) {
790             p = Pattern.compile("([0-9]+\\.){2}([0-9]|[A-Z]|[a-z]|[\\!\\(\\)\\*\\+\\-',:;=_]|(%(([0-9]|[A-F])|[a-f]){2}))+");
791         } else if ("sscc".equals(epcType)) {
792             p = Pattern.compile("[0-9]+\\.[0-9]+");
793         } else if ("giai".equals(epcType)) {
794             p = Pattern.compile("[0-9]+\\.([0-9]|[A-Z]|[a-z]|[\\!\\(\\)\\*\\+\\-',:;=_]|(%(([0-9]|[A-F])|[a-f]){2}))+");
795         } else {
796             throw new InvalidFormatException("Invalid 'pure identity' EPC format: unknown EPC type: " + epcType);
797         }
798         Matcher m = p.matcher(uri);
799         if (!m.matches()) {
800             throw new InvalidFormatException("Invalid 'pure identity' EPC format: pattern \"" + uri
801                     + "\" is invalid for EPC type \"" + epcType + "\" - check with Tag Data Standard");
802         }
803 
804         // check the number of digits for the different EPC types
805         boolean exceeded = false;
806         int count1 = uri.indexOf(".");
807         if ("sgtin".equals(epcType)) {
808             int count2 = uri.indexOf(".", count1 + 1) - (count1 + 1);
809             if (count1 + count2 > 13) {
810                 exceeded = true;
811             }
812         } else if ("sgln".equals(epcType)) {
813             int count2 = uri.indexOf(".", count1 + 1) - (count1 + 1);
814             if (count1 + count2 > 12) {
815                 exceeded = true;
816             }
817         } else if ("grai".equals(epcType)) {
818             int count2 = uri.indexOf(".", count1 + 1) - (count1 + 1);
819             if (count1 + count2 > 12) {
820                 exceeded = true;
821             }
822         } else if ("sscc".equals(epcType)) {
823             int count2 = uri.length() - (count1 + 1);
824             if (count1 + count2 > 17) {
825                 exceeded = true;
826             }
827         } else if ("giai".equals(epcType)) {
828             int count2 = uri.length() - (count1 + 1);
829             if (count1 + count2 > 30) {
830                 exceeded = true;
831             }
832         } else {
833             // nothing to count
834         }
835         if (exceeded) {
836             throw new InvalidFormatException(
837                     "Invalid 'pure identity' EPC format: check allowed number of characters for EPC type '" + epcType
838                             + "'");
839         }
840     }
