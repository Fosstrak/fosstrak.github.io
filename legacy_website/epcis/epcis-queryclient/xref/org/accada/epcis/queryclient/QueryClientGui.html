277     /**
278      * Constructs a new QueryClientGui which sends its queries to the given
279      * endpoint address. If no such address is provided, the properties file is
280      * checked; if there is an error with reading the properties, a default url
281      * will be provided.
282      * 
283      * @param address
284      *            The address to send the queries to.
285      */
286     public QueryClientGui(final String address) {
287         String url = DEFAULT_URL;
288         if (address != null) {
289             try {
290                 new URL(address);
291                 url = address;
292             } catch (MalformedURLException e) {
293                 // unable to parse address as url!
294                 // read properties
295             }
296         } else {
297             // read properties
298             Properties props = new Properties();
299             InputStream is = this.getClass().getResourceAsStream(PROPERTY_FILE);
300             if (is != null) {
301                 try {
302                     props.load(is);
303                 } catch (IOException e) {
304                     // unable to load properties
305                     // use default address
306                 }
307                 url = props.getProperty(PROPERTY_QUERY_URL);
308                 if (url == null) {
309                     url = DEFAULT_URL;
310                 }
311             }
312         }
313         this.queryUrl = url;
314         initWindow();
315     }

1092      * Handler for pressed "Run query" button.
1093      */
1094     private void mwQueryButtonPressed() {
1095         dwOutputTextArea.setText("");
1096         if (resultsWindow != null) {
1097             resultsWindow.dispose();
1098         }
1099         try {
1100             client.setEndpointAddress(mwServiceUrlTextField.getText());
1101             client.clearParameters();
1102 
1103             /* get event type selection from GUI */
1104             Vector<String> eventVector = new Vector<String>();
1105             if (mwObjectEventsCheckBox.isSelected()) {

1113             }
1114             if (mwTransactionEventsCheckBox.isSelected()) {
1115                 eventVector.add("TransactionEvent");
1116             }
1117             String[] eventArray = new String[eventVector.size()];
1118             eventVector.toArray(eventArray);
1119 
1120             ArrayOfString events = new ArrayOfString();
1121             events.setString(eventArray);
1122             client.addParameter(new QueryParam("eventType", events));
1123 
1124             String name;
1125             for (int i = 0; i < mwQueryArgumentTextFields.size() - 1; i++) {
1126                 name = ((JTextFieldEnhanced) mwQueryArgumentTextFields.get(i)).queryItem.getQueryText();
1127                 switch (((JTextFieldEnhanced) mwQueryArgumentTextFields.get(i)).queryItem.getParamType()) {
1128                 case ListOfString:
1129                     ArrayOfString valueArray = client.stringListToArray(((JTextField) mwQueryArgumentTextFields.get(i)).getText());
1130                     client.addParameter(new QueryParam(name, valueArray));
1131                     break;
1132                 case Int:
1133                     Integer valueInteger = Integer.decode(((JTextField) mwQueryArgumentTextFields.get(i)).getText());
1134                     client.addParameter(new QueryParam(name, valueInteger));
1135                     break;
1136                 case Time:
1137                     Calendar valueCalendar = Calendar.getInstance();
1138                     Date dateTemp;
1139                     try {
1140                         dateTemp = isoDateFormat.parse(((JTextField) mwQueryArgumentTextFields.get(i)).getText());
1141                     } catch (ParseException e) {
1142                         SimpleDateFormat isoDateFormatTemp = new SimpleDateFormat(
1143                                 "yyyy-MM-dd'T'HH:mm:ss");
1144                         dateTemp = isoDateFormatTemp.parse(((JTextField) mwQueryArgumentTextFields.get(i)).getText());
1145                     }
1146                     valueCalendar.setTime(dateTemp);
1147                     client.addParameter(new QueryParam(name, valueCalendar));
1148                     break;
1149                 default:
1150                     String value = ((JTextField) mwQueryArgumentTextFields.get(i)).getText();
1151                     client.addParameter(new QueryParam(name, value));
1152                     break;
1153                 }
1154             }
1155 
1156             if (isSubscribed.isSelected()) {
1157                 if (mwSubIdField.getText().equals("")) {
1158                     JFrame frame = new JFrame();
1159                     JOptionPane.showMessageDialog(frame,
1160                             "Please specify a SubscriptionID",
1161                             "Service is responding", JOptionPane.ERROR_MESSAGE);
1162                     return;
1163                 }
1164                 Subscribe subcr = new Subscribe();
1165                 subcr.setDest(new URI(mwDestUriTextField.getText()));
1166                 subcr.setQueryName("SimpleEventQuery");
1167                 subcr.setSubscriptionID(mwSubIdField.getText());
1168                 SubscriptionControls controls = new SubscriptionControls();
1169                 Calendar valueCalendar = Calendar.getInstance();
1170                 Date dateTemp;

1172                     dateTemp = isoDateFormat.parse(mwInitRecTimeField.getText());
1173                 } catch (ParseException e) {

1175                             "yyyy-MM-dd'T'HH:mm:ss");
1176                     dateTemp = isoDateFormatTemp.parse(mwInitRecTimeField.getText());
1177                 }
1178                 valueCalendar.setTime(dateTemp);
1179                 controls.setInitialRecordTime(valueCalendar);
1180                 controls.setReportIfEmpty(reportIf.isSelected());
1181                 QuerySchedule sched = new QuerySchedule();
1182

1190                     controls.setSchedule(sched);
1191                 } else {

1193                 }
1194                 subcr.setControls(controls);
1195                 client.subscribeQuery(subcr);
1196                 JFrame frame = new JFrame();

1203             }

1207                     && af.getFaultDetails()[1].getTextContent().endsWith(
1208                             "ImplementationException")) {
1209                 msg += "Reason: "
1210                         + af.getFaultDetails()[0].getChildNodes().item(0).getTextContent()
1211                         + "\n";
1212                 msg += "Severity: "
1213                         + af.getFaultDetails()[0].getChildNodes().item(1).getTextContent();
1214             } else {
1215                 msg += af.getFaultDetails()[0].getTextContent();
1216             }
1217             dwOutputTextArea.append("\nCould not execute query:\n");

1219             PrintWriter pw = new PrintWriter(detailed);

1248     private void mwQuerySelectComboBoxesChanged(final int i) {

1250

1252                 && (cb != mwQuerySelectComboBoxes.getLast())) {
1253             /*
1254              * the user selected "ignore" and this is not the last row, so
1255              * remove it

1257             removeArgumentRow(i);

1261             addArgumentRow(i);

1265         }
