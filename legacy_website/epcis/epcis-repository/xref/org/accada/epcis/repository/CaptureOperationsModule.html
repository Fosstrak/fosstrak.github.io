189     /**
190      * Implements the EPCIS capture operation. Takes HTTP POST request, extracts
191      * the payload into an XML document, validates the document against the
192      * EPCIS schema, and captures the EPCIS events given in the document. Errors
193      * are caught and returned as simple plaintext messages via HTTP.
194      * 
195      * @param req
196      *            The HttpServletRequest.
197      * @param rsp
198      *            The HttpServletResponse.
199      * @throws IOException
200      *             If an error occured while validating the request or writing
201      *             the response.
202      */
203     public void doPost(final HttpServletRequest req, final HttpServletResponse rsp) throws IOException {
204         LOG.info("EPCIS Capture Interface invoked.");
205         rsp.setContentType("text/plain");
206         final PrintWriter out = rsp.getWriter();
207 
208         InputStream is = null;
209         // check if we have a POST request with form parameters
210         if ("application/x-www-form-urlencoded".equalsIgnoreCase(req.getContentType())) {
211             // check if the 'event' or 'dbReset' form parameter are given
212             String event = req.getParameter("event");
213             String dbReset = req.getParameter("dbReset");
214             if (event != null) {
215                 LOG.info("Found deprecated 'event=' parameter. Refusing to process request.");
216                 String msg = "Starting from version 0.2.2, the EPCIS repository does not accept the EPCISDocument in the HTTP POST form parameter 'event' anymore. Please provide the EPCISDocument as HTTP POST payload instead.";
217                 rsp.setStatus(HttpServletResponse.SC_NOT_ACCEPTABLE);
218                 out.println(msg);
219             } else if (dbReset != null && dbReset.equalsIgnoreCase("true")) {
220                 LOG.debug("Found 'dbReset' parameter set to 'true'.");
221                 rsp.setContentType("text/plain");
222                 if (dbResetAllowed) {
223                     try {
224                         dbconnection = db.getConnection();
225                         LOG.debug("DB connection opened.");
226                         dbReset();
227                         LOG.debug("DB connection closed.");
228                         dbconnection.close();
229 
230                         String msg = "db reset successfull";
231                         LOG.info(msg);
232                         rsp.setStatus(HttpServletResponse.SC_OK);
233                         out.println(msg);
234                     } catch (final SQLException e) {
235                         String msg = "An error involving the database ocurred.";
236                         LOG.error(msg, e);
237                         rsp.setStatus(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
238                         out.println(msg);
239                     } catch (final Exception e) {
240                         String msg = "An unknown error ocurred.";
241                         LOG.error(msg, e);
242                         rsp.setStatus(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
243                         out.println(msg);
244                     }
245                 } else {
246                     String msg = "'dbReset' operation not allowed!";
247                     LOG.info(msg);
248                     rsp.setStatus(HttpServletResponse.SC_FORBIDDEN);
249                     out.println(msg);
250                 }
251             }
252             out.flush();
253             out.close();
254             return;
255         } else {
256             is = req.getInputStream();
257         }
258 
259         // parse the payload as XML document
260         try {
261             final DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
262             factory.setNamespaceAware(true);
263             final DocumentBuilder builder = factory.newDocumentBuilder();
264             document = builder.parse(is);
265         } catch (SAXException e) {
266             String msg = "Unable to parse the payload as XML document: "
267                     + (e.getException() == null ? e.getMessage() : e.getException().getMessage());
268             LOG.info(msg, e);
269             rsp.setStatus(HttpServletResponse.SC_BAD_REQUEST);
270             out.println("We're unable to capture the EPCIS events.");
271             out.println("The data you provided is not a valid XML document.");
272             out.flush();
273             out.close();
274             return;
275         } catch (ParserConfigurationException e) {
276             String msg = "XML document builder not correctly configured.";
277             LOG.error(msg, e);
278             rsp.setStatus(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
279             out.flush();
280             out.close();
281             return;
282         }
283         LOG.debug("Payload successfully parsed as XML document.");
284 
285         // validate the XML document against the EPCISDocument schema
286         if (validator != null) {
287             try {
288                 validator.validate(new DOMSource(document), null);
289             } catch (SAXException e) {
290                 String msg = "XML document is not a valid EPCISDocument";
291                 LOG.info(msg, e);
292                 rsp.setStatus(HttpServletResponse.SC_BAD_REQUEST);
293                 out.println("We're unable to capture the EPCIS events.");
294                 out.println("The XML document you provided is not conform to the EPCISDocument schema.");
295                 out.flush();
296                 out.close();
297                 return;
298             }
299             LOG.info("Incoming capture request was successfully validated against the EPCISDocument schema.");
300         } else {
301             LOG.warn("Schema validator unavailable. Unable to validate EPCIS capture event against schema!");
302         }
303 
304         // handle the capture operation
305         try {
306             dbconnection = db.getConnection();
307             LOG.debug("DB connection opened.");
308             handleDocument();
309             LOG.debug("DB connection closed.");
310             dbconnection.close();
311 
312             // return OK
313             LOG.info("EPCIS Capture Interface request succeeded.");
314             rsp.setStatus(HttpServletResponse.SC_OK);
315             out.println("Capture request succeeded.");
316         } catch (final SAXException e) {
317             String msg = "An error processing the XML document ocurred.";
318             LOG.error(msg, e);
319             rsp.setStatus(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
320             out.println(msg);
321         } catch (final SQLException e) {
322             String msg = "An error involving the database ocurred.";
323             LOG.error(msg, e);
324             rsp.setStatus(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
325             out.println(msg);
326         } catch (final UnsupportedOperationException e) {
327             String msg = e.getMessage();
328             LOG.info(msg, e);
329             rsp.setStatus(HttpServletResponse.SC_BAD_REQUEST);
330             out.println(msg);
331         } catch (final Exception e) {
332             String msg = "An unknown error ocurred.";
333             LOG.error(msg, e);
334             rsp.setStatus(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
335             out.println(msg);
336         }
337         out.flush();
338         out.close();
339     }

341     /**
342      * Cleans all event data from the epcis repository database.
343      * 
344      * @throws SQLException
345      *             If an error with the database occured.
346      * @throws Exception
347      *             If an exception reading from the SQL script occured.
348      */
349     private void dbReset() throws SQLException, Exception {
350         LOG.info("Running db reset script.");
351         Statement stmt = dbconnection.createStatement();
352         if (dbResetScript != null) {
353             BufferedReader reader = new BufferedReader(new FileReader(dbResetScript));
354             String line;
355             while ((line = reader.readLine()) != null) {
356                 stmt.addBatch(line);
357             }
358         }
359         stmt.executeBatch();
360     }

362     /**
363      * @see javax.servlet.GenericServlet#init()
364      * @throws ServletException
365      *             If the context could not be loaded.
366      */
367     public void init() throws ServletException {
368         // read configuration and set up database source
369         try {
370             Context initContext = new InitialContext();
371             Context env = (Context) initContext.lookup("java:comp/env");
372             db = (DataSource) env.lookup("jdbc/EPCISDB");
373         } catch (NamingException e) {
374             String msg = "Unable to read configuration, check META-INF/context.xml for Resource 'jdbc/EPCISDB'.";
375             LOG.error(msg, e);
376             throw new ServletException(msg, e);
377         }
378 
379         // load properties
380         String servletPath = getServletContext().getRealPath("/");
381         String appConfigFile = getServletContext().getInitParameter("appConfigFile");
382         Properties properties = new Properties();
383         try {
384             properties.load(new FileInputStream(servletPath + appConfigFile));
385         } catch (IOException e) {
386             LOG.error("Unable to load application properties from " + servletPath + appConfigFile);
387         }
388         insertMissingVoc = Boolean.parseBoolean(properties.getProperty("insertMissingVoc", "true"));
389         String dbResetAllowedStr = getServletContext().getInitParameter("dbResetAllowed");
390         dbResetAllowed = Boolean.parseBoolean(dbResetAllowedStr);
391         dbResetScript = servletPath + getServletContext().getInitParameter("dbResetScript");
392 
393         // load log4j config
394         String log4jConfigFile = getServletContext().getInitParameter("log4jConfigFile");
395         if (log4jConfigFile != null) {
396             // if no log4j properties file found, then do not try
397             // to load it (the application runs without logging)
398             PropertyConfigurator.configure(servletPath + log4jConfigFile);
399         }
400 
401         // load the schema validator
402         try {
403             String schemaPath = servletPath + getServletContext().getInitParameter("schemaPath");
404             String schemaFile = getServletContext().getInitParameter("schemaFile");
405             File xsd = new File(schemaPath + System.getProperty("file.separator") + schemaFile);
406             LOG.debug("Reading schema from '" + xsd.getAbsolutePath() + "'.");
407             if (!xsd.exists()) {
408                 LOG.warn("Unable to find the schema file (check "
409                         + "'pathToSchemaFiles' parameter in META-INF/context.xml)");
410                 LOG.warn("Schema validation will not be available!");
411                 validator = null;
412             } else {
413                 // load the schema to validate against
414                 SchemaFactory schemaFact = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
415                 Source schemaSrc = new StreamSource(xsd);
416                 Schema schema = schemaFact.newSchema(schemaSrc);
417 
418                 validator = schema.newValidator();
419             }
420         } catch (Exception e) {
421             LOG.warn("Unable to load the schema validator.", e);
422             LOG.warn("Schema validation will not be available!");
423             validator = null;
424         }
425     }

427     /**
428      * Parses the entire document and handles the supplied events.
429      * 
430      * @throws SQLException
431      *             If an error with the database occured.
432      * @throws SAXException
433      *             If an error parsing the document occured.
434      */
435     private void handleDocument() throws SQLException, SAXException {
436         NodeList eventList = document.getElementsByTagName("EventList");
437         NodeList events = eventList.item(0).getChildNodes();
438 
439         // walk through all supplied events
440         for (int i = 0; i < events.getLength(); i++) {
441             Node eventNode = events.item(i);
442             String nodeName = eventNode.getNodeName();
443 
444             if (nodeName.equals("ObjectEvent") || nodeName.equals("AggregationEvent")
445                     || nodeName.equals("QuantityEvent") || nodeName.equals("TransactionEvent")) {
446                 LOG.debug("processing event " + i + ": '" + nodeName + "'.");
447                 handleEvent(eventNode);
448             } else if (!nodeName.equals("#text") && !nodeName.equals("#comment")) {
449                 throw new SAXException("Encountered unknown event '" + nodeName + "'.");
450             }
451         }
452     }

454     /**
455      * Takes an XML document node, parses it as EPCIS event and inserts the data
456      * into the database. The parse routine is generic for all event types; the
457      * query generation part has some if/elses to take care of different event
458      * parameters.
459      * 
460      * @param eventNode
461      *            The current event node.
462      * @throws SAXException
463      *             If an error parsing the XML occurs.
464      * @throws SQLException
465      *             If an error connecting to the DB occurs.
466      */
467     private void handleEvent(final Node eventNode) throws SAXException, SQLException {
468         if (eventNode != null && eventNode.getChildNodes().getLength() == 0) {
469             throw new SAXException("Event element '" + eventNode.getNodeName() + "' has no children elements.");
470         }
471         Node curEventNode = null;
472 
473         // A lot of the initialized variables have type URI. This type isn't to
474         // compare with the URI-Type of the standard. In fact, most of the
475         // variables having type URI are declared as Vocabularies in the
476         // Standard. Commonly, we use String for the Standard-Type URI.
477 
478         Timestamp eventTime = null;
479         String eventTimeZoneOffset = null;
480         String action = null;
481         String parentID = null;
482         Long quantity = null;
483         URI bizStep = null;
484         URI disposition = null;
485         URI readPoint = null;
486         URI bizLocation = null;
487         URI epcClass = null;
488 
489         URI[] epcs = null;
490         List<BusinessTransactionType> bizTransactionList = null;
491         List<EventFieldExtension> fieldNameExtList = new ArrayList<EventFieldExtension>();
492 
493         try {
494             for (int i = 0; i < eventNode.getChildNodes().getLength(); i++) {
495                 curEventNode = eventNode.getChildNodes().item(i);
496                 String nodeName = curEventNode.getNodeName();
497 
498                 if (nodeName.equals("#text") || nodeName.equals("#comment")) {
499                     // ignore text or comments
500                     LOG.debug("  ignoring text or comment: '" + curEventNode.getTextContent().trim() + "'");
501                     continue;
502                 }
503 
504                 LOG.debug("  handling event field: '" + nodeName + "'");
505                 if (nodeName.equals("eventTime")) {
506                     String xmlTime = curEventNode.getTextContent();
507                     LOG.debug("    eventTime in xml is '" + xmlTime + "'");
508                     try {
509                         eventTime = TimeParser.parseAsTimestamp(xmlTime);
510                         int offset = TimeZone.getDefault().getRawOffset() + TimeZone.getDefault().getDSTSavings();
511                         eventTime = new Timestamp(eventTime.getTime() - offset);
512 
513                     } catch (ParseException e) {
514                         throw new SAXException("Invalid date/time (must be ISO8601).", e);
515                     }
516                     LOG.debug("    eventTime parsed as '" + eventTime + "'");
517                 } else if (nodeName.equals("eventTimeZoneOffset")) {
518                     eventTimeZoneOffset = curEventNode.getTextContent();
519                 } else if (nodeName.equals("epcList") || nodeName.equals("childEPCs")) {
520                     epcs = handleEpcs(curEventNode);
521                 } else if (nodeName.equals("bizTransactionList")) {
522                     bizTransactionList = handleBizTransactions(curEventNode);
523                 } else if (nodeName.equals("action")) {
524                     action = curEventNode.getTextContent();
525                     if (!action.equals("ADD") && !action.equals("OBSERVE") && !action.equals("DELETE")) {
526                         throw new SAXException("Encountered illegal 'action' value: " + action);
527                     }
528                 } else if (nodeName.equals("bizStep")) {
529                     bizStep = new URI(curEventNode.getTextContent());
530                 } else if (nodeName.equals("disposition")) {
531                     disposition = new URI(curEventNode.getTextContent());
532                 } else if (nodeName.equals("readPoint")) {
533                     Element attrElem = (Element) curEventNode;
534                     Node id = attrElem.getElementsByTagName("id").item(0);
535                     readPoint = new URI(id.getTextContent());
536                 } else if (nodeName.equals("bizLocation")) {
537                     Element attrElem = (Element) curEventNode;
538                     Node id = attrElem.getElementsByTagName("id").item(0);
539                     bizLocation = new URI(id.getTextContent());
540                 } else if (nodeName.equals("epcClass")) {
541                     epcClass = new URI(curEventNode.getTextContent());
542                 } else if (nodeName.equals("quantity")) {
543                     quantity = new Long(curEventNode.getTextContent());
544                 } else if (nodeName.equals("parentID")) {
545                     parentID = curEventNode.getTextContent();
546                 } else {
547                     String[] parts = nodeName.split(":");
548                     if (parts.length == 2) {
549                         LOG.debug("    treating unknown event field as extension.");
550                         String prefix = parts[0];
551                         String localname = parts[1];
552                         String namespace = document.getDocumentElement().getAttribute("xmlns:" + prefix);
553                         String value = curEventNode.getTextContent();
554                         fieldNameExtList.add(new EventFieldExtension(prefix, namespace, localname, value));
555                     } else {
556                         // this is not a valid extension
557                         throw new SAXException("    encountered unknown event field: '" + nodeName + "'.");
558                     }
559                 }
560             }
561         } catch (MalformedURIException e) {
562             throw new SAXException("  event field '" + curEventNode.getNodeName() + "' is not of type URI: "
563                     + curEventNode.getTextContent(), e);
564         }
565 
566         // preparing query
567         PreparedStatement ps;
568         String nodeName = eventNode.getNodeName();
569         if (nodeName.equals("AggregationEvent")) {
570             ps = dbconnection.prepareStatement(aggregationEventInsert);
571         } else if (nodeName.equals("ObjectEvent")) {
572             ps = dbconnection.prepareStatement(objectEventInsert);
573         } else if (nodeName.equals("QuantityEvent")) {
574             ps = dbconnection.prepareStatement(quantityEventInsert);
575         } else if (nodeName.equals("TransactionEvent")) {
576             ps = dbconnection.prepareStatement(transactionEventInsert);
577         } else {
578             throw new SAXException("Encountered unknown event element '" + nodeName + "'.");
579         }
580 
581         // parameters 1-7 of the sql query are shared by all events
582 
583         ps.setTimestamp(1, eventTime);
584         // according to specification: if recordTime is ommitted we have to
585         // include the capturing time, i.e. current time
586         ps.setTimestamp(2, new Timestamp(System.currentTimeMillis()));
587         // note: for the tests it is handy to set recordTime=eventTime
588         // ps.setTimestamp(2, eventTime);
589         ps.setString(3, eventTimeZoneOffset);
590         if (bizStep != null) {
591             ps.setLong(4, insertVocabulary("voc_BizStep", bizStep));
592         } else {
593             ps.setNull(4, java.sql.Types.BIGINT);
594         }
595         if (disposition != null) {
596             ps.setLong(5, insertVocabulary("voc_Disposition", disposition));
597         } else {
598             ps.setNull(5, java.sql.Types.BIGINT);
599         }
600         if (readPoint != null) {
601             ps.setLong(6, insertVocabulary("voc_ReadPoint", readPoint));
602         } else {
603             ps.setNull(6, java.sql.Types.BIGINT);
604         }
605         if (bizLocation != null) {
606             ps.setLong(7, insertVocabulary("voc_BizLoc", bizLocation));
607         } else {
608             ps.setNull(7, java.sql.Types.BIGINT);
609         }
610 
611         // special handling for QuantityEvent
612         if (nodeName.equals("QuantityEvent")) {
613             if (epcClass != null) {
614                 ps.setLong(8, insertVocabulary("voc_EPCClass", epcClass));
615             } else {
616                 ps.setNull(8, java.sql.Types.BIGINT);
617             }
618             if (quantity != null) {
619                 ps.setLong(9, quantity.longValue());
620             } else {
621                 ps.setNull(9, java.sql.Types.BIGINT);
622             }
623         } else {
624             // all other events have action
625             ps.setString(8, action);

772      * Parses the xml tree for epc nodes and returns a list of EPC URIs.
773      * 
774      * @param epcNode
775      *            The parent Node from which EPC URIs should be extracted.
776      * @return An array of Voci containing all the URIs found in the given node.

779      * @throws SAXParseException
780      *             If an unknown tag (no &lt;epc&gt;) is encountered.
781      */
782     private URI[] handleEpcs(final Node epcNode) throws MalformedURIException, SAXParseException {
783         List<URI> epcList = new ArrayList<URI>();
784 
785         for (int i = 0; i < epcNode.getChildNodes().getLength(); i++) {
786             Node curNode = epcNode.getChildNodes().item(i);
787             if (curNode.getNodeName().equals("epc")) {
788                 epcList.add(new URI(curNode.getTextContent()));
789             } else {
790                 if (curNode.getNodeName() != "#text" && curNode.getNodeName() != "#comment") {
791                     throw new SAXParseException("Unknown XML tag: " + curNode.getNodeName(), null);
792                 }
793             }
794         }
795 
796         URI[] epcs = new URI[epcList.size()];
797         epcList.toArray(epcs);
798 
799         return epcs;
800     }
