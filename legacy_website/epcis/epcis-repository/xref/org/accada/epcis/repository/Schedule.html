75      /**
76       * Constructor for creating a new schedule according to the parameters in
77       * the given QuerySchedule 'schedule'.
78       * 
79       * @param schedule
80       *            The EPCIS style schedule to be used for constructing this
81       *            schedule.
82       * @throws SubscriptionControlsException
83       *             If invalid data is part of the Schedule.
84       */
85      public Schedule(final QuerySchedule schedule)
86              throws SubscriptionControlsException {
87  
88          // ease handling of null values in the query schedule
89          if (schedule.getSecond() == null) {
90              schedule.setSecond("");
91          }
92          if (schedule.getMinute() == null) {
93              schedule.setMinute("");
94          }
95          if (schedule.getHour() == null) {
96              schedule.setHour("");
97          }
98          if (schedule.getDayOfMonth() == null) {
99              schedule.setDayOfMonth("");
100         }
101         if (schedule.getMonth() == null) {
102             schedule.setMonth("");
103         }
104         if (schedule.getDayOfWeek() == null) {
105             schedule.setDayOfWeek("");
106         }
107 
108         // retrieve the values from the given query schedule
109         String[] second = schedule.getSecond().split(",");
110         String[] minute = schedule.getMinute().split(",");
111         String[] hour = schedule.getHour().split(",");
112         String[] dayOfMonth = schedule.getDayOfMonth().split(",");
113         String[] month = schedule.getMonth().split(",");
114         String[] dayOfWeek = schedule.getDayOfWeek().split(",");
115 
116         // parse numbers and ranges, check and add values
117         handleValues(second, "second", 0, 59);
118         handleValues(minute, "minute", 0, 59);
119         handleValues(hour, "hour", 0, 23);
120         handleValues(dayOfMonth, "dayOfMonth", 1, 31);
121         handleValues(month, "month", 1, 12);
122         handleValues(dayOfWeek, "dayOfWeek", 1, 7);
123 
124         // check for invalid month/dayOfMonth combinations, e.g. 30.2., 31.4.
125         if (!months.isEmpty()
126                 && (months.first() == months.last() && months.first() == 1 && (daysOfMonth.first() == 30 || daysOfMonth.first() == 31))) {
127             throw new SubscriptionControlsException(
128                     "Invalid query schedule: impossible month/dayOfMonth combination, e.g. February 30.");
129         }
130         if (!months.isEmpty()
131                 && daysOfMonth.first() == 31
132                 && !months.contains(0) // months w. 31 days are always ok
133                 && !months.contains(2) && !months.contains(4)
134                 && !months.contains(6) && !months.contains(7)
135                 && !months.contains(9) && !months.contains(11)) {
136             throw new SubscriptionControlsException(
137                     "Invalid query schedule: impossible month/dayOfMonth combination, e.g. April 31.");
138         }
139     }

144      * @return The next scheduled time after now.

148     public GregorianCalendar nextScheduledTime() throws ImplementationException {
149         GregorianCalendar cal = new GregorianCalendar();
150         // start at the next second to avoid multiple results
151         cal.add(SECOND, 1);
152         return nextScheduledTime(cal);

155     /**
156      * Calculates the next scheduled time after the given time. Algorithm idea:<br> -
157      * start with biggest time unit (i.e. year) of the given time <br> - if the
158      * time unit is valid (e.g. the time unit matches the <br>
159      * scheduled time, this is implicitly true if the time in the <br>
160      * schedule was omitted) *and* there exists a valid smaller time <br>
161      * unit, *then* return this time unit <br> - do this recursively for all
162      * time units <br> - month needs to be special cased because of dayOfWeek
163      * 
164      * @param time
165      *            Time after which next scheduled time should be returned.
166      * @return The next scheduled time after 'time'.
167      * @throws ImplementationException
168      *             Almost any kind of error.
169      */
170     public GregorianCalendar nextScheduledTime(final GregorianCalendar time)
171             throws ImplementationException {
172         GregorianCalendar nextSchedule = (GregorianCalendar) time.clone();
173         // look at year
174         while (!monthMadeValid(nextSchedule)) {
175             nextSchedule.roll(YEAR, true);
176             setFieldsToMinimum(nextSchedule, MONTH);
177         }
178         return nextSchedule;
179     }

181     /**
182      * Returns true if the month and all smaller time units have been
183      * succesfully set to valid values.
184      * 
185      * @param nextSchedule
186      *            The current candidate for the result.
187      * @return True if month and smaller units succesfully set to valid values.
188      * @throws ImplementationException
189      *             Almost any kind of error.
190      */
191     private boolean monthMadeValid(final GregorianCalendar nextSchedule)
192             throws ImplementationException {
193         // check if the month of the current time is valid, i.e. there is a
194         // month value in the schedule equal to the month value of the current
195         // time
196         if (!months.isEmpty() && !months.contains(nextSchedule.get(MONTH))) {
197             // no, month value of the current time is invalid
198             // roll the month (set it to the next value)
199             if (!setFieldToNextValidRoll(nextSchedule, MONTH, DAY_OF_MONTH)) {
200                 return false;
201             }
202         }
203         // now we're in a valid month, make smaller units valid as well or go to
204         // next month
205         while (!dayMadeValid(nextSchedule)) {
206             // no valid day for this month, try next
207             if (!setFieldToNextValidRoll(nextSchedule, MONTH, DAY_OF_MONTH)) {
208                 return false;
209             }
210             // reset all smaller units to minimum
211             if (!setFieldsToMinimum(nextSchedule, DAY_OF_MONTH)) {
212                 return false;
213             }
214         }
215         return true;
216     }

218     /**
219      * Returns true if the day and all smaller units have been succesfully set
220      * to valid values within the set month.
221      * 
222      * @param nextSchedule
223      *            The current candidate for the result.
224      * @return True if day and smaller units succesfully set to valid values.
225      * @throws ImplementationException
226      *             Almost any kind of error.
227      */
228     private boolean dayMadeValid(final GregorianCalendar nextSchedule)
229             throws ImplementationException {
230         if (!daysOfMonth.contains(nextSchedule.get(DAY_OF_MONTH))
231                 && !daysOfMonth.isEmpty()) {
232             if (!setFieldToNextValidRoll(nextSchedule, DAY_OF_MONTH,
233                     HOUR_OF_DAY)) {
234                 return false;
235             }
236         }
237 
238         // Check and make this also a valid day of week.
239         while (!daysOfWeek.contains(nextSchedule.get(DAY_OF_WEEK))
240                 && !daysOfWeek.isEmpty()) {
241             if (!setFieldToNextValidRoll(nextSchedule, DAY_OF_MONTH,
242                     HOUR_OF_DAY)) {
243                 return false;
244             } else if (!daysOfWeek.contains(nextSchedule.get(DAY_OF_WEEK))) {
245                 dayMadeValid(nextSchedule);
246             }
247         }
248 
249         // Now we're in a valid day, make smaller units
250         // valid as well or go to next day.
251         while (!hourMadeValid(nextSchedule)) {
252             // No valid hour for this day, try next day.
253             if (!setFieldToNextValidRoll(nextSchedule, DAY_OF_MONTH,
254                     HOUR_OF_DAY)) {
255                 return false;
256             }
257             // Reset all smaller units to min.
258             if (!setFieldsToMinimum(nextSchedule, HOUR_OF_DAY)) {
259                 return false;
260             }
261         }
262         return true;
263     }

265     /**
266      * Returns true if the hour and all smaller units have been succesfully set
267      * to valid values within the set day.
268      * 
269      * @param nextSchedule
270      *            The current candidate for the result.
271      * @return True if hour and smaller units succesfully set to valid values.
272      * @throws ImplementationException
273      *             Almost any error.
274      */
275     private boolean hourMadeValid(final GregorianCalendar nextSchedule)
276             throws ImplementationException {
277         if (!hours.contains(nextSchedule.get(HOUR_OF_DAY)) && !hours.isEmpty()) {
278             if (!setFieldToNextValidRoll(nextSchedule, HOUR_OF_DAY, MINUTE)) {
279                 return false;
280             }
281         }
282 
283         // Now we're in a valid hour, make smaller units
284         // valid as well or go to next hour.
285         while (!minuteMadeValid(nextSchedule)) {
286             // No valid minute for this hour, try next hour.
287             if (!setFieldToNextValidRoll(nextSchedule, HOUR_OF_DAY, MINUTE)) {
288                 return false;
289             }
290             // Reset all smaller units to min.
291             if (!setFieldsToMinimum(nextSchedule, MINUTE)) {
292                 return false;
293             }
294         }
295         return true;
296     }

298     /**
299      * Returns true if the minute and all smaller units have been succesfully
300      * set to valid values within the set hour.
301      * 
302      * @param nextSchedule
303      *            The current candidate for the result.
304      * @return True if minute and smaller units succesfully set to valid values.
305      * @throws ImplementationException
306      *             Almost any error.
307      */
308     private boolean minuteMadeValid(final GregorianCalendar nextSchedule)
309             throws ImplementationException {
310         if (!minutes.contains(nextSchedule.get(MINUTE)) && !minutes.isEmpty()) {
311 
312             if (!setFieldToNextValidRoll(nextSchedule, MINUTE, SECOND)) {
313                 return false;
314             }
315         }
316 
317         // Now we're in a valid minute, make smaller units
318         // valid as well or go to next minute.
319         while (!secondMadeValid(nextSchedule)) {
320             // No valid second for this minute, try next minute.
321 
322             if (!setFieldToNextValidRoll(nextSchedule, MINUTE, SECOND)) {
323                 return false;
324             }
325             // Reset all smaller units to min.
326             if (!setFieldToMinimum(nextSchedule, SECOND)) {
327                 return false;
328             }
329         }
330         return true;
331     }

334      * Returns true if the second have been succesfully set to valid values
335      * within the set minute.
336      * 
337      * @param nextSchedule
338      *            The current candidate for the result.
339      * @return True if second succesfully set to valid values.
340      * @throws ImplementationException
341      *             Almost any error.
342      */
343     private boolean secondMadeValid(final GregorianCalendar nextSchedule)
344             throws ImplementationException {
345         // check whether the second value of the current time is a valid
346         // scheduled second
347         if (!seconds.isEmpty() && !seconds.contains(nextSchedule.get(SECOND))) {
348             // no current second is not scheduled
349             // set is to the next scheduled second
350             return setToNextScheduledValue(nextSchedule, SECOND);
351         }
352         return true;

355     /**
356      * Sets the specified field of the given callendar to the next scheduled
357      * value. Returns whether the new value has been set and is valid.
358      * 
359      * @param cal
360      *            Calendar to adjust.
361      * @param field
362      *            Field to adjust.
363      * @return Returns whether the new value has been set and is valid.
364      * @throws ImplementationException
365      *             Almost any error.
366      */
367     private boolean setToNextScheduledValue(final GregorianCalendar cal,
368             final int field) throws ImplementationException {
369         int next;
370         TreeSet<Integer> vals = getValues(field);
371         if (vals.isEmpty()) {
372             next = cal.get(field) + 1;
373         } else {
374             try {
375                 // get next scheduled value which is bigger than current
376                 int incrValue = cal.get(field) + 1;
377                 next = vals.tailSet(new Integer(incrValue)).first().intValue();
378             } catch (NoSuchElementException nse) {
379                 // there is no bigger scheduled value
380                 return false;
381             }
382         }
383         if (next > cal.getActualMaximum(field)
384                 || next < cal.getActualMinimum(field)) {
385             return false;
386         }
387         // all is well, set it to next
388         cal.set(field, next);
389         return true;
390     }

418     /**
419      * Sets the field of a GregorianCalender to its minimum, which is defined as
420      * the minimal possible value according to the calendar type possibly
421      * superseded by the defined values in the schedule we have. Returns whether
422      * the new value has been set and is valid.
423      * 
424      * @param cal
425      *            Calendar to adjust.
426      * @param field
427      *            Field to adjust.
428      * @return Returns whether the new value has been set and is valid.
429      * @throws ImplementationException
430      *             Almost any error.
431      */
432     private boolean setFieldToMinimum(final GregorianCalendar cal,
433             final int field) throws ImplementationException {
434         int min;
435         TreeSet<Integer> values = getValues(field);
436         if (values.isEmpty()) {
437             min = cal.getActualMinimum(field);
438         } else {
439             min = Math.max(values.first().intValue(),
440                     cal.getActualMinimum(field));
441             if (min > cal.getActualMaximum(field)) {
442                 min = cal.getActualMaximum(field);
443                 if (!values.contains(min) || min < cal.getActualMinimum(field)
444                         || min > cal.getActualMaximum(field)) {
445                     return false;
446                 }
447             }
448         }
449         cal.set(field, min);
450         return true;
451     }

453     /**
454      * Sets the given field of a GregorianCalender and all smaller fields (not
455      * WEEK_OF_DAY) to their minimum, which is defined as the minimal possible
456      * value according to the calendar type used possibly superseded by the
457      * defined values in the schedule we have. Returns whether the new values
458      * have been set and are all valid.
459      * 
460      * @param cal
461      *            The Calendar instance to adjust.
462      * @param largestField
463      *            This field and smaller ones are reset
464      * @return True if setting to min worked for all values.
465      * @throws ImplementationException
466      *             Various errors.
467      */
468     private boolean setFieldsToMinimum(final GregorianCalendar cal,
469             final int largestField) throws ImplementationException {
470         boolean result = true;
471         switch (largestField) {
472         case (MONTH):
473             result = setFieldToMinimum(cal, MONTH) && result;
474         case (DAY_OF_MONTH):
475             result = setFieldToMinimum(cal, DAY_OF_MONTH) && result;
476         case (HOUR_OF_DAY):
477             result = setFieldToMinimum(cal, HOUR_OF_DAY) && result;
478         case (MINUTE):
479             result = setFieldToMinimum(cal, MINUTE) && result;
480         case (SECOND):
481             result = setFieldToMinimum(cal, SECOND) && result;
482             break;
483         default:
484             String msg = "Invalid field: " + largestField;
485             ImplementationException iex = new ImplementationException();
486             iex.setSeverity(ImplementationExceptionSeverity.ERROR);
487             iex.setReason(msg);
488             LOG.error(msg, iex);
489             throw iex;
490 
491         }
492         return result;
493     }

530     /**
531      * Checks whether the given values, which are either numbers or ranges, are
532      * valid (parsable as Integer) and adds the value to the correct set of
533      * values (e.g. seconds).
534      * 
535      * @param values
536      *            The numbers and ranges to be checked and added.
537      * @param type
538      *            The name of the schedule element, e.g. 'second'.
539      * @param min
540      *            The minimum allowed value.
541      * @param max
542      *            The maximum allowed value.
543      * @throws SubscriptionControlsException
544      *             If one of the given values is invalid, i.e. does not lie
545      *             between the <code>min</code> and <code>max</code> value.
546      */
547     private void handleValues(final String[] values, final String type,
548             final int min, final int max) throws SubscriptionControlsException {
549         // we put values into this sorted set
550         TreeSet<Integer> vals = new TreeSet<Integer>();
551         for (String v : values) {
552             try {
553                 if (v.startsWith("[")) {
554                     // it's a range
555                     String[] range = v.substring(1, v.length() - 1).split("-");
556                     int start = Integer.parseInt(range[0]);
557                     int end = Integer.parseInt(range[1]);
558                     // check range
559                     if (start < min || end > max || start > end) {
560                         throw new SubscriptionControlsException(
561                                 "The value for '"
562                                         + type
563                                         + "' is out of range in the query schedule.");
564                     }
565                     // add all values in the range
566                     for (int value = start; value <= end; value++) {
567                         vals = addValue(value, type, vals);
568                     }
569                 } else if (!v.equals("")) {
570                     // it's a single value
571                     int value = Integer.parseInt(v);
572                     // check value
573                     if (value < min || value > max) {
574                         throw new SubscriptionControlsException(
575                                 "The value for '"
576                                         + type
577                                         + "' is out of range in the query schedule.");
578                     }
579                     // add value
580                     vals = addValue(value, type, vals);
581                 }
582             } catch (Exception e) {
583                 String msg = "The value '" + v + "' for parameter '" + type
584                         + "' is invalid in the query schedule.";
585                 LOG.info("USER ERROR: " + msg + e.getMessage());
586                 throw new SubscriptionControlsException(msg);
587             }
588         }
589 
590         if (type.equals("second")) {
591             this.seconds = vals;
592         } else if (type.equals("minute")) {
593             this.minutes = vals;
594         } else if (type.equals("hour")) {
595             this.hours = vals;
596         } else if (type.equals("dayOfMonth")) {
597             this.daysOfMonth = vals;
598         } else if (type.equals("month")) {
599             this.months = vals;
600         } else if (type.equals("dayOfWeek")) {
601             this.daysOfWeek = vals;
602         }
603     }

605     /**
606      * Adds a schedule value to the given set of values with some special
607      * treatment for 'month' and 'dayOfWeek'.
608      * 
609      * @param value
610      *            The value to be added.
611      * @param type
612      *            The name of the schedule element, e.g. 'second'.
613      * @param vals
614      *            The set of values to which the value should be added.
615      * @return The modified set of values.
616      */
617     private TreeSet<Integer> addValue(final int value, final String type,
618             final TreeSet<Integer> vals) {
619         if (type.equals("dayOfWeek")) {
620             vals.add(new Integer((value % 7) + 1));
621         } else if (type.equals("month")) {
622             vals.add(new Integer(value - 1));
623         } else {
624             vals.add(new Integer(value));
625         }
626         return vals;
627     }
