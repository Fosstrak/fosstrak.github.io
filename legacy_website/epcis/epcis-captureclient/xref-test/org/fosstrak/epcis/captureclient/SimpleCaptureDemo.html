66  	protected static final String LOCAL_EPCIS_CAPTURE_URL = "http://localhost:8080/epcis-repository/capture";
67  	protected static final String DEMO_EPCIS_CAPTURE_URL = "http://demo.fosstrak.org/epcis/capture";

69  	/**
70  	 * Reads a sample EPCIS capture request and submits it to an EPCIS
71  	 * repository instance.
72  	 * 
73  	 * @param args
74  	 *            Valid command-line argument is the URL of the EPCIS
75  	 *            repository (<code>url=</code>). A default values will be
76  	 *            provided if this argument is omitted.
77  	 */
78  	public static void main(String[] args) throws Exception {
79  		String captureUrl = LOCAL_EPCIS_CAPTURE_URL;
80  		if (args.length > 0) {
81  			// read command-line arguments
82  			for (int i = 0; i < args.length; i++) {
83  				if (args[i].startsWith("url=")) {
84  					captureUrl = args[i].substring(4);
85  				} else {
86  					System.out.println("ignoring command-line argument: " + args[i]);
87  				}
88  			}
89  		}
90  		System.out.println("using capture URL: " + captureUrl);
91          CaptureClient client = new CaptureClient(captureUrl);
92  
93  		// create and send the EPCIS capture request document
94  		EPCISDocumentType requestDoc = createEpcisDocument();
95  		printDoc(requestDoc);
96  		int httpResponseCode = client.capture(requestDoc);
97  
98  		if (httpResponseCode == 200) {
99  			System.out.println("capture of events successful");
100 		} else {
101 			System.err.println("HTTP response " + httpResponseCode);
102 		}
103 
104         // create and send an EPCIS MasterData capture request document
105         EPCISMasterDataDocumentType masterDataDoc = createEpcisMasterDataDocument();
106         printDoc(masterDataDoc);
107         httpResponseCode = client.capture(masterDataDoc);
108 
109         if (httpResponseCode == 200) {
110             System.out.println("capture of events successful");
111         } else {
112             System.err.println("HTTP response " + httpResponseCode);
113         }
114 	}

116 	/**
117 	 * Creates a simple EPCIS object event using the API objects from the EPCIS
118 	 * schema.
119 	 */
120 	private static EPCISDocumentType createEpcisDocument() {
121 		ObjectEventType objEvent = new ObjectEventType();
122 
123 		// set the eventTime and eventTimeZoneOffset
124 		XMLGregorianCalendar eventTime = getCurrentDateTime();
125 		String timeZoneOffset = getTimeZoneOffset(eventTime);
126 		objEvent.setEventTime(eventTime);
127 		objEvent.setEventTimeZoneOffset(timeZoneOffset);
128 
129 		// set the EPCs
130 		EPC epc = new EPC();
131 		epc.setValue("urn:epc:id:sgtin:0057000.123780.7788");
132 		EPCListType epcList = new EPCListType();
133 		epcList.getEpc().add(epc);
134 		objEvent.setEpcList(epcList);
135 
136 		// set the action
137 		objEvent.setAction(ActionType.ADD);
138 
139 		// set the bizStep
140 		objEvent.setBizStep("urn:fosstrak:demo:bizstep:fmcg:production");
141 
142 		// set the disposition
143 		objEvent.setDisposition("urn:fosstrak:demo:disp:fmcg:pendingQA");
144 
145 		// set the readPoint
146 		ReadPointType readPoint = new ReadPointType();
147 		readPoint.setId("urn:fosstrak:demo:fmcg:ssl:0037000.00729.210,432");
148 		objEvent.setReadPoint(readPoint);
149 
150 		// set the bizLocation
151 		BusinessLocationType bizLocation = new BusinessLocationType();
152 		bizLocation.setId("urn:fosstrak:demo:fmcg:ssl:0037000.00729.210");
153 		objEvent.setBizLocation(bizLocation);
154 
155 		// optionally add an event field extension
156 		addEventFieldExtension(objEvent);
157 
158 		// create the EPCISDocument containing the ObjectEvent
159 		EPCISDocumentType epcisDoc = new EPCISDocumentType();
160 		EPCISBodyType epcisBody = new EPCISBodyType();
161 		EventListType eventList = new EventListType();
162 		eventList.getObjectEventOrAggregationEventOrQuantityEvent().add(objEvent);
163 		epcisBody.setEventList(eventList);
164 		epcisDoc.setEPCISBody(epcisBody);
165 		epcisDoc.setSchemaVersion(new BigDecimal("1.0"));
166 		epcisDoc.setCreationDate(eventTime);
167 		return epcisDoc;
168 	}

171 	 * Returns an XMLGregorianCalendar representing the current date and time
172 	 */
173 	protected static XMLGregorianCalendar getCurrentDateTime() {
174 		XMLGregorianCalendar now = null;
175 		try {
176 			DatatypeFactory dataFactory = DatatypeFactory.newInstance();
177 			now = dataFactory.newXMLGregorianCalendar(new GregorianCalendar());
178 		} catch (DatatypeConfigurationException e) {
179 			System.err.println("unable to construct the date/time object");
180 			e.printStackTrace();
181 		}
182 		return now;
183 	}

185 	/**
186 	 * Retrieves the timezone from the given eventTime and formats it
187 	 * appropriately.
188 	 */
189 	protected static String getTimeZoneOffset(XMLGregorianCalendar eventTime) {
190 		int timezone = eventTime.getTimezone();
191 		int h = Math.abs(timezone / 60);
192 		int m = Math.abs(timezone % 60);
193 		DecimalFormat format = new DecimalFormat("00");
194 		String sign = (timezone < 0) ? "-" : "+";
195 		return sign + format.format(h) + ":" + format.format(m);
196 	}

199 		try {
200 			DocumentBuilder builder = DocumentBuilderFactory.newInstance().newDocumentBuilder();
201 			Element elem = builder.newDocument().createElementNS("http://www.example.com/epcis/extensions/", "temperature");
202 			elem.setPrefix("xyz");
203 			elem.setTextContent("30");
204 			objEvent.getAny().add(elem);

211 	/**
212 	 * Marshals the given EPCIS document into its XML representation and prints it to System.out.
213 	 */
214 	protected static void printDoc(Document epcisDoc) {
215 		try {
216 			ObjectFactory factory = new ObjectFactory();
217 			JAXBElement<? extends Document> item;
218 			JAXBContext context;
219 			if (epcisDoc instanceof EPCISDocumentType) {
220 			    item = factory.createEPCISDocument((EPCISDocumentType) epcisDoc);
221 	            context = JAXBContext.newInstance(EPCISDocumentType.class);
222 			} else {
223                 item = factory.createEPCISMasterDataDocument((EPCISMasterDataDocumentType) epcisDoc);
224                 context = JAXBContext.newInstance(EPCISMasterDataDocumentType.class);
225 			}
226 			Marshaller marshaller = context.createMarshaller();
227 			marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
228 			marshaller.setProperty(Marshaller.JAXB_FRAGMENT, Boolean.TRUE);
229 			marshaller.setProperty(Marshaller.JAXB_ENCODING, "UTF-8");
230 			marshaller.marshal(item, System.out);
231 		} catch (JAXBException e) {
232 			System.err.println("unable to print the EPCIS document");
233 			e.printStackTrace();
234 		}
235 	}

237     private static EPCISMasterDataDocumentType createEpcisMasterDataDocument() {
238         // create a <VocabularyElement> which contains two <attribute> elements
239         VocabularyElementType vocElem = new VocabularyElementType();
240         vocElem.setId("urn:epc:id:sgln:0037000.00729.0");
241         AttributeType attr1 = new AttributeType();
242         attr1.setId("urn:epcglobal:fmcg:mda:latitude");
243         attr1.getContent().add("+18.0000");
244         AttributeType attr2 = new AttributeType();
245         attr2.setId("urn:epcglobal:fmcg:mda:longitude");
246         attr2.getContent().add("-70.0000");
247         vocElem.getAttribute().add(attr1);
248         vocElem.getAttribute().add(attr2);
249 
250         // create a <Vocabulary> element of type 'BusinessLocation' and add a list of vocabulary elements
251         VocabularyType voc = new VocabularyType();
252         voc.setType("urn:epcglobal:epcis:vtype:BusinessLocation");
253         VocabularyElementListType vocElemList = new VocabularyElementListType();
254         vocElemList.getVocabularyElement().add(vocElem);
255         voc.setVocabularyElementList(vocElemList);
256         VocabularyListType vocList = new VocabularyListType();
257         vocList.getVocabulary().add(voc);
258         
259         // create the EPCISMasterDataDocument
260         EPCISMasterDataDocumentType masterDataDoc = new EPCISMasterDataDocumentType();
261         masterDataDoc.setSchemaVersion(new BigDecimal("1.0"));
262         masterDataDoc.setCreationDate(getCurrentDateTime());
263         EPCISMasterDataBodyType masterDataBody = new EPCISMasterDataBodyType();
264         masterDataBody.setVocabularyList(vocList);
265         masterDataDoc.setEPCISBody(masterDataBody);
266         return masterDataDoc;
267     }
