50  /**
51   *
52   *	<p>This class provides methods for translating an electronic
53   *	product code (EPC) between various levels of representation
54   *	including BINARY, TAG_ENCODING, PURE_IDENTITY and LEGACY
55   *	formats.  An additional output level ONS_HOSTNAME may be
56   *	defined for some coding schemes. When the class TDTEngine is
57   *	constructed, the path to a local directory must be specified,
58   *	by passing it as a single string parameter to the constructor
59   *	method (without any trailing slash or file separator).  e.g.
60   *      <pre><code>TDTEngine engine = new TDTEngine("/opt/TDT");</code></pre></p>
61   *      
62   *      <p>The specified directory must contain two subdirectories named
63   *      auxiliary and schemes.  The Tag Data Translation definition files
64   *      for the various coding schemes should be located inside the
65   *      subdirectory called <code>schemes</code>.  Any auxiliary lookup files
66   *      (such as <code>ManagerTranslation.xml</code>) should be located inside the
67   *      subdirectory called <code>auxiliary</code>.</p>
68   *
69   * @version $Revision: 1.9 $ $Date: 2006/01/27 13:42:32 $
70   */

194 	public PrefixMatch(Scheme s, Level level) { 
195 	    this.s = s;
196 	    this.level = level;
197 	}
198 	public Scheme getScheme() { return s; } 
199 	public Level getLevel() { return level; }

204     /** initialise various indices */ 
205     private void initFromTDT(EpcTagDataTranslation tdt) { 
206 	Scheme [] scheme = tdt.getScheme();
207 	for (Scheme ss : scheme) {
208 
209 	    // create an index so that we can find a scheme based on tag length
210 
211 	    for (Level level : ss.getLevel()) {
212 		String s = level.getPrefixMatch();
213 		if (s != null) 
214 		    {
215 			// insert into prefix tree according to level type.
216 			PrefixTree<PrefixMatch> prefix_tree = prefix_tree_map.get(level.getType());
217 			if (prefix_tree == null) {
218 			    prefix_tree = new PrefixTree<PrefixMatch>();
219 			    prefix_tree_map.put(level.getType(), prefix_tree);
220 			}
221 			prefix_tree.insert(s, new PrefixMatch(ss, level));
222 		    }
223 	    }
224 	    
225 	}
226     }

230     /** 
231      *     Given an input string, and optionally a tag length, find a
232      *     scheme / level with a matching prefix and tag length.
233      */
234     private PrefixMatch findPrefixMatch(String input, TagLengthList tagLength)
235     {
236 	List<PrefixMatch> match_list = new ArrayList<PrefixMatch>();
237 		    
238 	for (PrefixTree<PrefixMatch> tree : prefix_tree_map.values()) {
239 
240 	    List<PrefixMatch> list = tree.search(input);
241 	    if (! list.isEmpty()) {
242 		if (tagLength == null)  
243 		    match_list.addAll(list);
244 		else {
245 		    for (PrefixMatch match : list)
246 			if (match.getScheme().getTagLength() == tagLength)
247 			    match_list.add(match);
248 		}
249 	    }
250 	}
251 	
252 	if (match_list.isEmpty())
253 	    throw new TDTException("No schemes or levels matched the input value");
254 	else if (match_list.size() > 1) 
255 	    throw new TDTException("More than one scheme/level matched the input value");
256 	else
257 	    return match_list.get(0);
258     }

261     /** 
262      *     Given an input string, level, and optionally a tag length,
263      *     find a matching prefix.
264      */
265     private PrefixMatch findPrefixMatch(String input, TagLengthList tagLength, LevelTypeList level_type)
266     {
267 	List<PrefixMatch> match_list = new ArrayList<PrefixMatch>();
268 	PrefixTree<PrefixMatch> tree = prefix_tree_map.get(level_type);
269 	assert tree != null;
270 	List<PrefixMatch> list = tree.search(input);
271 	if (! list.isEmpty()) {
272 	    if (tagLength == null) 
273 		match_list.addAll(list);
274 	    else {
275 		for (PrefixMatch match : list)
276 		    if (match.getScheme().getTagLength() == tagLength)
277 			match_list.add(match);
278 	    }
279 	}
280 	if (match_list.isEmpty())
281 	    throw new TDTException("No schemes or levels matched the input value");
282 	else if (match_list.size() > 1) 
283 	    throw new TDTException("More than one scheme/level matched the input value");
284 	else
285 	    return match_list.get(0);
286     }

289     /** 
290      *	Translates the input string of a specified input level to a
291      *	specified outbound level of the same coding scheme.  For
292      *	example, the input string value may be a tag-encoding URI and
293      *	the outbound level specified by string outboundlevel may be
294      *	BINARY, in which case the return value is a binary
295      *	representation expressed as a string.
296      * 
297      *  <p>Note that this version of the method requires that the user
298      *  specify the input level, rather than searching for it. However
299      *  it still automatically finds the scheme used.</p>
300      *
301      *  @param input input tag coding
302      *  @param inputLevel level such as BINARY, or TAG_ENCODING.
303      *  @param tagLength tag length such as VALUE_64 or VALUE_96.
304      *  @param inputParameters a map with any additional properties.
305      *  @param outputLevel required output level.
306      *  @return output tag coding
307      */
308     public String convert(String input, 
309 			  LevelTypeList inputLevel, 
310 			  TagLengthList tagLength,
311 			  Map<String,String> inputParameters,
312 			  LevelTypeList outputLevel) {
313 	
314 	PrefixMatch match = findPrefixMatch(input, tagLength, inputLevel);
315 
316 	return convertLevel(match.getScheme(), match.getLevel(), input, inputParameters, outputLevel); 
317     }
318     /**
319      *	The convert method translates a String input to a specified
320      *	outbound level of the same coding scheme.  For example, the
321      *	input string value may be a tag-encoding URI and the outbound
322      *	level specified by string outboundlevel may be BINARY, in
323      *	which case the return value is a binary representation
324      *	expressed as a string.
325      *	@param input the identifier to be converted.
326      *	@param inputParameters additional parameters which need to be
327      *	provided because they cannot always be determined from the
328      *	input value alone.  Examples include the taglength,
329      *	companyprefixlength and filter values.
330      *	@param outputLevel the outbound level required for the ouput.
331      *	Permitted values include BINARY, TAG_ENCODING, PURE_IDENTITY,
332      *	LEGACY and ONS_HOSTNAME.
333      *	@return the identifier converted to the output level.
334      */
335     public String convert(String input,
336 			  Map<String,String> inputParameters,
337 			  LevelTypeList outputLevel) {
338 
339 	TagLengthList tagLength = null;
340 	if (inputParameters.containsKey("taglength")) {
341 	    // in principle, the user should provide a
342 	    // TagLengthList object in the parameter list.
343 	    String s = inputParameters.get("taglength");
344 	    tagLength = TagLengthList.valueOf(s);
345 	}
346 		
347 	PrefixMatch match = findPrefixMatch(input, tagLength);
348 	
349 	return convertLevel(match.getScheme(), match.getLevel(), input, inputParameters, outputLevel);
350     }

353     /** convert from a particular scheme / level
354      */
355     private String convertLevel(Scheme tdtscheme, 
356 				Level tdtlevel,
357 				String input, 
358 				Map<String, String> inputParameters, 
359 				LevelTypeList outboundlevel) {
360 	
361 	String outboundstring;
362 	Map<String, String> extraparams = 
363 	    //	    new NoisyMap
364 	    (new HashMap<String, String>(inputParameters));
365 		
366 		
367 	// get the scheme's option key, which is the name of a
368 	// parameter whose value is matched to the option key of the
369 	// level. 
370 
371 	String optionkey = tdtscheme.getOptionKey(); 
372 	String optionValue = extraparams.get(optionkey);
373 	// the name of a parameter which allows the appropriate option
374 	// to be selected
375 			
376 	// now consider the various options within the scheme and
377 	// level for each option element inside the level, check
378 	// whether the pattern attribute matches as a regular
379 	// expression
380 	
381 	String matchingOptionKey = null;
382 	Option matchingOption = null;
383 	Matcher prefixMatcher = null;
384 	for (Enumeration e = tdtlevel.enumerateOption(); e.hasMoreElements(); ) { 
385 	    Option opt = (Option) e.nextElement();
386 	    if (optionValue == null || optionValue.equals(opt.getOptionKey())) {
387 		// possible match 
388 
389 		Matcher matcher = Pattern.compile(opt.getPattern()).matcher(input);
390 		if (matcher.matches()) {
391 		    if (prefixMatcher != null) 
392 			throw new TDTException("Multiple patterns matched");
393 		    prefixMatcher = matcher;
394 		    matchingOptionKey = opt.getOptionKey();
395 		    matchingOption = opt;
396 		}
397 	    }
398 	}
399 	if (prefixMatcher == null) 
400 	    throw new TDTException("No patterns matched");
401 
402 	optionValue = matchingOptionKey;
403 
404 
405 	for (Enumeration e = matchingOption.enumerateField(); e.hasMoreElements();) { 
406 	    Field field = (Field) e.nextElement();
407 	    int seq = field.getSeq();
408 	    
409 	    String strfieldname = field.getName();
410 	    int fieldlength = field.getLength();
411 	    String strfieldvalue = prefixMatcher.group(seq);
412 	    //System.out.println("   processing field " + strfieldname + " = '" + strfieldvalue + "'");
413 		
414 						
415 	    if (field.getCompaction() == null) {
416 		// if compaction is null, treat field as an integer
417 
418 		if (field.getCharacterSet() != null) { // if the character set is specified
419 		    Matcher charsetmatcher = Pattern.compile("^"+field.getCharacterSet()+"$").matcher(strfieldvalue);
420 		    if (! charsetmatcher.matches()) {
421 			throw new TDTException("field "+strfieldname+" ("+strfieldvalue+") does not conform to the allowed character set ("+field.getCharacterSet()+") ");
422 		    }
423 		}
424 
425 		BigInteger bigvalue = null;
426 		
427 		if (tdtlevel.getType() == LevelTypeList.BINARY) { // if the input was BINARY
428 		    bigvalue = new BigInteger(strfieldvalue, 2);
429 		    extraparams.put(strfieldname, bigvalue.toString());
430 		}
431 		else {
432 		    if (field.getDecimalMinimum() != null || 
433 			field.getDecimalMaximum() != null)
434 			bigvalue = new BigInteger(strfieldvalue);
435 		    extraparams.put(strfieldname, strfieldvalue);
436 		}
437 
438 		if (field.getDecimalMinimum() != null) { // if the decimal minimum is specified
439 		    BigInteger bigmin = new BigInteger(field.getDecimalMinimum());
440 		    
441 		    if (bigvalue.compareTo(bigmin) == -1) { // throw an exception if the field value is less than the decimal minimum
442 			throw new TDTException("field "+strfieldname+" ("+bigvalue+") is less than DecimalMinimum ("+field.getDecimalMinimum()+") allowed");
443 		    }
444 		}
445 		
446 		if (field.getDecimalMaximum() != null) { // if the decimal maximum is specified
447 		    BigInteger bigmax = new BigInteger(field.getDecimalMaximum());
448 		    
449 		    if (bigvalue.compareTo(bigmax) == 1) { // throw an excpetion if the field value is greater than the decimal maximum
450 			throw new TDTException("field "+strfieldname+" ("+bigvalue+") is greater than DecimalMaximum ("+field.getDecimalMaximum()+") allowed");
451 		    }
452 		}
453 
454 		// after extracting the field, it may be necessary to pad it.
455 
456 		padField(extraparams, field);
457 								
458 	    } else {
459 		// compaction is specified - interpret binary as a string value using a truncated byte per character
460 						
461 		CompactionMethodList compaction = field.getCompaction();
462 		PadDirectionList padDir = field.getPadDir();
463 		String padchar = field.getPadChar();
464 		String s; 
465 		if (compaction == CompactionMethodList.VALUE_5) 
466 		    //"5-bit"
467 		    s = bin2uppercasefive(strfieldvalue);
468 		else if (compaction == CompactionMethodList.VALUE_4) 
469 		    // 6-bit
470 		    s = bin2alphanumsix(strfieldvalue);
471 		else if (compaction == CompactionMethodList.VALUE_3) 
472 		    // 7-bit
473 		    s = bin2asciiseven(strfieldvalue);
474 		else if (compaction == CompactionMethodList.VALUE_2)
475 		    // 8-bit
476 		    s = bin2bytestring(strfieldvalue);
477 		else 
478 		    throw new Error("unsupported compaction method " + compaction);
479 		extraparams.put(strfieldname, stripPadChar(s, padDir, padchar));
480 							
481 	    }
482 						
483 	} // for each field;
484 					
485 
486 	/**	the EXTRACT rules are performed after parsing the
487 	 *	input, in order to determine additional fields that
488 	 *	are to be derived from the fields obtained by the
489 	 *	pattern match process
490 	 */
491 			
492 	int seq = 0;
493 	for (Enumeration e = tdtlevel.enumerateRule(); e.hasMoreElements(); ) { 
494 	    Rule tdtrule = (Rule) e.nextElement();
495 	    if (tdtrule.getType() == ModeList.EXTRACT) {
496 		assert seq < tdtrule.getSeq() : "Rule out of sequence order";
497 		seq = tdtrule.getSeq();
498 		processRules(extraparams, tdtrule);
499 	    }
500 	}
501 
502 
503 			
504 	/**     Now we need to consider the corresponding output level
505 	 *	and output option.  The scheme must remain the same,
506 	 *	as must the value of optionkey (to select the
507 	 *	corresponding option element nested within the
508 	 *	required outbound level)
509 	 */
510 						
511 	Level tdtoutlevel = findLevel(tdtscheme, outboundlevel);
512 	Option tdtoutoption = findOption(tdtoutlevel, optionValue);
513 			
514 	/**	the FORMAT rules are performed before formatting the
515 	 *	output, in order to determine additional fields that
516 	 *	are required for preparation of the outbound format
517 	 */
518 			
519 	seq = 0;
520 	for (Enumeration e = tdtoutlevel.enumerateRule(); e.hasMoreElements(); ) { 
521 	    Rule tdtrule = (Rule) e.nextElement();
522 	    if (tdtrule.getType() == ModeList.FORMAT) {
523 		assert seq < tdtrule.getSeq() : "Rule out of sequence order";
524 		seq = tdtrule.getSeq();
525 		processRules(extraparams, tdtrule);
526 	    }
527 	}
528 			
529 	/**     Now we need to ensure that all fields required for the
530 	 *      outbound grammar are suitably padded etc.
531 	 *      processPadding takes care of firstly padding the
532 	 *      non-binary fields if padChar and padDir, length are
533 	 *      specified then (if necessary) converting to binary and
534 	 *      padding the binary representation to the left with
535 	 *      zeros if the bit string is has fewer bits than the
536 	 *      bitLength attribute specifies.  N.B. TDTv1.1 will be
537 	 *      more specific about bit-level padding rather than
538 	 *      assuming that it is always to the left with the zero
539 	 *      bit.
540 	 */
541 	
542 	//System.out.println(" prior to processPadding, " + extraparams);
543 	for (Enumeration e = tdtoutoption.enumerateField(); e.hasMoreElements(); ) {
544 	    Field field = (Field) e.nextElement();
545 	    //processPadding(extraparams, field, outboundlevel, tdtoutoption);
546 
547 	    padField(extraparams, field);
548 	    if (outboundlevel == LevelTypeList.BINARY)
549 		binaryPadding(extraparams, field);
550 
551 
552 	}
553 						
554 			
555 	/**    Construct the output from the specified grammar (in
556 	 *     ABNF format) together with the field values stored in
557 	 *     inputparams
558 	 */
559 			
560 	outboundstring = buildGrammar(tdtoutoption.getGrammar(), extraparams);
561 
562 	//System.out.println("final extraparams = " + extraparams);
563 	//System.out.println("returned " + outboundstring);
564 	return outboundstring;
565     }

605     /**
606      *	Returns a string built using a particular grammar.
607      *	Single-quotes strings are counted as literal strings, whereas
608      *	all other strings appearing in the grammar require
609      *	substitution with the corresponding value from the extraparams
610      *	hashmap.
611      */
612     private String buildGrammar(String grammar, Map<String, String> extraparams) {
613 	StringBuilder outboundstring = new StringBuilder();
614 	String[] fields = Pattern.compile("\\s+").split(grammar);
615 	for (int i = 0; i < fields.length; i++) {
616 	    if (fields[i].substring(0, 1).equals("'")) {
617 		outboundstring.append(fields[i].substring(1, fields[i].length()-1));
618 	    } else {
619 		outboundstring.append(extraparams.get(fields[i]));
620 	    }
621 	}
622 		
623 	return outboundstring.toString();
624     }

626     /**
627      *	
628      *	Converts the value of a specified fieldname from the
629      *	extraparams map into binary, either handling it as a large
630      *	integer or taking into account the compaction of each ASCII
631      *	byte that is specified in the TDT definition file for that
632      *	particular field
633      */
634     private String fieldToBinary(Field field, Map<String, String> extraparams) {
635 	// really need an index to find field number given fieldname;
636
