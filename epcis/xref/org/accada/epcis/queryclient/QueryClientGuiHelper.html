95      public String updateEndpointAddress(final String address) {
96          String endpointAddress;
97          if (address == null) {
98              Properties props = loadProperties();
99              endpointAddress = props.getProperty(PROP_QUERY_URL, DEFAULT_QUERY_URL);
100         } else {
101             endpointAddress = address;
102         }
103 
104         // check if the endpointAddress is valid
105         try {
106             new URL(endpointAddress);
107         } catch (Exception e) {
108             System.out.println("Invalid endpoint address provided. Using default: " + DEFAULT_QUERY_URL);
109             endpointAddress = DEFAULT_QUERY_URL;
110         }
111         setEndpointAddress(endpointAddress);
112         return endpointAddress;
113     }

135     private Properties loadProperties() {
136         InputStream is = getClass().getResourceAsStream(PROPERTY_FILE);
137         Properties props = new Properties();
138         try {

140             is.close();

148     /**
149      * Converts the values in a calendar object into a nicely formatted string.
150      * 
151      * @param cal
152      *            with the Calendar-Date
153      * @return String
154      */
155     private String prettyStringCalendar(final Calendar cal) {
156         if (cal == null) {
157             return null;
158         }
159         // set to current timezone
160         cal.setTimeZone(TimeZone.getDefault());
161         return TimeParser.format(cal);
162     }

398     /**
399      * Run the query with the currently set query arguments Returns the results
400      * in a format that is suitable for JTable.
401      * 
402      * @return The pretty-printed query results.
403      * @throws Exception
404      *             If any Exception occurred while invoking the query service.
405      */
406     public Object[][] runQuery() throws Exception {
407         QueryParams queryParams = new QueryParams();
408         queryParams.getParam().addAll(internalQueryParams);
409         debugTextArea.append("Number of query parameters: " + queryParams.getParam().size() + "\n");
410         for (QueryParam queryParam : internalQueryParams) {
411             debugTextArea.append(queryParam.getName() + " " + queryParam.getValue() + "\n");
412         }
413 
414         Poll poll = new Poll();
415         poll.setQueryName("SimpleEventQuery");
416         poll.setParams(queryParams);
417 
418         debugTextArea.append("running query...\n");
419         QueryResults results = queryClient.poll(poll);
420         debugTextArea.append("done\n");
421 
422         // print to debug window and return result
423         if (results != null && results.getResultsBody() != null && results.getResultsBody().getEventList() != null) {
424             return processEvents(results.getResultsBody().getEventList().getObjectEventOrAggregationEventOrQuantityEvent());
425         } else {
426             return null;
427         }
428     }

431      * Registers the Query by the server.
432      * 
433      * @param subscribe
434      *            The Subscribe object containing the query.
435      * @throws Exception
436      *             If any Exception occurred while invoking the query service.
437      */
438     public void subscribeQuery(final Subscribe subscribe) throws Exception {
439         QueryParams queryParams = new QueryParams();
440         queryParams.getParam().addAll(internalQueryParams);
441         debugTextArea.append("Number of query parameters: " + queryParams.getParam().size() + "\n");
442         for (QueryParam queryParam : internalQueryParams) {
443             debugTextArea.append(queryParam.getName() + " " + queryParam.getValue() + "\n");
444         }
445         subscribe.setParams(queryParams);
446         queryClient.subscribe(subscribe.getQueryName(), subscribe.getParams(), subscribe.getDest(),
447                 subscribe.getControls(), subscribe.getSubscriptionID());
448     }

450     /**
451      * Removes a registersQuery by the server.
452      * 
453      * @param subscriptionID
454      *            The ID of the query to be unsubscribed.
455      */
456     public void unsubscribeQuery(final String subscriptionID) {
457         try {
458             JFrame frame = new JFrame();
459             if (subscriptionID.equals("")) {
460                 JOptionPane.showMessageDialog(frame, "Please specify a SubscriptionID", "Service is responding",
461                         JOptionPane.ERROR_MESSAGE);
462                 return;
463             }
464             Unsubscribe parms = new Unsubscribe();
465             parms.setSubscriptionID(subscriptionID);
466             queryClient.unsubscribe(parms.getSubscriptionID());
467             JOptionPane.showMessageDialog(frame, "Successfully unsubscribed.", "Service is responding",
468                     JOptionPane.INFORMATION_MESSAGE);
469         } catch (Exception e) {
470             JFrame frame = new JFrame();
471             StringWriter sw = new StringWriter();
472             e.printStackTrace(new PrintWriter(sw));
473             String stacktrace = sw.toString();
474             JOptionPane.showMessageDialog(frame, "Sorry, the Service returned an Error:\n" + stacktrace,
475                     "Service is not responding", JOptionPane.ERROR_MESSAGE);
476             e.printStackTrace();
477         }

492     /**
493      * Query the service for subscriptions.
494      */
495     public void querySubscriptionIDs() {
496         String title = "Service is responding";
497         StringBuilder msg = new StringBuilder();
498         try {
499             GetSubscriptionIDs parms = new GetSubscriptionIDs();
500             parms.setQueryName("SimpleEventQuery");
501             List<String> subscriptionIDs = queryClient.getSubscriptionIds(parms.getQueryName());
502             if (subscriptionIDs != null && !subscriptionIDs.isEmpty()) {
503                 msg.append("The following subscription IDs were found in the repository:\n");
504                 for (String s : subscriptionIDs) {
505                     msg.append("- ").append(s).append("\n");
506                 }
507             } else {
508                 msg.append("There are no subscribed queries.");
509             }
510         } catch (Exception e) {
511             title = "Service is not responding";
512             msg.append("Could not retrieve subscription IDs from repository.");
513             e.printStackTrace();
514         }
515         JFrame frame = new JFrame();
516         JOptionPane.showMessageDialog(frame, msg.toString(), title, JOptionPane.INFORMATION_MESSAGE);
517     }
