915     /**
916      * The user pushed the Generate event-button. This method converts the data
917      * from the user interface to XML, POSTs it to the server and displays the
918      * answer to the user. It also does some simple client-side checks to see if
919      * all necessary fields are filled.
920      */
921     private void mwGenerateEventButtonPressed() {
922         dwOutputTextArea.setText("");
923         /* used later for user interaction */
924         JFrame frame = new JFrame();
925 
926         try {
927             /* DOM-tree stuff */
928             Document document = null;
929             DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
930             DocumentBuilder builder = factory.newDocumentBuilder();
931             DOMImplementation impl = builder.getDOMImplementation();
932 
933             /* temporary element variable */
934             Element element = null;
935 
936             /* create empty document and fetch root */
937             document = impl.createDocument("urn:epcglobal:epcis:xsd:1", "epcis:EPCISDocument", null);
938             Element root = document.getDocumentElement();
939 
940             root.setAttribute("creationDate", format(Calendar.getInstance()));
941             root.setAttribute("xmlns:xsi", "http://www.w3.org/2001/XMLSchema-instance");
942             root.setAttribute("xmlns:epcis", "urn:epcglobal:epcis:xsd:1");
943             root.setAttribute("schemaVersion", "1.0");
944             element = document.createElement("EPCISBody");
945             root.appendChild(element);
946             root = element;
947             element = document.createElement("EventList");
948             root.appendChild(element);
949             root = element;
950             element = document.createElement(xmlEventNode[mwEventTypeChooserComboBox.getSelectedIndex()]);
951             root.appendChild(element);
952             root = element;
953 
954             // eventTime
955             if (!addEventTime(document, root)) {
956                 JOptionPane.showMessageDialog(frame, "Please specify the event time "
957                         + "(e.g. 2005-07-18T17:33:20.231Z)", "Error", JOptionPane.ERROR_MESSAGE);
958                 return;
959             }
960 
961             // eventTimeZoneOffset
962             if (!addEventTimeZoneOffset(document, root)) {
963                 JOptionPane.showMessageDialog(frame, "Please specify the event timezone offset "
964                         + "(e.g. +00:00 or -06:30)", "Error", JOptionPane.ERROR_MESSAGE);
965                 return;
966             }
967 
968             // recordTime is set by the capture-Interface
969 
970             if (xmlEventNode[mwEventTypeChooserComboBox.getSelectedIndex()].equals("ObjectEvent")) {
971                 if (!addEpcList(document, root)) {
972                     JOptionPane.showMessageDialog(frame, "Please specify at least one EPC", "Error",
973                             JOptionPane.ERROR_MESSAGE);
974                     return;
975                 }
976                 addAction(document, root);
977                 addBizStep(document, root);
978                 addDisposition(document, root);
979                 addReadPoint(document, root);
980                 addBizLocation(document, root);
981                 addBizTransactionList(document, root);
982             } else if (xmlEventNode[mwEventTypeChooserComboBox.getSelectedIndex()].equals("AggregationEvent")) {
983                 if (!addParentId(document, root) && !mwActionComboBox.getSelectedItem().equals("OBSERVE")) {
984                     JOptionPane.showMessageDialog(frame, "Because action is OBSERVE, it's required to "
985                             + "specify a ParentID", "Error", JOptionPane.ERROR_MESSAGE);
986                     return;
987                 }
988                 if (!addChildEpcList(document, root)) {
989                     JOptionPane.showMessageDialog(frame, "Please specify at least one EPC", "Error",
990                             JOptionPane.ERROR_MESSAGE);
991                     return;
992                 }
993                 addAction(document, root);
994                 addBizStep(document, root);
995                 addDisposition(document, root);
996                 addReadPoint(document, root);
997                 addBizLocation(document, root);
998                 addBizTransactionList(document, root);
999             } else if (xmlEventNode[mwEventTypeChooserComboBox.getSelectedIndex()].equals("QuantityEvent")) {
1000                 if (!addEpcClass(document, root)) {
1001                     JOptionPane.showMessageDialog(frame, "Please specify an EPC class (URI)", "Error",
1002                             JOptionPane.ERROR_MESSAGE);
1003                     return;
1004                 }
1005                 if (!addQuantity(document, root)) {
1006                     JOptionPane.showMessageDialog(frame, "Please specify a quantity value (integer number)", "Error",
1007                             JOptionPane.ERROR_MESSAGE);
1008                     return;
1009                 }
1010                 addBizStep(document, root);
1011                 addDisposition(document, root);
1012                 addReadPoint(document, root);
1013                 addBizLocation(document, root);
1014                 addBizTransactionList(document, root);
1015             } else if (xmlEventNode[mwEventTypeChooserComboBox.getSelectedIndex()].equals("TransactionEvent")) {
1016                 if (!addBizTransactionList(document, root)) {
1017                     JOptionPane.showMessageDialog(frame, "Please specify at least one business "
1018                             + "transaction (ID, Type)", "Error", JOptionPane.ERROR_MESSAGE);
1019                     return;
1020                 }
1021                 addParentId(document, root);
1022                 if (!addEpcList(document, root)) {
1023                     JOptionPane.showMessageDialog(frame, "Please specify at least one EPC", "Error",
1024                             JOptionPane.ERROR_MESSAGE);
1025                     return;
1026                 }
1027                 addAction(document, root);
1028                 addBizStep(document, root);
1029                 addDisposition(document, root);
1030                 addReadPoint(document, root);
1031                 addBizLocation(document, root);
1032             }
1033 
1034             DOMSource domsrc = new DOMSource(document);
1035 
1036             StringWriter out = new StringWriter();
1037             StreamResult streamResult = new StreamResult(out);
1038             TransformerFactory tf = TransformerFactory.newInstance();
1039             Transformer serializer = tf.newTransformer();
1040 
1041             serializer.setOutputProperty(OutputKeys.INDENT, "yes");
1042             serializer.transform(domsrc, streamResult);
1043 
1044             String eventXml = out.toString();
1045             String postData = eventXml;
1046 
1047             dwOutputTextArea.append("sending HTTP POST data:\n");
1048             dwOutputTextArea.append(postData);
1049 
1050             /* connect the service, write out xml and get response */
1051             int response = client.capture(postData);
1052 
1053             if (response == 200) {
1054                 JOptionPane.showMessageDialog(frame, "Capture request succeeded.", "Success",
1055                         JOptionPane.INFORMATION_MESSAGE);
1056             } else {
1057                 JOptionPane.showMessageDialog(frame, "Capture request failed (HTTP response code " + response + ").",
1058                         "Error", JOptionPane.ERROR_MESSAGE);
1059             }
1060 
1061         } catch (MalformedURLException mfue) {
1062             JOptionPane.showMessageDialog(frame, mfue.getMessage(), "Error", JOptionPane.ERROR_MESSAGE);
1063             StringWriter detailed = new StringWriter();
1064             PrintWriter pw = new PrintWriter(detailed);
1065             mfue.printStackTrace(pw);
1066             dwOutputTextArea.append(detailed.toString());
1067         } catch (IOException ioe) {
1068             JOptionPane.showMessageDialog(frame, ioe.getMessage(), "Error", JOptionPane.ERROR_MESSAGE);
1069             StringWriter detailed = new StringWriter();
1070             PrintWriter pw = new PrintWriter(detailed);
1071             ioe.printStackTrace(pw);
1072             dwOutputTextArea.append(detailed.toString());
1073         } catch (TransformerException te) {
1074             JOptionPane.showMessageDialog(frame, te.getMessage(), "Error", JOptionPane.ERROR_MESSAGE);
1075             StringWriter detailed = new StringWriter();
1076             PrintWriter pw = new PrintWriter(detailed);
1077             te.printStackTrace(pw);
1078             dwOutputTextArea.append(detailed.toString());
1079         } catch (ParserConfigurationException pce) {
1080             JOptionPane.showMessageDialog(frame, pce.getMessage(), "Error", JOptionPane.ERROR_MESSAGE);
1081             StringWriter detailed = new StringWriter();
1082             PrintWriter pw = new PrintWriter(detailed);
1083             pce.printStackTrace(pw);
1084             dwOutputTextArea.append(detailed.toString());
1085         } catch (Exception e) {
1086             JOptionPane.showMessageDialog(frame, e.getMessage(), "Error", JOptionPane.ERROR_MESSAGE);
1087             StringWriter detailed = new StringWriter();
1088             PrintWriter pw = new PrintWriter(detailed);
1089             e.printStackTrace(pw);
1090             dwOutputTextArea.append(detailed.toString());
1091         }

1767     /**
1768      * Formats a <code>Calendar</code> value into an ISO8601-compliant
1769      * date/time string.
1770      * 
1771      * @param cal
1772      *            The time value to be formatted into a date/time string.
1773      * @return The formatted date/time string.
1774      */
1775     private static String format(final Calendar cal) {
1776         if (cal == null) {
1777             throw new IllegalArgumentException("argument can not be null");
1778         }
1779 
1780         // determine era and adjust year if necessary
1781         int year = cal.get(Calendar.YEAR);
1782         if (cal.isSet(Calendar.ERA) && cal.get(Calendar.ERA) == GregorianCalendar.BC) {
1783             /**
1784              * calculate year using astronomical system: year n BCE =>
1785              * astronomical year -n + 1
1786              */
1787             year = 0 - year + 1;
1788         }
1789 
1790         /**
1791          * the format of the date/time string is: YYYY-MM-DDThh:mm:ss.SSSTZD
1792          * note that we cannot use java.text.SimpleDateFormat for formatting
1793          * because it can't handle years <= 0 and TZD's
1794          */
1795         StringBuilder buf = new StringBuilder();
1796         // year ([-]YYYY)
1797         buf.append(XXXX_FORMAT.format(year));
1798         buf.append('-');
1799         // month (MM)
1800         buf.append(XX_FORMAT.format(cal.get(Calendar.MONTH) + 1));
1801         buf.append('-');
1802         // day (DD)
1803         buf.append(XX_FORMAT.format(cal.get(Calendar.DAY_OF_MONTH)));
1804         buf.append('T');
1805         // hour (hh)
1806         buf.append(XX_FORMAT.format(cal.get(Calendar.HOUR_OF_DAY)));
