130             final Boolean reportIfEmpty, final GregorianCalendar initialRecordTime,
131             final GregorianCalendar lastTimeExecuted, final String queryName) {
132         LOG.debug("Constructing Query Subscription with ID '" + subscriptionID + "'");
133         this.queryParams = queryParams;
134         this.subscriptionID = subscriptionID;
135         this.dest = dest;
136         this.initialRecordTime = initialRecordTime;
137         this.reportIfEmpty = reportIfEmpty;
138         this.queryName = queryName;
139         this.lastTimeExecuted = lastTimeExecuted;

141         // update/add GE_recordTime restriction to query params (we only need to
142         // return results not previously returned!)
143         updateRecordTime(queryParams, initialRecordTime);

146     /**
147      * Updates the subscription in the database. This is required in order to
148      * correctly re-initialize the subscriptions, especially the
149      * lastTimeExecuted field, after a context restart.
150      * <p>
151      * TODO: This is a back-end method: move this method to the
152      * QueryOperationsBackend and delegate to it (thus we would need a reference
153      * to the QueryOperationsBackend in this class).
154      * 
155      * @param lastTimeExecuted
156      *            The new lastTimeExecuted.
157      */
158     private void updateSubscription(final GregorianCalendar lastTimeExecuted) {
159         String jndiName = getProperties().getProperty("jndi.datasource.name", "java:comp/env/jdbc/EPCISDB");
160         try {
161             // open a database connection
162             Context ctx = new InitialContext();
163             DataSource db = (DataSource) ctx.lookup(jndiName);
164             Connection dbconnection = db.getConnection();
165 
166             // update the subscription in the database
167             String update = "UPDATE subscription SET lastexecuted=(?), params=(?)" + " WHERE subscriptionid=(?);";
168             PreparedStatement stmt = dbconnection.prepareStatement(update);
169             LOG.debug("SQL: " + update);
170             Timestamp ts = new Timestamp(lastTimeExecuted.getTimeInMillis());
171             String time = ts.toString();
172             stmt.setString(1, time);
173             LOG.debug("       query param 1: " + time);
174             ByteArrayOutputStream outStream = new ByteArrayOutputStream();
175             ObjectOutput out = new ObjectOutputStream(outStream);
176             out.writeObject(queryParams);
177             ByteArrayInputStream inStream = new ByteArrayInputStream(outStream.toByteArray());
178             stmt.setBinaryStream(2, inStream, inStream.available());
179             LOG.debug("       query param 2: [" + inStream.available() + " bytes]");
180             stmt.setString(3, subscriptionID);
181             LOG.debug("       query param 3: " + subscriptionID);
182             stmt.executeUpdate();
183             dbconnection.commit();
184 
185             // close the database connection
186             dbconnection.close();
187         } catch (SQLException e) {
188             String msg = "An SQL error occurred while updating the subscriptions in the database.";
189             LOG.error(msg, e);
190         } catch (IOException e) {
191             String msg = "Unable to update the subscription in the database: " + e.getMessage();
192             LOG.error(msg, e);
193         } catch (NamingException e) {
194             String msg = "Unable to find JNDI data source with name " + jndiName;
195             LOG.error(msg, e);
196         }
197     }

199     /**
200      * Updates or adds the 'GE_recordTime' query parameter in the given query
201      * parameter array and sets its value to the given time.
202      * 
203      * @param queryParams
204      *            The (old) query parameter array.
205      * @param initialRecordTime
206      *            The time to which the 'GE_recordTime' parameter will be
207      *            updated.
208      */
209     private void updateRecordTime(final QueryParams queryParams, final GregorianCalendar initialRecordTime) {
210         // update or add GE_recordTime restriction
211         boolean foundRecordTime = false;
212         for (QueryParam p : this.queryParams.getParam()) {
213             if (p.getName().equalsIgnoreCase("GE_recordTime")) {
214                 LOG.debug("Updating query parameter 'GE_recordTime' with value '" + initialRecordTime.getTime() + "'.");
215                 p.setValue(initialRecordTime);
216                 foundRecordTime = true;
217                 break;
218             }
219         }
220         if (!foundRecordTime) {
221             LOG.debug("Adding query parameter 'GE_recordTime' with value '" + initialRecordTime.getTime() + "'.");
222             QueryParam newParam = new QueryParam();
223             newParam.setName("GE_recordTime");
224             newParam.setValue(initialRecordTime);
225             this.queryParams.getParam().add(newParam);
226         }
227 
228         // update the subscription in the db
229         updateSubscription(initialRecordTime);
230     }

232     /**
233      * Runs the query assigned to this subscription. Advances lastTimeExecuted.
234      */
235     public void executeQuery() {
236         if (LOG.isDebugEnabled()) {
237             LOG.debug("--------------------------------------------");
238             LOG.debug("Executing subscribed query '" + subscriptionID + "' with " + queryParams.getParam().size()
239                     + " parameters:");
240             for (QueryParam p : queryParams.getParam()) {
241                 LOG.debug(" param name:  " + p.getName());
242                 Object val = p.getValue();
243                 if (val instanceof GregorianCalendar) {
244                     LOG.debug(" param value: " + ((GregorianCalendar) val).getTime());
245                 } else {
246                     LOG.debug(" param value: " + val);
247                 }
248             }
249         }
250 
251         // poll the query
252         Poll poll = new Poll();
253         poll.setQueryName(queryName);
254         poll.setParams(queryParams);
255         QueryResults result = null;
256         try {
257             // get current time and send the query
258             GregorianCalendar cal = new GregorianCalendar();
259             result = executePoll(poll);
260             LOG.debug("Subscribed query '" + subscriptionID + "' has been executed");
261 
262             // set new lastTimeExecuted (must be <= to time when query is
263             // executed, otherwise we loose results)
264             cal.add(Calendar.SECOND, 1);
265             this.lastTimeExecuted = cal;
266         } catch (QueryTooLargeExceptionResponse e) {
267             // send exception back to client
268             QueryTooLargeException qtle = e.getFaultInfo();
269             if (qtle == null) {
270                 qtle = new QueryTooLargeException();
271                 qtle.setQueryName(queryName);
272                 qtle.setSubscriptionID(subscriptionID);
273                 qtle.setReason(e.getMessage());
274                 LOG.info("USER ERROR: " + qtle.getReason());
275             }
276             callbackQueryTooLargeException(qtle);
277             return;
278         } catch (ImplementationExceptionResponse e) {
279             // send exception back to client
280             ImplementationException ie = e.getFaultInfo();
281             if (ie == null) {
282                 ie = new ImplementationException();
283                 ie.setQueryName(queryName);
284                 ie.setReason(e.getMessage());
285                 ie.setSubscriptionID(subscriptionID);
286                 LOG.info("USER ERROR: " + ie.getReason());
287             }
288             callbackImplementationException(ie);
289             return;
290         } catch (Exception e) {
291             String msg = "An unexpected error occurred while executing a subscribed query";
292             LOG.error(msg + ": " + e.getMessage(), e);
293             // send exception back to client
294             ImplementationException ie = new ImplementationException();
295             ie.setQueryName(queryName);
296             ie.setReason(msg);
297             ie.setSubscriptionID(subscriptionID);
298             callbackImplementationException(ie);
299             return;
300         }
301         result.setSubscriptionID(subscriptionID);
302         EventListType eventList = result.getResultsBody().getEventList();
303 
304         // check if we have an empty result list
305         boolean isEmpty = false;
306         isEmpty = (eventList == null) ? true : eventList.getObjectEventOrAggregationEventOrQuantityEvent().isEmpty();
307         if (!reportIfEmpty.booleanValue() && isEmpty) {
308             LOG.debug("Subscribed query '" + subscriptionID + "' returned no results, nothing to report.");
309             return;
310         }
311 
312         callbackResults(result);
313 
314         // update query params with new lastTimeExecuted
315         updateRecordTime(queryParams, lastTimeExecuted);
316     }

318     /**
319      * Poll a query using local transport.
320      */
321     protected QueryResults executePoll(Poll poll) throws ImplementationExceptionResponse,
322             QueryTooComplexExceptionResponse, QueryTooLargeExceptionResponse, SecurityExceptionResponse,
323             ValidationExceptionResponse, NoSuchNameExceptionResponse, QueryParameterExceptionResponse {
324         // we use CXF's local transport feature here
325         EPCglobalEPCISService service = new EPCglobalEPCISService();
326         QName portName = new QName("urn:epcglobal:epcis:wsdl:1", "EPCglobalEPCISServicePortLocal");
327         service.addPort(portName, "http://schemas.xmlsoap.org/soap/", "local://query");
328         EPCISServicePortType servicePort = service.getPort(portName, EPCISServicePortType.class);
329 
330         // the same using CXF API
331         // JaxWsProxyFactoryBean factory = new JaxWsProxyFactoryBean();
332         // factory.setAddress("local://query");
333         // factory.setServiceClass(EPCISServicePortType.class);
334         // EPCISServicePortType servicePort = (EPCISServicePortType)
335         // factory.create();
336 
337         return servicePort.poll(poll);
338     }

361     /**
362      * Serializes and sends the given object back to the client. The Object must
363      * be an instance of QueryResults, QueryTooLargeException, or
364      * ImplementationException.
365      * 
366      * @param o
367      *            The object to be sent back to the client. An instance of
368      *            QueryResults, QueryTooLargeException, or
369      *            ImplementationException.
370      */
371     private void callbackObject(final Object o) {
372         if (LOG.isDebugEnabled()) {
373             LOG.debug("Callback " + o + " at " + new Date());
374         }
375         // create the EPCIS document
376         EPCISQueryDocumentType epcisDoc = new EPCISQueryDocumentType();
377         epcisDoc.setSchemaVersion(BigDecimal.valueOf(1.0));
378         try {
379             DatatypeFactory dataFactory = DatatypeFactory.newInstance();
380             XMLGregorianCalendar now = dataFactory.newXMLGregorianCalendar(new GregorianCalendar());
381             epcisDoc.setCreationDate(now);
382         } catch (DatatypeConfigurationException e) {
383             // oh well - don't care about setting the creation date
384         }
385         EPCISQueryBodyType epcisBody = new EPCISQueryBodyType();
386         if (o instanceof QueryResults) {
387             epcisBody.setQueryResults((QueryResults) o);
388         } else if (o instanceof QueryTooLargeException) {
389             epcisBody.setQueryTooLargeException((QueryTooLargeException) o);
390         } else if (o instanceof ImplementationException) {
391             epcisBody.setImplementationException((ImplementationException) o);
392         } else {
393             epcisBody = null;
394         }
395         epcisDoc.setEPCISBody(epcisBody);
396 
397         // serialize the response
398         String data;
399         try {
400             data = marshalQueryDoc(epcisDoc);
401         } catch (JAXBException e) {
402             String msg = "An error serializing contents occurred: " + e.getMessage();
403             LOG.error(msg, e);
404             return;
405         }
406 
407         // set up connection and send data to given destination
408         try {
409             URL serviceUrl = new URL(dest.toString());
410             if (LOG.isDebugEnabled()) {
411                 LOG.debug("Sending results of subscribed query '" + subscriptionID + "' to '" + serviceUrl + "'");
412                 if (data.length() < 10 * 1024) {
413                     LOG.debug("Sending data:\n" + data);
414                 } else {
415                     LOG.debug("Sending data: [" + data.length() + " bytes]");
416                 }
417             }
418             int responseCode;
419             try {
420                 responseCode = sendData(serviceUrl, data);
421             } catch (Exception e) {
422                 LOG.warn("Unable to send results of subscribed query '" + subscriptionID + "' to '" + serviceUrl
423                         + "', retrying in 3 sec ...");
424                 // wait 3 seconds and try again
425                 try {
426                     Thread.sleep(3000);
427                 } catch (InterruptedException e1) {
428                     // never mind
429                 }
430                 try {
431                     responseCode = sendData(serviceUrl, data);
432                 } catch (Exception e2) {
433                     LOG.warn("Unable to send results of subscribed query '" + subscriptionID + "' to '" + serviceUrl
434                             + "', retrying in 3 sec ...");
435                     // wait 3 seconds and try again
436                     try {
437                         Thread.sleep(3000);
438                     } catch (InterruptedException e1) {
439                         // never mind
440                     }
441                     responseCode = sendData(serviceUrl, data);
442                 }
443             }
444             LOG.debug("Response " + responseCode);
445         } catch (IOException e) {
446             String msg = "Unable to send results of subscribed query '" + subscriptionID + "' to '" + dest + "': "
447                     + e.getMessage();
448             LOG.error(msg, e);
449             return;
450         }
451     }

454      * Marshals the given EPCIS query document into it's XML representation.
455      * 
456      * @param epcisDoc
457      *            The EPCISQueryDocumentType to marshal.
458      * @return The marshaled EPCISQueryDocumentType XML String.
459      */
460     private String marshalQueryDoc(EPCISQueryDocumentType epcisDoc) throws JAXBException {
461         ObjectFactory objectFactory = new ObjectFactory();
462         JAXBContext context = JAXBContext.newInstance("org.fosstrak.epcis.model");
463         JAXBElement<EPCISQueryDocumentType> item = objectFactory.createEPCISQueryDocument(epcisDoc);
464         LOG.debug("Serializing " + item + " into XML");
465         StringWriter writer = new StringWriter();
466         Marshaller marshaller = context.createMarshaller();
467         marshaller.setProperty(Marshaller.JAXB_ENCODING, "UTF-8");
468         marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
469         marshaller.marshal(item, writer);
470         return writer.toString();
471     }

473     /**
474      * Sends the given data String to the specified URL.
475      * 
476      * @param url
477      *            The URL to send the data to.
478      * @param data
479      *            The data to send.
480      * @return The HTTP response code.
481      * @throws IOException
482      *             If a communication error occurred.
483      */
484     private int sendData(final URL url, final String data) throws IOException {
485         HttpURLConnection connection;
486         if ("HTTPS".equalsIgnoreCase(url.getProtocol()) && trustAllCertificates()) {
487             connection = getAllTrustingConnection(url);
488         } else {
489             connection = getConnection(url);
490         }
491         connection.setRequestMethod("POST");
492         connection.setRequestProperty("content-type", "text/xml");
493         connection.setRequestProperty("content-length", "" + data.length());
494         connection.setDoOutput(true);
495         connection.setDoInput(true);
496 
497         // send data
498         Writer out = new OutputStreamWriter(connection.getOutputStream());
499         out.write(data);
500         out.flush();
501         out.close();
502 
503         // get response code
504         int responseCode = connection.getResponseCode();
505 
506         // disconnect and return
507         connection.disconnect();
508         return responseCode;
509     }

537     /**
538      * Retrieves an "all-trusting" HTTP URL connection object, by disabling the
539      * validation of certificates and overriding the default trust manager with
540      * one that trusts all certificates.
541      * 
542      * @param url
543      *            The URL on which a connection will be opened.
544      * @return A HttpURLConnection connection object.
545      * @throws IOException
546      *             If an I/O error occurred.
547      */
548     private HttpURLConnection getAllTrustingConnection(URL url) throws IOException {
549         // Create a trust manager that does not validate certificate chains
550         TrustManager[] trustAllCerts = new TrustManager[] { new X509TrustManager() {
551             public java.security.cert.X509Certificate[] getAcceptedIssuers() {
552                 return null;
553             }
554 
555             public void checkClientTrusted(java.security.cert.X509Certificate[] certs, String authType) {
556             }
557 
558             public void checkServerTrusted(java.security.cert.X509Certificate[] certs, String authType) {
559             }
560         } };
561 
562         // Install the all-trusting trust manager
563         try {
564             SSLContext sc = SSLContext.getInstance("SSL");
565             sc.init(null, trustAllCerts, new java.security.SecureRandom());
566             HttpsURLConnection.setDefaultSSLSocketFactory(sc.getSocketFactory());
567         } catch (Exception e) {
568             LOG.error("Unable to install the all-trusting trust manager", e);
569         }
570         return getConnection(url);
571     }
