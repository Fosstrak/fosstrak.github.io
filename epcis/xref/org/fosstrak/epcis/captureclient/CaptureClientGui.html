975             root.setAttribute("xmlns:xsi", "http://www.w3.org/2001/XMLSchema-instance");
976             root.setAttribute("xmlns:epcis", "urn:epcglobal:epcis:xsd:1");
977             root.setAttribute("schemaVersion", "1.0");

979             root.appendChild(element);
980             root = element;
981             element = document.createElement("EventList");
982             root.appendChild(element);
983             root = element;
984             element = document.createElement(xmlEventNode[mwEventTypeChooserComboBox.getSelectedIndex()]);
985             root.appendChild(element);
986             root = element;
987 
988             // eventTime
989             if (!addEventTime(document, root)) {
990                 JOptionPane.showMessageDialog(frame, "Please specify the event time "
991                         + "(e.g. 2005-07-18T17:33:20.231Z)", "Error", JOptionPane.ERROR_MESSAGE);
992                 return;
993             }
994 
995             // eventTimeZoneOffset
996             if (!addEventTimeZoneOffset(document, root)) {
997                 JOptionPane.showMessageDialog(frame, "Please specify the event timezone offset "
998                         + "(e.g. +00:00 or -06:30)", "Error", JOptionPane.ERROR_MESSAGE);
999                 return;

1003

1006                     JOptionPane.showMessageDialog(frame, "Please specify at least one EPC", "Error",

1014                 addBizLocation(document, root);
1015                 addBizTransactionList(document, root);
1016             } else if (xmlEventNode[mwEventTypeChooserComboBox.getSelectedIndex()].equals("AggregationEvent")) {
1017                 if (!addParentId(document, root) && !mwActionComboBox.getSelectedItem().equals("OBSERVE")) {
1018                     JOptionPane.showMessageDialog(frame, "Because action is OBSERVE, it's required to "
1019                             + "specify a ParentID", "Error", JOptionPane.ERROR_MESSAGE);
1020                     return;
1021                 }
1022                 if (!addChildEpcList(document, root)) {
1023                     JOptionPane.showMessageDialog(frame, "Please specify at least one EPC", "Error",
1024                             JOptionPane.ERROR_MESSAGE);
1025                     return;
1026                 }
1027                 addAction(document, root);
1028                 addBizStep(document, root);
1029                 addDisposition(document, root);
1030                 addReadPoint(document, root);
1031                 addBizLocation(document, root);
1032                 addBizTransactionList(document, root);
1033             } else if (xmlEventNode[mwEventTypeChooserComboBox.getSelectedIndex()].equals("QuantityEvent")) {
1034                 if (!addEpcClass(document, root)) {

1036                             JOptionPane.ERROR_MESSAGE);

1041                             JOptionPane.ERROR_MESSAGE);
1042                     return;
1043                 }
1044                 addBizStep(document, root);
1045                 addDisposition(document, root);
1046                 addReadPoint(document, root);
1047                 addBizLocation(document, root);
1048                 addBizTransactionList(document, root);
1049             } else if (xmlEventNode[mwEventTypeChooserComboBox.getSelectedIndex()].equals("TransactionEvent")) {
1050                 if (!addBizTransactionList(document, root)) {

1054                 }
1055                 addParentId(document, root);
1056                 if (!addEpcList(document, root)) {
1057                     JOptionPane.showMessageDialog(frame, "Please specify at least one EPC", "Error",
1058                             JOptionPane.ERROR_MESSAGE);
1059                     return;
1060                 }
1061                 addAction(document, root);
1062                 addBizStep(document, root);
1063                 addDisposition(document, root);
1064                 addReadPoint(document, root);
1065                 addBizLocation(document, root);
1066             }
1067 
1068             DOMSource domsrc = new DOMSource(document);
1069 
1070             StringWriter out = new StringWriter();
1071             StreamResult streamResult = new StreamResult(out);
1072             TransformerFactory tf = TransformerFactory.newInstance();
1073             Transformer serializer = tf.newTransformer();
1074 
1075             serializer.setOutputProperty(OutputKeys.INDENT, "yes");
1076             serializer.transform(domsrc, streamResult);

1080

1082             dwOutputTextArea.append(postData);

1089                         JOptionPane.INFORMATION_MESSAGE);

1092                         "Error", JOptionPane.ERROR_MESSAGE);

1765     /**
1766      * Formats a <code>Calendar</code> value into an ISO8601-compliant
1767      * date/time string.
1768      * 
1769      * @param cal
1770      *            The time value to be formatted into a date/time string.
1771      * @return The formatted date/time string.
1772      */
1773     private static String format(final Calendar cal) {
1774         if (cal == null) {
1775             throw new IllegalArgumentException("argument can not be null");
1776         }
1777 
1778         // determine era and adjust year if necessary
1779         int year = cal.get(Calendar.YEAR);
1780         if (cal.isSet(Calendar.ERA) && cal.get(Calendar.ERA) == GregorianCalendar.BC) {
1781             /**
1782              * calculate year using astronomical system: year n BCE =>
1783              * astronomical year -n + 1
1784              */
1785             year = 0 - year + 1;
1786         }
1787 
1788         /**
1789          * the format of the date/time string is: YYYY-MM-DDThh:mm:ss.SSSTZD
1790          * note that we cannot use java.text.SimpleDateFormat for formatting
1791          * because it can't handle years <= 0 and TZD's
1792          */
1793         StringBuilder buf = new StringBuilder();
1794         // year ([-]YYYY)
1795         buf.append(XXXX_FORMAT.format(year));
1796         buf.append('-');
1797         // month (MM)
1798         buf.append(XX_FORMAT.format(cal.get(Calendar.MONTH) + 1));
1799         buf.append('-');
1800         // day (DD)
1801         buf.append(XX_FORMAT.format(cal.get(Calendar.DAY_OF_MONTH)));
1802         buf.append('T');
1803         // hour (hh)
1804         buf.append(XX_FORMAT.format(cal.get(Calendar.HOUR_OF_DAY)));
1805         buf.append(':');
1806         // minute (mm)
1807         buf.append(XX_FORMAT.format(cal.get(Calendar.MINUTE)));
1808         buf.append(':');
1809         // second (ss)
1810         buf.append(XX_FORMAT.format(cal.get(Calendar.SECOND)));
1811         buf.append('.');
1812         // millisecond (SSS)
1813         buf.append(XXX_FORMAT.format(cal.get(Calendar.MILLISECOND)));
1814         // time zone designator (+/-hh:mm)
1815         buf.append(getTimeZone(cal));
1816         return buf.toString();
1817     }

1819     /**
1820      * Returns the time zone designator in a ISO6601-compliant format from the
1821      * given <code>Calendar</code> value.
1822      * 
1823      * @param cal
1824      *            The Calendar to be formatted.
1825      * @return The time zone designator from the given Calendar.
1826      */
1827     private static String getTimeZone(final Calendar cal) {
1828         StringBuilder buf = new StringBuilder();
1829         TimeZone tz = cal.getTimeZone();
1830         // determine offset of timezone from UTC (incl. daylight saving)
1831         int offset = tz.getOffset(cal.getTimeInMillis());
