96       * @param url The URL at which the capture service listens.
97       * @param authOptions The authentication options:
98       *            <p>
99       *            <table border="1">
100      *            <tr>
101      *            <td><code>authenticationOptions[0]</code></td>
102      *            <td><code>[1]</code></td>

104      *            </tr>

114      *            </tr>
115      *            </table>

117     public CaptureClient(final String url, Object[] authOptions) {
118     	// set the URL
119     	if (url != null) {
120     		captureUrl = url;
121     	} else {
122     		Properties props = loadProperties();
123     		if (props != null) {
124     			captureUrl = props.getProperty(PROPERTY_CAPTURE_URL);
125     		}
126     		if (captureUrl == null) {
127     			captureUrl = DEFAULT_CAPTURE_URL;
128     		}
129     	}
130     	this.authOptions = authOptions;
131     }

136     private Properties loadProperties() {
137         Properties props = new Properties();
138         InputStream is = getClass().getResourceAsStream(PROPERTY_FILE);
139         if (is != null) {
140 	        try {
141 	            props.load(is);
142 	            is.close();
143 	        } catch (IOException e) {
144 	            System.out.println("Unable to load properties from "
145 	                    + PROPERTY_FILE + ". Using defaults.");
146 	        }
147         }
148         else {
149             System.out.println("Unable to load properties from file "
150                     + PROPERTY_FILE + ". Using defaults.");
151         }
152         return props;

184     /**
185      * Send the given EPCISDocumentType to the repository for capturing.
186      * 
187      * @param epcisDoc
188      *            The EPCISDocument containing a list of events inside the
189      *            EPCISBody element.
190      * @return The HTTP response code from the repository.
191      * @throws IOException
192      *             If an error sending the document occurred.
193      * @throws JAXBException
194      *             If an error serializing the given document into XML occurred.
195      */
196     public int capture(final EPCISDocumentType epcisDoc) throws Exception {
197         StringWriter writer = new StringWriter();
198         ObjectFactory objectFactory = new ObjectFactory();
199         JAXBContext context = JAXBContext.newInstance("org.fosstrak.epcis.model");
200         JAXBElement<EPCISDocumentType> item = objectFactory.createEPCISDocument(epcisDoc);
201         Marshaller marshaller = context.createMarshaller();
202         marshaller.setProperty(Marshaller.JAXB_ENCODING, "UTF-8");
203         marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
204         marshaller.marshal(item, writer);
205         return capture(writer.toString());
206     }

227     /**
228      * Opens a connection to the capture module.
229      * 
230      * @param contentType
231      *            The HTTP content-type, e.g., <code>text/xml</code>
232      * @return The HTTP connection object.
233      * @throws IOException
234      */
235     private HttpURLConnection getConnection(final String contentType) throws Exception {
236 
237     	URL serviceUrl = new URL(captureUrl);
238     	HttpURLConnection connection;
239     	SSLContext sslContext = null;
240     	
241     	if (authOptions != null) {
242 
243     		if (AuthenticationType.BASIC.equals(authOptions[0])) {
244 
245     			// logger.debug("Authenticating via Basic as: " +
246     			// authenticationOptions[1]);
247 
248     			final String username = (String)authOptions[1];
249     			final String password = (String)authOptions[2];
250     			
251     			if (isEmpty(username) || isEmpty(password)) {
252     				throw new Exception("Authentication method " + authOptions[0] + " requires a valid user name and password");
253     			}
254 
255     			Authenticator.setDefault(new Authenticator() {
256     				@Override
257     				protected PasswordAuthentication getPasswordAuthentication() {
258     					return new PasswordAuthentication(username, password.toCharArray());
259     				}
260     			});
261     			
262     		}
263     		else if (AuthenticationType.HTTPS_WITH_CLIENT_CERT.equals(authOptions[0])) {
264 
265     			// logger.debug("Authenticating with certificate in file: " +
266     			// authenticationOptions[1]);
267 
268     			if (!"HTTPS".equalsIgnoreCase(serviceUrl.getProtocol())) {
269     				throw new Exception("Authentication method " + authOptions[0] + " requires the use of HTTPS");
270     			}
271 
272     			String keyStoreFile = (String)authOptions[1];
273     			String password = (String)authOptions[2];
274     			
275     			if (isEmpty(keyStoreFile) || isEmpty(password)) {
276     				throw new Exception("Authentication method " + authOptions[0] + " requires a valid keystore (PKCS12 or JKS) and password");
277     			}
278 
279     			KeyStore keyStore = KeyStore.getInstance(keyStoreFile.endsWith(".p12") ? "PKCS12" : "JKS");
280     			keyStore.load(new FileInputStream(new File(keyStoreFile)), password.toCharArray());
281 
282     			Authenticator.setDefault(null);
283     	    	sslContext = getSSLContext(keyStore, password.toCharArray());
284     	    	
285     		}
286     		else {
287     			Authenticator.setDefault(null);
288     		}
289     	}
290     	else {
291     		Authenticator.setDefault(null);
292     	}
293 
294     	connection = (HttpURLConnection)serviceUrl.openConnection();
295     	if (sslContext != null && connection instanceof HttpsURLConnection) {
296     		HttpsURLConnection httpsConnection = (HttpsURLConnection)connection;
297     		httpsConnection.setHostnameVerifier(this);
298 			httpsConnection.setSSLSocketFactory(sslContext.getSocketFactory());
299     	}
300     	connection.setRequestProperty("content-type", contentType);
301     	connection.setRequestMethod("POST");
302     	connection.setDoInput(true);
303     	connection.setDoOutput(true);
304     	
305     	return connection;
306     }

308     /**
309      * Send data to the repository's capture operation using HTTP POST. The data
310      * will be sent using the given content-type.
311      * 
312      * @param data
313      *            The data to send.
314      * @return The HTTP response message
315      * @throws IOException
316      *             If an error on the HTTP layer occurred.
317      */
318     private int doPost(final String data, final String contentType) throws Exception {
319         HttpURLConnection connection = getConnection(contentType);
320         // write the data
321         OutputStreamWriter wr = new OutputStreamWriter(connection.getOutputStream());
322         wr.write(data);
323         wr.flush();
324         wr.close();
325 
326         return connection.getResponseCode();
327     }

329     /**
330      * Send data to the repository's capture operation using HTTP POST. The data
331      * will be sent using the given content-type.
332      * 
333      * @param data
334      *            The data to send.
335      * @return The HTTP response message from the repository.
336      * @throws IOException
337      *             If an error on the HTTP layer occurred.
338      */
339     private int doPost(final InputStream data, final String contentType) throws Exception {
340         HttpURLConnection connection = getConnection(contentType);
341         // read from input and write to output
342         OutputStream os = connection.getOutputStream();
343         int b;
344         while ((b = data.read()) != -1) {
345             os.write(b);
346         }
347         os.flush();
348         os.close();
349 
350         return connection.getResponseCode();
351     }

386 	private SSLContext getSSLContext(KeyStore keyStore, char[] password) throws Exception {
387 		KeyManagerFactory keyManagerFactory = KeyManagerFactory.getInstance("SunX509");
388 		keyManagerFactory.init(keyStore, password);
389 		SSLContext context = SSLContext.getInstance("TLS");
390 		context.init(keyManagerFactory.getKeyManagers(), new TrustManager[] { this }, new SecureRandom());
391 		return context;
