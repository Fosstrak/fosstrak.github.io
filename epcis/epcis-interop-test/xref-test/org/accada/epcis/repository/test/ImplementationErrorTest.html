37  /**
38   * Tests for some ImplementationException with severity SEVERE (SE50). Note: to
39   * get an implementation exception, for example mysql instance must be shut
40   * down.
41   * 
42   * @author Marco Steybe
43   */
44  public class ImplementationErrorTest extends TestCase {
45  
46      private static final String PATH = "src/test/resources/queries/webservice/requests/";
47  
48      private QueryControlClient client = new QueryControlClient();
49  
50      /**
51       * No testing, just print a message that reminds that the setup for an
52       * ImplementationException must be given.
53       */
54      public void testSetup() {
55          // the easiest (and maybe currently only) way to test for an
56          // ImplementationException is when an EPC is not in URI format.
57          System.out.println("SETUP: modify the URI of an ObjectEvent EPC in the DB so that it is not valid anymore!");
58      }
59  
60      /**
61       * Tests if ImplementationException is raised.
62       * 
63       * @throws ServiceException
64       *             If an error in the EPCIS query service occured.
65       * @throws IOException
66       *             If an I/O error occured.
67       */
68      public void testSE51() throws IOException, ServiceException {
69          final String query = "Test-EPCIS10-SE51-Request-1-Poll.xml";
70          InputStream fis = new FileInputStream(PATH + query);
71          try {
72              client.poll(fis);
73              fis.close();
74              fail("ImplementationException expected");
75          } catch (ImplementationException e) {
76              // ok
77              fis.close();
78              assertEquals(e.getSeverity(),
79                      ImplementationExceptionSeverity.ERROR);
80          }
81      }
82  
83      /**
84       * Tests if ImplementationException is raised (callback).
85       * 
86       * @throws ServiceException
87       *             If an error in the EPCIS query service occured.
88       * @throws IOException
89       *             If an I/O error occured.
90       */
91      public void testSE69() throws IOException, ServiceException {
92          // subscribe query
93          final String query = "Test-EPCIS10-SE69-Request-1-Subscribe.xml";
94          InputStream fis = new FileInputStream(PATH + query);
95          client.subscribe(fis);
96          fis.close();
97  
98          // start subscription response listener
99          QueryCallbackListener listener = QueryCallbackListener.getInstance();
100         if (!listener.isRunning()) {
101             listener.start();
102         }
103         System.out.println("waiting ...");
104         synchronized (listener) {
105             try {
106                 listener.wait(60000);
107             } catch (InterruptedException e) {
108                 e.printStackTrace();
109             }
110         }
111         String resp = listener.fetchResponse();
112         assertNotNull(resp);
113 
114         client.unsubscribe("QuerySE69"); // clean up
115         assertTrue(resp.contains("ImplementationException"));
116     }
117 
118     /**
119      * {@inheritDoc}
120      */
121     protected void tearDown() throws Exception {
122         // make sure the query is unsubscribed!
123         try {
124             client.unsubscribe("QuerySE69");
125         } catch (NoSuchSubscriptionException e) {
126         }
127     }
