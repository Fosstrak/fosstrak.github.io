92      /**
93       * A helper method to parse and convert the XML representation of an EPCIS
94       * query results into a QueryResults object.
95       * 
96       * @param xmlQueryResults
97       *            The InputStream containing the XML representation of a
98       *            QueryResults object.
99       * @return The parsed QueryResults object.
100      */
101     public static QueryResults parseQueryResults(
102             final InputStream xmlQueryResults) {
103         QueryResults queryResults = null;
104 
105         DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
106         Document epcisq;
107         try {
108             DocumentBuilder builder = factory.newDocumentBuilder();
109             epcisq = builder.parse(xmlQueryResults);
110         } catch (Exception e) {
111             throw new RuntimeException("Unable to parse the XML query.", e);
112         }
113         Element queryName = (Element) epcisq.getElementsByTagName("queryName").item(
114                 0);
115 
116         if (!queryName.getTextContent().equals("SimpleEventQuery")) {
117             Element vocList = (Element) epcisq.getElementsByTagName(
118                     "VocabularyList").item(0);
119             VocabularyType[] vocs = null;
120             if (vocList != null) {
121                 NodeList vocsList = vocList.getElementsByTagName("Vocabulary");
122                 if (vocsList != null && vocsList.getLength() != 0) {
123                     vocs = handleVocabularies(vocsList);
124                 }
125             }
126 
127             EventListType eventList = null;
128             QueryResultsBody queryResultsBody = new QueryResultsBody(eventList,
129                     vocs);
130 
131             // no extensions implemented!
132             QueryResultsExtensionType queryResultsExtension = null;
133 
134             String queryNam = queryName.getTextContent();
135             String subscriptionId = null;
136             MessageElement[] message = null;
137             queryResults = new QueryResults(queryNam, subscriptionId,
138                     queryResultsBody, queryResultsExtension, message);
139         } else {
140             Element eventList = (Element) epcisq.getElementsByTagName(
141                     "EventList").item(0);
142             ObjectEventType[] objectEvents = null;
143             AggregationEventType[] aggrEvents = null;
144             QuantityEventType[] quantEvents = null;
145             TransactionEventType[] transEvents = null;
146             if (eventList != null) {
147                 NodeList objectEventsList = eventList.getElementsByTagName("ObjectEvent");
148                 if (objectEventsList != null
149                         && objectEventsList.getLength() != 0) {
150                     objectEvents = handleObjectEvents(objectEventsList);
151                 }
152 
153                 NodeList aggregationEventsList = eventList.getElementsByTagName("AggregationEvent");
154                 if (aggregationEventsList != null
155                         && aggregationEventsList.getLength() != 0) {
156                     aggrEvents = handleAggregationEvents(aggregationEventsList);
157                 }
158 
159                 NodeList quantityEventsList = eventList.getElementsByTagName("QuantityEvent");
160                 if (quantityEventsList != null
161                         && quantityEventsList.getLength() != 0) {
162                     quantEvents = handleQuantityEvents(quantityEventsList);
163                 }
164 
165                 NodeList transactionEventsList = eventList.getElementsByTagName("TransactionEvent");
166                 if (transactionEventsList != null
167                         && transactionEventsList.getLength() != 0) {
168                     transEvents = handleTransactionEvents(transactionEventsList);
169                 }
170             }
171 
172             EPCISEventListExtensionType epcisEventList = null;
173             MessageElement[] message = null;
174             EventListType eventListType = new EventListType(objectEvents,
175                     aggrEvents, quantEvents, transEvents, epcisEventList,
176                     message);
177 
178             VocabularyType[] vocabulary = null;
179             QueryResultsBody queryResultsBody = new QueryResultsBody(
180                     eventListType, vocabulary);
181 
182             // no extensions implemented!
183             QueryResultsExtensionType queryResultsExtension = null;
184 
185             String queryNam = queryName.getTextContent();
186             String subscriptionId = null;
187             queryResults = new QueryResults(queryNam, subscriptionId,
188                     queryResultsBody, queryResultsExtension, message);
189         }
190         return queryResults;
191     }

193     /**
194      * Parses the object event inside the QueryResults event list.
195      * 
196      * @param objectEventList
197      *            The NodeList parsed from the XML containing the object events.
198      * @return An array of ObjectEventType.
199      */
200     private static ObjectEventType[] handleObjectEvents(
201             final NodeList objectEventList) {
202         Vector<ObjectEventType> list = new Vector<ObjectEventType>();
203 
204         for (int i = 0; i < objectEventList.getLength(); i++) {
205             Element objectEvent = (Element) objectEventList.item(i);
206 
207             // parse <eventTime>
208             Node eventTimeNode = objectEvent.getElementsByTagName("eventTime").item(
209                     0);
210             Calendar eventTime = handleTime(eventTimeNode);
211 
212             // parse <recordTime>
213             Node recordTimeNode = objectEvent.getElementsByTagName("recordTime").item(
214                     0);
215             Calendar recordTime = handleTime(recordTimeNode);
216 
217             // parse <eventTimeZoneOffset>
218             Node eventTimeZoneOffsetNode = objectEvent.getElementsByTagName(
219                     "eventTimeZoneOffset").item(0);
220             String eventTimeZoneOffset = eventTimeZoneOffsetNode.getTextContent();
221 
222             // TODO parse extension
223             EPCISEventExtensionType baseExtension = null;
224 
225             // TODO parse message element
226             List<MessageElement> messages = new ArrayList<MessageElement>();
227             Node temperature = objectEvent.getElementsByTagName(
228                     "hls:temperature").item(0);
229             if (temperature != null) {
230                 MessageElement me = new MessageElement("temperature", "hls",
231                         "http://schema.hls.com/extension");
232                 me.setValue(temperature.getTextContent());
233                 messages.add(me);
234             }
235             Node batchNumber = objectEvent.getElementsByTagName(
236                     "hls:batchNumber").item(0);
237             if (batchNumber != null) {
238                 MessageElement me = new MessageElement("batchNumber", "hls",
239                         "http://schema.hls.com/extension");
240                 me.setValue(batchNumber.getTextContent());
241                 messages.add(me);
242             }
243             MessageElement[] message = new MessageElement[messages.size()];
244             message = messages.toArray(message);
245 
246             // parse <epcList>
247             Node epcListNode = objectEvent.getElementsByTagName("epcList").item(
248                     0);
249             EPC[] epcList = handleEpcList(epcListNode);
250 
251             // parse <action>
252             Node actionNode = objectEvent.getElementsByTagName("action").item(0);
253             ActionType action = handleAction(actionNode);
254 
255             // parse <bizStep>
256             Node bizStepNode = objectEvent.getElementsByTagName("bizStep").item(
257                     0);
258             URI bizStep = handleUri(bizStepNode);
259 
260             // parse <disposition>
261             Node dispNode = objectEvent.getElementsByTagName("disposition").item(
262                     0);
263             URI disposition = handleUri(dispNode);
264 
265             // parse <readPoint>
266             Node readPointNode = objectEvent.getElementsByTagName("readPoint").item(
267                     0);
268             ReadPointType readPoint = null;
269             if (readPointNode != null) {
270                 Element readPointElement = (Element) readPointNode;
271                 Node idNode = readPointElement.getElementsByTagName("id").item(
272                         0);
273                 URI id = handleUri(idNode);
274                 readPoint = new ReadPointType(id, null, null);
275             }
276 
277             // parse <bizLocation>
278             Node bizLocNode = objectEvent.getElementsByTagName("bizLocation").item(
279                     0);
280             BusinessLocationType bizLocation = null;
281             if (bizLocNode != null) {
282                 Element bizLocElement = (Element) bizLocNode;
283                 Node idNode = bizLocElement.getElementsByTagName("id").item(0);
284                 URI id = handleUri(idNode);
285                 bizLocation = new BusinessLocationType(id, null, null);
286             }
287 
288             // parse <bizTransactionList>
289             Node bizTransListNode = objectEvent.getElementsByTagName(
290                     "bizTransactionList").item(0);
291             BusinessTransactionType[] bizTransList = handleBizTransList(bizTransListNode);
292 
293             // TODO parse extension
294             ObjectEventExtensionType extension = null;
295 
296             ObjectEventType objectEventType = new ObjectEventType(eventTime,
297                     recordTime, eventTimeZoneOffset, baseExtension, epcList,
298                     action, bizStep, disposition, readPoint, bizLocation,
299                     bizTransList, extension, message);
300             list.add(objectEventType);
301         }
302         ObjectEventType[] objectEvent = new ObjectEventType[list.size()];
303         return list.toArray(objectEvent);
304     }

613     /**
614      * Parses the vocabularies of an event.
615      * 
616      * @param vocList
617      *            The NodeList parsed from the XML containing the vocabularies .
618      * @return An array of VocabularyType.
619      */
620     private static VocabularyType[] handleVocabularies(final NodeList vocList) {
621         List<VocabularyType> vocabularies = new ArrayList<VocabularyType>();
622 
623         for (int i = 0; i < vocList.getLength(); i++) {
624             Element vocElem = (Element) vocList.item(i);
625 
626             // get 'type' attribute
627             Node vocTypeNode = vocElem.getAttributeNode("type");
628             URI vocType = handleUri(vocTypeNode);
629 
630             // parse <VocabularyElement>
631             List<VocabularyElementType> vocElementList = new ArrayList<VocabularyElementType>();
632             NodeList vocElementNodeList = vocElem.getElementsByTagName("VocabularyElement");
633             for (int j = 0; j < vocElementNodeList.getLength(); j++) {
634                 Element vocElementElem = (Element) vocElementNodeList.item(j);
635 
636                 Node vocIdNode = vocElementElem.getAttributeNode("id");
637                 URI vocId = handleUri(vocIdNode);
638 
639                 // parse <attribute>
640                 List<AttributeType> attrList = new ArrayList<AttributeType>();
641                 NodeList attrNodeList = vocElementElem.getElementsByTagName("attribute");
642                 for (int k = 0; k < attrNodeList.getLength(); k++) {
643                     Element attrElem = (Element) attrNodeList.item(k);
644 
645                     Node attrIdNode = attrElem.getAttributeNode("id");
646                     URI attrId = handleUri(attrIdNode);
647 
648                     String attrStringVal = attrElem.getTextContent();
649                     MessageElement[] attrVal = new MessageElement[] {
650                         new MessageElement(new Text(attrStringVal))
651                     };
652 
653                     AttributeType attr = new AttributeType();
654                     attr.setId(attrId);
655                     attr.set_any(attrVal);
656                     attrList.add(attr);
657                 }
658                 AttributeType[] attrs = null;
659                 if (attrList.size() > 0) {
660                     attrs = new AttributeType[attrList.size()];
661                     attrs = attrList.toArray(attrs);
662                 }
