79       *             If an I/O error occured.

81      public void testSE53() throws IOException, ServiceException {
82          // subscribe first query
83          final String query = "Test-EPCIS10-SE53-Request-1-Subscribe.xml";
84          InputStream fis = new FileInputStream(PATH + query);
85          client.subscribe(fis);
86          fis.close();
87  
88          // subscribe second query
89          final String query2 = "Test-EPCIS10-SE53-Request-2-Subscribe.xml";
90          fis = new FileInputStream(PATH + query2);
91          try {
92              client.subscribe(fis);
93              // fail
94              fis.close();
95              client.unsubscribe("QuerySE53"); // clean up
96              fail("DuplicateSubscriptionException expected");
97          } catch (DuplicateSubscriptionException e) {
98              fis.close();
99              client.unsubscribe("QuerySE53"); // clean up
100             assertEquals(
101                     "SubscriptionID 'QuerySE53' already exists. Choose a different subscriptionID.",
102                     e.getReason());
103         }
104     }
