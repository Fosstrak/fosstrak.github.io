230     /**
231      * Formats a <code>Calendar</code> value into an ISO8601-compliant
232      * date/time string. This method is copied from
233      * org.fosstrak.epcis.utils.TimeParser (module epcis-commons).
234      * 
235      * @see org.fosstrak.epcis.utils.TimeParser#format(Calendar)
236      * @param cal
237      *            The time value to be formatted into a date/time string.
238      * @return The formatted date/time string.
239      */
240     private static String format(final Calendar cal) {
241         if (cal == null) {
242             throw new IllegalArgumentException("argument can not be null");
243         }
244 
245         // determine era and adjust year if necessary
246         int year = cal.get(Calendar.YEAR);
247         if (cal.isSet(Calendar.ERA) && cal.get(Calendar.ERA) == GregorianCalendar.BC) {
248             /**
249              * calculate year using astronomical system: year n BCE =>
250              * astronomical year -n + 1
251              */
252             year = 0 - year + 1;
253         }
254 
255         /**
256          * the format of the date/time string is: YYYY-MM-DDThh:mm:ss.SSSTZD
257          * note that we cannot use java.text.SimpleDateFormat for formatting
258          * because it can't handle years <= 0 and TZD's
259          */
260         StringBuilder buf = new StringBuilder();
261         // year ([-]YYYY)
262         buf.append(XXXX_FORMAT.format(year));
263         buf.append('-');
264         // month (MM)
265         buf.append(XX_FORMAT.format(cal.get(Calendar.MONTH) + 1));
266         buf.append('-');
267         // day (DD)
268         buf.append(XX_FORMAT.format(cal.get(Calendar.DAY_OF_MONTH)));
269         buf.append('T');
270         // hour (hh)
271         buf.append(XX_FORMAT.format(cal.get(Calendar.HOUR_OF_DAY)));
272         buf.append(':');
273         // minute (mm)
274         buf.append(XX_FORMAT.format(cal.get(Calendar.MINUTE)));
275         buf.append(':');
276         // second (ss)
277         buf.append(XX_FORMAT.format(cal.get(Calendar.SECOND)));
278         buf.append('.');
279         // millisecond (SSS)
280         buf.append(XXX_FORMAT.format(cal.get(Calendar.MILLISECOND)));
281         // time zone designator (+/-hh:mm)
282         buf.append(getTimeZone(cal));
283         return buf.toString();
284     }

286     /**
287      * Returns the time zone designator in a ISO6601-compliant format from the
288      * given <code>Calendar</code> value.
289      * 
290      * @param cal
291      *            The Calendar to be formatted.
292      * @return The time zone designator from the given Calendar.
293      */
294     private static String getTimeZone(final Calendar cal) {
295         StringBuilder buf = new StringBuilder();
296         TimeZone tz = cal.getTimeZone();
297         // determine offset of timezone from UTC (incl. daylight saving)
298         int offset = tz.getOffset(cal.getTimeInMillis());
299         int hours = Math.abs((offset / (60 * 1000)) / 60);
300         int minutes = Math.abs((offset / (60 * 1000)) % 60);
301         buf.append(offset < 0 ? '-' : '+');
302         buf.append(XX_FORMAT.format(hours));
303         buf.append(':');
304         buf.append(XX_FORMAT.format(minutes));
305         return buf.toString();
306     }
