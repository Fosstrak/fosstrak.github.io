34  /**
35   * TCPIPProtocol is an interface to an RFID-Reader. It allows to send a request
36   * to the reader. The sendRequest method then locks until a response from the
37   * reader has arrived or a timeout occured.
38   * When terminating the session with the reader one should close the socket with
39   * the close method.
40   *
41   * @author Jonas Haller
42   */

105     * Sends a request Frame to the reader and returns the response Frame.
106     * This method method locks until a response from the reader has arrived or
107     * a timeout occured.
108     *
109     * @param frame
110     *          the Frame to be sent to the reader
111     * @return
112     *          the response from the reader as Frame
113     * @throws IOException
114     *          in case of a timeout or any other io exception.
115     */
116    public synchronized Frame sendRequest(Frame frame)
117          throws IOException {
118 
119       // create byte array from Frame
120       byte[] request = frame.getFrame();
121 
122       // send request and receive response
123       byte[] response = sendRequestBytes(request);
124 
125       // create Frame from byte array
126       Frame responseFrame;
127       try {
128          responseFrame = new Frame(response);
129       } catch (FrameException e) {
130          throw new IOException(e.getMessage());
131       }
132 
133       return responseFrame;
134    }

235    /**
236     * Receives a specific notification Frame from the reader and returns the
237     * Frame.
238     * This method method locks until a notification from the reader has arrived
239     * or a timeout occured.
240     *
241     * @return
242     *          the notification from the reader as Frame
243     * @throws IOException
244     *          in case of a timeout or any other io exception.
245     */
246    public synchronized Frame receiveNotification(Category aCategory,
247          int aMessageID) throws IOException {
248 
249       // check message ID
250       if ((aMessageID < 0x00) || (aMessageID > 0xFF)) {
251          throw new IOException("Can not receive notification, MessageID out of"
252                + " range [0x00, 0xFF]");
253       }
254 
255       // total timeout and start time
256       long totalTimeout = timeout;
257       long t0 = System.currentTimeMillis();
258       long time = 0;
259 
260       byte[] notification = null;
261       try {
262          do {
263             // handle omitted notifications
264             if (notification != null) {
265                handleUnusedFrame(notification);
266             }
267 
268             // set timeout to remaining time
269             time = System.currentTimeMillis() - t0;
270             timeout = totalTimeout - time;
271 
272             // receive notification
273             notification = receiveNotificationBytes();
274          } while (((notification[1] & 0x0F) != aCategory.toInt()) ||
275                ((notification[2] & 0xFF) != aMessageID));
276       } catch (IOException ioe) {
277          timeout = totalTimeout;
278          throw ioe;
279       }
280       timeout = totalTimeout;
281 
282       // create Frame from byte array
283       Frame notificationFrame;
284       try {
285          notificationFrame = new Frame(notification);
286       } catch (FrameException e) {
287          throw new IOException(e.getMessage());
288       }
289 
290       return notificationFrame;
291    }

293    /**
294     * Sends a request on the socket and returns the response.
295     * This method method locks until a response from the reader has arrived or
296     * a timeout occured.
297     *
298     * @param request
299     *          the frame to send as byte array
300     * @return
301     *          the response from the reader as byte array
302     * @throws IOException
303     *            in case of a timeout or any other io exception
304     */
305 	private synchronized byte[] sendRequestBytes(byte[] request)
306          throws IOException {
307 
308 		while (sendReceiveActive) {
309 			try {
310 				wait();
311 			} catch (InterruptedException e) {
312 			}
313 		}
314 		sendReceiveActive = true;
315 
316 		// send request
317 		out.write(request);
318 
319       // for debugging only...
320 		// log.debug(" sent frame: " +
321 		// ByteBlock.byteArrayToHexString(request));
322 
323       // receive response frame (omit notification frames)
324       long totalTimeout = timeout;
325       long t0 = System.currentTimeMillis();
326       long time = 0;
327 
328       byte[] response = null;
329       try {
330          do {
331             // handle omitted frames (notifications)
332             if (response != null) {
333                handleUnusedFrame(response);
334             }
335 
336             // set timeout to remaining time
337             time = System.currentTimeMillis() - t0;
338             timeout = totalTimeout - time;
339 
340             // receive frame bytes
341             response = receiveFrameBytes();
342          } while ((response[1] & 0x20) != 0x00);
343       } catch (IOException ioe) {
344          timeout = totalTimeout;
345          throw ioe;
346       }
347       timeout = totalTimeout;
348 
349 
350       if (((response[1] & 0x2F) != (request[1] & 0x0F)) ||
351             (response[2] != request[2])) {
352          throw new IOException ("Can not receive response.");
353       }
354 
355 		//  for debugging only...
356       // log.debug(" received frame: " +
357       // ByteBlock.byteArrayToHexString(response));
358 
359 		// allow another send
360 		sendReceiveActive = false;
361 		notifyAll();
362 		return response;
363 	}

367     * This method method locks until a notification from the reader has arrived or
368     * a timeout occured.

371     *          the notification from the reader as byte array

375    private synchronized byte[] receiveNotificationBytes()
376          throws IOException {
377 
378       while (sendReceiveActive) {
379          try {
380             wait();
381          } catch (InterruptedException e) {
382          }
383       }
384       sendReceiveActive = true;
385 
386       // receive frame
387       byte[] notification = receiveFrameBytes();
388       if ((notification[1] & 0x20) == 0x00) {
389          throw new IOException ("Can not receive notification.");
390       }
391 
392       //  for debugging only...
393       // log.debug(" received frame: " +
394       // ByteBlock.byteArrayToHexString(response));
395 
396       // allow another send
397       sendReceiveActive = false;
398       notifyAll();
399       return notification;

402 	/**
403 	 * This method receives a frame from the reader (specific to Impinj Frame
404     * format).
405     *
406     * @return
407     *          the frame from the reader as byte array
408     * @throws IOException
409     *          in case of a timeout or any other io exception
410 	 */
411 	private synchronized byte[] receiveFrameBytes() throws IOException {
412 
413 		byte[] buf = null;
414       final int InitBytes = 5; // (length from bytes 1,3 and 4)
415 
416 		// receiving data...
417       // wait for at least InitBytes bytes of response
418       long t0 = System.currentTimeMillis();
419       long time = 0;
420       while (in.available() < InitBytes) {
421          try { Thread.sleep(10); } catch (InterruptedException ie) {}
422          time = System.currentTimeMillis() - t0;
423          if (time >= timeout) {
424             throw new IOException("Timeout occured, no response for "
425                   + timeout + "ms.");
426          }
427       }
428 
429 		// read first InitBytes bytes
430 		byte[] initBytesArray = new byte[InitBytes];
431 		int stat = in.read(initBytesArray, 0, InitBytes);
432 		if (stat != InitBytes) {
433 			throw new IOException("No data available.");
434 		}
435 
436 		// get length of response frame
437       int msb = ByteBlock.byteToNumber(initBytesArray[3]) & 0x03;
438       int lsb = ByteBlock.byteToNumber(initBytesArray[4]);
439       int headerlen = 6;
440       int tslen = 0;
441       if ((initBytesArray[1] & 0x40) > 0) {
442          tslen = 8;
443       }
444 		int len = (msb << 8) + lsb + headerlen + tslen;
445 
446       // wait for (len - InitBytes) bytes (rest of response)
447       while (in.available() < (len - InitBytes)) {
448          try { Thread.sleep(10); } catch (InterruptedException ie) {}
449          time = System.currentTimeMillis() - t0;
450          if (time >= timeout) {
451             throw new IOException("Timeout occured, response not completely "
452                   + "arrived for " + timeout + "ms.");
453          }
454       }
455 
456       // read (len - InitBytes) bytes
457       buf = new byte[len - InitBytes];
458       stat = in.read(buf, 0, len - InitBytes);
459       if (stat != (len - InitBytes)) {
460          throw new IOException("No data available.");
461       }
462 
463       // compose init bytes and rest of frame
464       byte[] frame = new byte[len];
465       System.arraycopy(initBytesArray, 0, frame, 0, InitBytes);
466       System.arraycopy(buf, 0, frame, InitBytes, len - InitBytes);
467 
468 		// verify checksum
469       byte crc = CRC8.crc8(frame);
470       if (crc != 0x00) {
471          throw new IOException("CRC error in response.");
472       }
473 
474       return frame;
475 	}

477    /**
478     * Handles frames that are not used otherwise (logging).
479     *
480     * @param framebytes
481     *          the frame to handle
482     */
483    public void handleUnusedFrame(byte[] framebytes) {
484       String framestring = ByteBlock.byteArrayToHexString(framebytes);
485       Frame frame;
486       try {
487          frame = new Frame(framebytes);
488       } catch (FrameException e) {
489          log.info("Unable to pack unused frame: " + framestring);
490          return;
491       }
492 
493       if (frame.getType() == Type.NOTIFICATION) {
494          if (frame.getCategory() == Category.OPERATING_MODEL) {
495             switch (frame.getMessageID()) {
496             case 0x01:
497             case 0x02:
498                // InventoryNtf or InventoryStatusNtf get omitted on stopping
499                // inventory, do not log these notifications.
500                break;
501             default:
502                log.info("Received unused Operating Command Set notification: " +
503                      framestring);
504             }
505          } else if (frame.getCategory() == Category.MODEM_MANAGEMENT) {
506             switch (frame.getMessageID()) {
507             case 0x00:
508                log.info("Received SocketConnectionStatus notification, an " +
509                      "other client tried to connect to the reader.");
510                break;
511             case 0x01:
512                log.info("Received SystemError notification: " + framestring);
513                break;
514             default:
515                log.info("Received unused Management Command Set notification: " +
516                      framestring);
517                break;
518             }
519          } else {
520             log.info("Received unused notification: " + framestring);
521          }
522       }
523       if (frame.getType() == Type.RESPONSE) {
524          log.info("Received unused response: " + framestring);
525       }
526    }
