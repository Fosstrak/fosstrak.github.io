147     private Feed getFeedInRightFormat(String urlId, String urlSize, int format) {
148         String id   = urlId;
149         int    size = Config.DEFAULT_FEED_SIZE;
150 
151         try {
152             size = Integer.parseInt(urlSize);
153         } catch (Exception ex) {
154             ex.printStackTrace();
155 
156             // continue with default size
157         }
158 
159         Feed   feed   = null;
160         Abdera abdera = AtomPub.getAbdera();
161 
162         feed = abdera.getFactory().newFeed();
163         feed.setTitle("Unspecified Feed");
164         feed.setSubtitle("");
165         feed.addAuthor("");
166 
167         boolean isFeedExisting = false;
168 
169         for (String feedID : FEED_DB.getFeedIDs()) {
170             if (feedID.equals(id)) {
171                 isFeedExisting = true;
172             }
173         }
174 
175         if (isFeedExisting) {
176             FeedBean feedBean = FEED_DB.getFeed(id);
177 
178             feed.setId(id);
179             feed.setTitle(feedBean.getTitle());
180             feed.setSubtitle(feedBean.getSubtitle());
181             feed.addAuthor(feedBean.getAuthor());
182             feed.addCategory(feedBean.getCategory());
183             feed.addContributor(feedBean.getContributor());
184             feed.addLink(feedBean.getLink());
185             feed.setLogo(feedBean.getLogo());
186 
187             Date myUpdatedDate = new Date();
188 
189             feed.setUpdated(myUpdatedDate);
190 
191             int entrySize = java.lang.Math.min(size, feedBean.getEntries().size());
192 
193             for (int i = 0; i < entrySize; i++) {
194 
195                 // for (EntryBean e : feedBean.getEntries()) {
196                 EntryBean entryBean = feedBean.getEntries().get(i);
197                 Entry     entry     = feed.addEntry();
198 
199                 entry.setId(entryBean.getId());
200                 entry.setTitle(entryBean.getTitle());
201 
202                 if (format == HUMAN_READABLE) {
203                     entry.setSummaryAsHtml(entryBean.getSummary());
204                 }
205 
206                 if (format == MACHINE_READABLE) {
207                     entry.setSummaryAsHtml(entryBean.getSummaryXML());
208                 }
209 
210                 // entry.setUpdated(e.getUpdated());
211                 // entry.setPublished(e.getPublished());
212                 entry.addLink(entryBean.getLink());
213             }
214 
215             if ((feed.getEntries().size() == 0) && (size != 0)) {
216                 Entry entry = feed.addEntry();
217 
218                 entry.setId("nosubsription");
219                 entry.setTitle("Empty Feed");
220                 entry.setSummaryAsHtml("<h2>No events captured for this feed yet.</h2>");
221                 entry.setUpdated(new Date());
222                 entry.setPublished(new Date());
223                 entry.addLink(Config.GET_RESTFUL_EPCIS_URL());
224             }
225         } else {
226             Entry entry = feed.addEntry();
227 
228             entry.setId("unspecified");
229             entry.setTitle("Unspecified Feed");
230             entry.setSummaryAsHtml("<h2>There is no subscription for this feed yet.</h2>");
231             entry.setUpdated(new Date());
232             entry.setPublished(new Date());
233             entry.addLink(Config.GET_RESTFUL_EPCIS_URL());
234         }
235 
236         return feed;
237     }

248     @PUT

250     public Response addFeed(@PathParam(ID) String urlId, Feed myNewFeed) {
251         String id       = myNewFeed.getId() + "";
252         String title    = myNewFeed.getTitle();
253         String subtitle = myNewFeed.getSubtitle();
254         String author   = "";
255 
256         try {
257             author = myNewFeed.getAuthors().get(0).getName();
258         } catch (Exception ex) {
259             ex.printStackTrace();
260 
261             // continue
262         }
263 
264         String category = "";
265 
266         try {
267             category = myNewFeed.getCategories().get(0).getTerm();
268         } catch (Exception ex) {
269             ex.printStackTrace();
270 
271             // continue
272         }
273 
274         String contributor = "";
275 
276         try {
277             contributor = myNewFeed.getContributors().get(0).getName();
278         } catch (Exception ex) {
279             ex.printStackTrace();
280 
281             // continue
282         }
283 
284         String link = "";
285 
286         try {
287             link = myNewFeed.getLinks().get(0).getHref().toASCIIString();
288         } catch (Exception ex) {
289             ex.printStackTrace();
290 
291             // continue
292         }
293 
294         String   logo     = myNewFeed.getLogo().toASCIIString();
295 
296         FeedBean feedBean = new FeedBean(id, title, subtitle, author, category, contributor, link, logo);
297 
298         FEED_DB.addFeed(feedBean);
299 
300         return Response.ok().build();
301     }

320     /**
321      * Add an HTML Entry to a Feed
322      *
323      *
324      * @param entry as Feed Entry
325      * @param id
326      *
327      * @return
328      */
329     @POST
330     @Path(FEED_ID_URI)
331     public Response addEntry(Entry entry, @PathParam("id") String id) {
332 
333         Feed    feed           = null;
334 
335         boolean isFeedExisting = false;
336 
337         for (String feedId : FEED_DB.getFeedIDs()) {
338             if (feedId.equals(id)) {
339                 isFeedExisting = true;
340             }
341         }
342 
343         if (!isFeedExisting) {
344             Abdera abdera = AtomPub.getAbdera();
345 
346             feed = abdera.getFactory().newFeed();
347             feed.setTitle("Unspecified Feed");
348             feed.setSubtitle("Feed with ID " + id);
349         }
350 
351         int    entryID = FEED_DB.getNextEntryID(id);
352         String link    = "";
353 
354         try {
355             link = entry.getLinks().get(0).getHref().getPath();
356         } catch (Exception ex) {
357 
358             // ok, just continue
359         }
360 
361         EntryBean entryBean = new EntryBean(entryID + "", entry.getTitle(), entry.getSummary(), entry.getContent(), entry.getUpdated().toString(), entry.getPublished().toString(), link);
362 
363         FEED_DB.addEntryToFeed(id, entryBean);
364 
365         return Response.ok("OK").build();
366     }
