24  /**
25   * This class loads the user-configurable properties from the config.property
26   * file located in /WEB-INF/classes/.
27   * @author <a href="http://www.guinard.org">Dominique Guinard</a>
28   */
29  import java.io.IOException;
30  import java.io.InputStream;
31  import java.net.URL;
32  import java.util.Properties;
33  import java.util.logging.Level;
34  import java.util.logging.Logger;
35  
36  /**
37   * This is the ConfiguLoader Singleton which which loads all
38   * user-configurable properties from the config.property
39   * file located in /WEB-INF/classes/.
40   * @author <a href="http://www.guinard.org">Dominique Guinard</a>
41   */
42  public class ConfigLoader extends Properties {
43  
44      private static ConfigLoader UNIQUE_CONFIGURATOR = null;
45  
46      /**
47       * The ConfigLoader is a Singleton, thus the private constructor.
48       */
49      private ConfigLoader() {
50          String propsFileName = "config.properties";
51          try {
52              InputStream propsStream = this.getClass().getClassLoader().getResourceAsStream(propsFileName);
53              URL loc = this.getClass().getClassLoader().getResource(propsFileName);
54              System.out.println(loc);
55              super.load(propsStream);
56          } catch (IOException ex) {
57              Logger.getLogger(ConfigLoader.class.getName()).log(Level.SEVERE, "Unable to load the property file: " + propsFileName, ex);
58          }
59      }
60  
61      public static ConfigLoader getInstance() {
62          if (UNIQUE_CONFIGURATOR == null) {
63              return UNIQUE_CONFIGURATOR = new ConfigLoader();
64          } else {
65              return UNIQUE_CONFIGURATOR;
66          }
67      }
68  }
