21  package org.accada.ale.util;
22  
23  import java.io.ByteArrayInputStream;
24  import java.io.InputStream;
25  
26  /**
27   * This class provides some methods to packaging a message into a soap envelope.
28   * 
29   * @author regli
30   */
31  public class EnveloperUtil {
32  
33  	/** xml version tag */
34  	private static final String XML_VERSION = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>";
35  	/** opening soap envelope and body tag */
36  	private static final String ENVELOPE_OPEN = "<soap:Envelope xmlns:soap=\"http://www.w3.org/2003/05/soap-envelope\"><soap:Body>";
37  	/** closing soap body and envelope tag */
38  	private static final String ENVELOPE_CLOSE = "</soap:Body></soap:Envelope>";
39  
40  	/**
41  	 * This method packaging a message from an input stream into a soap envelope and returns it as an input stream.
42  	 * 
43  	 * @param in input stream with the message to envelope
44  	 * @return input stream with the soap message
45  	 */
46  	public static InputStream envelope(InputStream in) {
47  		
48  		return new ByteArrayInputStream(envelope(StreamUtil.inputStream2String(in)).getBytes());
49  		
50  	}
51  	
52  	/**
53  	 * This method packaging a message from a string into a soap envelope and returns it as a string.
54  	 * 
55  	 * @param s string with the message to envelope
56  	 * @return string with the soap message
57  	 */
58  	public static String envelope(String s) {
59  		
60  		String xmlVersion;
61  		String content;
62  		int index;
63  		if ((index = s.indexOf("?>") + 2) > 2) {
64  			xmlVersion = s.substring(0, index);
65  			content = s.substring(index);
66  		} else {
67  			xmlVersion = XML_VERSION;
68  			content = s;
69  		}
70  		
71  		StringBuffer buf = new StringBuffer();
72  		buf.append(xmlVersion);
73  		buf.append(ENVELOPE_OPEN);
74  		buf.append(content);
75  		buf.append(ENVELOPE_CLOSE);
76  		
77  		return buf.toString();
78  		
79  	}
80  	
81  }
