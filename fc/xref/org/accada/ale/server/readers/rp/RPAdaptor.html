119 	 * this method calls the URL constructor with a string.
120 	 * @param urlString the URL to be converted into a java.net.URL
121 	 * @param comment "NotificationPoint" or " ConnectionPoint" 
122 	 * @return a URL created from urlString
123 	 * @throws ImplementationException when a MalformedURLException is thrown
124 	 */
125 	private URL toURL(String urlString, String comment) throws ImplementationExceptionResponse {
126 		URL url = null;
127 		try {
128 			url = new URL(urlString);

131 			throw new ImplementationExceptionResponse("Could not extract " + comment + " from LRProperty");
132 		}
133 		return url;

136 	/**
137 	 * this method extracts the connection settings from the LRProperty. when there 
138 	 * is the need to disconnect the reader (when something in the connection to the 
139 	 * reader has been changed in the LRSpec), true will be returned false otherwise.
140 	 * if necessary the reader will be restarted
141 	 * @throws ImplementationException whenever an error occurs
142 	 * @return returns true if the connection to the reader needs to be reconnected
143 	 */
144 	private boolean extractConnectionSettings() throws ImplementationExceptionResponse {
145 		URL connectionPoint = toURL(logicalReaderProperties.get("ConnectionPoint"), "ConnectionPoint");
146 		URL notificationPoint = toURL(logicalReaderProperties.get("NotificationPoint"), "NotificationPoint");
147 		String interval = logicalReaderProperties.get("ReadTimeInterval");
148 	
149 		setReadTimeInterval(-1);
150 		try {
151 			setReadTimeInterval(Integer.parseInt(interval));
152 		} catch (Exception ne) {
153 			LOG.error("could not extract readTimeIntervall from LRPropery");
154 			throw new ImplementationExceptionResponse("could not extract notificationPoint from LRPropery");
155 		}
156 		
157 		// assert that the readTimeInterval is not -1
158 		if (readTimeInterval == -1) {
159 			LOG.error("ReadTimeInterval not set - assuming 2000ms");
160 			setReadTimeInterval(DEFAULT_READTIME_INTERVALL);
161 		}
162 		
163 		boolean disconnect = false;
164 		
165 		// compare the new sources string to the old sources string 
166 		String nsources = logicalReaderProperties.get("PhysicalReaderSource");
167 		if (!nsources.equalsIgnoreCase(sourcesString)) {
168 			disconnect = true;
169 		}		
170 		
171 		// disconnect if command channel host has changed
172 		if (!connectionPoint.getHost().equalsIgnoreCase(commandChannelHost)) {
173 			disconnect = true;
174 		}
175 		
176 		// disconnect if notificationChannelHost has changed
177 		if (!notificationPoint.getHost().equalsIgnoreCase(notificationChannelHost)) {
178 			disconnect = true;
179 		}
180 		
181 		// disconnect when the port has changed
182 		if (connectionPoint.getPort() != commandChannelPort) {
183 			disconnect = true;
184 		}
185 			
186 		// disconnect when the port has changed
187 		if (notificationPoint.getPort() != notificationChannelPort) {
188 			disconnect = true;
189 		}
190 		// disconnect when the reader was not connected before
191 		if (inputGenerator != null && !inputGenerator.isReady()) {
192 			disconnect = true;
193 		}
194 		
195 		LOG.debug("readTimeInterval " + readTimeInterval);
196 		LOG.debug(String.format("commandChannelHost %s on port %d", 
197 				connectionPoint.getHost(), 
198 				connectionPoint.getPort()));
199 		LOG.debug(String.format("notificationChannelHost %s on port %d", 
200 				notificationPoint.getHost(), 
201 				notificationPoint.getPort()));
202 		
203 		// set the new commandChannelSettings
204 		setCommandChannelHost(connectionPoint.getHost());
205 		setCommandChannelPort(connectionPoint.getPort());
206 		
207 		// set the new notificationChannelSettings
208 		setNotificationChannelHost(notificationPoint.getHost());
209 		setNotificationChannelPort(notificationPoint.getPort());
210 		
211 		// set the new sources
212 		sourcesString = nsources;
213 		setSourcesFromArray(sourcesString.split("[,]"));
214 		
215 		return disconnect;
216 	}

219 	/**
220 	 * sets up a reader. if the adaptor can be connected to the rp-proxy 
221 	 * the adaptor will be set to connected.
222 	 * @throws ImplementationException whenever an internal error occured
223 	 */
224 	@Override
225 	public void connectReader() throws ImplementationExceptionResponse {
226 		if (isConnected()) {
227 			return;
228 		}
229 		
230 		try {
231 			inputGenerator = new InputGenerator(this);
232 			
233 		} catch (ImplementationExceptionResponse e) {
234 			setDisconnected();
235 			throw e;
236 		}
237 		
238 		LOG.debug("setup identifyThread on RPAdaptor " + getName());
239 		// setup the polling thread
240 		identifyThread = new IdentifyThread(this);
241 		identifyThread.setPollingFrequency(readTimeInterval);
242 		identifyThread.start();
243 		
244 		// suspend the polling thread to the beginning
245 		identifyThread.suspendIdentify();		
246 		
247 		setConnected();
248 	}

250 	/**
251 	 * destroys a reader.
252 	 * @throws ImplementationException whenever an internal error occured
253 	 */
254 	@Override
255 	public void disconnectReader() throws ImplementationExceptionResponse {
256 		
257 		setCommandChannelHost(null);
258 		setNotificationChannelHost(null);
259 		setCommandChannelPort(-1);
260 		setNotificationChannelPort(-1);
261 		setReadTimeInterval(-1);
262 		
263 		if (inputGenerator != null) {
264 			inputGenerator.remove();
265 			
266 			inputGenerator = null;
267 		}
268 		
269 		
270 		if (identifyThread != null) {
271 			identifyThread.stopIdentify();
272 			identifyThread = null;
273 		}
274 		setDisconnected();
275 		setStopped();
276 	}

303 	/**
304 	 * starts a base reader to read tags. if the reader 
305 	 * is not yet connected this command will connect the reader 
306 	 * immediately and then start it.
307 	 * If the reader cannot be connected then the reader is 
308 	 * not started as well.
309 	 *
310 	 */
311 	@Override
312 	public synchronized void start() {
313 		if (!isConnected()) {
314 			try {
315 				connectReader();
316 			} catch (ImplementationExceptionResponse e) {
317 				e.printStackTrace();
318 				
319 				setDisconnected();
320 			}
321 		}
322 		
323 		// reader is not connected so it is not started as well
324 		if (!isConnected()) {
325 			setStopped();
326 		} else {
327 			LOG.debug("identifyThread starting to identify");
328 			identifyThread.resumeIdentify();
329 			setStarted();
330 		}
331 	}

344 	/**
345 	 * updates a reader according the specified LRSpec.
346 	 * @param spec LRSpec for the reader
347 	 * @throws ImplementationException whenever an internal error occurs
348 	 */
349 	@Override
350 	public synchronized void update(LRSpec spec) throws ImplementationExceptionResponse {
351 		stop();
352 		
353 		setLRSpec(spec);
354 		if (extractConnectionSettings()) {
355 			LOG.debug("restarting reader " + getName());
356 			disconnectReader();
357 			// set the connection settings again and then start the reader
358 			extractConnectionSettings();
359 			connectReader();
360 		}
361 		start();
362 	}

364 	/**
365 	 * Triggers the identification of all tags that are currently available 
366 	 * on the reader. this method is used when the IdentifyThread is used to poll the adaptor.
367 	 * @param readPointNames the readers/sources that have to be polled
368 	 * @return a set of Observations
369 	 * @throws HardwareException whenever an internal hardware error occurs (as reader not available...)
370 	 */
371 	@Override
372 	public Observation[] identify(String[] readPointNames)
373 			throws HardwareException {
374 		
375 		//LOG.debug("identify called an RPAdaptor " + getName());
376 		
377 		if ((inputGenerator != null) && (countObservers() > 0)) {
378 			
379 			if (inputGenerator.isReady()) {
380 			
381 				try {
382 					inputGenerator.poll();
383 					
384 				} catch (RPProxyException e) {
385 					e.printStackTrace();
386 					throw new HardwareException("Could not poll the adaptor " + getName());
387 				}
388 			} else {
389 				LOG.debug("rp-proxy not ready (yet)");
390 			}
391 		}
392 		
393 		return null;
394 	}
