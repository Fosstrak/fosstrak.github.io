183 	 * This method returns the new ec report.
184 	 * 
185 	 * @return ec report
186 	 * @throws ECSpecValidationException if a tag is invalid
187 	 * @throws ImplementationException if an implementation exception occurs
188 	 */
189 	public ECReport getECReport() throws ECSpecValidationExceptionResponse, ImplementationExceptionResponse {
190 
191 		//generate new ECReport
192 		if (reportType.equalsIgnoreCase(ECReportSetEnum.ADDITIONS)) {
193 			
194 			// get additional tags
195 			Map<String, Tag> reportTags = new HashMap<String, Tag>();
196 
197 			// add tags from current EventCycle 
198 			for (Tag tag : currentEventCycle.getTags()) {
199 				reportTags.put(tag.getTagIDAsPureURI(), tag);
200 			}
201 				
202 			// remove tags from last EventCycle
203 			if (currentEventCycle.getLastEventCycleTags() != null) {
204 				for (Tag tag : currentEventCycle.getLastEventCycleTags()) {
205 					reportTags.remove(tag.getTagIDAsPureURI());
206 				}
207 			}
208 				
209 			// add tags to report with filtering
210 			readyToCreateReport = true;
211 
212 			for (Tag tag : reportTags.values()) {
213 				addTag(tag);
214 			}
215 			//writeDebugInformation(reportTags);
216 	
217 		} else if (reportType.equalsIgnoreCase(ECReportSetEnum.CURRENT)) {
218 
219 			// get tags from current EventCycle 
220 			for (Tag tag : currentEventCycle.getTags()) {
221 				addTag(tag);
222 			}
223 
224 		//} else if (reportType == ECReportSetEnum.DELETIONS) {
225 		} else if (reportType.equalsIgnoreCase(ECReportSetEnum.DELETIONS)) {
226 			
227 			// get removed tags
228 			Map<String, Tag> reportTags = new HashMap<String, Tag>();
229 				
230 			// add tags from last EventCycle
231 			if (currentEventCycle.getLastEventCycleTags() != null) {
232 				for (Tag tag : currentEventCycle.getLastEventCycleTags()) {
233 					reportTags.put(tag.getTagIDAsPureURI(), tag);
234 				}
235 			}
236 				
237 			// remove tags from current EventCycle
238 			for (Tag tag : currentEventCycle.getTags()) {
239 				reportTags.remove(tag.getTagIDAsPureURI());
240 			}
241 				
242 			// add tags to report with filtering
243 			readyToCreateReport = true;
244 			for (Tag tag : reportTags.values()) {
245 				addTag(tag);
246 			}
247 			//writeDebugInformation(reportTags);
248 		}
249 
250 		if (reportSpec.isReportIfEmpty() || !isEmpty()) {
251 			ECReport temp = report;	
252 			report = new ECReport();
253 			report.setReportName(name);	
254 			return temp;
255 		} else {
256 			report = new ECReport();
257 			report.setReportName(name);
258 			return null;
259 		}
260 	}

267 	 * This method initializes the filter patterns on the basis of the ec report specification.
268 	 */
269 	private void initFilterPatterns() {
270 	
271 		LOG.debug("Init filter patterns");
272 		
273 		// get filter spec
274 		ECFilterSpec filterSpec = reportSpec.getFilterSpec();
275 		if (filterSpec != null) {
276 			
277 			// add ECIncludePatterns from spec to includePatterns set
278 			List<String> ecIncludePatterns = filterSpec.getIncludePatterns().getIncludePattern();
279 			if (ecIncludePatterns != null) {
280 				for (String pattern : ecIncludePatterns) {
281 					try {
282 						includePatterns.add(new Pattern(pattern, PatternUsage.FILTER));
283 					} catch (ECSpecValidationExceptionResponse e) {
284 						e.printStackTrace();
285 					}
286 				}
287 			}
288 			
289 			// add ECExcludePatterns from spec to excludePatterns set
290 			List<String> ecExcludePatterns = filterSpec.getExcludePatterns().getExcludePattern();
291 			if (ecExcludePatterns != null) {
292 				for (String pattern : ecExcludePatterns) {
293 					try {
294 						excludePatterns.add(new Pattern(pattern, PatternUsage.FILTER));
295 					} catch (ECSpecValidationExceptionResponse e) {
296 						e.printStackTrace();
297 					}
298 				}
299 			}
300 		}

324 	/**
325 	 * This method checks on the basis of the filter patterns if the specified tag could be a member of this report.
326 	 * 
327 	 * @param tagURI to check for possible membership
328 	 * @return true if the tag could be a member of this report and false otherwise
329 	 * @throws ECSpecValidationException if the tag is invalid
330 	 * @throws ImplementationException if an implementation exception occurs
331 	 */
332 	private boolean isMember(String tagURI) throws ECSpecValidationExceptionResponse, ImplementationExceptionResponse {
333 				
334 		if (reportType.equalsIgnoreCase(ECReportSetEnum.ADDITIONS)) {
335 		
336 			// if report type is additions the tag is only a member if it wasn't a member of the last event cycle	
337 			Set<Tag> tags = currentEventCycle.getLastEventCycleTags();
338 			if (tags != null) {
339 				for (Tag tag : tags) {
340 					if (tag.getTagIDAsPureURI().equals(tagURI)) {
341 						return false;
342 					}
343 				}
344 			}
345 		}			
346 
347 		// check if tagURI is member of an exclude pattern
348 		for (Pattern pattern : excludePatterns) {
349 			if (pattern.isMember(tagURI)) {
350 				return false;
351 			}
352 		}
353 		
354 		// check if there are include patterns specified
355 		if (includePatterns.size() == 0) {
356 			return true;
357 		} else {
358 			
359 			// check if tagURI is a member of an include pattern
360 			for (Pattern pattern : includePatterns) {
361 				if (pattern.isMember(tagURI)) {
362 					return true;
363 				}
364 			}
365 			return false;
366 		}
367 	}

369 	/**
370 	 * This method adds a tag to the matching group of the report.
371 	 * 
372 	 * @param tag to add
373 	 * @throws ECSpecValidationException if the tag is invalid
374 	 * @throws ImplementationException if an implementation exception occurs
375 	 */
376 	private void addTagToReportGroup(Tag tag) throws ImplementationExceptionResponse, ECSpecValidationExceptionResponse {
377 		
378 		// get tag URI
379 		String tagURI = tag.getTagIDAsPureURI();
380 		
381 		// get group name (use group patterns)
382 		String groupName = getGroupName(tagURI);
383 		
384 		LOG.debug("The group name for tag '" + tagURI + "' is '" + groupName + "'");
385 		
386 		// get matching group
387 		ECReportGroup matchingGroup = null;
388 		List<ECReportGroup> groups = report.getGroup();
389 		//if (groups == null) {
390 		if (groups.isEmpty()) {
391 			matchingGroup = null;
392 		} else {
393 			for (ECReportGroup group : groups) {
394 				if (groupName == null) {
395 					if (group.getGroupName() == null) {
396 						matchingGroup = group;
397 					}
398 				} else {
399 					if (groupName.equals(group.getGroupName())) {
400 						matchingGroup = group;
401 					}
402 				}
403 			}
404 		}
405 	
406 		// create group if group does not already exist
407 		if (matchingGroup == null) {
408 			
409 			LOG.debug("Group '" + groupName + "' does not already exist, create it");
410 						
411 			// create group
412 			matchingGroup = new ECReportGroup();
413 			
414 			// set name
415 			matchingGroup.setGroupName(groupName);
416 			
417 			// set count
418 			if (reportSpec.getOutput().isIncludeCount()) {
419 				ECReportGroupCount groupCount = new ECReportGroupCount();
420 				groupCount.setCount(0);
421 				matchingGroup.setGroupCount(groupCount);
422 			}
423 			
424 			// create and set group list
425 			matchingGroup.setGroupList(new ECReportGroupList());
426 			
427 			// add to groups
428 			report.getGroup().add(matchingGroup);
429 			
430 		}
431 		
432 		// create group list member
433 		ECReportGroupListMember groupMember = new ECReportGroupListMember();
434 		if (reportSpec.getOutput().isIncludeRawDecimal()) {
435 			EPC epc = new EPC();
436 			epc.setValue(tag.getTagIDAsPureURI());
437 			groupMember.setRawDecimal(epc);
438 		}
439 		if (reportSpec.getOutput().isIncludeTag()) {
440 			EPC epc = new EPC();
441 			epc.setValue(tag.getTagIDAsPureURI());
442 			groupMember.setTag(epc);
443 		}
444 		if (reportSpec.getOutput().isIncludeRawHex()) {
445 			if (tag.getTagID() != null) {
446 				EPC epc = new EPC();
447 				epc.setValue(HexUtil.byteArrayToHexString(tag.getTagID()));
448 				groupMember.setRawHex(epc);
449 			}
450 		}
451 		
452 		// add list member to group list
453 		List<ECReportGroupListMember> members = matchingGroup.getGroupList().getMember();		members.add(groupMember);
454 		
455 		// increment group counter
456 		if (reportSpec.getOutput().isIncludeCount()) {
457 			matchingGroup.getGroupCount().setCount(matchingGroup.getGroupCount().getCount() + 1);
458 		}
459 		
460 		LOG.debug("Tag '" + tagURI + "' successfully added to group '" + groupName + "' of report '" + name + "'");
461 		
462 	}

465 	 * This method get the matching group of this report for the specified tag.
466 	 * 
467 	 * @param tagURI to search group for
468 	 * @return group name
469 	 * @throws ECSpecValidationException if the tag is invalid
470 	 * @throws ImplementationException if an implementation exception occurs
471 	 */
472 	private String getGroupName(String tagURI) throws ImplementationExceptionResponse, ECSpecValidationExceptionResponse {
473 		
474 		for (Pattern pattern : groupPatterns) {
475 			if (pattern.isMember(tagURI)) {
476 				return pattern.getGroupName(tagURI);
477 			}
478 		}
479 			
480 		return null;
481 		
482 	}
