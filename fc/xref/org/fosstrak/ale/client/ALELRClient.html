135 	/**
136 	 * This method initializes the class, by loading properties, texts and setting the endpoint.
137 	 * 
138 	 * @throws IOException
139 	 */
140 	public void initialize() throws IOException {
141 
142 		// load properties
143 		InputStream inputStream = this.getClass().getResourceAsStream(PROPERTIES_FILE_LOCATION);
144 		if (inputStream == null) {
145 			throw new IllegalArgumentException("Could not load properties from classpath (" + PROPERTIES_FILE_LOCATION + ")");
146 		}
147 		props.load(inputStream);
148 		
149 		// load text
150 		InputStream languageStream = null;
151 		
152 		// try to get language form property file
153 		if (props.containsKey("Language")) {
154 			languageStream = this.getClass().getResourceAsStream("/props/ALELRClient_" + props.getProperty("Language") + ".lang");
155 		}
156 		
157 		// try system default language
158 		if (languageStream == null) {
159 			languageStream = this.getClass().getResourceAsStream("/props/ALELRClient_" + SYSTEM_DEFAULT_LOCALE.getLanguage() + ".lang");
160 		}
161 		
162 		// try default language
163 		if (languageStream == null) {
164 			languageStream = this.getClass().getResourceAsStream("/props/ALELRClient_" + DEFAULT_LOCALE.getLanguage() + ".lang");
165 		}
166 		
167 		if (languageStream == null) {
168 			throw new IllegalArgumentException("Could not load language package from classpath (" + DEFAULT_LOCALE + ")");
169 		}
170 		guiText = new PropertyResourceBundle(languageStream);
171 
172 		//--------------------
173 		JaxWsProxyFactoryBean factory = new JaxWsProxyFactoryBean();
174         factory.setServiceClass(ALELRServicePortType.class);
175         String address = props.getProperty("EndPoint");
176         if (address == null) {
177             throw new IOException("no endpoint defined!");
178         }
179         factory.setAddress(address);
180         aleProxy = (ALELRServicePortType) factory.create();
181         
182 		initializeGUI();
183 		
184 	}

245 	 * This method sets the command selection panel depending on the command id.
246 	 * 
247 	 * @param command id
248 	 */
249 	private void setCommandPanel(int command) {
250 		
251 		if (command == -1) {
252 			commandPanel.removeAll();
253 			commandPanel.setBorder(null);
254 			this.setVisible(false);
255 			this.setVisible(true);
256 			return;
257 		}
258 		
259 		commandPanel.setBorder(BorderFactory.createCompoundBorder(BorderFactory.createTitledBorder(guiText.getString("Command" + command)), BorderFactory.createEmptyBorder(5,5,5,5)));
260 		commandPanel.removeAll();
261 
262 		switch(command) {
263 		
264 			case 4: //getLRSpecNames
265 			case 11: // getStandardVersion
266 			case 12: // getVendorVersion
267 				commandPanel.setLayout(new GridLayout(1, 1, 5, 0));
268 				break;
269 	
270 			case 2: // undefine
271 			case 5: // getLRSpec
272 				commandPanel.setLayout(new GridLayout(5, 1, 5, 0));
273 				addLRSpecNamesComboBox(commandPanel);
274 				addSeparator(commandPanel);
275 				break;
276 
277 			case 10: // getPropertyValue
278 				commandPanel.setLayout(new GridLayout(7, 1, 5, 0));
279 				addLRSpecNamesComboBox(commandPanel);
280 				addPropertyValueField(commandPanel);
281 				addSeparator(commandPanel);
282 				break;
283 				
284 			case 3: // update
285 			case 6: // addReaders
286 			case 7: // setReaders
287 			case 8: // removeReaders
288 			case 9: // setProperties
289 				commandPanel.setLayout(new GridLayout(8, 1, 5, 0));
290 				addLRSpecNamesComboBox(commandPanel);
291 				addChooseFileField(commandPanel);
292 				addSeparator(commandPanel);
293 				break;
294 				
295 			case 1: // define
296 				commandPanel.setLayout(new GridLayout(8, 1, 5, 0));
297 				addReaderNameValueField(commandPanel); 
298 				addChooseFileField(commandPanel);
299 				addSeparator(commandPanel);
300 				break;
301 		}
302 		
303 		addExecuteButton(commandPanel);
304 		
305 		this.setVisible(false);
306 		this.setVisible(true);
307 		
308 	}

446 	/**
447 	 * This method executes the command which is selected in the command selection combobox
448 	 * with the parameters which are set in the corresponding fields.
449 	 * To execute the commands, the methods of the ale proxy will be invoked. 
450 	 */
451 	private void executeCommand() {
452 		
453 		Object result = null;
454 		String specName = null;
455 		String textParameter = null;
456 		try {
457 			
458 			switch(commandSelection.getSelectedIndex()) {
459 			
460 				case 4: //getLogicalReaderNames
461 					result = aleProxy.getLogicalReaderNames(new EmptyParms());
462 					break;
463 					
464 				case 11: // getStandardVersion
465 					result = aleProxy.getStandardVersion(new EmptyParms());
466 					break;
467 					
468 				case 12: // getVendorVersion
469 					result = aleProxy.getVendorVersion(new EmptyParms());
470 					break;
471 		
472 				case 2: // undefine
473 				case 5: // getLRSpec
474 					// get specName
475 					specName = (String)specNameComboBox.getSelectedItem();
476 					if (specName == null || "".equals(specName)) {
477 						showExcpetionDialog(guiText.getString("SpecNameNotSpecifiedDialog"));
478 						break;
479 					}
480 					
481 					switch(commandSelection.getSelectedIndex()) {
482 					
483 						case 2: // undefine
484 							Undefine undefineParms = new Undefine();
485 							undefineParms.setName(specName);
486 							aleProxy.undefine(undefineParms);
487 							result = guiText.getString("SuccessfullyUndefinedMessage");
488 							break;
489 							
490 						case 5: // getLRSpec
491 							GetLRSpec getLRSpecParms = new GetLRSpec();
492 							getLRSpecParms.setName(specName);
493 							result = aleProxy.getLRSpec(getLRSpecParms);
494 							break;
495 						
496 					}
497 					
498 					break;
499 					
500 				case 10: // getPropertyValue
501 					// get specName
502 					
503 					// get the l
504 					textParameter = propertyValueField.getText();
505 					
506 					specName = (String)specNameComboBox.getSelectedItem();
507 					
508 					if (specName == null || "".equals(specName)) {
509 						showExcpetionDialog(guiText.getString("SpecNameNotSpecifiedDialog"));
510 						break;
511 					}
512 				
513 					GetPropertyValue getPropertyValue = new GetPropertyValue();
514 					getPropertyValue.setName(specName);
515 					getPropertyValue.setPropertyName(textParameter);
516 					result = aleProxy.getPropertyValue(getPropertyValue);
517 					
518 					break;
519 					
520 				case 3: // update
521 				case 6: // addReaders
522 				case 7: // setReaders
523 				case 8: // removeReaders
524 				case 9: // setProperties
525 				case 1: // define
526 
527 					if (commandSelection.getSelectedIndex() != 1) {
528 						// get specName
529 						specName = (String)specNameComboBox.getSelectedItem();
530 						if (specName == null || "".equals(specName)) {
531 							showExcpetionDialog(guiText.getString("SpecNameNotSpecifiedDialog"));
532 							break;
533 						}
534 					}
535 					
536 					// get filePath
537 					String filePath = filePathField.getText();
538 					if (filePath == null || "".equals(filePath)) {
539 						showExcpetionDialog(guiText.getString("FilePathNotSpecifiedDialog"));
540 						break;
541 					}
542 					try {
543 						switch (commandSelection.getSelectedIndex()) {
544 						case 1: // define
545 						case 3: // update
546 							// get the LRSpec
547 							LRSpec spec = getLRSpecFromFile(filePath);
548 							if (commandSelection.getSelectedIndex() == 1) {
549 								textParameter = readerNameValueField.getText();
550 								Define define = new Define();
551 								define.setName(textParameter);
552 								define.setSpec(spec);
553 								aleProxy.define(define);
554 								result = guiText.getString("SuccessfullyDefinedMessage");
555 							} else {
556 								Update update = new Update();
557 								update.setName(specName);
558 								update.setSpec(spec);
559 								aleProxy.update(update);
560 								result = guiText.getString("SuccessfullyUpdateMessage");
561 							}
562 							break;
563 							
564 						case 6: // addReaders
565 							AddReaders addReaders = DeserializerUtil.deserializeAddReaders(filePath);
566 							addReaders.setName(specName);
567 							aleProxy.addReaders(addReaders);
568 							result = guiText.getString("SuccessfullyAddReadersMessage");
569 							break;
570 						case 7: // setReaders
571 							SetReaders setReaders = DeserializerUtil.deserializeSetReaders(filePath);
572 							setReaders.setName(specName);
573 							aleProxy.setReaders(setReaders);
574 							result = guiText.getString("SuccessfullySetReadersMessage");
575 							break;
576 						case 8: // removeReaders
577 							RemoveReaders removeReaders = DeserializerUtil.deserializeRemoveReaders(filePath);
578 							removeReaders.setName(specName);
579 							aleProxy.removeReaders(removeReaders);
580 							result = guiText.getString("SuccessfullyRemoveReadersMessage");
581 							break;
582 						case 9: // setProperties
583 							SetProperties setProperties = DeserializerUtil.deserializeSetProperties(filePath);
584 							setProperties.setName(specName);
585 							aleProxy.setProperties(setProperties);
586 							result = guiText.getString("SuccessfullySetPropertiesMessage");
587 							break;
588 							
589 						}
590 						
591 					} catch(FileNotFoundException e) {
592 						showExcpetionDialog(guiText.getString("FileNotFoundDialog"));
593 						break;
594 					} catch(Exception e) {
595 						showExcpetionDialog(guiText.getString("UnexpectedFileFormatDialog"));
596 						break;
597 					}
598 					break;
599 			}
600 		} catch (Exception e) {
601 			e.printStackTrace();		
602 			String reason = e.getMessage();
603 			if (e instanceof ImplementationExceptionResponse) {
604 				showExcpetionDialog(guiText.getString("ImplementationExceptionDialog"), reason);
605 			} else if (e instanceof SecurityExceptionResponse) {
606 				showExcpetionDialog(guiText.getString("SecurityExceptionDialog"), reason);
607 			} else if (e instanceof InUseExceptionResponse) {
608 				showExcpetionDialog(guiText.getString("InUseExceptionDialog"), reason);
609 			} else if (e instanceof NoSuchNameExceptionResponse) {
610 				showExcpetionDialog(guiText.getString("NoSuchNameExceptionDialog"), reason);
611 			} else if (e instanceof ImmutableReaderExceptionResponse) {
612 				showExcpetionDialog(guiText.getString("ImmutableReaderExceptionDialog"), reason);
613 			}
614 		} 
615 		
616 		showResult(result);
617 		
618 		// update spec name combobox
619 		List<String> logicalReaderNames = null;
620 		try {
621 			logicalReaderNames = aleProxy.getLogicalReaderNames(new EmptyParms()).getString();
622 		} catch (Exception e) {}
623 		if (logicalReaderNames != null && specNameComboBox != null && specNameComboBox.getSelectedObjects() != null && specNameComboBox.getSelectedObjects().length > 0) {
624 			String current = (String)specNameComboBox.getSelectedObjects()[0];
625 			specNameComboBox.removeAllItems();
