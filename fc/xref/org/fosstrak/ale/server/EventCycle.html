125 	/**
126 	 * Constructor sets parameter and starts thread.
127 	 * 
128 	 * @param generator to which this event cycle belongs to
129 	 * @throws ImplementationException if an implementation exception occurs
130 	 */
131 	public EventCycle(ReportsGenerator generator) 
132 		throws ImplementationExceptionResponse {
133 		
134 		// set name
135 		name = generator.getName() + "_" + number++;
136 		
137 		// set ReportGenerator
138 		this.generator = generator;
139 		
140 		// set spec
141 		spec = generator.getSpec();
142 		
143 		// get report specs and create a report for each spec
144 		for (ECReportSpec reportSpec : spec.getReportSpecs().getReportSpec()) {
145 			
146 			// add report spec and report to reports
147 			reports.add(new Report(reportSpec, this));
148 			
149 		}
150 		
151 		// init BoundarySpec values
152 		durationValue = getDurationValue();
153 		
154 		LOG.debug(String.format("durationValue: %s\n",
155 				durationValue));
156 		
157 		setAcceptTags(false);
158 		
159 		LOG.debug("adding logicalReaders to EventCycle");
160 		// get LogicalReaderStubs
161 		if (spec.getLogicalReaders() != null) {
162 			List<String> logicalReaderNames = 
163 				spec.getLogicalReaders().getLogicalReader();
164 			for (String logicalReaderName : logicalReaderNames) {
165 				LOG.debug("retrieving logicalReader " + logicalReaderName);
166 				LogicalReader logicalReader = 
167 					LogicalReaderManager.getLogicalReader(logicalReaderName);
168 				
169 				if (logicalReader != null) {
170 					LOG.debug("adding logicalReader " + 
171 							logicalReader.getName() + " to EventCycle " + name);
172 					logicalReaders.add(logicalReader);
173 				}
174 			}
175 		} else {
176 			LOG.error("ECSpec contains no readers");
177 		}
178 		
179 		for (LogicalReader logicalReader : logicalReaders) {
180 			
181 			// subscribe this event cycle to the logical readers
182 			LOG.debug(
183 					"registering EventCycle " + name + " on reader " + 
184 					logicalReader.getName());
185 			
186 			logicalReader.addObserver(this);
187 		}
188 		
189 		rounds = 0;
190 		
191 		// create and start Thread
192 		thread = new Thread(this);
193 		thread.start();
194 		
195 		LOG.debug("New EventCycle  '" + name + "' created.");
196 
197 	}

327 	 * implementation of the observer interface for tags.
328 	 * @param o an observable object that triggered the update
329 	 * @param arg the arguments passed by the observable
330 	 */
331 	public synchronized void update(Observable o, Object arg) {
332 		LOG.debug("EventCycle "+ getName() + ": Update notification received. ");
333 		if (!isAcceptingTags()) {
334 			LOG.debug("EventCycle "+ getName() + ": Not accepting notification.");
335 			return;
336 		}
337 		if (arg instanceof Tag) {
338 			
339 			// process one tag
340 			Tag tag = (Tag) arg;
341 			//tag.prettyPrint(LOG);
342 			try {
343 				addTag(tag);
344 			} catch (ImplementationExceptionResponse ie) {
345 				ie.printStackTrace();
346 			} catch (ECSpecValidationExceptionResponse ive) {
347 				ive.printStackTrace();
348 			}
349 		} else if (arg instanceof List) {
350 			// process multiple tags at once
351 			
352 			List<Tag> tagList = (List<Tag>) arg;
353 			LOG.debug("EventCycle "+ getName() + ": Received list of tags :");
354 			for (Tag tag : tagList) {
355 				try {
356 					addTag(tag);
357 				} catch (ImplementationExceptionResponse ie) {
358 					ie.printStackTrace();
359 				} catch (ECSpecValidationExceptionResponse ive) {
360 					ive.printStackTrace();
361 				}
362 			}
363 		}
364 	}

369 	public void stop() {
370 		// unsubscribe this event cycle from logical readers
371 		for (LogicalReader logicalReader : logicalReaders) {
372 			//logicalReader.unsubscribeEventCycle(this);
373 			logicalReader.deleteObserver(this);
374 		}
375 		
376 		if (thread.isAlive()) {
377 			thread.interrupt();
378 			
379 			// stop EventCycle
380 			LOG.debug("EventCycle '" + name + "' stopped.");
381 		}
382 		
383 		isTerminated = true;
384 		
385 		synchronized (this) {
386 			this.notifyAll();
387 		}		
388 	}

408 	/**
409 	 * This method is the main loop of the event cycle in which the tags will be collected.
410 	 * At the end the reports will be generated and the subscribers will be notified.
411 	 */
412 	public void run() {
413 		
414 		// wait for the start
415 		// running will be set by the ReportsGenerator when the EventCycle
416 		// has a subscriber
417 		if (!running) {
418 			synchronized (this) {
419 				try {
420 					this.wait();
421 				} catch (InterruptedException e) {
422 					e.printStackTrace();
423 				}
424 			}
425 		}
426 		
427 		while (running) {
428 			rounds ++;
429 			synchronized (lock) {
430 				roundOver = false;
431 			}
432 			LOG.info("EventCycle "+ getName() + ": Starting (Round " + rounds + ").");
433 			
434 			// set start time
435 			long startTime = System.currentTimeMillis();
436 	
437 			// accept tags
438 			setAcceptTags(true);
439 			
440 			//------------------------------ run for the specified time
441 			try {
442 				
443 				if (durationValue > 0) {
444 					
445 					// if durationValue is specified and larger than zero, 
446 					// wait for notify or durationValue elapsed.
447 					synchronized (this) {
448 						long dt = (System.currentTimeMillis() - startTime);
449 						this.wait(Math.max(1, durationValue - dt));
450 						terminationCondition = ECTerminationCondition.DURATION;
451 					}
452 				} else {
453 					
454 					// if durationValue is not specified or smaller than zero, 
455 					// wait for notify.
456 					synchronized (this) {
457 						this.wait();
458 					}
459 				}
460 			
461 			} catch (InterruptedException e) {
462 				
463 				// if Thread is stopped with method stop(), 
464 				// then return without notify subscribers.
465 				return;
466 			}
467 			
468 			// dont accept tags anymore
469 			setAcceptTags(false);
470 			//------------------------ generate the reports
471 			
472 			// get reports
473 			try {
474 				// compute total time
475 				totalTime = System.currentTimeMillis() - startTime;
476 				
477 				LOG.info("EventCycle "+ getName() + 
478 						": Number of Tags read in the current EventCyle.java: " 
479 						+ tags.size());
480 				
481 				ECReports ecReports = getECReports();
482 				
483 				// notifySubscribers
484 				generator.notifySubscribers(ecReports);
485 				
486 				// store the current tags into the old tags
487 				// explicitly clear the tags
488 				if (lastEventCycleTags != null) {
489 					lastEventCycleTags.clear();
490 				}
491 				lastEventCycleTags = tags;
492 				tags = new HashSet<Tag>();
493 				
494 			} catch (ECSpecValidationExceptionResponse e) {
495 				LOG.error("EventCycle "+ getName() + 
496 						": Could not create ECReports (" + e.getMessage() + ")");
497 			} catch (ImplementationExceptionResponse e) {
498 				LOG.error("EventCycle "+ getName() + 
499 						": Could not create ECReports (" + e.getMessage() + ")");
500 			} catch (Exception e) {
501 				e.printStackTrace();
502 			}
503 			
504 			
505 			LOG.info("EventCycle "+ getName() + 
506 					": EventCycle finished (Round " + rounds + ").");
507 			try {
508 				// inform possibly waiting workers about the finish
509 				synchronized (lock) {
510 					roundOver = true;
511 					lock.notifyAll();
512 				}
513 				// wait until reschedule.
514 				synchronized (this) {	
515 					this.wait();
516 				}
517 				LOG.debug("eventcycle continues");
518 			} catch (InterruptedException e) {
519 				LOG.error("eventcycle got interrupted");
520 			}			
521 		}
522 			
523 			
524 		// stop EventCycle
525 		stop();
526 		
527 	}

559 	/**
560 	 * This method returns the duration value extracted from the event cycle 
561 	 * specification. 
562 	 * @return duration value in milliseconds
563 	 * @throws ImplementationException if an implementation exception occurs
564 	 */
565 	private long getDurationValue() throws ImplementationExceptionResponse {
566 		if (spec.getBoundarySpec() != null) {
567 			ECTime duration = spec.getBoundarySpec().getDuration();
568 			if (duration != null) {
569 				if (duration.getUnit().compareToIgnoreCase(ECTimeUnit.MS) == 0) {
570 					return duration.getValue();
571 				} else {
572 					throw new ImplementationExceptionResponse(
573 							"The only ECTimeUnit allowed is milliseconds (MS).");
574 				}
575 			}
576 		}
577 		return -1;
