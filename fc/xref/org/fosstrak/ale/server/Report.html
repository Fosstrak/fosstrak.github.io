135 	 * @param tag to add
136 	 * @throws ECSpecValidationException if the tag is invalid
137 	 * @throws ImplementationException if an implementation exception occurs
138 	 */
139 	public void addTag(Tag tag) throws ECSpecValidationExceptionResponse, ImplementationExceptionResponse {
140 
141 		// get tag URI
142 		String tagURI = tag.getTagIDAsPureURI();
143 	
144 		// check if the tag is a member of this report (use filter patterns and set spec)
145 		if (isMember(tagURI)) {
146 	
147 				LOG.debug("Event '" + tag + "' is member of report '" + name + "'");
148 			
149 				// add tag to report
150 				addTagToReportGroup(tag);
151 		}
152 	}

173 		out +=  '\t' + "eventcycle " + currentEventCycle.getName() + '\n';
174 		out +=  '\t' + "round " + currentEventCycle.getRounds() + '\n';
175 		if (reportTags == null) {
176 			out += '\t' + "no tags" + '\n';
177 			out +=  "+++++++++++++++++++++++++++++++++++++++++++++++++++++" + '\n';
178 			LOG.info(out);
179 			return;
180 		}
181 		
182 		
183 		for (Tag tag : reportTags.values()) {
184 			out += '\t' + tag.getTagIDAsPureURI() + '\n';
185 		}

187 		LOG.info(out);

190 	/**
191 	 * This method returns the new ec report.
192 	 * 
193 	 * @return ec report
194 	 * @throws ECSpecValidationException if a tag is invalid
195 	 * @throws ImplementationException if an implementation exception occurs
196 	 */
197 	public ECReport getECReport() throws ECSpecValidationExceptionResponse, ImplementationExceptionResponse {
198 
199 		//generate new ECReport
200 		if (reportType.equalsIgnoreCase(ECReportSetEnum.ADDITIONS)) {
201 			
202 			// get additional tags
203 			Map<String, Tag> reportTags = new HashMap<String, Tag>();
204 
205 			// add tags from current EventCycle 
206 			for (Tag tag : currentEventCycle.getTags()) {
207 				reportTags.put(tag.getTagIDAsPureURI(), tag);
208 			}
209 				
210 			// remove tags from last EventCycle
211 			if (currentEventCycle.getLastEventCycleTags() != null) {
212 				for (Tag tag : currentEventCycle.getLastEventCycleTags()) {
213 					reportTags.remove(tag.getTagIDAsPureURI());
214 				}
215 			}
216 				
217 			// add tags to report with filtering
218 			readyToCreateReport = true;
219 
220 			for (Tag tag : reportTags.values()) {
221 				addTag(tag);
222 			}
223 			//writeDebugInformation(reportTags);
224 	
225 		} else if (reportType.equalsIgnoreCase(ECReportSetEnum.CURRENT)) {
226 
227 			// get tags from current EventCycle 
228 			for (Tag tag : currentEventCycle.getTags()) {
229 				addTag(tag);
230 			}
231 
232 		//} else if (reportType == ECReportSetEnum.DELETIONS) {
233 		} else if (reportType.equalsIgnoreCase(ECReportSetEnum.DELETIONS)) {
234 			
235 			// get removed tags
236 			Map<String, Tag> reportTags = new HashMap<String, Tag>();
237 				
238 			// add tags from last EventCycle
239 			if (currentEventCycle.getLastEventCycleTags() != null) {
240 				for (Tag tag : currentEventCycle.getLastEventCycleTags()) {
241 					reportTags.put(tag.getTagIDAsPureURI(), tag);
242 				}
243 			}
244 				
245 			// remove tags from current EventCycle
246 			for (Tag tag : currentEventCycle.getTags()) {
247 				reportTags.remove(tag.getTagIDAsPureURI());
248 			}
249 				
250 			// add tags to report with filtering
251 			readyToCreateReport = true;
252 			for (Tag tag : reportTags.values()) {
253 				addTag(tag);
254 			}
255 			//writeDebugInformation(reportTags);
256 		}
257 
258 		if (reportSpec.isReportIfEmpty() || !isEmpty()) {
259 			ECReport temp = report;	
260 			report = new ECReport();
261 			report.setReportName(name);	
262 			return temp;
263 		} else {
264 			report = new ECReport();
265 			report.setReportName(name);
266 			return null;
267 		}
268 	}

275 	 * This method initializes the filter patterns on the basis of the ec report specification.
276 	 */
277 	private void initFilterPatterns() {
278 	
279 		LOG.debug("Init filter patterns");
280 		
281 		// get filter spec
282 		ECFilterSpec filterSpec = reportSpec.getFilterSpec();
283 		if (filterSpec != null) {
284 			
285 			// add ECIncludePatterns from spec to includePatterns set
286 			List<String> ecIncludePatterns = filterSpec.getIncludePatterns().getIncludePattern();
287 			if (ecIncludePatterns != null) {
288 				for (String pattern : ecIncludePatterns) {
289 					try {
290 						includePatterns.add(new Pattern(pattern, PatternUsage.FILTER));
291 					} catch (ECSpecValidationExceptionResponse e) {
292 						e.printStackTrace();
293 					}
294 				}
295 			}
296 			
297 			// add ECExcludePatterns from spec to excludePatterns set
298 			List<String> ecExcludePatterns = filterSpec.getExcludePatterns().getExcludePattern();
299 			if (ecExcludePatterns != null) {
300 				for (String pattern : ecExcludePatterns) {
301 					try {
302 						excludePatterns.add(new Pattern(pattern, PatternUsage.FILTER));
303 					} catch (ECSpecValidationExceptionResponse e) {
304 						e.printStackTrace();
305 					}
306 				}
307 			}
308 		}
309 		
310 	}

315 	private void initGroupPatterns() {
316 		
317 		LOG.debug("Init group patterns");
318 		if (reportSpec.getGroupSpec() != null) {
319 			// get group spec
320 			List<String> groupSpec = reportSpec.getGroupSpec().getPattern();
321 			// add ECGroupPatterns from spec to groupPatterns set
322 			for (String pattern : groupSpec) {
323 				try {
324 					groupPatterns.add(new Pattern(pattern, PatternUsage.GROUP));
325 				} catch (ECSpecValidationExceptionResponse e) {
326 					e.printStackTrace();
327 				}	
328 			}
329 		}

332 	/**
333 	 * This method checks on the basis of the filter patterns if the specified tag could be a member of this report.
334 	 * 
335 	 * @param tagURI to check for possible membership
336 	 * @return true if the tag could be a member of this report and false otherwise
337 	 * @throws ECSpecValidationException if the tag is invalid
338 	 * @throws ImplementationException if an implementation exception occurs
339 	 */
340 	private boolean isMember(String tagURI) throws ECSpecValidationExceptionResponse, ImplementationExceptionResponse {
341 				
342 		if (reportType.equalsIgnoreCase(ECReportSetEnum.ADDITIONS)) {
343 		
344 			// if report type is additions the tag is only a member if it wasn't a member of the last event cycle	
345 			Set<Tag> tags = currentEventCycle.getLastEventCycleTags();
346 			if (tags != null) {
347 				for (Tag tag : tags) {
348 					if (tag.getTagIDAsPureURI().equals(tagURI)) {
349 						return false;
350 					}
351 				}
352 			}
353 		}			
354 
355 		// check if tagURI is member of an exclude pattern
356 		for (Pattern pattern : excludePatterns) {
357 			if (pattern.isMember(tagURI)) {
358 				return false;
359 			}
360 		}
361 		
362 		// check if there are include patterns specified
363 		if (includePatterns.size() == 0) {
364 			return true;
365 		} else {
366 			
367 			// check if tagURI is a member of an include pattern
368 			for (Pattern pattern : includePatterns) {
369 				if (pattern.isMember(tagURI)) {
370 					return true;
371 				}
372 			}
373 			return false;
374 		}
375 	}

377 	/**
378 	 * This method adds a tag to the matching group of the report.
379 	 * 
380 	 * @param tag to add
381 	 * @throws ECSpecValidationException if the tag is invalid
382 	 * @throws ImplementationException if an implementation exception occurs
383 	 */
384 	private void addTagToReportGroup(Tag tag) throws ImplementationExceptionResponse, ECSpecValidationExceptionResponse {
385 		
386 		// get tag URI
387 		String tagURI = tag.getTagIDAsPureURI();
388 		// if this one is null, try something different to compense crashes...
389 		if (null == tagURI) {
390 			tagURI = Tag.getTDTEngine().bin2hex(tag.getTagAsBinary());
391 		}
392 		
393 		// get group name (use group patterns)
394 		String groupName = getGroupName(tagURI);
395 		
396 		LOG.debug("The group name for tag '" + tagURI + "' is '" + groupName + "'");
397 		
398 		// get matching group
399 		ECReportGroup matchingGroup = null;
400 		List<ECReportGroup> groups = report.getGroup();
401 		//if (groups == null) {
402 		if (groups.isEmpty()) {
403 			matchingGroup = null;
404 		} else {
405 			for (ECReportGroup group : groups) {
406 				if (groupName == null) {
407 					if (group.getGroupName() == null) {
408 						matchingGroup = group;
409 					}
410 				} else {
411 					if (groupName.equals(group.getGroupName())) {
412 						matchingGroup = group;
413 					}
414 				}
415 			}
416 		}
417 	
418 		// create group if group does not already exist
419 		if (matchingGroup == null) {
420 			
421 			LOG.debug("Group '" + groupName + "' does not already exist, create it");
422 						
423 			// create group
424 			matchingGroup = new ECReportGroup();
425 			
426 			// set name
427 			matchingGroup.setGroupName(groupName);
428 			
429 			// set count
430 			if (reportSpec.getOutput().isIncludeCount()) {
431 				ECReportGroupCount groupCount = new ECReportGroupCount();
432 				groupCount.setCount(0);
433 				matchingGroup.setGroupCount(groupCount);
434 			}
435 			
436 			// create and set group list
437 			matchingGroup.setGroupList(new ECReportGroupList());
438 			
439 			// add to groups
440 			report.getGroup().add(matchingGroup);
441 			
442 		}
443 		
444 		// create group list member
445 		ECReportGroupListMember groupMember = new ECReportGroupListMember();
446 		
447 		// RAW DECIMAL
448 		TDTEngine tdt = Tag.getTDTEngine();
449 		try {
450 			if (reportSpec.getOutput().isIncludeRawDecimal()) {
451 				String bin = tag.getTagAsBinary();
452 				if (null != bin) {
453 					EPC epc = new EPC();
454 					epc.setValue(
455 							createRepresentationAsRawDecimal(
456 									bin.length(), 
457 									tdt.bin2dec(bin)));
458 					groupMember.setRawDecimal(epc);
459 				} else {
460 					throw new Exception("missing binary representation of the tag");
461 				}
462 			}
463 		} catch (Exception e) {
464 			LOG.debug("could not put tag into report as format 'RawDecimal'");
465 		}
466 		// TAG FORMAT
467 		try {
468 			if (reportSpec.getOutput().isIncludeTag()) {
469 				if (null != tag.getTagAsBinary()) {
470 					EPC epc = new EPC();
471 					epc.setValue(Tag.convert_to_TAG_ENCODING(
472 							tag.getTagLength(), 
473 							tag.getFilter(), 
474 							tag.getCompanyPrefixLength(), 
475 							tag.getTagAsBinary()));
476 					groupMember.setTag(epc);	
477 				} else {
478 					throw new Exception("missing binary representation of the tag");
479 				}
480 			}
481 		} catch (Exception e) {
482 			LOG.debug("could not put tag into report as format 'Tag'");
483 
484 			e.printStackTrace();
485 		}
486 		// RAW HEX
487 		try {
488 			if (reportSpec.getOutput().isIncludeRawHex()) {
489 				String bin = tag.getTagAsBinary();
490 				if (null != bin) {
491 					EPC epc = new EPC();
492 					epc.setValue(
493 							createRepresentationAsRawHex(
494 									bin.length(), 
495 									tdt.bin2hex(bin)));
496 					groupMember.setRawHex(epc);
497 				} else {
498 					throw new Exception("missing binary representation of the tag");
499 				}
500 			}
501 		} catch (Exception e) {
502 			LOG.debug("could not put tag into report as format 'RawHex'");
503 		}
504 		// EPC
505 		try {
506 			if (reportSpec.getOutput().isIncludeEPC()) {
507 				String bin = tag.getTagAsBinary();
508 				try {
509 					if (null != bin) {
510 						EPC epc = new EPC();
511 						epc.setValue(Tag.convert_to_PURE_IDENTITY(
512 								tag.getTagLength(), 
513 								tag.getFilter(), 
514 								tag.getCompanyPrefixLength(), 
515 								bin));
516 						groupMember.setEpc(epc);
517 					}
518 				} catch (Exception ex) {
519 					LOG.debug("extracting epc via binary failed.");
520 				}
521 				if ((null == groupMember.getEpc()) && (null != tag.getTagIDAsPureURI())) {
522 					EPC epc = new EPC();
523 					epc.setValue(tag.getTagIDAsPureURI());
524 					groupMember.setEpc(epc);
525 				}
526 			}
527 		} catch (Exception e) {
528 			LOG.debug("could not put tag into report as format 'EPC'");
529 		}
530 		
531 		// check if we need to add tag stats
532 		ECReportSpecExtension ecReportSpecExtension = reportSpec.getExtension();
533 		if ((null != ecReportSpecExtension) && 
534 				(null != ecReportSpecExtension.getStatProfileNames())) {
535 			
536 			LOG.debug("adding stat profile");
537 			addStatProfiles(
538 					tag,
539 					groupMember,
540 					ecReportSpecExtension.
541 						getStatProfileNames().getStatProfileName());
542 		}
543 		
544 		// add list member to group list
545 		List<ECReportGroupListMember> members = matchingGroup.getGroupList().getMember();		
546 		members.add(groupMember);
547 		
548 		// increment group counter
549 		if (reportSpec.getOutput().isIncludeCount()) {
550 			matchingGroup.getGroupCount().setCount(matchingGroup.getGroupCount().getCount() + 1);
551 		}
552 		
553 		LOG.debug("Tag '" + tagURI + "' successfully added to group '" + groupName + "' of report '" + name + "'");
554 		
555 	}

559 	 * @param bitStringLength the length of the bit-string (binary representation).
560 	 * @param dec the decimal representation of the tag.
561 	 * @return the formatted epc-decimal representation.
562 	 */
563 	private String createRepresentationAsRawDecimal(int bitStringLength, String dec) {
564 		return String.format("urn:epc:raw:%d.%s", 
565 				bitStringLength, 
566 				dec);

581 	/**
582 	 * for each statistics profile name add the respective statistics profile.
583 	 * @param tag the tag holding information the statistics.
584 	 * @param groupMember the group member where to add the statistics.
585 	 * @param statProfileName a list of statistic profile names.
586 	 */
587 	private void addStatProfiles(Tag tag, ECReportGroupListMember groupMember,
588 			List<String> statProfileName) {
589 
590 		ECReportGroupListMemberExtension extension = 
591 			new ECReportGroupListMemberExtension();
592 		groupMember.setExtension(extension);
593 		
594 		extension.setStats(new Stats());
595 		List<ECTagStat> ecTagStats = extension.getStats().getStat();
596 		for (String profile : statProfileName) {		
597 			LOG.debug("adding stat profile: " + profile);
598 		
599 			ECTagStat ecTagStat = new ECTagStat();
600 			ecTagStats.add(ecTagStat);
601 			
602 			ecTagStat.setProfile(profile);
603 			ecTagStat.setStatBlocks(new StatBlocks());
604 			ECReaderStat readerStat = new ECReaderStat();
605 			ecTagStat.getStatBlocks().getStatBlock().add(readerStat);
606 			
607 			readerStat.setReaderName(tag.getReader());
608 			readerStat.setSightings(new Sightings());
609 			readerStat.getSightings().getSighting().add(new ECSightingStat());
610 		}
611 	}

615 	 * 
616 	 * @param tagURI to search group for
617 	 * @return group name
618 	 * @throws ECSpecValidationException if the tag is invalid
619 	 * @throws ImplementationException if an implementation exception occurs
620 	 */
621 	private String getGroupName(String tagURI) throws ImplementationExceptionResponse, ECSpecValidationExceptionResponse {
622 		
623 		for (Pattern pattern : groupPatterns) {
624 			if (pattern.isMember(tagURI)) {
625 				return pattern.getGroupName(tagURI);
626 			}
627 		}
628 			
629 		return null;

633 	/**
634 	 * This method indicates if the report contains any tags.
635 	 * 
636 	 * @return true if the report is empty and false otherwise
637 	 */
638 	private boolean isEmpty() {
